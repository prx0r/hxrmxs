import json
import torch
from typing import List, Dict, Any, Optional
from sentence_transformers import SentenceTransformer
import chromadb
from src.utils.logging import get_logger

logger = get_logger(__name__)

class TCLRepository:
    """Enhanced repository for therapeutic components with embeddings and learning"""
    
    def __init__(self, chromadb_client, tcl_json_path: str):
        self.chromadb_client = chromadb_client
        self.tcl_json_path = tcl_json_path
        self.embedding_model = SentenceTransformer("all-MiniLM-L6-v2")
        self.vector_collection = None
        self.components = []
        
        # Effectiveness tracking for components
        self.effectiveness_scores = {}
        
    async def initialize(self):
        """Initialize the repository with components"""
        try:
            self.vector_collection = self.chromadb_client.get_or_create_collection("tcl_components")
            await self.load_tcl_components()
            logger.info(f"TCL Repository initialized with {len(self.components)} components")
        except Exception as e:
            logger.error(f"TCL Repository initialization failed: {e}")
            raise
    
    async def load_tcl_components(self):
        """Load therapeutic components with embeddings"""
        try:
            with open(self.tcl_json_path, 'r', encoding='utf-8') as f:
                data = json.load(f)
            
            for component in data.get('components', []):
                # Generate embedding for skill purpose
                skill_purpose = component.get('skill_definition', {}).get('skill_purpose', '')
                if skill_purpose:
                    embedding = self.embedding_model.encode(skill_purpose).tolist()
                    
                    # Store component with embedding
                    self.vector_collection.add(
                        embeddings=[embedding],
                        metadatas=[component],
                        ids=[component['component_id']]
                    )
                    
                    self.components.append(component)
                    
                    # Initialize effectiveness score
                    self.effectiveness_scores[component['component_id']] = {
                        'success_rate': 0.5,  # Default
                        'usage_count': 0,
                        'feedback_scores': []
                    }
            
            logger.info(f"Loaded {len(self.components)} TCL components with embeddings")
            
        except Exception as e:
            logger.error(f"Failed to load TCL components: {e}")
            raise
    
    async def get_recommended_components(self, query: str, user_cluster: str = None, 
                                       n_results: int = 3) -> List[Dict[str, Any]]:
        """Get components with effectiveness scoring for user cluster"""
        try:
            # Generate query embedding
            query_embedding = self.embedding_model.encode(query).tolist()
            
            # Retrieve similar components
            results = self.vector_collection.query(
                query_embeddings=[query_embedding],
                n_results=n_results * 2,  # Get more for filtering
                include=["metadatas", "distances"]
            )
            
            enhanced_components = []
            if results and results.get("metadatas"):
                for i, metadata in enumerate(results["metadatas"][0]):
                    component_id = metadata["component_id"]
                    distance = results["distances"][0][i]
                    
                    # Get effectiveness for this component
                    effectiveness = self.effectiveness_scores.get(component_id, {})
                    
                    # Calculate combined score (similarity + effectiveness)
                    similarity_score = 1.0 - distance
                    effectiveness_score = effectiveness.get('success_rate', 0.5)
                    combined_score = 0.7 * similarity_score + 0.3 * effectiveness_score
                    
                    enhanced_component = {
                        **metadata,
                        "similarity_score": similarity_score,
                        "effectiveness_score": effectiveness_score,
                        "combined_score": combined_score,
                        "usage_count": effectiveness.get('usage_count', 0)
                    }
                    enhanced_components.append(enhanced_component)
            
            # Sort by combined score and return top results
            enhanced_components.sort(key=lambda x: x["combined_score"], reverse=True)
            return enhanced_components[:n_results]
            
        except Exception as e:
            logger.error(f"Failed to get recommended components: {e}")
            return []
    
    async def update_component_effectiveness(self, component_id: str, 
                                           success: bool, feedback_score: float = None):
        """Update effectiveness metrics for a component"""
        try:
            if component_id not in self.effectiveness_scores:
                self.effectiveness_scores[component_id] = {
                    'success_rate': 0.5,
                    'usage_count': 0,
                    'feedback_scores': []
                }
            
            effectiveness = self.effectiveness_scores[component_id]
            effectiveness['usage_count'] += 1
            
            if feedback_score is not None:
                effectiveness['feedback_scores'].append(feedback_score)
                # Update success rate based on feedback
                avg_feedback = sum(effectiveness['feedback_scores']) / len(effectiveness['feedback_scores'])
                effectiveness['success_rate'] = avg_feedback / 10.0  # Normalize to 0-1
            elif success is not None:
                # Simple success/failure update
                current_rate = effectiveness['success_rate']
                count = effectiveness['usage_count']
                # Exponential moving average
                alpha = 0.1
                new_rate = alpha * (1.0 if success else 0.0) + (1 - alpha) * current_rate
                effectiveness['success_rate'] = new_rate
            
            logger.info(f"Updated effectiveness for {component_id}: {effectiveness}")
            
        except Exception as e:
            logger.error(f"Failed to update component effectiveness: {e}")

import numpy as np
import torch
from sklearn.cluster import HDBSCAN
from sklearn.metrics.pairwise import cosine_similarity
from typing import List, Dict, Any, Optional, Tuple
from sqlalchemy.ext.asyncio import AsyncSession

class SimilarUserPatternService:
    """Discovers patterns across similar user profiles for therapy prediction"""
    
    def __init__(self, db_session_factory):
        self.db_session_factory = db_session_factory
        self.clusterer = HDBSCAN(min_cluster_size=50, metric="cosine")
        self.user_clusters = {}  # user_id -> cluster_id
        self.cluster_preferences = {}  # cluster_id -> therapy preferences
        self.user_vectors_cache = {}  # user_id -> vector
        
    async def find_similar_profiles(self, user_vector: torch.Tensor, 
                                  user_id: str = None, n_similar: int = 10) -> List[Dict[str, Any]]:
        """Find users with similar personality profiles and successful outcomes"""
        try:
            async with self.db_session_factory() as session:
                # Get all user vectors from cache or compute
                if not self.user_vectors_cache:
                    await self._build_user_vector_cache(session)
                
                # Convert query vector to numpy
                query_vector = user_vector.cpu().numpy().reshape(1, -1)
                
                # Find similar users
                similar_users = []
                for uid, vector in self.user_vectors_cache.items():
                    if uid != user_id:  # Exclude self
                        similarity = cosine_similarity(query_vector, vector.reshape(1, -1))[0][0]
                        similar_users.append({
                            'user_id': uid,
                            'similarity': similarity,
                            'cluster_id': self.user_clusters.get(uid),
                            'vector': vector
                        })
                
                # Sort by similarity and return top N
                similar_users.sort(key=lambda x: x['similarity'], reverse=True)
                return similar_users[:n_similar]
                
        except Exception as e:
            logger.error(f"Failed to find similar profiles: {e}")
            return []
    
    async def get_cluster_therapy_preferences(self, user_vector: torch.Tensor) -> Dict[str, float]:
        """Get therapy preferences based on user cluster"""
        try:
            # Find user's cluster
            cluster_id = await self._predict_user_cluster(user_vector)
            
            if cluster_id in self.cluster_preferences:
                return self.cluster_preferences[cluster_id]
            else:
                # Return default preferences
                return {
                    'cognitive_restructuring': 0.5,
                    'emotional_support': 0.7,
                    'behavioral_guidance': 0.4,
                    'crisis_intervention': 0.3
                }
                
        except Exception as e:
            logger.error(f"Failed to get cluster preferences: {e}")
            return {}
    
    async def update_cluster_preferences(self, outcome_data: List[Dict[str, Any]]):
        """Update cluster-based therapy preferences from successful outcomes"""
        try:
            # Group outcomes by cluster
            cluster_outcomes = {}
            for outcome in outcome_data:
                cluster_id = outcome.get('cluster_id')
                if cluster_id is not None:
                    if cluster_id not in cluster_outcomes:
                        cluster_outcomes[cluster_id] = []
                    cluster_outcomes[cluster_id].append(outcome)
            
            # Update preferences for each cluster
            for cluster_id, outcomes in cluster_outcomes.items():
                preferences = self._calculate_cluster_preferences(outcomes)
                self.cluster_preferences[cluster_id] = preferences
                
            logger.info(f"Updated preferences for {len(cluster_outcomes)} clusters")
            
        except Exception as e:
            logger.error(f"Failed to update cluster preferences: {e}")
    
    async def _build_user_vector_cache(self, session: AsyncSession):
        """Build cache of user vectors"""
        try:
            # This would integrate with DMPVN service to get user vectors
            # For now, simulate with random vectors
            users = await session.execute(select(User.id))
            user_ids = [row[0] for row in users.fetchall()]
            
            for user_id in user_ids:
                # Simulate user vector (32-dimensional)
                vector = np.random.rand(32)
                self.user_vectors_cache[user_id] = vector
            
            # Perform clustering
            if len(self.user_vectors_cache) >= 50:
                vectors = np.array(list(self.user_vectors_cache.values()))
                cluster_labels = self.clusterer.fit_predict(vectors)
                
                for i, user_id in enumerate(self.user_vectors_cache.keys()):
                    self.user_clusters[user_id] = cluster_labels[i]
            
            logger.info(f"Built user vector cache with {len(self.user_vectors_cache)} users")
            
        except Exception as e:
            logger.error(f"Failed to build user vector cache: {e}")
    
    async def _predict_user_cluster(self, user_vector: torch.Tensor) -> int:
        """Predict cluster for new user vector"""
        try:
            # Find closest cluster centroid
            query_vector = user_vector.cpu().numpy()
            
            # Calculate cluster centroids
            cluster_centroids = {}
            for user_id, cluster_id in self.user_clusters.items():
                if cluster_id not in cluster_centroids:
                    cluster_centroids[cluster_id] = []
                cluster_centroids[cluster_id].append(self.user_vectors_cache[user_id])
            
            # Find closest centroid
            best_cluster = 0
            best_similarity = -1
            
            for cluster_id, vectors in cluster_centroids.items():
                centroid = np.mean(vectors, axis=0)
                similarity = cosine_similarity(query_vector.reshape(1, -1), centroid.reshape(1, -1))[0][0]
                
                if similarity > best_similarity:
                    best_similarity = similarity
                    best_cluster = cluster_id
            
            return best_cluster
            
        except Exception as e:
            logger.error(f"Failed to predict user cluster: {e}")
            return 0  # Default cluster
    
    def _calculate_cluster_preferences(self, outcomes: List[Dict[str, Any]]) -> Dict[str, float]:
        """Calculate therapy preferences from cluster outcomes"""
        preferences = {}
        
        # Aggregate successful therapy types
        therapy_success = {}
        for outcome in outcomes:
            therapy_type = outcome.get('therapy_type')
            success = outcome.get('success', False)
            
            if therapy_type:
                if therapy_type not in therapy_success:
                    therapy_success[therapy_type] = {'success': 0, 'total': 0}
                
                therapy_success[therapy_type]['total'] += 1
                if success:
                    therapy_success[therapy_type]['success'] += 1
        
        # Calculate preference scores
        for therapy_type, stats in therapy_success.items():
            if stats['total'] > 0:
                preferences[therapy_type] = stats['success'] / stats['total']
            else:
                preferences[therapy_type] = 0.5  # Default
        
        return preferences

# src/discovery/juncture_detector.py
import torch
import torch.nn as nn
from transformers import AutoModel, AutoTokenizer
from typing import Dict, Any
from src.utils.logging import get_logger

logger = get_logger(__name__)

class JunctureDetector(nn.Module):
    """Identifies therapeutically significant moments for prioritized storage"""
    
    def __init__(self, model_name: str = "cardiffnlp/tweet-topic-21-multi"):
        super().__init__()
        self.model_name = model_name
        
        try:
            self.bert_model = AutoModel.from_pretrained(model_name)
            self.tokenizer = AutoTokenizer.from_pretrained(model_name)
            
            # Classification layers
            self.classifier = nn.Sequential(
                nn.Linear(768 + 4, 256),  # BERT embedding + user state
                nn.ReLU(),
                nn.Dropout(0.1),
                nn.Linear(256, 64),
                nn.ReLU(),
                nn.Dropout(0.1),
                nn.Linear(64, 1),
                nn.Sigmoid()
            )
            
            self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
            self.to(self.device)
            
            # Set to eval mode (would load trained weights in production)
            self.eval()
            
            logger.info(f"JunctureDetector initialized with {model_name}")
            
        except Exception as e:
            logger.error(f"Failed to initialize JunctureDetector: {e}")
            # Fallback to simple keyword-based detection
            self.bert_model = None
            self.tokenizer = None
            self.classifier = None
    
    def forward(self, text: str, user_state: Dict[str, float]) -> float:
        """Predict therapeutic significance score (0-1)"""
        try:
            if self.bert_model is None:
                # Fallback to keyword-based detection
                return self._keyword_based_detection(text, user_state)
            
            # Tokenize text
            inputs = self.tokenizer(
                text, 
                return_tensors="pt", 
                truncation=True, 
                padding=True, 
                max_length=512
            ).to(self.device)
            
            # Get BERT embeddings
            with torch.no_grad():
                bert_output = self.bert_model(**inputs)
                text_embedding = bert_output.last_hidden_state.mean(dim=1)  # [1, 768]
            
            # Prepare user state tensor
            state_tensor = torch.tensor([
                user_state.get('valence', 0.5),
                user_state.get('arousal', 0.5),
                user_state.get('dominance', 0.5),
                user_state.get('crisis_indicators', 0.0)
            ], device=self.device).unsqueeze(0)  # [1, 4]
            
            # Combine embeddings
            combined_input = torch.cat([text_embedding, state_tensor], dim=1)
            
            # Predict significance
            with torch.no_grad():
                significance_score = self.classifier(combined_input).item()
            
            return significance_score
            
        except Exception as e:
            logger.error(f"JunctureDetector forward pass failed: {e}")
            return self._keyword_based_detection(text, user_state)
    
    def _keyword_based_detection(self, text: str, user_state: Dict[str, float]) -> float:
        """Fallback keyword-based significance detection"""
        try:
            text_lower = text.lower()
            
            # Crisis keywords (high significance)
            crisis_keywords = [
                "help", "crisis", "emergency", "suicide", "kill", "die", "death",
                "hurt", "harm", "can't cope", "give up", "hopeless", "worthless"
            ]
            
            # Breakthrough keywords (medium-high significance)
            breakthrough_keywords = [
                "realize", "understand", "breakthrough", "clarity", "insight",
                "finally", "suddenly", "changed", "different"
            ]
            
            # Emotional intensity keywords (medium significance)
            intensity_keywords = [
                "overwhelming", "intense", "powerful", "deep", "profound",
                "scared", "angry", "devastated", "elated", "ecstatic"
            ]
            
            # Count keyword matches
            crisis_count = sum(1 for kw in crisis_keywords if kw in text_lower)
            breakthrough_count = sum(1 for kw in breakthrough_keywords if kw in text_lower)
            intensity_count = sum(1 for kw in intensity_keywords if kw in text_lower)
            
            # Calculate significance score
            score = 0.0
            score += crisis_count * 0.3
            score += breakthrough_count * 0.2
            score += intensity_count * 0.1
            
            # Factor in user state
            if user_state.get('arousal', 0.5) > 0.7:  # High arousal
                score += 0.1
            if user_state.get('valence', 0.5) < 0.3:  # Negative valence
                score += 0.1
            
            return min(score, 1.0)
            
        except Exception as e:
            logger.error(f"Keyword-based detection failed: {e}")
            return 0.5  # Default significance

    def detect_juncture_category(self, significance_score: float) -> str:
        """Categorize juncture based on significance score"""
        if significance_score >= 0.8:
            return "crisis"
        elif significance_score >= 0.6:
            return "pivotal"
        elif significance_score >= 0.3:
            return "contextual"
        else:
            return "ephemeral"

class TherapeuticDiscoveryEngine:
    """Mines successful therapeutic combinations from user interactions"""
    
    def __init__(self, db_session_factory):
        self.db_session_factory = db_session_factory
        self.discovered_combinations = {}
        self.validation_threshold = 0.05  # p-value threshold
        
    async def discover_novel_combinations(self, min_support: int = 10) -> List[Dict[str, Any]]:
        """Find frequent successful framework combinations"""
        try:
            async with self.db_session_factory() as session:
                # Get successful therapeutic interactions
                successful_interactions = await self._get_successful_interactions(session)
                
                if len(successful_interactions) < min_support:
                    logger.warning(f"Insufficient data for discovery: {len(successful_interactions)} interactions")
                    return []
                
                # Extract therapy combinations
                therapy_combinations = self._extract_therapy_combinations(successful_interactions)
                
                # Apply market basket analysis
                frequent_combinations = self._market_basket_analysis(
                    therapy_combinations, min_support
                )
                
                # Validate combinations
                validated_combinations = []
                for combination in frequent_combinations:
                    if await self._validate_combination_significance(combination, session):
                        validated_combinations.append(combination)
                
                logger.info(f"Discovered {len(validated_combinations)} validated combinations")
                return validated_combinations
                
        except Exception as e:
            logger.error(f"Novel combination discovery failed: {e}")
            return []
    
    async def validate_combination_effectiveness(self, combination: List[str], 
                                               baseline_effectiveness: float) -> Dict[str, Any]:
        """Statistically validate new therapeutic combinations"""
        try:
            async with self.db_session_factory() as session:

                combination_outcomes = await self._get_combination_outcomes(combination, session)
                
                if len(combination_outcomes) < 10:
                    return {
                        'validated': False,
                        'reason': 'insufficient_data',
                        'sample_size': len(combination_outcomes)
                    }
                
                # Calculate effectiveness metrics
                success_rate = sum(outcome['success'] for outcome in combination_outcomes) / len(combination_outcomes)
                
                # Cohen's d effect size calculation
                successes = [1 if outcome['success'] else 0 for outcome in combination_outcomes]
                effect_size = self._calculate_cohens_d(successes, baseline_effectiveness)
                
                # Statistical significance test (chi-square)
                observed_successes = sum(successes)
                expected_successes = len(successes) * baseline_effectiveness
                
                # Chi-square test
                observed = [observed_successes, len(successes) - observed_successes]
                expected = [expected_successes, len(successes) - expected_successes]
                
                if expected[0] > 5 and expected[1] > 5:  # Chi-square validity check
                    chi2_stat, p_value = chi2_contingency([observed, expected])[:2]
                else:
                    p_value = 1.0  # Not enough data for chi-square
                
                # Bootstrap confidence intervals
                confidence_interval = self._bootstrap_confidence_interval(successes)
                
                validation_result = {
                    'validated': p_value < self.validation_threshold and effect_size > 0.2,
                    'success_rate': success_rate,
                    'baseline_rate': baseline_effectiveness,
                    'effect_size': effect_size,
                    'p_value': p_value,
                    'confidence_interval': confidence_interval,
                    'sample_size': len(combination_outcomes)
                }
                
                logger.info(f"Validation result for {combination}: {validation_result}")

    def _market_basket_analysis(self, combinations: List[List[str]], min_support: int) -> List[Dict[str, Any]]:
        """Apply market basket analysis to find frequent combinations"""
        # Count combination frequencies
        combination_counts = Counter(tuple(combo) for combo in combinations)
        
        # Filter by minimum support
        frequent_combinations = []
        total_transactions = len(combinations)
        
        for combo, count in combination_counts.items():
            if count >= min_support:
                support = count / total_transactions
                frequent_combinations.append({
                    'combination': list(combo),
                    'support': support,
                    'frequency': count,
                    'total_transactions': total_transactions
                })
        
        # Sort by support
        frequent_combinations.sort(key=lambda x: x['support'], reverse=True)
        return frequent_combinations

    def _calculate_cohens_d(self, sample: List[int], population_mean: float) -> float:
        """Calculate Cohen's d effect size"""
        try:
            sample_mean = np.mean(sample)
            sample_std = np.std(sample, ddof=1)
            
            if sample_std == 0:
                return 0.0
            
            cohens_d = (sample_mean - population_mean) / sample_std
            return cohens_d
            
        except Exception as e:
            logger.error(f"Cohen's d calculation failed: {e}")
            return 0.0
    
    def _bootstrap_confidence_interval(self, sample: List[int], 
                                     confidence: float = 0.95, 
                                     n_bootstrap: int = 1000) -> Tuple[float, float]:
        """Calculate bootstrap confidence interval"""
        try:
            bootstrap_means = []
            
            for _ in range(n_bootstrap):
                bootstrap_sample = np.random.choice(sample, size=len(sample), replace=True)
                bootstrap_means.append(np.mean(bootstrap_sample))
            
            alpha = 1 - confidence
            lower_percentile = (alpha / 2) * 100
            upper_percentile = (1 - alpha / 2) * 100
            
            lower_bound = np.percentile(bootstrap_means, lower_percentile)
            upper_bound = np.percentile(bootstrap_means, upper_percentile)
            
            return (lower_bound, upper_bound)
            
        except Exception as e:
            logger.error(f"Bootstrap CI calculation failed: {e}")
            return (0.0, 1.0)

            reverse=True
        )

        new_score = (1 - learning_rate) * current_score + learning_rate * score
        self.therapy_preferences[therapy_type] = new_score
