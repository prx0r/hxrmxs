def create_parser_from_config(config_class):
    """Create an argument parser from a configuration dataclass."""
    import argparse
    from dataclasses import fields
    
    parser = argparse.ArgumentParser(description=config_class.__doc__)
    for field in fields(config_class):
        field_type = field.type
        if hasattr(field_type, "__origin__") and field_type.__origin__ is Union:
            # Handle Optional types
            field_type = field_type.__args__[0]
        
        # Special handling for List, Dict, Tuple
        if hasattr(field_type, "__origin__"):
            if field_type.__origin__ is list:
                field_type = str  # Treat lists as comma-separated strings
            elif field_type.__origin__ is dict:
                field_type = str  # Treat dicts as json strings
            elif field_type.__origin__ is tuple:
                field_type = str  # Treat tuples as comma-separated strings
        
        parser.add_argument(
            f"--{field.name}",
            type=field_type,
            default=field.default,
            help=f"{field.name} parameter"
        )
    
    return parser

@dataclass
class DNAConstraints:
    """Constraints for DNA sequence generation."""
    gc_min: float = 0.4           # Minimum GC content
    gc_max: float = 0.6           # Maximum GC content
    max_homopolymer: int = 4      # Maximum homopolymer length (e.g., AAAA)
    primer_length: int = 23       # Length of primers
    target_density: float = 1.5   # Target information density in bits/nt
    salt_bits: int = 8            # Number of bits for salt in HEDGES
    max_length: int = 1000        # Maximum sequence length

class DNAEncoder:
    """DNA sequence encoding and manipulation with HEDGES constraints."""
    
    def __init__(self, config, device=None):
        """
        Initialize the DNA encoder.
        
        Args:
            config: Configuration object with DNA parameters
            device: Torch device to use (default: None, uses config.device)
        """
        self.config = config
        self.device = device or config.device
        
        # Initialize constraints
        self.constraints = DNAConstraints(
            gc_min=config.gc_min,
            gc_max=config.gc_max,
            max_homopolymer=config.max_homopolymer,
            primer_length=config.primer_length,
            target_density=config.target_density,
            salt_bits=config.salt_bits
        )
        
        # Initialize Evo2 model if available
        if EVO2_AVAILABLE:
            logger.info(f"Initializing Evo2 model: {config.evo2_model_name}")
            self.evo2_model = Evo2(config.evo2_model_name).to(self.device)
        else:
            logger.warning("Using mock Evo2 model")
            self.evo2_model = MockEvo2(config.evo2_model_name)
        
        # Cache for generated sequences to avoid regeneration
        self.sequence_cache = {}
    
    def generate_sequence(self, prompt_seq: str, n_tokens: int = 400, temperature: float = 1.0, top_k: int = 4) -> str:
        """
        Generate a DNA sequence using Evo-2 1B model.
        
        Args:
            prompt_seq: Prompt sequence for the model
            n_tokens: Number of tokens to generate
            temperature: Sampling temperature
            top_k: Top-k sampling parameter
            
        Returns:
            Generated DNA sequence
        """
        # Check cache first
        cache_key = f"{prompt_seq}_{n_tokens}_{temperature}_{top_k}"
        if cache_key in self.sequence_cache:
            logger.debug(f"Using cached sequence for: {prompt_seq[:20]}...")
            return self.sequence_cache[cache_key]
        
        logger.info(f"Generating sequence with prompt: {prompt_seq[:20]}...")
        output = self.evo2_model.generate(
            prompt_seqs=[prompt_seq], 
            n_tokens=n_tokens, 
            temperature=temperature, 
            top_k=top_k
        )
        sequence = output.sequences[0]
        
        # Apply HEDGES encoding
        encoded_sequence = self.hedges_encode(sequence)
        
        # Cache the result
        self.sequence_cache[cache_key] = encoded_sequence
        
        return encoded_sequence
    
    def validate_sequence(self, sequence: str) -> bool:
        """
        Validate a DNA sequence against HEDGES constraints.
        
        Args:
            sequence: DNA sequence to validate
            
        Returns:
            True if the sequence meets all constraints, False otherwise
        """
        if not sequence or not all(c in 'ACGT' for c in sequence):
            logger.debug("Invalid characters in sequence")
            return False
        
        # Check GC content
        gc_count = sum(1 for c in sequence if c in 'GC')
        gc_ratio = gc_count / len(sequence)
        if not (self.constraints.gc_min <= gc_ratio <= self.constraints.gc_max):
            logger.debug(f"GC ratio {gc_ratio:.2f} outside range [{self.constraints.gc_min:.2f}, {self.constraints.gc_max:.2f}]")
            return False
        
        # Check homopolymer runs
        prev_nt = None
        run_length = 1
        for nt in sequence:
            if nt == prev_nt:
                run_length += 1
                if run_length > self.constraints.max_homopolymer:
                    logger.debug(f"Homopolymer run of {run_length} exceeds maximum {self.constraints.max_homopolymer}")
                    return False
            else:
                run_length = 1
            prev_nt = nt
        
        return True
    
    def fix_sequence(self, sequence: str) -> str:
        """
        Fix a DNA sequence to meet HEDGES constraints.
        
        Args:
            sequence: DNA sequence to fix
            
        Returns:
            Fixed DNA sequence
        """
        if not sequence:
            return ""
        
        # Fix invalid characters
        sequence = ''.join(c if c in 'ACGT' else np.random.choice(['A', 'C', 'G', 'T']) for c in sequence)
        
        # Fix homopolymer runs
        fixed_seq = ""
        prev_nt = None
        run_length = 1
        
        for i, nt in enumerate(sequence):
            if nt == prev_nt:
                run_length += 1
                if run_length > self.constraints.max_homopolymer:
                    replacements = [n for n in 'ACGT' if n != prev_nt]
                    nt = replacements[i % len(replacements)]
                    run_length = 1
            else:
                run_length = 1
            fixed_seq += nt
            prev_nt = nt
        
        # Fix GC content
        gc_count = sum(1 for c in fixed_seq if c in 'GC')
        gc_ratio = gc_count / len(fixed_seq)
        
        if gc_ratio < self.constraints.gc_min:
            # Increase GC content
            needed = int((self.constraints.gc_min - gc_ratio) * len(fixed_seq)) + 1
            at_positions = [i for i, c in enumerate(fixed_seq) if c in 'AT']
            if at_positions:
                to_replace = np.random.choice(at_positions, size=min(needed, len(at_positions)), replace=False)
                fixed_seq = list(fixed_seq)
                for pos in to_replace:
                    fixed_seq[pos] = np.random.choice(['G', 'C'])
                fixed_seq = ''.join(fixed_seq)
        
        elif gc_ratio > self.constraints.gc_max:
            # Decrease GC content
            needed = int((gc_ratio - self.constraints.gc_max) * len(fixed_seq)) + 1
            gc_positions = [i for i, c in enumerate(fixed_seq) if c in 'GC']
            if gc_positions:
                to_replace = np.random.choice(gc_positions, size=min(needed, len(gc_positions)), replace=False)
                fixed_seq = list(fixed_seq)
                for pos in to_replace:
                    fixed_seq[pos] = np.random.choice(['A', 'T'])
                fixed_seq = ''.join(fixed_seq)
        
        return fixed_seq
    
    def hedges_encode(self, sequence: str) -> str:
        """
        Encode information in a DNA sequence using HEDGES algorithm.
        
        HEDGES (High-Efficiency DNA-based GPU-Encoded Storage) uses a salted hash
        to modify a bitstream for robust DNA storage.
        
        Args:
            sequence: Original DNA sequence
            
        Returns:
            HEDGES-encoded DNA sequence
        """
        # Convert sequence to bits (simplified)
        encoding = {'A': '00', 'C': '01', 'G': '10', 'T': '11'}
        bits = [int(b) for nt in sequence for b in encoding.get(nt, '00')]
        
        # Generate salt
        salt = np.random.randint(0, 2**self.constraints.salt_bits, dtype=np.uint8).tobytes()
        
        # Encode using HEDGES algorithm
        encoded = ""
        for i in range(0, len(bits), 2):
            if i+1 >= len(bits):
                bit_pair = [bits[i], 0]  # Pad with 0 if needed
            else:
                bit_pair = [bits[i], bits[i+1]]
            
            index = i.to_bytes(4, 'big')
            prev = bits[max(0, i-8):i] if i > 0 else []
            prev_bytes = bytes([sum(prev[j:j+8]) for j in range(0, len(prev), 8)]) if prev else b''
            
            hash_input = salt + index + prev_bytes
            hash_val = int(sha256(hash_input).hexdigest(), 16) % 4
            
            encoded_val = (bit_pair[0] * 2 + bit_pair[1] + hash_val) % 4
            encoded += ['A', 'C', 'G', 'T'][encoded_val]
        
        # Add primers
        primer_5 = ''.join(np.random.choice(['A', 'C', 'G', 'T']) for _ in range(self.constraints.primer_length))
        primer_3 = ''.join(np.random.choice(['A', 'C', 'G', 'T']) for _ in range(self.constraints.primer_length))
        
        full_sequence = primer_5 + encoded + primer_3
        
        # Validate and fix if necessary
        if not self.validate_sequence(full_sequence):
            full_sequence = self.fix_sequence(full_sequence)
        
        return full_sequence
    
    def hedges_decode(self, encoded_sequence: str) -> str:
        """
        Decode a HEDGES-encoded DNA sequence.
        
        Note: This is a simplified decoder for demonstration. In a real system,
        you would need to know the original salt used for encoding.
        
        Args:
            encoded_sequence: HEDGES-encoded DNA sequence
            
        Returns:
            Approximately decoded sequence
        """
        # Remove primers
        encoded = encoded_sequence[self.constraints.primer_length:-self.constraints.primer_length]
        
        # Simplified decoding (approximate without knowing the salt)
        decoding = {'A': 'A', 'C': 'C', 'G': 'G', 'T': 'T'}
        
        return ''.join(decoding[nt] for nt in encoded)
    
    def get_information_density(self, sequence: str) -> float:
        """
        Calculate the information density of a DNA sequence in bits/nt.
        
        Args:
            sequence: DNA sequence
            
        Returns:
            Information density in bits per nucleotide
        """
        if not sequence:
            return 0.0
        
        # Simple Shannon entropy calculation
        nucleotides = ['A', 'C', 'G', 'T']
        counts = {nt: sequence.count(nt) for nt in nucleotides}
        total = sum(counts.values())
        
        if total == 0:
            return 0.0
        
        entropy = 0.0
        for count in counts.values():
            if count > 0:
                p = count / total
                entropy -= p * np.log2(p)
        
        # Adjust for HEDGES encoding overhead
        adjusted_density = entropy * (1 - (2 * self.constraints.primer_length / len(sequence)))
        
        return adjusted_density
    
    def generate_architecture_dna(self, architecture_params: Dict) -> str:
        """
        Generate DNA sequence encoding a neural architecture.
        
        Args:
            architecture_params: Dictionary of architecture parameters
            
        Returns:
            DNA sequence encoding the architecture
        """
        # Convert architecture parameters to a prompt string
        param_strings = []
        for key, value in sorted(architecture_params.items()):
            param_strings.append(f"{key}:{value}")
        
        prompt = "Architecture: " + ", ".join(param_strings)
        
        # Generate DNA encoding of the architecture
        return self.generate_sequence(prompt, n_tokens=200)
    
    def save_sequence(self, sequence: str, metadata: Dict, filepath: str) -> None:
        """
        Save a DNA sequence with metadata to a file.
        
        Args:
            sequence: DNA sequence
            metadata: Dictionary of metadata
            filepath: Path to save the sequence
        """
        data = {
            "sequence": sequence,
            "metadata": metadata,
            "information_density": self.get_information_density(sequence),
            "length": len(sequence),
            "gc_content": sum(1 for c in sequence if c in 'GC') / len(sequence) if sequence else 0
        }
        
        with open(filepath, 'w') as f:
            json.dump(data, f, indent=2)
    
    def load_sequence(self, filepath: str) -> Tuple[str, Dict]:
        """
        Load a DNA sequence with metadata from a file.
        
        Args:
            filepath: Path to load the sequence from
            
        Returns:
            Tuple of (sequence, metadata)
        """
        with open(filepath, 'r') as f:
            data = json.load(f)
        
        return data["sequence"], data["metadata"]

                 complex_valued: bool = True):
        """
        Initialize a Liquid ODE component.
        
        Args:
            state_size: Dimension of the state vectors
            hidden_size: Dimension of hidden representation (default: state_size)
            tau: Time constant for exponential decay
            activation: Activation function ('tanh', 'relu', 'sigmoid', 'gelu')
            complex_valued: Whether to use complex-valued weights
        """
        super().__init__()
        self.state_size = state_size
        self.hidden_size = hidden_size if hidden_size is not None else state_size
        self.tau = nn.Parameter(torch.tensor(tau))
        self.complex_valued = complex_valued
        
        # Select activation function
        if activation == 'tanh':
            self.activation = torch.tanh
        elif activation == 'relu':
            self.activation = F.relu
        elif activation == 'sigmoid':
            self.activation = torch.sigmoid
        elif activation == 'gelu':
            self.activation = F.gelu
        else:
            raise ValueError(f"Unsupported activation function: {activation}")
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Neural ODE weights
        self.W = nn.Parameter(torch.randn(self.hidden_size, self.state_size, dtype=dtype) * 0.01)
        self.bias = nn.Parameter(torch.zeros(self.hidden_size, dtype=dtype))
        
        # Recurrent connection weights
        self.R = nn.Parameter(torch.randn(self.hidden_size, self.hidden_size, dtype=dtype) * 0.01)
        
        # Input projection for external inputs
        self.U = nn.Parameter(torch.randn(self.hidden_size, self.state_size, dtype=dtype) * 0.01)
        
        # Damping factor
        self.damping = nn.Parameter(torch.ones(self.hidden_size, dtype=dtype) * 0.1)
        
    def forward(self, x: torch.Tensor, h: Optional[torch.Tensor] = None, dt: float = 0.1) -> Tuple[torch.Tensor, torch.Tensor]:
        """
        Compute the next state using liquid neural network dynamics.
        
        Args:
            x: Input tensor of shape (batch_size, state_size)
            h: Previous hidden state (batch_size, hidden_size) or None for zeros
            dt: Time step for numerical integration
            
        Returns:
            Tuple of (new_hidden_state, output)
        """
        batch_size = x.shape[0]
        
        # Initialize hidden state if None
        if h is None:
            dtype = torch.complex64 if self.complex_valued else torch.float32
            h = torch.zeros(batch_size, self.hidden_size, dtype=dtype, device=x.device)
        
        # Make input complex if using complex mode
        if self.complex_valued and not torch.is_complex(x):
            x = x.to(torch.complex64)
        
        # Compute ODE dynamics (dh/dt)
        input_projection = torch.matmul(x, self.U.t())
        recurrent_influence = torch.matmul(h, self.R.t())
        damping_term = -self.damping * h / self.tau
        
        # Full dynamics equation: dh/dt = -h/tau + activation(W·x + R·h + bias) + damping_term
        dh = (-h / self.tau + 
              self.activation(input_projection + recurrent_influence + self.bias) + 
              damping_term)
        
        # Integrate using Euler method: h_new = h + dh * dt
        h_new = h + dh * dt
        
        # Project to output space
        output = torch.matmul(h_new, self.W)
        
        # If complex-valued, return real part for output
        if self.complex_valued:
            output = output.real
            
        return h_new, output
    
    def forward_euler_steps(self, x: torch.Tensor, h: Optional[torch.Tensor] = None, 
                           steps: int = 10, dt: float = 0.1) -> Tuple[torch.Tensor, torch.Tensor]:
        """
        Run multiple Euler integration steps.
        
        Args:
            x: Input tensor of shape (batch_size, state_size)
            h: Previous hidden state or None
            steps: Number of integration steps
            dt: Time step for each integration
            
        Returns:
            Tuple of (final_hidden_state, final_output)
        """
        curr_h = h
        curr_x = x
        
        for _ in range(steps):
            curr_h, curr_x = self.forward(curr_x, curr_h, dt)
            
        return curr_h, curr_x


class HarmonicFunction(nn.Module):
    """
    Implements harmonic function approximation using holomorphic functions.
    
    This module enforces harmonicity by using complex-valued operations that
    satisfy the Cauchy-Riemann equations, ensuring the resulting function
    is a valid solution to Laplace's equation.
    """
    
    def __init__(self, 
                 input_dim: int,
                 output_dim: int,
                 hidden_dim: int = 128,
                 num_layers: int = 3,
                 activation: str = 'celu'):
        """
        Initialize a Harmonic Function approximator.
        
        Args:
            input_dim: Dimension of input features
            output_dim: Dimension of output
            hidden_dim: Dimension of hidden layers
            num_layers: Number of hidden layers
            activation: Activation function ('celu', 'crelu', 'tanh')
        """
        super().__init__()
        self.input_dim = input_dim
        self.output_dim = output_dim
        self.hidden_dim = hidden_dim
        self.num_layers = num_layers
        
        # Input projection to complex domain
        self.input_proj = nn.Linear(input_dim, hidden_dim, dtype=torch.complex64)
        
        # Hidden layers with complex weights
        self.hidden_layers = nn.ModuleList([
            nn.Linear(hidden_dim, hidden_dim, dtype=torch.complex64)
            for _ in range(num_layers)
        ])
        
        # Output projection
        self.output_proj = nn.Linear(hidden_dim, output_dim, dtype=torch.complex64)
        
        # Select activation function (must be holomorphic for harmonicity)
        if activation == 'celu':
            self.activation = self._complex_elu
        elif activation == 'crelu':
            self.activation = self._complex_relu
        elif activation == 'tanh':
            self.activation = torch.tanh
        else:
            raise ValueError(f"Unsupported activation: {activation}")
    
    def _complex_elu(self, x: torch.Tensor) -> torch.Tensor:
        """Complex ELU activation (approximately holomorphic)."""
        return F.elu(x.real) + 1j * F.elu(x.imag)
    
    def _complex_relu(self, x: torch.Tensor) -> torch.Tensor:
        """Complex ReLU activation (not holomorphic, but useful in practice)."""
        return F.relu(x.real) + 1j * F.relu(x.imag)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Forward pass through the harmonic function approximator.
        
        Args:
            x: Input tensor (batch_size, input_dim)
            
        Returns:
            Complex-valued output tensor (batch_size, output_dim)
        """
        # Convert to complex if needed
        if not torch.is_complex(x):
            x = torch.complex(x, torch.zeros_like(x))
        
        # Input projection
        h = self.input_proj(x)
        
        # Hidden layers
        for layer in self.hidden_layers:
            h = self.activation(layer(h))
        
        # Output projection
        output = self.output_proj(h)
        
        return output

class HPULayer(nn.Module):
    """
    Layer of Harmonic Processing Units with graph connectivity.
    
    This layer connects multiple HPUs in a graph structure, allowing for
    communication and coordination between units.
    """
    
    def __init__(self,
                num_units: int = 8,
                unit_dim: int = 17,
                sparsity: float = 0.2,
                bioelectric_kappa: float = 0.1,
                liquid_tau: float = 1.0,
                or_frequency: float = 40.0):
        """
        Initialize an HPU layer.
        
        Args:
            num_units: Number of HPUs in the layer
            unit_dim: Dimension of each HPU
            sparsity: Connection sparsity (0.0 = fully connected, 1.0 = no connections)
            bioelectric_kappa: Scale factor for bioelectric potential
            liquid_tau: Time constant for liquid neural dynamics
            or_frequency: Orchestrated objective reduction frequency in Hz
        """
        super().__init__()
        self.num_units = num_units
        self.unit_dim = unit_dim
        self.sparsity = sparsity
        
        # Create HPUs
        self.units = nn.ModuleList([
            HarmonicProcessingUnit(
                dim=unit_dim,
                bioelectric_kappa=bioelectric_kappa,
                liquid_tau=liquid_tau,
                or_frequency=or_frequency
            ) for _ in range(num_units)
        ])
        
        # Initialize adjacency matrix with specified sparsity
        adjacency = torch.rand(num_units, num_units)
        adjacency = (adjacency > sparsity).float()
        
        # Remove self-connections
        adjacency.fill_diagonal_(0)
        
        # Make symmetric (undirected graph)
        adjacency = torch.triu(adjacency) + torch.triu(adjacency, 1).t()
        
        # Normalize rows
        row_sums = adjacency.sum(dim=1, keepdim=True)
        adjacency = adjacency / (row_sums + 1e-8)
        
        self.register_buffer('adjacency', adjacency)
        
    def forward(self, inputs: torch.Tensor, t: float = 0.0, dt: float = 0.01) -> torch.Tensor:
        """
        Process inputs through the HPU layer.
        
        Args:
            inputs: Input tensor (batch_size, num_units, unit_dim)
            t: Current simulation time
            dt: Time step
            
        Returns:
            Processed outputs (batch_size, num_units, unit_dim)
        """
        batch_size = inputs.shape[0]
        outputs = []
        
        for i, unit in enumerate(self.units):
            # Get input for this unit
            unit_input = inputs[:, i]
            
            # Get adjacency row for this unit
            unit_adjacency = self.adjacency[i]
            
            # Process through HPU
            unit_output = unit(
                unit_input,
                neighbors=self.units,
                adjacency=unit_adjacency,
                t=t,
                dt=dt
            )
            
            outputs.append(unit_output)
        
        # Stack outputs
        return torch.stack(outputs, dim=1)

        log_centers = np.log(centers[mask])
        
        # Simple linear regression in log-log space
        if len(log_centers) > 1:
            A = np.vstack([log_centers, np.ones(len(log_centers))]).T
            m, c = np.linalg.lstsq(A, log_hist, rcond=None)[0]
            # Power law exponent is negative slope
            return -m
        else:
            return self.target_exponent
    
    def reset_avalanche_history(self) -> None:
        """Reset the avalanche history."""
        self.avalanche_sizes = []


class Morphosome(nn.Module):
    """
    Morphogenetic growth guided by DNA sequences.
    
    The Morphosome drives growth and development of the cell architecture
    based on encoded DNA sequences from Evo-2 model with HEDGES constraints.
    """
    
    def __init__(self,
                in_channels: int,
                out_channels: int,
                num_rules: int = 10,
                complex_valued: bool = True):
        """
        Initialize a Morphosome.
        
        Args:
            in_channels: Number of input channels
            out_channels: Number of output channels
            num_rules: Number of morphogenetic rules
            complex_valued: Whether to use complex-valued weights
        """
        super().__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.num_rules = num_rules
        self.complex_valued = complex_valued
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Multiple morphogenetic rule pathways
        self.rules = nn.ModuleList([
            nn.Linear(in_channels, out_channels, dtype=dtype)
            for _ in range(num_rules)
        ])
        
        # Adaptive weights for rules
        self.rule_weights = nn.Parameter(torch.ones(num_rules, dtype=dtype) / num_rules)
        
        # DNA encoder (for processing DNA sequence information)
        self.dna_encoder = nn.Linear(in_channels, out_channels, dtype=dtype)
        
        # Growth inhibitor (for controlled growth)
        self.inhibitor = nn.Parameter(torch.zeros(out_channels, dtype=dtype))
        
        # Growth activator
        self.activator = nn.Parameter(torch.ones(out_channels, dtype=dtype) * 0.1)
        
        # Activation function
        self.activation = nn.ReLU()
        
        # Morphogenetic state
        self.register_buffer('morpho_state', torch.zeros(1, out_channels, dtype=dtype))
        
        # DNA sequence buffer
        self.dna_sequence = None
    
    def set_dna_sequence(self, sequence: torch.Tensor) -> None:
        """
        Set the DNA sequence to guide morphogenetic growth.
        
        Args:
            sequence: Encoded DNA sequence tensor
        """
        self.dna_sequence = sequence
    
    def encode_dna(self, x: torch.Tensor) -> torch.Tensor:
        """
        Encode DNA information into morphogenetic signals.
        
        Args:
            x: Input tensor
            
        Returns:
            DNA-encoded tensor
        """
        if self.dna_sequence is not None:
            # Extract features from DNA sequence
            dna_features = self.dna_encoder(self.dna_sequence)
            
            # Combine with input
            return x + 0.1 * dna_features
        else:
            return x
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Process input through the Morphosome.
        
        Args:
            x: Input tensor (batch_size, in_channels)
            
        Returns:
            Morphogenetic output tensor (batch_size, out_channels)
        """
        batch_size = x.shape[0]
        
        # Ensure morpho state matches batch size
        if self.morpho_state.shape[0] != batch_size:
            self.morpho_state = self.morpho_state.expand(batch_size, -1)
        
        # Convert input to complex if needed and using complex mode
        if self.complex_valued and not torch.is_complex(x):
            x = torch.complex(x, torch.zeros_like(x))
        
        # Encode DNA information
        x_dna = self.encode_dna(x)
        
        # Apply each rule
        rule_outputs = []
        for rule in self.rules:
            rule_outputs.append(rule(x_dna))
        
        # Stack and weight rule outputs
        rule_tensor = torch.stack(rule_outputs, dim=0)
        weighted_output = torch.sum(self.rule_weights.view(-1, 1, 1) * rule_tensor, dim=0)
        
        # Apply reaction-diffusion style dynamics
        # Growth = activator - inhibitor
        growth_signal = self.activator * weighted_output - self.inhibitor * self.morpho_state
        
        # Update morphogenetic state
        self.morpho_state = self.morpho_state + 0.1 * growth_signal
        
        # Apply activation function
        if self.complex_valued:
            output = self.activation(self.morpho_state.real)
        else:
            output = self.activation(self.morpho_state)
        
        return output


class Synthosome(nn.Module):
    """
    Protein synthesis with harmonic graph metanetworks.
    
    The Synthosome synthesizes "proteins" (feature vectors) using
    harmonic processing and graph neural network mechanisms.
    """
    
    def __init__(self,
                in_channels: int,
                out_channels: int,
                num_proteins: int = 20,
                complex_valued: bool = True):
        """
        Initialize a Synthosome.
        
        Args:
            in_channels: Number of input channels
            out_channels: Number of output channels
            num_proteins: Number of protein types to synthesize
            complex_valued: Whether to use complex-valued weights
        """
        super().__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.num_proteins = num_proteins
        self.complex_valued = complex_valued
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Protein synthesis templates
        self.templates = nn.Parameter(torch.randn(num_proteins, in_channels, dtype=dtype) * 0.1)
        
        # Protein encoders (different for each protein type)
        self.protein_encoders = nn.ModuleList([
            nn.Linear(in_channels, out_channels // num_proteins, dtype=dtype)
            for _ in range(num_proteins)
        ])
        
        # Attention mechanism for protein selection
        self.attention_key = nn.Linear(in_channels, num_proteins, dtype=dtype)
        self.attention_query = nn.Parameter(torch.randn(num_proteins, dtype=dtype))
        
        # Protein combiner
        self.combiner = nn.Linear(out_channels, out_channels, dtype=dtype)
        
        # Activation function
        self.activation = nn.ReLU()
        
        # Protein repository
        self.register_buffer('protein_repository', torch.zeros(1, num_proteins, out_channels // num_proteins, dtype=dtype))
    
    def synthesize_protein(self, x: torch.Tensor, protein_idx: int) -> torch.Tensor:
        """
        Synthesize a specific protein.
        
        Args:
            x: Input tensor
            protein_idx: Index of protein to synthesize
            
        Returns:
            Synthesized protein features
        """
        template = self.templates[protein_idx]
        encoder = self.protein_encoders[protein_idx]
        
        # Match input to template
        similarity = F.cosine_similarity(x, template.unsqueeze(0), dim=-1, eps=1e-8)
        
        # Scale input by similarity
        scaled_input = x * similarity.unsqueeze(-1)
        
        # Encode protein
        protein = encoder(scaled_input)
        
        return protein
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Process input through the Synthosome.
        
        Args:
            x: Input tensor (batch_size, in_channels)
            
        Returns:
            Synthesized proteins tensor (batch_size, out_channels)
        """
        batch_size = x.shape[0]
        
        # Ensure protein repository matches batch size
        if self.protein_repository.shape[0] != batch_size:
            self.protein_repository = self.protein_repository.expand(batch_size, -1, -1)
        
        # Convert input to complex if needed and using complex mode
        if self.complex_valued and not torch.is_complex(x):
            x = torch.complex(x, torch.zeros_like(x))
        
        # Compute attention scores for protein selection
        attention_scores = torch.matmul(self.attention_key(x), self.attention_query)
        attention_weights = F.softmax(attention_scores.real if self.complex_valued else attention_scores, dim=-1)
        
        # Synthesize each protein type
        proteins = []
        for i in range(self.num_proteins):
            protein = self.synthesize_protein(x, i)
            proteins.append(protein)
            
            # Update protein repository
            self.protein_repository[:, i] = 0.9 * self.protein_repository[:, i] + 0.1 * protein
        
        # Combine proteins based on attention weights
        protein_tensor = torch.stack(proteins, dim=1)
        weighted_proteins = protein_tensor * attention_weights.unsqueeze(-1)
        combined_proteins = weighted_proteins.reshape(batch_size, -1)
        
        # Final protein processing
        output = self.combiner(combined_proteins)
        
        # Apply activation function
        if self.complex_valued:
            return self.activation(output.real)
        else:
            return self.activation(output)


class Fermentosome(nn.Module):
    """
    Metabolic optimization with Monod kinetics.
    
    The Fermentosome optimizes resource utilization using 
    liquid neural dynamics and bio-inspired metabolic equations.
    """
    
    def __init__(self,
                in_channels: int,
                out_channels: int,
                num_metabolites: int = 10,
                complex_valued: bool = True):
        """
        Initialize a Fermentosome.
        
        Args:
            in_channels: Number of input channels
            out_channels: Number of output channels
            num_metabolites: Number of metabolite types
            complex_valued: Whether to use complex-valued weights
        """
        super().__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.num_metabolites = num_metabolites
        self.complex_valued = complex_valued
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Metabolite processing pathways
        self.pathways = nn.ModuleList([
            nn.Linear(in_channels, out_channels // num_metabolites, dtype=dtype)
            for _ in range(num_metabolites)
        ])
        
        # Maximum growth rates (μmax in Monod equation)
        self.max_growth_rates = nn.Parameter(torch.ones(num_metabolites, dtype=dtype) * 0.1)
        
        # Half-saturation constants (Ks in Monod equation)
        self.saturation_constants = nn.Parameter(torch.ones(num_metabolites, dtype=dtype) * 0.5)
        
        # Liquid neural network weights for dynamic adaptation
        self.liquid_weights = nn.Parameter(torch.randn(out_channels, out_channels, dtype=dtype) * 0.01)
        self.tau = nn.Parameter(torch.tensor(1.0, dtype=dtype))
        
        # Metabolite combiner
        self.combiner = nn.Linear(out_channels, out_channels, dtype=dtype)
        
        # Activation function
        self.activation = nn.Sigmoid()
        
        # Metabolite concentrations
        self.register_buffer('metabolite_conc', torch.ones(1, num_metabolites, dtype=dtype) * 0.5)
    
    def monod_kinetics(self, substrate_conc: torch.Tensor) -> torch.Tensor:
        """
        Apply Monod kinetics to calculate growth rates.
        
        The Monod equation: μ = μmax * [S] / (Ks + [S])
        where μ is growth rate, μmax is maximum growth rate,
        [S] is substrate concentration, and Ks is half-saturation constant.
        
        Args:
            substrate_conc: Substrate concentrations
            
        Returns:
            Growth rates
        """
        return self.max_growth_rates * substrate_conc / (self.saturation_constants + substrate_conc)
    
    def forward(self, x: torch.Tensor, dt: float = 0.01) -> torch.Tensor:
        """
        Process input through the Fermentosome.
        
        Args:
            x: Input tensor (batch_size, in_channels)
            dt: Time step
            
        Returns:
            Metabolic output tensor (batch_size, out_channels)
        """
        batch_size = x.shape[0]
        
        # Ensure metabolite concentrations match batch size
        if self.metabolite_conc.shape[0] != batch_size:
            self.metabolite_conc = self.metabolite_conc.expand(batch_size, -1)
        
        # Convert input to complex if needed and using complex mode
        if self.complex_valued and not torch.is_complex(x):
            x = torch.complex(x, torch.zeros_like(x))
        
        # Extract substrate concentrations from input
        substrate_conc = F.softplus(x.real if self.complex_valued else x)
        substrate_conc = substrate_conc.mean(dim=-1, keepdim=True).expand(-1, self.num_metabolites)
        
        # Calculate growth rates using Monod kinetics
        growth_rates = self.monod_kinetics(substrate_conc)
        
        # Update metabolite concentrations
        metabolite_change = growth_rates - 0.1 * self.metabolite_conc
        self.metabolite_conc = torch.clamp(self.metabolite_conc + metabolite_change * dt, min=0.01, max=0.99)
        
        # Process each metabolic pathway
        pathway_outputs = []
        for i, pathway in enumerate(self.pathways):
            # Scale input by metabolite concentration
            scaled_input = x * self.metabolite_conc[:, i:i+1]
            pathway_output = pathway(scaled_input)
            pathway_outputs.append(pathway_output)
        
        # Combine pathway outputs
        metabolism = torch.cat(pathway_outputs, dim=-1)
        
        # Apply liquid neural dynamics
        dx_dt = -metabolism / self.tau + torch.matmul(metabolism, self.liquid_weights)
        metabolism = metabolism + dx_dt * dt
        
        # Final combination
        output = self.combiner(metabolism)
        
        # Apply activation function
        if self.complex_valued:
            return self.activation(output.real)
        else:
            return self.activation(output)


class Quantosome(nn.Module):
    """
    Quantum coherence simulation.
    
    The Quantosome simulates quantum-like effects such as superposition,
    entanglement, and collapse dynamics inspired by Orchestrated Objective
    Reduction (Orch OR) theory.
    """
    
    def __init__(self,
                state_size: int = 16,
                collapse_frequency: float = 40.0,
                complex_valued: bool = True):
        """
        Initialize a Quantosome.
        
        Args:
            state_size: Dimension of quantum state
            collapse_frequency: Frequency of quantum state collapse in Hz (40Hz for Orch OR)
            complex_valued: Whether to use complex-valued weights
        """
        super().__init__()
        self.state_size = state_size
        self.collapse_frequency = collapse_frequency
        self.complex_valued = complex_valued
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Quantum state projector
        self.projector = nn.Linear(state_size, state_size, dtype=dtype)
        
        # Operators for quantum simulation
        self.hadamard = nn.Parameter(self._create_hadamard(state_size, dtype))
        self.phase = nn.Parameter(self._create_phase(state_size, dtype))
        
        # Entanglement matrix
        self.entanglement = nn.Parameter(torch.eye(state_size, dtype=dtype) * 0.1)
        
        # Normalizer for probability distribution
        self.normalizer = nn.Softmax(dim=-1)
        
        # Quantum state
        self.register_buffer('quantum_state', self._create_initial_state(1, state_size, dtype))
        
        # Time tracking for collapse
        self.register_buffer('time_since_collapse', torch.zeros(1))
    
    def _create_hadamard(self, size: int, dtype: torch.dtype) -> torch.Tensor:
        """Create a Hadamard-like operator."""
        h = torch.ones((size, size), dtype=dtype) / np.sqrt(size)
        for i in range(size):
            for j in range(size):
                if (i & j).bit_count() % 2 == 1:
                    h[i, j] *= -1
        return h
    
    def _create_phase(self, size: int, dtype: torch.dtype) -> torch.Tensor:
        """Create a phase operator."""
        phase = torch.eye(size, dtype=dtype)
        for i in range(size):
            phase[i, i] = torch.exp(1j * torch.tensor(np.pi * i / size))
        return phase
    
    def _create_initial_state(self, batch_size: int, state_size: int, dtype: torch.dtype) -> torch.Tensor:
        """Create initial quantum state (|0> state)."""
        state = torch.zeros((batch_size, state_size), dtype=dtype)
        state[:, 0] = 1.0
        return state
    
    def apply_gate(self, state: torch.Tensor, gate: torch.Tensor) -> torch.Tensor:
        """Apply quantum gate to state."""
        return torch.matmul(state, gate.t())
    
    def forward(self, x: torch.Tensor, dt: float = 0.01) -> torch.Tensor:
        """
        Process input through the Quantosome.
        
        Args:
            x: Input tensor (batch_size, state_size)
            dt: Time step
            
        Returns:
            Quantum processed tensor (batch_size, state_size)
        """
        batch_size = x.shape[0]
        
        # Ensure quantum state matches batch size
        if self.quantum_state.shape[0] != batch_size:
            self.quantum_state = self._create_initial_state(batch_size, self.state_size, 
                                                           self.quantum_state.dtype)
        
        # Convert input to complex if needed and using complex mode
        if self.complex_valued and not torch.is_complex(x):
            x = torch.complex(x, torch.zeros_like(x))
        
        # Project input to quantum state
        input_state = self.projector(x)
        
        # Apply Hadamard gate for superposition
        superposition = self.apply_gate(input_state, self.hadamard)
        
        # Apply phase gate
        phased = self.apply_gate(superposition, self.phase)
        
        # Apply entanglement
        entangled = torch.matmul(phased, self.entanglement)
        
        # Update time since last collapse
        self.time_since_collapse += dt
        
        # Check for collapse (40Hz frequency)
        if self.time_since_collapse >= 1.0 / self.collapse_frequency:
            # Reset timer
            self.time_since_collapse.zero_()
            
            # Calculate collapse probabilities
            probs = self.normalizer(torch.abs(entangled) ** 2)
            
            # Sample collapse outcome
            collapsed_idx = torch.multinomial(probs, 1)
            
            # Create collapsed state
            collapsed = torch.zeros_like(entangled)
            collapsed.scatter_(1, collapsed_idx, 1.0)
            
            # Update quantum state with collapse
            self.quantum_state = collapsed
        else:
            # Without collapse, update quantum state with new entangled state
            self.quantum_state = 0.9 * self.quantum_state + 0.1 * entangled
        
        # Normalize state
        norm = torch.norm(self.quantum_state, dim=-1, keepdim=True)
        self.quantum_state = self.quantum_state / (norm + 1e-8)
        
        # Output probabilities
        if self.complex_valued:
            output = torch.abs(self.quantum_state) ** 2
        else:
            output = self.quantum_state ** 2
        
        return output


class Metabolosome(nn.Module):
    """
    Resource allocation using liquid dynamics.
    
    The Metabolosome allocates resources within the cell using
    liquid neural dynamics for adaptive response to changing conditions.
    """
    
    def __init__(self,
                in_channels: int,
                out_channels: int,
                num_resources: int = 8,
                complex_valued: bool = True):
        """
        Initialize a Metabolosome.
        
        Args:
            in_channels: Number of input channels
            out_channels: Number of output channels
            num_resources: Number of resource types
            complex_valued: Whether to use complex-valued weights
        """
        super().__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.num_resources = num_resources
        self.complex_valued = complex_valued
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Resource allocation network
        self.resource_allocator = nn.Linear(in_channels, out_channels, dtype=dtype)
        
        # Demand estimator
        self.demand_estimator = nn.Sequential(
            nn.Linear(in_channels, num_resources, dtype=dtype),
            nn.Softplus()
        )
        
        # Supply estimator
        self.supply_estimator = nn.Sequential(
            nn.Linear(in_channels, num_resources, dtype=dtype),
            nn.Softplus()
        )
        
        # Liquid neural network weights for dynamic adaptation
        self.liquid_weights = nn.Parameter(torch.randn(out_channels, out_channels, dtype=dtype) * 0.01)
        self.tau = nn.Parameter(torch.tensor(1.0, dtype=dtype))
        
        # Resource combiner
        self.combiner = nn.Linear(num_resources, out_channels, dtype=dtype)
        
        # Activation function
        self.activation = nn.Softmax(dim=-1)
        
        # Resource state
        self.register_buffer('resource_state', torch.ones(1, num_resources, dtype=dtype) / num_resources)
    
    def forward(self, x: torch.Tensor, dt: float = 0.01) -> torch.Tensor:
        """
        Process input through the Metabolosome.
        
        Args:
            x: Input tensor (batch_size, in_channels)
            dt: Time step
            
        Returns:
            Resource allocation tensor (batch_size, out_channels)
        """
        batch_size = x.shape[0]
        
        # Ensure resource state matches batch size
        if self.resource_state.shape[0] != batch_size:
            self.resource_state = self.resource_state.expand(batch_size, -1)
        
        # Convert input to complex if needed and using complex mode
        if self.complex_valued and not torch.is_complex(x):
            x = torch.complex(x, torch.zeros_like(x))
        
        # Estimate supply and demand
        demand = self.demand_estimator(x)
        supply = self.supply_estimator(x)
        
        # Calculate resource allocation based on supply and demand
        allocation = supply / (demand + 1e-8)
        
        # Apply liquid neural dynamics to resource state
        dx_dt = -self.resource_state / self.tau + torch.matmul(self.resource_state, self.liquid_weights[:self.num_resources, :self.num_resources])
        self.resource_state = self.resource_state + dx_dt * dt
        
        # Update resource state with new allocation
        self.resource_state = 0.9 * self.resource_state + 0.1 * allocation
        
        # Ensure resource state is normalized
        if self.complex_valued:
            sum_resources = torch.sum(self.resource_state.real, dim=-1, keepdim=True)
        else:
            sum_resources = torch.sum(self.resource_state, dim=-1, keepdim=True)
        self.resource_state = self.resource_state / (sum_resources + 1e-8)
        
        # Combine resources into final output
        output = self.combiner(self.resource_state)
        
        # Process through resource allocator
        final_allocation = self.resource_allocator(output)
        
        # Apply activation function
        if self.complex_valued:
            return self.activation(final_allocation.real)
        else:
            return self.activation(final_allocation)

        # Create organelles
        self.organelles = nn.ModuleDict(
            create_organelles(
                in_channels=config.d_model,
                out_channels=config.d_model,
                complex_valued=self.complex_valued
            )
        )
        
        # Internal state
        self.state = nn.Parameter(torch.randn(config.d_model, dtype=self.dtype) * 0.01)
        
        # DNA encoder for processing genetic information
        if dna_sequence:
            # Convert DNA sequence to embedding
            self.dna_embedding = self._encode_dna(dna_sequence)
            # Set DNA sequence in morphosome
            if hasattr(self.organelles, 'morphosome'):
                self.organelles.morphosome.set_dna_sequence(self.dna_embedding)
        else:
            self.dna_embedding = nn.Parameter(torch.zeros(config.d_model, dtype=self.dtype))
        
        # Cell membrane (controls information flow in/out)
        self.membrane_in = nn.Linear(config.d_model, config.d_model, dtype=self.dtype)
        self.membrane_out = nn.Linear(config.d_model, config.d_model, dtype=self.dtype)
        
        # Growth state (for cell division and architecture expansion)
        self.register_buffer('growth_state', torch.zeros(1, dtype=torch.float32))
        self.division_threshold = 0.9  # Threshold for cell division
        
        # Cell age (used for lifecycle management)
        self.register_buffer('age', torch.zeros(1, dtype=torch.float32))
        self.max_age = 100.0  # Maximum cell age
        
        # Health metrics
        self.register_buffer('health', torch.ones(1, dtype=torch.float32))
        self.register_buffer('energy', torch.ones(1, dtype=torch.float32))
        
        # Activation function
        self.activation = nn.Tanh()
    
    def _encode_dna(self, dna_sequence: str) -> torch.Tensor:
        """
        Encode DNA sequence into an embedding.
        
        Args:
            dna_sequence: DNA sequence string
            
        Returns:
            Encoded DNA embedding
        """
        # Simple encoding: convert ACGT to one-hot vectors
        encoding = {'A': 0, 'C': 1, 'G': 2, 'T': 3}
        one_hot = torch.zeros(len(dna_sequence), 4, dtype=self.dtype)
        
        for i, nucleotide in enumerate(dna_sequence):
            if nucleotide in encoding:
                one_hot[i, encoding[nucleotide]] = 1.0
        
        # Project to embedding space
        projection = nn.Linear(4 * len(dna_sequence), self.d_model, dtype=self.dtype)
        embedding = projection(one_hot.flatten().unsqueeze(0)).squeeze(0)
        
        return embedding
    
    def forward(self, 
               inputs: torch.Tensor,
               neighbors: List['DigitalCell'] = None,
               adjacency: torch.Tensor = None,
               t: float = 0.0,
               dt: float = 0.01) -> Tuple[torch.Tensor, Dict]:
        """
        Process inputs through the digital cell.
        
        Args:
            inputs: Input tensor (batch_size, d_model)
            neighbors: List of neighboring cells
            adjacency: Adjacency matrix for neighbor connections
            t: Current simulation time
            dt: Time step
            
        Returns:
            Tuple of (output tensor, cell state metrics)
        """
        batch_size = inputs.shape[0]
        
        # Convert input to complex if needed and using complex mode
        if self.complex_valued and not torch.is_complex(inputs):
            inputs = torch.complex(inputs, torch.zeros_like(inputs))
        
        # Process through cell membrane
        membrane_inputs = self.membrane_in(inputs)
        
        # Process through HPU
        hpu_output = self.hpu(
            membrane_inputs,
            neighbors=[cell.hpu for cell in neighbors] if neighbors else None,
            adjacency=adjacency,
            t=t,
            dt=dt
        )
        
        # Process through each organelle
        organelle_outputs = {}
        
        # Energosome: Energy allocation
        energy_allocation = self.organelles["energosome"](hpu_output, t)
        organelle_outputs["energy"] = energy_allocation
        
        # Criticosome: Critical dynamics
        critical_output = self.organelles["criticosome"](hpu_output, dt)
        organelle_outputs["critical"] = critical_output
        
        # Morphosome: Morphogenetic growth
        morpho_output = self.organelles["morphosome"](hpu_output)
        organelle_outputs["morpho"] = morpho_output
        
        # Update growth state based on morphosome output
        growth_signal = morpho_output.mean().real if self.complex_valued else morpho_output.mean()
        self.growth_state = self.growth_state + 0.01 * growth_signal
        
        # Synthosome: Protein synthesis
        synth_output = self.organelles["synthosome"](hpu_output)
        organelle_outputs["synth"] = synth_output
        
        # Fermentosome: Metabolic optimization
        ferment_output = self.organelles["fermentosome"](hpu_output, dt)
        organelle_outputs["ferment"] = ferment_output
        
        # Update energy based on fermentosome output
        energy_update = ferment_output.mean().real if self.complex_valued else ferment_output.mean()
        self.energy = torch.clamp(self.energy + 0.01 * (energy_update - 0.5), 0.1, 1.0)
        
        # Quantosome: Quantum coherence
        quant_output = self.organelles["quantosome"](hpu_output, dt)
        organelle_outputs["quant"] = quant_output
        
        # Metabolosome: Resource allocation
        meta_output = self.organelles["metabolosome"](hpu_output, dt)
        organelle_outputs["meta"] = meta_output
        
        # Adaptosome: Environmental sensing
        adapt_output = self.organelles["adaptosome"](hpu_output, dt)
        organelle_outputs["adapt"] = adapt_output
        
        # Regulosome: Regulatory feedback
        regulo_output = self.organelles["regulosome"](hpu_output, dt)
        organelle_outputs["regulo"] = regulo_output
        
        # Communisome: Inter-cell communication
        if neighbors:
            # Get messages from neighbors
            neighbor_states = torch.stack([
                neighbor.state.expand(batch_size, -1) for neighbor in neighbors
            ], dim=1)
            
            # Process messages through communisome
            comm_input = torch.cat([hpu_output, neighbor_states.mean(dim=1)], dim=-1)
            comm_output = self.organelles["communisome"](comm_input[:, :self.d_model])
        else:
            comm_output = self.organelles["communisome"](hpu_output)
        
        organelle_outputs["comm"] = comm_output
        
        # Combine all organelle outputs (weighted by energy allocation)
        combined_output = torch.zeros_like(hpu_output)
        for i, (name, output) in enumerate(organelle_outputs.items()):
            if i < energy_allocation.shape[1]:
                weight = energy_allocation[:, i:i+1]
                combined_output = combined_output + output * weight
        
        # Update internal state
        self.state.data = 0.95 * self.state.data + 0.05 * combined_output.mean(dim=0)
        
        # Process through membrane for output
        output = self.membrane_out(combined_output)
        
        # Apply activation
        if self.complex_valued:
            output = self.activation(output.real)
        else:
            output = self.activation(output)
        
        # Update age
        self.age += dt
        
        # Update health based on energy and age
        age_factor = 1.0 - (self.age / self.max_age)
        self.health = 0.95 * self.health + 0.05 * self.energy * age_factor
        
        # Prepare metrics dictionary
        metrics = {
            "growth": self.growth_state.item(),
            "energy": self.energy.item(),
            "health": self.health.item(),
            "age": self.age.item(),
            "ready_to_divide": self.growth_state > self.division_threshold
        }
        
        return output, metrics
    
    def should_divide(self) -> bool:
        """
        Check if cell should divide (for growth).
        
        Returns:
            True if cell should divide, False otherwise
        """
        return self.growth_state > self.division_threshold and self.health > 0.5
    
    def divide(self) -> 'DigitalCell':
        """
        Divide this cell to create a daughter cell.
        
        Returns:
            Newly created daughter cell
        """
        # Create daughter cell with mutation of DNA
        if self.dna_sequence:
            # Introduce small mutations in the DNA sequence
            daughter_dna = self._mutate_dna(self.dna_sequence)
        else:
            daughter_dna = None
        
        # Create daughter cell
        daughter_cell = DigitalCell(
            config=self.config,
            cell_id=self.cell_id + 1000,  # Arbitrary offset for new ID
            dna_sequence=daughter_dna
        )
        
        # Share half of energy with daughter
        daughter_cell.energy = self.energy * 0.5
        self.energy = self.energy * 0.5
        
        # Reset growth state
        self.growth_state.zero_()
        
        return daughter_cell
    
    def _mutate_dna(self, sequence: str) -> str:
        """
        Introduce small mutations in DNA sequence.
        
        Args:
            sequence: Original DNA sequence
            
        Returns:
            Mutated DNA sequence
        """
        nucleotides = ['A', 'C', 'G', 'T']
        mutation_rate = 0.01  # 1% mutation rate
        
        mutated = list(sequence)
        for i in range(len(mutated)):
            if np.random.random() < mutation_rate:
                # Choose a different nucleotide
                current = mutated[i]
                options = [nt for nt in nucleotides if nt != current]
                mutated[i] = np.random.choice(options)
        
        return ''.join(mutated)
    
    def get_state_vector(self) -> torch.Tensor:
        """
        Get the cell's state vector for external use.
        
        Returns:
            Cell state vector
        """
        return self.state.detach()
    
    def set_dna(self, dna_sequence: str) -> None:
        """
        Set or update the cell's DNA sequence.
        
        Args:
            dna_sequence: New DNA sequence
        """
        self.dna_sequence = dna_sequence
        self.dna_embedding = self._encode_dna(dna_sequence)
        
        # Update morphosome with new DNA
        if hasattr(self.organelles, 'morphosome'):
            self.organelles.morphosome.set_dna_sequence(self.dna_embedding)


class CellularTissue(nn.Module):
    """
    A collection of interconnected digital cells forming a tissue.
    
    The tissue organizes cells in a graph structure, enabling
    communication, coordinated behavior, and emergent properties.
    """
    
    def __init__(self, 
                config,
                initial_cells: int = 10,
                sparsity: float = 0.2,
                dna_encoder: Optional[DNAEncoder] = None):
        """
        Initialize a cellular tissue.
        
        Args:
            config: Configuration object
            initial_cells: Number of cells to create initially
            sparsity: Connection sparsity (0.0 = fully connected, 1.0 = no connections)
            dna_encoder: DNA encoder for generating cell DNA sequences
        """
        super().__init__()
        self.config = config
        self.sparsity = sparsity
        self.dna_encoder = dna_encoder
        
        # Create cells
        self.cells = nn.ModuleList([
            DigitalCell(
                config=config,
                cell_id=i,
                dna_sequence=self._generate_cell_dna(i) if dna_encoder else None
            ) for i in range(initial_cells)
        ])
        
        # Create graph of cell connections
        self.graph = nx.Graph()
        self.initialize_graph()
        
        # Convert graph to adjacency matrix
        self.register_buffer('adjacency', self._graph_to_adjacency())
        
        # Growth state
        self.register_buffer('growth_tracker', torch.zeros(1))
        
        # Time tracker
        self.register_buffer('time', torch.zeros(1))
    
    def _generate_cell_dna(self, cell_id: int) -> str:
        """
        Generate DNA sequence for a cell using the DNA encoder.
        
        Args:
            cell_id: Cell identifier
            
        Returns:
            Generated DNA sequence
        """
        if self.dna_encoder:
            # Generate a prompt based on cell_id
            prompt = f"Cell architecture parameters for cell {cell_id}"
            
            # Generate DNA sequence
            return self.dna_encoder.generate_sequence(prompt, n_tokens=200)
        else:
            return None
    
    def initialize_graph(self) -> None:
        """Initialize the graph structure connecting cells."""
        # Add nodes for all cells
        for i, cell in enumerate(self.cells):
            self.graph.add_node(i, cell=cell)
        
        # Add edges based on sparsity parameter
        for i in range(len(self.cells)):
            for j in range(i+1, len(self.cells)):
                if np.random.random() > self.sparsity:
                    # Add edge with random weight
                    weight = np.random.uniform(0.1, 0.5)
                    self.graph.add_edge(i, j, weight=weight)
        
        logger.info(f"Initialized graph with {len(self.cells)} cells and {self.graph.number_of_edges()} connections")
    
    def _graph_to_adjacency(self) -> torch.Tensor:
        """
        Convert networkx graph to adjacency tensor.
        
        Returns:
            Adjacency tensor
        """
        # Get number of cells
        n = len(self.cells)
        
        # Create adjacency matrix
        adjacency = torch.zeros(n, n)
        
        # Fill in edge weights
        for i, j in self.graph.edges():
            weight = self.graph[i][j]['weight']
            adjacency[i, j] = weight
            adjacency[j, i] = weight  # Symmetric (undirected graph)
        
        return adjacency
    
    def update_adjacency(self) -> None:
        """Update the adjacency tensor from the graph."""
        self.adjacency = self._graph_to_adjacency().to(self.adjacency.device)
    
    def forward(self, inputs: torch.Tensor, dt: float = 0.01) -> Tuple[torch.Tensor, Dict]:
        """
        Process inputs through the cellular tissue.
        
        Args:
            inputs: Input tensor (batch_size, num_cells, d_model)
            dt: Time step
            
        Returns:
            Tuple of (output tensors, tissue state metrics)
        """
        batch_size = inputs.shape[0]
        num_cells = min(inputs.shape[1], len(self.cells))
        
        # Update time
        self.time += dt
        
        # Process each cell
        outputs = []
        metrics = []
        
        for i, cell in enumerate(self.cells):
            if i < num_cells:
                # Get input for this cell
                cell_input = inputs[:, i]
                
                # Get neighboring cells
                neighbors = [self.cells[j] for j in self.graph.neighbors(i)]
                
                # Get adjacency row for this cell
                cell_adjacency = self.adjacency[i]
                
                # Process through cell
                cell_output, cell_metrics = cell(
                    cell_input,
                    neighbors=neighbors,
                    adjacency=cell_adjacency,
                    t=self.time.item(),
                    dt=dt
                )
                
                outputs.append(cell_output)
                metrics.append(cell_metrics)
        
        # Stack outputs
        stacked_outputs = torch.stack(outputs, dim=1)
        
        # Check for cell division
        self._check_cell_division()
        
        # Aggregate metrics
        tissue_metrics = self._aggregate_metrics(metrics)
        
        return stacked_outputs, tissue_metrics
    
    def _check_cell_division(self) -> None:
        """Check if any cells should divide and handle division."""
        cells_to_add = []
        
        for i, cell in enumerate(self.cells):
            if cell.should_divide():
                # Create daughter cell
                daughter = cell.divide()
                cells_to_add.append(daughter)
                
                # Track growth
                self.growth_tracker += 1
                
                logger.info(f"Cell {i} divided, creating new cell")
        
        # Add new cells to module list
        if cells_to_add:
            self.cells.extend(cells_to_add)
            
            # Update graph with new cells
            self._update_graph_after_division(len(self.cells) - len(cells_to_add), len(self.cells))
            
            # Update adjacency tensor
            self.update_adjacency()
    
    def _update_graph_after_division(self, start_idx: int, end_idx: int) -> None:
        """
        Update graph after cell division to include new cells.
        
        Args:
            start_idx: Starting index of new cells
            end_idx: Ending index of new cells
        """
        # Add nodes for new cells
        for i in range(start_idx, end_idx):
            self.graph.add_node(i, cell=self.cells[i])
        
        # Connect new cells with preferential attachment
        # (cells with more connections are more likely to connect to new cells)
        for i in range(start_idx, end_idx):
            # Calculate connection probabilities based on current degrees
            degrees = np.array([self.graph.degree(j) for j in range(start_idx)])
            probabilities = degrees / degrees.sum() if degrees.sum() > 0 else np.ones(start_idx) / start_idx
            
            # Determine number of connections for this new cell
            num_connections = max(1, int((1 - self.sparsity) * 5))
            
            # Select cells to connect to
            connect_to = np.random.choice(
                range(start_idx),
                size=min(num_connections, start_idx),
                replace=False,
                p=probabilities
            )
            
            # Add edges
            for j in connect_to:
                weight = np.random.uniform(0.1, 0.5)
                self.graph.add_edge(i, j, weight=weight)
    
    def _aggregate_metrics(self, cell_metrics: List[Dict]) -> Dict:
        """
        Aggregate metrics from all cells.
        
        Args:
            cell_metrics: List of metric dictionaries from cells
            
        Returns:
            Aggregated metrics dictionary
        """
        # Initialize aggregate metrics
        aggregated = {
            "num_cells": len(self.cells),
            "avg_energy": 0.0,
            "avg_health": 0.0,
            "avg_age": 0.0,
            "growth_events": self.growth_tracker.item(),
            "connectivity": 0.0,
            "ready_to_divide": 0
        }
        
        # Aggregate metrics across cells
        if cell_metrics:
            aggregated["avg_energy"] = sum(m["energy"] for m in cell_metrics) / len(cell_metrics)
            aggregated["avg_health"] = sum(m["health"] for m in cell_metrics) / len(cell_metrics)
            aggregated["avg_age"] = sum(m["age"] for m in cell_metrics) / len(cell_metrics)
            aggregated["ready_to_divide"] = sum(1 for m in cell_metrics if m["ready_to_divide"])
        
        # Calculate graph connectivity
        if self.graph.number_of_nodes() > 1:
            aggregated["connectivity"] = 2 * self.graph.number_of_edges() / (self.graph.number_of_nodes() * (self.graph.number_of_nodes() - 1))
        
        return aggregated


class OrganismLayer(nn.Module):
    """
    A layer of tissues forming a complete organism.
    
    The organism layer organizes multiple tissues into a coherent system,
    enabling hierarchical information processing and emergent behaviors.
    """
    
    def __init__(self,
                config,
                num_tissues: int = 4,
                cells_per_tissue: int = 10,
                tissue_sparsity: float = 0.2,
                inter_tissue_sparsity: float = 0.5,
                dna_encoder: Optional[DNAEncoder] = None):
        """
        Initialize an organism layer.
        
        Args:
            config: Configuration object
            num_tissues: Number of tissues
            cells_per_tissue: Number of cells per tissue
            tissue_sparsity: Connection sparsity within tissues
            inter_tissue_sparsity: Connection sparsity between tissues
            dna_encoder: DNA encoder for generating cell DNA sequences
        """
        super().__init__()
        self.config = config
        self.num_tissues = num_tissues
        self.inter_tissue_sparsity = inter_tissue_sparsity
        
        # Create tissues
        self.tissues = nn.ModuleList([
            CellularTissue(
                config=config,
                initial_cells=cells_per_tissue,
                sparsity=tissue_sparsity,
                dna_encoder=dna_encoder
            ) for _ in range(num_tissues)
        ])
        
        # Create graph connecting tissues
        self.tissue_graph = nx.Graph()
        self.initialize_tissue_graph()
        
        # Convert tissue graph to adjacency matrix
        self.register_buffer('tissue_adjacency', self._tissue_graph_to_adjacency())
        
        # Create tissue-to-tissue projections
        self.tissue_projections = nn.ModuleList([
            nn.Linear(config.d_model, config.d_model)
            for _ in range(self.tissue_graph.number_of_edges())
        ])
        
        # Tissue output combination
        self.tissue_combiner = nn.Linear(config.d_model * num_tissues, config.d_model)
        
        # Time tracker
        self.register_buffer('time', torch.zeros(1))
    
    def initialize_tissue_graph(self) -> None:
        """Initialize the graph structure connecting tissues."""
        # Add nodes for all tissues
        for i in range(self.num_tissues):
            self.tissue_graph.add_node(i, tissue=self.tissues[i])
        
        # Add edges based on inter-tissue sparsity parameter
        edge_counter = 0
        for i in range(self.num_tissues):
            for j in range(i+1, self.num_tissues):
                if np.random.random() > self.inter_tissue_sparsity:
                    # Add edge with random weight
                    weight = np.random.uniform(0.1, 0.5)
                    self.tissue_graph.add_edge(i, j, weight=weight, projection_idx=edge_counter)
                    edge_counter += 1
        
        logger.info(f"Initialized tissue graph with {self.num_tissues} tissues and {self.tissue_graph.number_of_edges()} connections")
    
    def _tissue_graph_to_adjacency(self) -> torch.Tensor:
        """
        Convert tissue graph to adjacency tensor.
        
        Returns:
            Adjacency tensor
        """
        # Create adjacency matrix
        adjacency = torch.zeros(self.num_tissues, self.num_tissues)
        
        # Fill in edge weights
        for i, j in self.tissue_graph.edges():
            weight = self.tissue_graph[i][j]['weight']
            adjacency[i, j] = weight
            adjacency[j, i] = weight  # Symmetric (undirected graph)
        
        return adjacency
    
    def forward(self, inputs: torch.Tensor, dt: float = 0.01) -> Tuple[torch.Tensor, Dict]:
        """
        Process inputs through the organism layer.
        
        Args:
            inputs: Input tensor (batch_size, num_tissues, max_cells_per_tissue, d_model)
            dt: Time step
            
        Returns:
            Tuple of (output tensor, organism state metrics)
        """
        batch_size = inputs.shape[0]
        
        # Update time
        self.time += dt
        
        # Process each tissue
        tissue_outputs = []
        tissue_metrics = []
        
        for i, tissue in enumerate(self.tissues):
            if i < inputs.shape[1]:
                # Get input for this tissue
                tissue_input = inputs[:, i]
                
                # Process through tissue
                output, metrics = tissue(tissue_input, dt)
                
                tissue_outputs.append(output)
                tissue_metrics.append(metrics)
        
        # Process inter-tissue communication
        processed_outputs = []
        for i, output in enumerate(tissue_outputs):
            # Start with original output
            processed = output
            
            # Add contributions from connected tissues
            for j in self.tissue_graph.neighbors(i):
                if j < len(tissue_outputs):
                    # Get edge data
                    edge_data = self.tissue_graph[i][j]
                    weight = edge_data['weight']
                    projection_idx = edge_data['projection_idx']
                    
                    # Project neighboring tissue's output
                    neighbor_output = tissue_outputs[j]
                    projected = self.tissue_projections[projection_idx](neighbor_output)
                    
                    # Add weighted contribution
                    processed = processed + weight * projected
            
            processed_outputs.append(processed)
        
        # Combine tissue outputs
        # First, ensure all outputs have the same number of cells
        max_cells = max(output.shape[1] for output in processed_outputs)
        padded_outputs = []
        
        for output in processed_outputs:
            if output.shape[1] < max_cells:
                # Pad with zeros
                padding = torch.zeros(batch_size, max_cells - output.shape[1], output.shape[2], device=output.device)
                padded = torch.cat([output, padding], dim=1)
                padded_outputs.append(padded)
            else:
                padded_outputs.append(output)
        
        # Stack and combine
        stacked = torch.cat([output.mean(dim=1) for output in padded_outputs], dim=1)
        combined_output = self.tissue_combiner(stacked)
        
        # Aggregate metrics
        organism_metrics = self._aggregate_tissue_metrics(tissue_metrics)
        
        return combined_output, organism_metrics
    
    def _aggregate_tissue_metrics(self, tissue_metrics: List[Dict]) -> Dict:
        """
        Aggregate metrics from all tissues.
        
        Args:
            tissue_metrics: List of metric dictionaries from tissues
            
        Returns:
            Aggregated metrics dictionary
        """
        # Initialize aggregate metrics
        aggregated = {
            "num_tissues": self.num_tissues,
            "total_cells": sum(m["num_cells"] for m in tissue_metrics),
            "avg_energy": 0.0,
            "avg_health": 0.0,
            "total_growth_events": sum(m["growth_events"] for m in tissue_metrics),
            "inter_tissue_connectivity": 0.0
        }
        
        # Aggregate metrics across tissues
        if tissue_metrics:
            # Weighted average by number of cells in each tissue
            total_cells = aggregated["total_cells"]
            if total_cells > 0:
                aggregated["avg_energy"] = sum(m["avg_energy"] * m["num_cells"] for m in tissue_metrics) / total_cells
                aggregated["avg_health"] = sum(m["avg_health"] * m["num_cells"] for m in tissue_metrics) / total_cells
        
        # Calculate inter-tissue connectivity
        if self.tissue_graph.number_of_nodes() > 1:
            aggregated["inter_tissue_connectivity"] = (
                2 * self.tissue_graph.number_of_edges() / 
                (self.tissue_graph.number_of_nodes() * (self.tissue_graph.number_of_nodes() - 1))
            )
        
        return aggregated

class MultiheadAttention(nn.Module):
    """
    Multi-head attention mechanism with complex-valued weights option.
    
    A more efficient implementation of multihead attention designed for
    harmonic processing.
    """
    
    def __init__(self,
                d_model: int,
                num_heads: int = 8,
                dropout: float = 0.1,
                complex_valued: bool = True):
        """
        Initialize a multihead attention module.
        
        Args:
            d_model: Dimension of model
            num_heads: Number of attention heads
            dropout: Dropout probability
            complex_valued: Whether to use complex-valued weights
        """
        super().__init__()
        assert d_model % num_heads == 0, "d_model must be divisible by num_heads"
        
        self.d_model = d_model
        self.num_heads = num_heads
        self.head_dim = d_model // num_heads
        self.complex_valued = complex_valued
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Linear projections
        self.q_proj = nn.Linear(d_model, d_model, dtype=dtype)
        self.k_proj = nn.Linear(d_model, d_model, dtype=dtype)
        self.v_proj = nn.Linear(d_model, d_model, dtype=dtype)
        self.out_proj = nn.Linear(d_model, d_model, dtype=dtype)
        
        # Dropout
        self.dropout = nn.Dropout(dropout)
        
        # Initialize weights
        self._reset_parameters()
    
    def _reset_parameters(self):
        """Initialize weights with appropriate scaling."""
        # Xavier initialization
        gain = 1.0 / math.sqrt(2.0) if self.complex_valued else 1.0
        
        nn.init.xavier_uniform_(self.q_proj.weight, gain=gain)
        nn.init.xavier_uniform_(self.k_proj.weight, gain=gain)
        nn.init.xavier_uniform_(self.v_proj.weight, gain=gain)
        nn.init.xavier_uniform_(self.out_proj.weight, gain=gain)
        
        nn.init.zeros_(self.q_proj.bias)
        nn.init.zeros_(self.k_proj.bias)
        nn.init.zeros_(self.v_proj.bias)
        nn.init.zeros_(self.out_proj.bias)
    
    def forward(self, 
               query: torch.Tensor,
               key: Optional[torch.Tensor] = None,
               value: Optional[torch.Tensor] = None,
               attn_mask: Optional[torch.Tensor] = None) -> Tuple[torch.Tensor, torch.Tensor]:
        """
        Forward pass for multihead attention.
        
        Args:
            query: Query tensor (batch_size, seq_len_q, d_model)
            key: Key tensor (batch_size, seq_len_k, d_model)
            value: Value tensor (batch_size, seq_len_v, d_model)
            attn_mask: Attention mask
            
        Returns:
            Tuple of (output tensor, attention weights)
        """
        batch_size = query.shape[0]
        
        # Default key and value to query if not provided
        if key is None:
            key = query
        if value is None:
            value = key
        
        # Convert inputs to complex if needed and using complex mode
        if self.complex_valued and not torch.is_complex(query):
            query = torch.complex(query, torch.zeros_like(query))
            key = torch.complex(key, torch.zeros_like(key))
            value = torch.complex(value, torch.zeros_like(value))
        
        # Linear projections
        q = self.q_proj(query)
        k = self.k_proj(key)
        v = self.v_proj(value)
        
        # Reshape for multi-head attention
        q = q.view(batch_size, -1, self.num_heads, self.head_dim).transpose(1, 2)
        k = k.view(batch_size, -1, self.num_heads, self.head_dim).transpose(1, 2)
        v = v.view(batch_size, -1, self.num_heads, self.head_dim).transpose(1, 2)
        
        # Compute attention scores
        scores = torch.matmul(q, k.transpose(-2, -1)) / math.sqrt(self.head_dim)
        
        # Apply mask if provided
        if attn_mask is not None:
            scores = scores.masked_fill(attn_mask == 0, -1e9)
        
        # Apply attention weights
        if self.complex_valued:
            # For complex values, use real part for attention weights
            attn_weights = F.softmax(scores.real, dim=-1)
        else:
            attn_weights = F.softmax(scores, dim=-1)
        
        # Apply dropout to attention weights
        attn_weights = self.dropout(attn_weights)
        
        # Apply attention to values
        if self.complex_valued:
            # For complex values, convert attention weights back to complex
            attn_weights = torch.complex(attn_weights, torch.zeros_like(attn_weights))
        
        context = torch.matmul(attn_weights, v)
        
        # Reshape back
        context = context.transpose(1, 2).contiguous().view(batch_size, -1, self.d_model)
        
        # Final projection
        output = self.out_proj(context)
        
        return output, attn_weights


class TransformerEncoderLayer(nn.Module):
    """
    Transformer encoder layer with complex-valued option.
    
    Implements a single layer of the transformer encoder architecture
    with support for complex-valued computations.
    """
    
    def __init__(self,
                d_model: int,
                nhead: int = 8,
                dim_feedforward: int = 2048,
                dropout: float = 0.1,
                activation: str = 'gelu',
                complex_valued: bool = True,
                norm_first: bool = False):
        """
        Initialize a transformer encoder layer.
        
        Args:
            d_model: Dimension of model
            nhead: Number of attention heads
            dim_feedforward: Dimension of feedforward network
            dropout: Dropout probability
            activation: Activation function
            complex_valued: Whether to use complex-valued weights
            norm_first: Whether to apply normalization before attention/feedforward
        """
        super().__init__()
        self.d_model = d_model
        self.nhead = nhead
        self.dim_feedforward = dim_feedforward
        self.dropout_prob = dropout
        self.complex_valued = complex_valued
        self.norm_first = norm_first
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Self-attention
        self.self_attn = MultiheadAttention(
            d_model=d_model,
            num_heads=nhead,
            dropout=dropout,
            complex_valued=complex_valued
        )
        
        # Feedforward network
        self.linear1 = nn.Linear(d_model, dim_feedforward, dtype=dtype)
        self.dropout = nn.Dropout(dropout)
        self.linear2 = nn.Linear(dim_feedforward, d_model, dtype=dtype)
        
        # Layer normalization
        self.norm1 = nn.LayerNorm(d_model, elementwise_affine=True)
        self.norm2 = nn.LayerNorm(d_model, elementwise_affine=True)
        
        # Dropout
        self.dropout1 = nn.Dropout(dropout)
        self.dropout2 = nn.Dropout(dropout)
        
        # Activation function
        if activation == 'relu':
            self.activation = F.relu
        elif activation == 'gelu':
            self.activation = F.gelu
        elif activation == 'tanh':
            self.activation = torch.tanh
        else:
            raise ValueError(f"Unsupported activation: {activation}")
    
    def _sa_block(self, x: torch.Tensor, attn_mask: Optional[torch.Tensor] = None) -> torch.Tensor:
        """Self-attention block."""
        x_attend, _ = self.self_attn(x, attn_mask=attn_mask)
        return self.dropout1(x_attend)
    
    def _ff_block(self, x: torch.Tensor) -> torch.Tensor:
        """Feedforward block."""
        if self.complex_valued:
            # For complex values, apply activation to real and imaginary parts
            x = self.linear1(x)
            real = self.activation(x.real)
            imag = self.activation(x.imag)
            x = torch.complex(real, imag)
            x = self.dropout(x)
            x = self.linear2(x)
        else:
            x = self.linear2(self.dropout(self.activation(self.linear1(x))))
        return self.dropout2(x)
    
    def forward(self, 
               src: torch.Tensor, 
               src_mask: Optional[torch.Tensor] = None) -> torch.Tensor:
        """
        Forward pass for transformer encoder layer.
        
        Args:
            src: Source tensor (batch_size, seq_len, d_model)
            src_mask: Source mask
            
        Returns:
            Output tensor
        """
        # Convert input to complex if needed and using complex mode
        if self.complex_valued and not torch.is_complex(src):
            src = torch.complex(src, torch.zeros_like(src))
        
        # Apply layer norm first if specified
        if self.norm_first:
            # Apply self-attention with normalization
            if self.complex_valued:
                x = src + self._sa_block(self._complex_layer_norm(self.norm1, src), src_mask)
                x = x + self._ff_block(self._complex_layer_norm(self.norm2, x))
            else:
                x = src + self._sa_block(self.norm1(src), src_mask)
                x = x + self._ff_block(self.norm2(x))
        else:
            # Apply self-attention followed by normalization
            if self.complex_valued:
                x = self._complex_layer_norm(self.norm1, src + self._sa_block(src, src_mask))
                x = self._complex_layer_norm(self.norm2, x + self._ff_block(x))
            else:
                x = self.norm1(src + self._sa_block(src, src_mask))
                x = self.norm2(x + self._ff_block(x))
        
        return x
    
    def _complex_layer_norm(self, norm: nn.LayerNorm, x: torch.Tensor) -> torch.Tensor:
        """Apply layer normalization to complex tensor."""
        return torch.complex(
            norm(x.real),
            norm(x.imag)
        )


class TransformerEncoder(nn.Module):
    """
    Transformer encoder with multiple layers.
    
    Stacks multiple transformer encoder layers for deep representation learning.
    """
    
    def __init__(self,
                num_layers: int,
                d_model: int,
                nhead: int = 8,
                dim_feedforward: int = 2048,
                dropout: float = 0.1,
                activation: str = 'gelu',
                complex_valued: bool = True,
                norm_first: bool = False):
        """
        Initialize a transformer encoder.
        
        Args:
            num_layers: Number of encoder layers
            d_model: Dimension of model
            nhead: Number of attention heads
            dim_feedforward: Dimension of feedforward network
            dropout: Dropout probability
            activation: Activation function
            complex_valued: Whether to use complex-valued weights
            norm_first: Whether to apply normalization before attention/feedforward
        """
        super().__init__()
        
        # Create layers
        self.layers = nn.ModuleList([
            TransformerEncoderLayer(
                d_model=d_model,
                nhead=nhead,
                dim_feedforward=dim_feedforward,
                dropout=dropout,
                activation=activation,
                complex_valued=complex_valued,
                norm_first=norm_first
            ) for _ in range(num_layers)
        ])
        
        # Layer normalization
        self.norm = nn.LayerNorm(d_model)
        self.complex_valued = complex_valued
    
    def forward(self, 
               src: torch.Tensor,
               mask: Optional[torch.Tensor] = None) -> torch.Tensor:
        """
        Forward pass for transformer encoder.
        
        Args:
            src: Source tensor (batch_size, seq_len, d_model)
            mask: Source mask
            
        Returns:
            Output tensor
        """
        output = src
        
        # Process through each layer
        for layer in self.layers:
            output = layer(output, src_mask=mask)
        
        # Apply final normalization
        if self.complex_valued:
            output = torch.complex(
                self.norm(output.real),
                self.norm(output.imag)
            )
        else:
            output = self.norm(output)
        
        return output


class StripedHyenaAttention(nn.Module):
    """
    Striped Hyena Attention for efficient long-range dependencies.
    
    This is an implementation inspired by Striped Hyena 2, which uses
    a combination of attention patterns to efficiently capture long-range
    dependencies in sequences.
    """
    
    def __init__(self,
                d_model: int,
                num_heads: int = 8,
                long_range_size: int = 1024,
                short_range_size: int = 128,
                dropout: float = 0.1,
                complex_valued: bool = True):
        """
        Initialize a Striped Hyena Attention module.
        
        Args:
            d_model: Dimension of model
            num_heads: Number of attention heads
            long_range_size: Size of long-range attention window
            short_range_size: Size of short-range attention window
            dropout: Dropout probability
            complex_valued: Whether to use complex-valued weights
        """
        super().__init__()
        self.d_model = d_model
        self.num_heads = num_heads
        self.long_range_size = long_range_size
        self.short_range_size = short_range_size
        self.complex_valued = complex_valued
        
        # Split heads evenly between long and short range
        self.long_heads = num_heads // 2
        self.short_heads = num_heads - self.long_heads
        
        # Regular multihead attention for short-range
        self.short_attn = MultiheadAttention(
            d_model=d_model,
            num_heads=self.short_heads,
            dropout=dropout,
            complex_valued=complex_valued
        )
        
        # Long-range attention mechanism
        self.long_q = nn.Linear(d_model, d_model, dtype=torch.complex64 if complex_valued else torch.float32)
        self.long_k = nn.Linear(d_model, d_model, dtype=torch.complex64 if complex_valued else torch.float32)
        self.long_v = nn.Linear(d_model, d_model, dtype=torch.complex64 if complex_valued else torch.float32)
        
        # Strided convolutions for long-range
        self.conv_q = nn.Conv1d(
            d_model, d_model, kernel_size=3, stride=1, padding=1,
            dtype=torch.complex64 if complex_valued else torch.float32
        )
        self.conv_k = nn.Conv1d(
            d_model, d_model, kernel_size=3, stride=1, padding=1,
            dtype=torch.complex64 if complex_valued else torch.float32
        )
        
        # Output projection
        self.out_proj = nn.Linear(d_model, d_model, dtype=torch.complex64 if complex_valued else torch.float32)
        
        # Dropout
        self.dropout = nn.Dropout(dropout)
    
    def _create_striped_mask(self, seq_len: int, device: torch.device) -> torch.Tensor:
        """
        Create a striped attention mask.
        
        Args:
            seq_len: Sequence length
            device: Device for tensor
            
        Returns:
            Striped attention mask
        """
        # Create base mask (causal)
        mask = torch.triu(torch.ones(seq_len, seq_len, device=device), diagonal=1)
        mask = mask.masked_fill(mask == 1, float('-inf'))
        
        # Add stripes for long-range attention
        stride = self.long_range_size // 8
        for i in range(1, 9):
            offset = i * stride
            if offset < seq_len:
                for j in range(seq_len - offset):
                    mask[j, j + offset] = 0
        
        return mask
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Forward pass for Striped Hyena Attention.
        
        Args:
            x: Input tensor (batch_size, seq_len, d_model)
            
        Returns:
            Output tensor
        """
        batch_size, seq_len, _ = x.shape
        
        # Convert input to complex if needed and using complex mode
        if self.complex_valued and not torch.is_complex(x):
            x = torch.complex(x, torch.zeros_like(x))
        
        # Short-range attention
        # Create local attention mask (attend only to nearby tokens)
        short_mask = torch.ones(seq_len, seq_len, device=x.device)
        for i in range(seq_len):
            start = max(0, i - self.short_range_size)
            end = min(seq_len, i + self.short_range_size + 1)
            short_mask[i, start:end] = 0
        short_mask = short_mask.bool()
        
        short_out, _ = self.short_attn(x, attn_mask=~short_mask)
        
        # Long-range attention with strided convolutions
        # Project to query, key, value
        q = self.long_q(x)
        k = self.long_k(x)
        v = self.long_v(x)
        
        # Apply strided convolutions
        q = q.transpose(1, 2)  # [batch, d_model, seq_len]
        k = k.transpose(1, 2)
        
        q = self.conv_q(q).transpose(1, 2)  # [batch, seq_len, d_model]
        k = self.conv_k(k).transpose(1, 2)
        
        # Create striped mask for long-range attention
        striped_mask = self._create_striped_mask(seq_len, x.device)
        
        # Compute attention scores
        scores = torch.matmul(q, k.transpose(-2, -1)) / math.sqrt(self.d_model)
        scores = scores + striped_mask
        
        # Apply attention weights
        if self.complex_valued:
            # For complex values, use real part for attention weights
            attn_weights = F.softmax(scores.real, dim=-1)
            # Convert attention weights back to complex
            attn_weights = torch.complex(attn_weights, torch.zeros_like(attn_weights))
        else:
            attn_weights = F.softmax(scores, dim=-1)
        
        # Apply dropout to attention weights
        attn_weights = self.dropout(attn_weights)
        
        # Apply attention to values
        long_out = torch.matmul(attn_weights, v)
        
        # Combine short and long range attention
        combined = short_out + long_out
        
        # Final projection
        output = self.out_proj(combined)
        
        return output


class NeuralController(nn.Module):
    """
    Neural Controller for the LH-LLM.
    
    The Neural Controller directs the overall operation of the model,
    predicting optimal actions and coordinating cell behavior.
    """
    
    def __init__(self,
                d_model: int,
                num_heads: int = 8,
                num_layers: int = 4,
                dim_feedforward: int = 2048,
                dropout: float = 0.1,
                activation: str = 'gelu',
                complex_valued: bool = True,
                use_hyena: bool = True):
        """
        Initialize a Neural Controller.
        
        Args:
            d_model: Dimension of model
            num_heads: Number of attention heads
            num_layers: Number of transformer layers
            dim_feedforward: Dimension of feedforward network
            dropout: Dropout probability
            activation: Activation function
            complex_valued: Whether to use complex-valued weights
            use_hyena: Whether to use Striped Hyena Attention
        """
        super().__init__()
        self.d_model = d_model
        self.complex_valued = complex_valued
        self.use_hyena = use_hyena
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Input encoding
        self.input_encoding = nn.Linear(d_model, d_model, dtype=dtype)
        
        # Positional encoding
        self.pos_encoder = nn.Parameter(torch.randn(1, 1000, d_model, dtype=dtype) * 0.02)
        
        # Transformer or Striped Hyena
        if use_hyena:
            self.attention_layers = nn.ModuleList([
                StripedHyenaAttention(
                    d_model=d_model,
                    num_heads=num_heads,
                    dropout=dropout,
                    complex_valued=complex_valued
                ) for _ in range(num_layers)
            ])
            
            # Feedforward layers for Hyena
            self.ff_layers = nn.ModuleList([
                nn.Sequential(
                    nn.Linear(d_model, dim_feedforward, dtype=dtype),
                    nn.LayerNorm(dim_feedforward) if not complex_valued else nn.Identity(),
                    self._get_activation_fn(activation),
                    nn.Linear(dim_feedforward, d_model, dtype=dtype),
                ) for _ in range(num_layers)
            ])
            
            # Layer norms
            self.layer_norms1 = nn.ModuleList([
                nn.LayerNorm(d_model) for _ in range(num_layers)
            ])
            self.layer_norms2 = nn.ModuleList([
                nn.LayerNorm(d_model) for _ in range(num_layers)
            ])
        else:
            self.transformer = TransformerEncoder(
                num_layers=num_layers,
                d_model=d_model,
                nhead=num_heads,
                dim_feedforward=dim_feedforward,
                dropout=dropout,
                activation=activation,
                complex_valued=complex_valued
            )
        
        # Action prediction head
        self.action_head = nn.Linear(d_model, 3, dtype=dtype)  # 3 actions: evolve, replicate, optimize
        
        # Additional prediction heads
        self.fitness_head = nn.Linear(d_model, 1, dtype=dtype)
        self.coherence_head = nn.Linear(d_model, 1, dtype=dtype)
        self.quantum_head = nn.Linear(d_model, 1, dtype=dtype)
        
        # Activation for outputs
        self.activation = nn.Softmax(dim=-1)
    
    def _get_activation_fn(self, activation: str) -> Callable:
        """Get activation function."""
        if activation == 'relu':
            return nn.ReLU()
        elif activation == 'gelu':
            return nn.GELU()
        elif activation == 'tanh':
            return nn.Tanh()
        else:
            raise ValueError(f"Unsupported activation: {activation}")
    
    def _apply_complex_layernorm(self, norm: nn.LayerNorm, x: torch.Tensor) -> torch.Tensor:
        """Apply layer norm to complex tensor."""
        if self.complex_valued:
            return torch.complex(
                norm(x.real),
                norm(x.imag)
            )
        else:
            return norm(x)
    
    def forward(self, inputs: torch.Tensor) -> Dict[str, torch.Tensor]:
        """
        Forward pass for the Neural Controller.
        
        Args:
            inputs: Input tensor (batch_size, seq_len, d_model)
            
        Returns:
            Dictionary of output tensors
        """
        # Convert input to complex if needed and using complex mode
        if self.complex_valued and not torch.is_complex(inputs):
            inputs = torch.complex(inputs, torch.zeros_like(inputs))
        
        # Input encoding
        x = self.input_encoding(inputs)
        
        # Add positional encoding
        seq_len = x.shape[1]
        if seq_len <= self.pos_encoder.shape[1]:
            x = x + self.pos_encoder[:, :seq_len]
        else:
            # If sequence is longer than positional encoding, repeat
            repeat_times = (seq_len // self.pos_encoder.shape[1]) + 1
            extended_pos = self.pos_encoder.repeat(1, repeat_times, 1)
            x = x + extended_pos[:, :seq_len]
        
        # Pass through transformer or hyena layers
        if self.use_hyena:
            for i, (attn, ff, norm1, norm2) in enumerate(zip(
                self.attention_layers, self.ff_layers, self.layer_norms1, self.layer_norms2
            )):
                # Attention with residual connection and layer norm
                x_norm = self._apply_complex_layernorm(norm1, x)
                x = x + attn(x_norm)
                
                # Feedforward with residual connection and layer norm
                x_norm = self._apply_complex_layernorm(norm2, x)
                if self.complex_valued:
                    x_real = ff[0](x_norm.real)
                    x_imag = ff[0](x_norm.imag)
                    x_real = ff[2](x_real)
                    x_imag = ff[2](x_imag)
                    x_real = ff[3](x_real)
                    x_imag = ff[3](x_imag)
                    ff_out = torch.complex(x_real, x_imag)
                else:
                    ff_out = ff(x_norm)
                
                x = x + ff_out
        else:
            x = self.transformer(x)
        
        # Mean pooling over sequence dimension
        x_mean = x.mean(dim=1)
        
        # Prediction heads
        if self.complex_valued:
            action_logits = self.action_head(x_mean).real
            fitness = self.fitness_head(x_mean).real
            coherence = self.coherence_head(x_mean).real
            quantum_fidelity = self.quantum_head(x_mean).real
        else:
            action_logits = self.action_head(x_mean)
            fitness = self.fitness_head(x_mean)
            coherence = self.coherence_head(x_mean)
            quantum_fidelity = self.quantum_head(x_mean)
        
        # Apply activation
        action_probs = self.activation(action_logits)
        
        return {
            "action_probs": action_probs,
            "fitness": fitness,
            "coherence": coherence,
            "quantum_fidelity": quantum_fidelity
        }


class GraphMessagePassing(nn.Module):
    """
    Graph message passing for the Graph Metanetwork.
    
    Implements message passing between nodes in a graph neural network.
    """
    
    def __init__(self,
                in_channels: int,
                out_channels: int,
                aggr: str = 'add',
                complex_valued: bool = True):
        """
        Initialize a graph message passing module.
        
        Args:
            in_channels: Number of input channels
            out_channels: Number of output channels
            aggr: Aggregation method ('add', 'mean', or 'max')
            complex_valued: Whether to use complex-valued weights
        """
        super().__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.aggr = aggr
        self.complex_valued = complex_valued
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Node transformation
        self.node_transform = nn.Linear(in_channels, out_channels, dtype=dtype)
        
        # Edge transformation
        self.edge_transform = nn.Linear(in_channels + 1, out_channels, dtype=dtype)
        
        # Message transformation
        self.message_transform = nn.Linear(in_channels * 2 + 1, out_channels, dtype=dtype)
        
        # Update function
        self.update_fn = nn.Sequential(
            nn.Linear(in_channels + out_channels, out_channels, dtype=dtype),
            nn.ReLU() if not complex_valued else nn.Identity()
        )
        
        # Activation
        self.activation = nn.ReLU()
    
    def forward(self, 
               x: torch.Tensor, 
               edge_index: torch.Tensor, 
               edge_attr: torch.Tensor) -> torch.Tensor:
        """
        Forward pass for graph message passing.
        
        Args:
            x: Node features (num_nodes, in_channels)
            edge_index: Edge indices (2, num_edges)
            edge_attr: Edge attributes (num_edges, 1)
            
        Returns:
            Updated node features
        """
        # Convert inputs to complex if needed and using complex mode
        if self.complex_valued and not torch.is_complex(x):
            x = torch.complex(x, torch.zeros_like(x))
            edge_attr = torch.complex(edge_attr, torch.zeros_like(edge_attr))
        
        # Node transformation
        transformed_nodes = self.node_transform(x)
        
        # Compute messages
        row, col = edge_index
        
        # Edge transformation
        edge_features = torch.cat([x[row], edge_attr], dim=-1)
        transformed_edges = self.edge_transform(edge_features)
        
        # Message computation
        messages = torch.cat([x[row], x[col], edge_attr], dim=-1)
        transformed_messages = self.message_transform(messages)
        
        # Apply edge weights to messages
        weighted_messages = transformed_messages * transformed_edges
        
        # Aggregate messages
        aggregated_messages = self._aggregate(weighted_messages, row, x.size(0))
        
        # Update node features
        updated_nodes = torch.cat([x, aggregated_messages], dim=-1)
        new_features = self.update_fn(updated_nodes)
        
        # Add residual connection
        output = transformed_nodes + new_features
        
        # Apply activation
        if self.complex_valued:
            return torch.complex(
                self.activation(output.real),
                self.activation(output.imag)
            )
        else:
            return self.activation(output)
    
    def _aggregate(self, messages: torch.Tensor, row: torch.Tensor, num_nodes: int) -> torch.Tensor:
        """
        Aggregate messages per node.
        
        Args:
            messages: Message features
            row: Source node indices
            num_nodes: Number of nodes
            
        Returns:
            Aggregated messages
        """
        # Initialize output
        aggr_out = torch.zeros(num_nodes, messages.shape[-1], dtype=messages.dtype, device=messages.device)
        
        # Aggregate messages
        if self.aggr == 'add' or self.aggr == 'sum':
            aggr_out.index_add_(0, row, messages)
        elif self.aggr == 'mean':
            aggr_out.index_add_(0, row, messages)
            count = torch.zeros(num_nodes, dtype=torch.float, device=messages.device)
            count.index_add_(0, row, torch.ones_like(row, dtype=torch.float))
            count[count == 0] = 1
            aggr_out = aggr_out / count.unsqueeze(-1)
        elif self.aggr == 'max':
            # For max aggregation, use scatter_reduce
            # Note: scatter_reduce with 'amax' requires PyTorch 1.13+
            aggr_out = torch.scatter_reduce(
                messages, row, 0, 'amax', num_nodes
            )
        
        return aggr_out


class GraphMetanetwork(nn.Module):
    """
    Graph Metanetwork for the LH-LLM.
    
    The Graph Metanetwork manages communication and information flow
    between digital cells in a graph structure.
    """
    
    def __init__(self,
                in_channels: int,
                hidden_channels: int,
                out_channels: int,
                num_layers: int = 3,
                dropout: float = 0.1,
                complex_valued: bool = True):
        """
        Initialize a Graph Metanetwork.
        
        Args:
            in_channels: Number of input channels
            hidden_channels: Number of hidden channels
            out_channels: Number of output channels
            num_layers: Number of message passing layers
            dropout: Dropout probability
            complex_valued: Whether to use complex-valued weights
        """
        super().__init__()
        self.in_channels = in_channels
        self.hidden_channels = hidden_channels
        self.out_channels = out_channels
        self.num_layers = num_layers
        self.complex_valued = complex_valued
        
        # Input projection
        self.input_proj = nn.Linear(
            in_channels, hidden_channels, 
            dtype=torch.complex64 if complex_valued else torch.float32
        )
        
        # Message passing layers
        self.mp_layers = nn.ModuleList([
            GraphMessagePassing(
                in_channels=hidden_channels,
                out_channels=hidden_channels,
                complex_valued=complex_valued
            ) for _ in range(num_layers)
        ])
        
        # Output projection
        self.output_proj = nn.Linear(
            hidden_channels, out_channels,
            dtype=torch.complex64 if complex_valued else torch.float32
        )
        
        # Layer normalization
        self.norms = nn.ModuleList([
            nn.LayerNorm(hidden_channels) for _ in range(num_layers)
        ])
        
        # Dropout
        self.dropout = nn.Dropout(dropout)
    
    def forward(self, 
               x: torch.Tensor, 
               edge_index: torch.Tensor, 
               edge_attr: torch.Tensor) -> torch.Tensor:
        """
        Forward pass for Graph Metanetwork.
        
        Args:
            x: Node features (batch_size, num_nodes, in_channels)
            edge_index: Edge indices (2, num_edges)
            edge_attr: Edge attributes (num_edges, 1)
            
        Returns:
            Updated node features
        """
        batch_size, num_nodes, _ = x.shape
        
        # Process each batch item independently
        outputs = []
        for i in range(batch_size):
            # Get features for this batch item
            node_features = x[i]
            
            # Convert to complex if needed and using complex mode
            if self.complex_valued and not torch.is_complex(node_features):
                node_features = torch.complex(node_features, torch.zeros_like(node_features))
            
            # Input projection
            h = self.input_proj(node_features)
            
            # Message passing layers
            for j, (mp_layer, norm) in enumerate(zip(self.mp_layers, self.norms)):
                # Apply message passing
                h_new = mp_layer(h, edge_index, edge_attr)
                
                # Apply normalization
                if self.complex_valued:
                    h_real = norm(h_new.real)
                    h_imag = norm(h_new.imag)
                    h_new = torch.complex(h_real, h_imag)
                else:
                    h_new = norm(h_new)
                
                # Apply dropout
                h_new = self.dropout(h_new)
                
                # Residual connection
                h = h + h_new
            
            # Output projection
            output = self.output_proj(h)
            
            outputs.append(output)
        
        # Stack outputs
        return torch.stack(outputs, dim=0)


class DiffusionEvolution(nn.Module):
    """
    Diffusion Evolution for parameter optimization.
    
    Uses latent space diffusion to evolve model parameters,
    producing diverse and optimized solutions.
    """
    
    def __init__(self,
                latent_dim: int,
                param_dim: int,
                num_steps: int = 50,
                beta_schedule: str = 'linear',
                min_beta: float = 0.1,
                max_beta: float = 0.9):
        """
        Initialize a Diffusion Evolution module.
        
        Args:
            latent_dim: Dimension of latent space
            param_dim: Dimension of parameter space
            num_steps: Number of diffusion steps
            beta_schedule: Schedule for noise scale ('linear', 'cosine', or 'quadratic')
            min_beta: Minimum noise scale
            max_beta: Maximum noise scale
        """
        super().__init__()
        self.latent_dim = latent_dim
        self.param_dim = param_dim
        self.num_steps = num_steps
        
        # Create beta schedule
        if beta_schedule == 'linear':
            self.betas = torch.linspace(min_beta, max_beta, num_steps)
        elif beta_schedule == 'cosine':
            self.betas = torch.cos(torch.linspace(0, math.pi/2, num_steps)) * (max_beta - min_beta) + min_beta
        elif beta_schedule == 'quadratic':
            self.betas = torch.linspace(min_beta**0.5, max_beta**0.5, num_steps) ** 2
        else:
            raise ValueError(f"Unsupported beta schedule: {beta_schedule}")
        
        # Precompute diffusion parameters
        self.alphas = 1.0 - self.betas
        self.alphas_cumprod = torch.cumprod(self.alphas, dim=0)
        self.alphas_cumprod_prev = F.pad(self.alphas_cumprod[:-1], (1, 0), value=1.0)
        self.sqrt_alphas_cumprod = torch.sqrt(self.alphas_cumprod)
        self.sqrt_one_minus_alphas_cumprod = torch.sqrt(1.0 - self.alphas_cumprod)
        
        # Encoder (parameter space -> latent space)
        self.encoder = nn.Sequential(
            nn.Linear(param_dim, latent_dim * 2),
            nn.SiLU(),
            nn.Linear(latent_dim * 2, latent_dim)
        )
        
        # Decoder (latent space -> parameter space)
        self.decoder = nn.Sequential(
            nn.Linear(latent_dim, latent_dim * 2),
            nn.SiLU(),
            nn.Linear(latent_dim * 2, param_dim)
        )
        
        # Denoiser (predicts noise)
        self.denoiser = nn.Sequential(
            nn.Linear(latent_dim + 1, latent_dim * 2),  # +1 for time step embedding
            nn.SiLU(),
            nn.Linear(latent_dim * 2, latent_dim * 2),
            nn.SiLU(),
            nn.Linear(latent_dim * 2, latent_dim)
        )
    
    def encode(self, params: torch.Tensor) -> torch.Tensor:
        """
        Encode parameters to latent space.
        
        Args:
            params: Parameter tensor
            
        Returns:
            Latent representation
        """
        return self.encoder(params)
    
    def decode(self, latent: torch.Tensor) -> torch.Tensor:
        """
        Decode latent representation to parameters.
        
        Args:
            latent: Latent tensor
            
        Returns:
            Parameter tensor
        """
        return self.decoder(latent)
    
    def diffuse(self, latent: torch.Tensor, t: int) -> Tuple[torch.Tensor, torch.Tensor]:
        """
        Apply diffusion for a specific time step.
        
        Args:
            latent: Latent tensor
            t: Time step
            
        Returns:
            Tuple of (noisy latent, noise)
        """
        noise = torch.randn_like(latent)
        sqrt_alphas_cumprod_t = self.sqrt_alphas_cumprod[t].view(-1, 1)
        sqrt_one_minus_alphas_cumprod_t = self.sqrt_one_minus_alphas_cumprod[t].view(-1, 1)
        
        noisy_latent = sqrt_alphas_cumprod_t * latent + sqrt_one_minus_alphas_cumprod_t * noise
        return noisy_latent, noise

        return denoised_latent

class MultiheadAttention(nn.Module):
    """
    Multi-head attention mechanism with complex-valued weights option.
    
    A more efficient implementation of multihead attention designed for
    harmonic processing.
    """
    
    def __init__(self,
                d_model: int,
                num_heads: int = 8,
                dropout: float = 0.1,
                complex_valued: bool = True):
        """
        Initialize a multihead attention module.
        
        Args:
            d_model: Dimension of model
            num_heads: Number of attention heads
            dropout: Dropout probability
            complex_valued: Whether to use complex-valued weights
        """
        super().__init__()
        assert d_model % num_heads == 0, "d_model must be divisible by num_heads"
        
        self.d_model = d_model
        self.num_heads = num_heads
        self.head_dim = d_model // num_heads
        self.complex_valued = complex_valued
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Linear projections
        self.q_proj = nn.Linear(d_model, d_model, dtype=dtype)
        self.k_proj = nn.Linear(d_model, d_model, dtype=dtype)
        self.v_proj = nn.Linear(d_model, d_model, dtype=dtype)
        self.out_proj = nn.Linear(d_model, d_model, dtype=dtype)
        
        # Dropout
        self.dropout = nn.Dropout(dropout)
        
        # Initialize weights
        self._reset_parameters()
    
    def _reset_parameters(self):
        """Initialize weights with appropriate scaling."""
        # Xavier initialization
        gain = 1.0 / math.sqrt(2.0) if self.complex_valued else 1.0
        
        nn.init.xavier_uniform_(self.q_proj.weight, gain=gain)
        nn.init.xavier_uniform_(self.k_proj.weight, gain=gain)
        nn.init.xavier_uniform_(self.v_proj.weight, gain=gain)
        nn.init.xavier_uniform_(self.out_proj.weight, gain=gain)
        
        nn.init.zeros_(self.q_proj.bias)
        nn.init.zeros_(self.k_proj.bias)
        nn.init.zeros_(self.v_proj.bias)
        nn.init.zeros_(self.out_proj.bias)
    
    def forward(self, 
               query: torch.Tensor,
               key: Optional[torch.Tensor] = None,
               value: Optional[torch.Tensor] = None,
               attn_mask: Optional[torch.Tensor] = None) -> Tuple[torch.Tensor, torch.Tensor]:
        """
        Forward pass for multihead attention.
        
        Args:
            query: Query tensor (batch_size, seq_len_q, d_model)
            key: Key tensor (batch_size, seq_len_k, d_model)
            value: Value tensor (batch_size, seq_len_v, d_model)
            attn_mask: Attention mask
            
        Returns:
            Tuple of (output tensor, attention weights)
        """
        batch_size = query.shape[0]
        
        # Default key and value to query if not provided
        if key is None:
            key = query
        if value is None:
            value = key
        
        # Convert inputs to complex if needed and using complex mode
        if self.complex_valued and not torch.is_complex(query):
            query = torch.complex(query, torch.zeros_like(query))
            key = torch.complex(key, torch.zeros_like(key))
            value = torch.complex(value, torch.zeros_like(value))
        
        # Linear projections
        q = self.q_proj(query)
        k = self.k_proj(key)
        v = self.v_proj(value)
        
        # Reshape for multi-head attention
        q = q.view(batch_size, -1, self.num_heads, self.head_dim).transpose(1, 2)
        k = k.view(batch_size, -1, self.num_heads, self.head_dim).transpose(1, 2)
        v = v.view(batch_size, -1, self.num_heads, self.head_dim).transpose(1, 2)
        
        # Compute attention scores
        scores = torch.matmul(q, k.transpose(-2, -1)) / math.sqrt(self.head_dim)
        
        # Apply mask if provided
        if attn_mask is not None:
            scores = scores.masked_fill(attn_mask == 0, -1e9)
        
        # Apply attention weights
        if self.complex_valued:
            # For complex values, use real part for attention weights
            attn_weights = F.softmax(scores.real, dim=-1)
        else:
            attn_weights = F.softmax(scores, dim=-1)
        
        # Apply dropout to attention weights
        attn_weights = self.dropout(attn_weights)
        
        # Apply attention to values
        if self.complex_valued:
            # For complex values, convert attention weights back to complex
            attn_weights = torch.complex(attn_weights, torch.zeros_like(attn_weights))
        
        context = torch.matmul(attn_weights, v)
        
        # Reshape back
        context = context.transpose(1, 2).contiguous().view(batch_size, -1, self.d_model)
        
        # Final projection
        output = self.out_proj(context)
        
        return output, attn_weights


class TransformerEncoderLayer(nn.Module):
    """
    Transformer encoder layer with complex-valued option.
    
    Implements a single layer of the transformer encoder architecture
    with support for complex-valued computations.
    """
    
    def __init__(self,
                d_model: int,
                nhead: int = 8,
                dim_feedforward: int = 2048,
                dropout: float = 0.1,
                activation: str = 'gelu',
                complex_valued: bool = True,
                norm_first: bool = False):
        """
        Initialize a transformer encoder layer.
        
        Args:
            d_model: Dimension of model
            nhead: Number of attention heads
            dim_feedforward: Dimension of feedforward network
            dropout: Dropout probability
            activation: Activation function
            complex_valued: Whether to use complex-valued weights
            norm_first: Whether to apply normalization before attention/feedforward
        """
        super().__init__()
        self.d_model = d_model
        self.nhead = nhead
        self.dim_feedforward = dim_feedforward
        self.dropout_prob = dropout
        self.complex_valued = complex_valued
        self.norm_first = norm_first
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Self-attention
        self.self_attn = MultiheadAttention(
            d_model=d_model,
            num_heads=nhead,
            dropout=dropout,
            complex_valued=complex_valued
        )
        
        # Feedforward network
        self.linear1 = nn.Linear(d_model, dim_feedforward, dtype=dtype)
        self.dropout = nn.Dropout(dropout)
        self.linear2 = nn.Linear(dim_feedforward, d_model, dtype=dtype)
        
        # Layer normalization
        self.norm1 = nn.LayerNorm(d_model, elementwise_affine=True)
        self.norm2 = nn.LayerNorm(d_model, elementwise_affine=True)
        
        # Dropout
        self.dropout1 = nn.Dropout(dropout)
        self.dropout2 = nn.Dropout(dropout)
        
        # Activation function
        if activation == 'relu':
            self.activation = F.relu
        elif activation == 'gelu':
            self.activation = F.gelu
        elif activation == 'tanh':
            self.activation = torch.tanh
        else:
            raise ValueError(f"Unsupported activation: {activation}")
    
    def _sa_block(self, x: torch.Tensor, attn_mask: Optional[torch.Tensor] = None) -> torch.Tensor:
        """Self-attention block."""
        x_attend, _ = self.self_attn(x, attn_mask=attn_mask)
        return self.dropout1(x_attend)
    
    def _ff_block(self, x: torch.Tensor) -> torch.Tensor:
        """Feedforward block."""
        if self.complex_valued:
            # For complex values, apply activation to real and imaginary parts
            x = self.linear1(x)
            real = self.activation(x.real)
            imag = self.activation(x.imag)
            x = torch.complex(real, imag)
            x = self.dropout(x)
            x = self.linear2(x)
        else:
            x = self.linear2(self.dropout(self.activation(self.linear1(x))))
        return self.dropout2(x)
    
    def forward(self, 
               src: torch.Tensor, 
               src_mask: Optional[torch.Tensor] = None) -> torch.Tensor:
        """
        Forward pass for transformer encoder layer.
        
        Args:
            src: Source tensor (batch_size, seq_len, d_model)
            src_mask: Source mask
            
        Returns:
            Output tensor
        """
        # Convert input to complex if needed and using complex mode
        if self.complex_valued and not torch.is_complex(src):
            src = torch.complex(src, torch.zeros_like(src))
        
        # Apply layer norm first if specified
        if self.norm_first:
            # Apply self-attention with normalization
            if self.complex_valued:
                x = src + self._sa_block(self._complex_layer_norm(self.norm1, src), src_mask)
                x = x + self._ff_block(self._complex_layer_norm(self.norm2, x))
            else:
                x = src + self._sa_block(self.norm1(src), src_mask)
                x = x + self._ff_block(self.norm2(x))
        else:
            # Apply self-attention followed by normalization
            if self.complex_valued:
                x = self._complex_layer_norm(self.norm1, src + self._sa_block(src, src_mask))
                x = self._complex_layer_norm(self.norm2, x + self._ff_block(x))
            else:
                x = self.norm1(src + self._sa_block(src, src_mask))
                x = self.norm2(x + self._ff_block(x))
        
        return x
    
    def _complex_layer_norm(self, norm: nn.LayerNorm, x: torch.Tensor) -> torch.Tensor:
        """Apply layer normalization to complex tensor."""
        return torch.complex(
            norm(x.real),
            norm(x.imag)
        )


class TransformerEncoder(nn.Module):
    """
    Transformer encoder with multiple layers.
    
    Stacks multiple transformer encoder layers for deep representation learning.
    """
    
    def __init__(self,
                num_layers: int,
                d_model: int,
                nhead: int = 8,
                dim_feedforward: int = 2048,
                dropout: float = 0.1,
                activation: str = 'gelu',
                complex_valued: bool = True,
                norm_first: bool = False):
        """
        Initialize a transformer encoder.
        
        Args:
            num_layers: Number of encoder layers
            d_model: Dimension of model
            nhead: Number of attention heads
            dim_feedforward: Dimension of feedforward network
            dropout: Dropout probability
            activation: Activation function
            complex_valued: Whether to use complex-valued weights
            norm_first: Whether to apply normalization before attention/feedforward
        """
        super().__init__()
        
        # Create layers
        self.layers = nn.ModuleList([
            TransformerEncoderLayer(
                d_model=d_model,
                nhead=nhead,
                dim_feedforward=dim_feedforward,
                dropout=dropout,
                activation=activation,
                complex_valued=complex_valued,
                norm_first=norm_first
            ) for _ in range(num_layers)
        ])
        
        # Layer normalization
        self.norm = nn.LayerNorm(d_model)
        self.complex_valued = complex_valued
    
    def forward(self, 
               src: torch.Tensor,
               mask: Optional[torch.Tensor] = None) -> torch.Tensor:
        """
        Forward pass for transformer encoder.
        
        Args:
            src: Source tensor (batch_size, seq_len, d_model)
            mask: Source mask
            
        Returns:
            Output tensor
        """
        output = src
        
        # Process through each layer
        for layer in self.layers:
            output = layer(output, src_mask=mask)
        
        # Apply final normalization
        if self.complex_valued:
            output = torch.complex(
                self.norm(output.real),
                self.norm(output.imag)
            )
        else:
            output = self.norm(output)
        
        return output


class StripedHyenaAttention(nn.Module):
    """
    Striped Hyena Attention for efficient long-range dependencies.
    
    This is an implementation inspired by Striped Hyena 2, which uses
    a combination of attention patterns to efficiently capture long-range
    dependencies in sequences.
    """
    
    def __init__(self,
                d_model: int,
                num_heads: int = 8,
                long_range_size: int = 1024,
                short_range_size: int = 128,
                dropout: float = 0.1,
                complex_valued: bool = True):
        """
        Initialize a Striped Hyena Attention module.
        
        Args:
            d_model: Dimension of model
            num_heads: Number of attention heads
            long_range_size: Size of long-range attention window
            short_range_size: Size of short-range attention window
            dropout: Dropout probability
            complex_valued: Whether to use complex-valued weights
        """
        super().__init__()
        self.d_model = d_model
        self.num_heads = num_heads
        self.long_range_size = long_range_size
        self.short_range_size = short_range_size
        self.complex_valued = complex_valued
        
        # Split heads evenly between long and short range
        self.long_heads = num_heads // 2
        self.short_heads = num_heads - self.long_heads
        
        # Regular multihead attention for short-range
        self.short_attn = MultiheadAttention(
            d_model=d_model,
            num_heads=self.short_heads,
            dropout=dropout,
            complex_valued=complex_valued
        )
        
        # Long-range attention mechanism
        self.long_q = nn.Linear(d_model, d_model, dtype=torch.complex64 if complex_valued else torch.float32)
        self.long_k = nn.Linear(d_model, d_model, dtype=torch.complex64 if complex_valued else torch.float32)
        self.long_v = nn.Linear(d_model, d_model, dtype=torch.complex64 if complex_valued else torch.float32)
        
        # Strided convolutions for long-range
        self.conv_q = nn.Conv1d(
            d_model, d_model, kernel_size=3, stride=1, padding=1,
            dtype=torch.complex64 if complex_valued else torch.float32
        )
        self.conv_k = nn.Conv1d(
            d_model, d_model, kernel_size=3, stride=1, padding=1,
            dtype=torch.complex64 if complex_valued else torch.float32
        )
        
        # Output projection
        self.out_proj = nn.Linear(d_model, d_model, dtype=torch.complex64 if complex_valued else torch.float32)
        
        # Dropout
        self.dropout = nn.Dropout(dropout)
    
    def _create_striped_mask(self, seq_len: int, device: torch.device) -> torch.Tensor:
        """
        Create a striped attention mask.
        
        Args:
            seq_len: Sequence length
            device: Device for tensor
            
        Returns:
            Striped attention mask
        """
        # Create base mask (causal)
        mask = torch.triu(torch.ones(seq_len, seq_len, device=device), diagonal=1)
        mask = mask.masked_fill(mask == 1, float('-inf'))
        
        # Add stripes for long-range attention
        stride = self.long_range_size // 8
        for i in range(1, 9):
            offset = i * stride
            if offset < seq_len:
                for j in range(seq_len - offset):
                    mask[j, j + offset] = 0
        
        return mask
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Forward pass for Striped Hyena Attention.
        
        Args:
            x: Input tensor (batch_size, seq_len, d_model)
            
        Returns:
            Output tensor
        """
        batch_size, seq_len, _ = x.shape
        
        # Convert input to complex if needed and using complex mode
        if self.complex_valued and not torch.is_complex(x):
            x = torch.complex(x, torch.zeros_like(x))
        
        # Short-range attention
        # Create local attention mask (attend only to nearby tokens)
        short_mask = torch.ones(seq_len, seq_len, device=x.device)
        for i in range(seq_len):
            start = max(0, i - self.short_range_size)
            end = min(seq_len, i + self.short_range_size + 1)
            short_mask[i, start:end] = 0
        short_mask = short_mask.bool()
        
        short_out, _ = self.short_attn(x, attn_mask=~short_mask)
        
        # Long-range attention with strided convolutions
        # Project to query, key, value
        q = self.long_q(x)
        k = self.long_k(x)
        v = self.long_v(x)
        
        # Apply strided convolutions
        q = q.transpose(1, 2)  # [batch, d_model, seq_len]
        k = k.transpose(1, 2)
        
        q = self.conv_q(q).transpose(1, 2)  # [batch, seq_len, d_model]
        k = self.conv_k(k).transpose(1, 2)
        
        # Create striped mask for long-range attention
        striped_mask = self._create_striped_mask(seq_len, x.device)
        
        # Compute attention scores
        scores = torch.matmul(q, k.transpose(-2, -1)) / math.sqrt(self.d_model)
        scores = scores + striped_mask
        
        # Apply attention weights
        if self.complex_valued:
            # For complex values, use real part for attention weights
            attn_weights = F.softmax(scores.real, dim=-1)
            # Convert attention weights back to complex
            attn_weights = torch.complex(attn_weights, torch.zeros_like(attn_weights))
        else:
            attn_weights = F.softmax(scores, dim=-1)
        
        # Apply dropout to attention weights
        attn_weights = self.dropout(attn_weights)
        
        # Apply attention to values
        long_out = torch.matmul(attn_weights, v)
        
        # Combine short and long range attention
        combined = short_out + long_out
        
        # Final projection
        output = self.out_proj(combined)
        
        return output


class NeuralController(nn.Module):
    """
    Neural Controller for the LH-LLM.
    
    The Neural Controller directs the overall operation of the model,
    predicting optimal actions and coordinating cell behavior.
    """
    
    def __init__(self,
                d_model: int,
                num_heads: int = 8,
                num_layers: int = 4,
                dim_feedforward: int = 2048,
                dropout: float = 0.1,
                activation: str = 'gelu',
                complex_valued: bool = True,
                use_hyena: bool = True):
        """
        Initialize a Neural Controller.
        
        Args:
            d_model: Dimension of model
            num_heads: Number of attention heads
            num_layers: Number of transformer layers
            dim_feedforward: Dimension of feedforward network
            dropout: Dropout probability
            activation: Activation function
            complex_valued: Whether to use complex-valued weights
            use_hyena: Whether to use Striped Hyena Attention
        """
        super().__init__()
        self.d_model = d_model
        self.complex_valued = complex_valued
        self.use_hyena = use_hyena
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Input encoding
        self.input_encoding = nn.Linear(d_model, d_model, dtype=dtype)
        
        # Positional encoding
        self.pos_encoder = nn.Parameter(torch.randn(1, 1000, d_model, dtype=dtype) * 0.02)
        
        # Transformer or Striped Hyena
        if use_hyena:
            self.attention_layers = nn.ModuleList([
                StripedHyenaAttention(
                    d_model=d_model,
                    num_heads=num_heads,
                    dropout=dropout,
                    complex_valued=complex_valued
                ) for _ in range(num_layers)
            ])
            
            # Feedforward layers for Hyena
            self.ff_layers = nn.ModuleList([
                nn.Sequential(
                    nn.Linear(d_model, dim_feedforward, dtype=dtype),
                    nn.LayerNorm(dim_feedforward) if not complex_valued else nn.Identity(),
                    self._get_activation_fn(activation),
                    nn.Linear(dim_feedforward, d_model, dtype=dtype),
                ) for _ in range(num_layers)
            ])
            
            # Layer norms
            self.layer_norms1 = nn.ModuleList([
                nn.LayerNorm(d_model) for _ in range(num_layers)
            ])
            self.layer_norms2 = nn.ModuleList([
                nn.LayerNorm(d_model) for _ in range(num_layers)
            ])
        else:
            self.transformer = TransformerEncoder(
                num_layers=num_layers,
                d_model=d_model,
                nhead=num_heads,
                dim_feedforward=dim_feedforward,
                dropout=dropout,
                activation=activation,
                complex_valued=complex_valued
            )
        
        # Action prediction head
        self.action_head = nn.Linear(d_model, 3, dtype=dtype)  # 3 actions: evolve, replicate, optimize
        
        # Additional prediction heads
        self.fitness_head = nn.Linear(d_model, 1, dtype=dtype)
        self.coherence_head = nn.Linear(d_model, 1, dtype=dtype)
        self.quantum_head = nn.Linear(d_model, 1, dtype=dtype)
        
        # Activation for outputs
        self.activation = nn.Softmax(dim=-1)
    
    def _get_activation_fn(self, activation: str) -> Callable:
        """Get activation function."""
        if activation == 'relu':
            return nn.ReLU()
        elif activation == 'gelu':
            return nn.GELU()
        elif activation == 'tanh':
            return nn.Tanh()
        else:
            raise ValueError(f"Unsupported activation: {activation}")
    
    def _apply_complex_layernorm(self, norm: nn.LayerNorm, x: torch.Tensor) -> torch.Tensor:
        """Apply layer norm to complex tensor."""
        if self.complex_valued:
            return torch.complex(
                norm(x.real),
                norm(x.imag)
            )
        else:
            return norm(x)
    
    def forward(self, inputs: torch.Tensor) -> Dict[str, torch.Tensor]:
        """
        Forward pass for the Neural Controller.
        
        Args:
            inputs: Input tensor (batch_size, seq_len, d_model)
            
        Returns:
            Dictionary of output tensors
        """
        # Convert input to complex if needed and using complex mode
        if self.complex_valued and not torch.is_complex(inputs):
            inputs = torch.complex(inputs, torch.zeros_like(inputs))
        
        # Input encoding
        x = self.input_encoding(inputs)
        
        # Add positional encoding
        seq_len = x.shape[1]
        if seq_len <= self.pos_encoder.shape[1]:
            x = x + self.pos_encoder[:, :seq_len]
        else:
            # If sequence is longer than positional encoding, repeat
            repeat_times = (seq_len // self.pos_encoder.shape[1]) + 1
            extended_pos = self.pos_encoder.repeat(1, repeat_times, 1)
            x = x + extended_pos[:, :seq_len]
        
        # Pass through transformer or hyena layers
        if self.use_hyena:
            for i, (attn, ff, norm1, norm2) in enumerate(zip(
                self.attention_layers, self.ff_layers, self.layer_norms1, self.layer_norms2
            )):
                # Attention with residual connection and layer norm
                x_norm = self._apply_complex_layernorm(norm1, x)
                x = x + attn(x_norm)
                
                # Feedforward with residual connection and layer norm
                x_norm = self._apply_complex_layernorm(norm2, x)
                if self.complex_valued:
                    x_real = ff[0](x_norm.real)
                    x_imag = ff[0](x_norm.imag)
                    x_real = ff[2](x_real)
                    x_imag = ff[2](x_imag)
                    x_real = ff[3](x_real)
                    x_imag = ff[3](x_imag)
                    ff_out = torch.complex(x_real, x_imag)
                else:
                    ff_out = ff(x_norm)
                
                x = x + ff_out
        else:
            x = self.transformer(x)
        
        # Mean pooling over sequence dimension
        x_mean = x.mean(dim=1)
        
        # Prediction heads
        if self.complex_valued:
            action_logits = self.action_head(x_mean).real
            fitness = self.fitness_head(x_mean).real
            coherence = self.coherence_head(x_mean).real
            quantum_fidelity = self.quantum_head(x_mean).real
        else:
            action_logits = self.action_head(x_mean)
            fitness = self.fitness_head(x_mean)
            coherence = self.coherence_head(x_mean)
            quantum_fidelity = self.quantum_head(x_mean)
        
        # Apply activation
        action_probs = self.activation(action_logits)
        
        return {
            "action_probs": action_probs,
            "fitness": fitness,
            "coherence": coherence,
            "quantum_fidelity": quantum_fidelity
        }


class GraphMessagePassing(nn.Module):
    """
    Graph message passing for the Graph Metanetwork.
    
    Implements message passing between nodes in a graph neural network.
    """
    
    def __init__(self,
                in_channels: int,
                out_channels: int,
                aggr: str = 'add',
                complex_valued: bool = True):
        """
        Initialize a graph message passing module.
        
        Args:
            in_channels: Number of input channels
            out_channels: Number of output channels
            aggr: Aggregation method ('add', 'mean', or 'max')
            complex_valued: Whether to use complex-valued weights
        """
        super().__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.aggr = aggr
        self.complex_valued = complex_valued
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Node transformation
        self.node_transform = nn.Linear(in_channels, out_channels, dtype=dtype)
        
        # Edge transformation
        self.edge_transform = nn.Linear(in_channels + 1, out_channels, dtype=dtype)
        
        # Message transformation
        self.message_transform = nn.Linear(in_channels * 2 + 1, out_channels, dtype=dtype)
        
        # Update function
        self.update_fn = nn.Sequential(
            nn.Linear(in_channels + out_channels, out_channels, dtype=dtype),
            nn.ReLU() if not complex_valued else nn.Identity()
        )
        
        # Activation
        self.activation = nn.ReLU()
    
    def forward(self, 
               x: torch.Tensor, 
               edge_index: torch.Tensor, 
               edge_attr: torch.Tensor) -> torch.Tensor:
        """
        Forward pass for graph message passing.
        
        Args:
            x: Node features (num_nodes, in_channels)
            edge_index: Edge indices (2, num_edges)
            edge_attr: Edge attributes (num_edges, 1)
            
        Returns:
            Updated node features
        """
        # Convert inputs to complex if needed and using complex mode
        if self.complex_valued and not torch.is_complex(x):
            x = torch.complex(x, torch.zeros_like(x))
            edge_attr = torch.complex(edge_attr, torch.zeros_like(edge_attr))
        
        # Node transformation
        transformed_nodes = self.node_transform(x)
        
        # Compute messages
        row, col = edge_index
        
        # Edge transformation
        edge_features = torch.cat([x[row], edge_attr], dim=-1)
        transformed_edges = self.edge_transform(edge_features)
        
        # Message computation
        messages = torch.cat([x[row], x[col], edge_attr], dim=-1)
        transformed_messages = self.message_transform(messages)
        
        # Apply edge weights to messages
        weighted_messages = transformed_messages * transformed_edges
        
        # Aggregate messages
        aggregated_messages = self._aggregate(weighted_messages, row, x.size(0))
        
        # Update node features
        updated_nodes = torch.cat([x, aggregated_messages], dim=-1)
        new_features = self.update_fn(updated_nodes)
        
        # Add residual connection
        output = transformed_nodes + new_features
        
        # Apply activation
        if self.complex_valued:
            return torch.complex(
                self.activation(output.real),
                self.activation(output.imag)
            )
        else:
            return self.activation(output)
    
    def _aggregate(self, messages: torch.Tensor, row: torch.Tensor, num_nodes: int) -> torch.Tensor:
        """
        Aggregate messages per node.
        
        Args:
            messages: Message features
            row: Source node indices
            num_nodes: Number of nodes
            
        Returns:
            Aggregated messages
        """
        # Initialize output
        aggr_out = torch.zeros(num_nodes, messages.shape[-1], dtype=messages.dtype, device=messages.device)
        
        # Aggregate messages
        if self.aggr == 'add' or self.aggr == 'sum':
            aggr_out.index_add_(0, row, messages)
        elif self.aggr == 'mean':
            aggr_out.index_add_(0, row, messages)
            count = torch.zeros(num_nodes, dtype=torch.float, device=messages.device)
            count.index_add_(0, row, torch.ones_like(row, dtype=torch.float))
            count[count == 0] = 1
            aggr_out = aggr_out / count.unsqueeze(-1)
        elif self.aggr == 'max':
            # For max aggregation, use scatter_reduce
            # Note: scatter_reduce with 'amax' requires PyTorch 1.13+
            aggr_out = torch.scatter_reduce(
                messages, row, 0, 'amax', num_nodes
            )
        
        return aggr_out


class GraphMetanetwork(nn.Module):
    """
    Graph Metanetwork for the LH-LLM.
    
    The Graph Metanetwork manages communication and information flow
    between digital cells in a graph structure.
    """
    
    def __init__(self,
                in_channels: int,
                hidden_channels: int,
                out_channels: int,
                num_layers: int = 3,
                dropout: float = 0.1,
                complex_valued: bool = True):
        """
        Initialize a Graph Metanetwork.
        
        Args:
            in_channels: Number of input channels
            hidden_channels: Number of hidden channels
            out_channels: Number of output channels
            num_layers: Number of message passing layers
            dropout: Dropout probability
            complex_valued: Whether to use complex-valued weights
        """
        super().__init__()
        self.in_channels = in_channels
        self.hidden_channels = hidden_channels
        self.out_channels = out_channels
        self.num_layers = num_layers
        self.complex_valued = complex_valued
        
        # Input projection
        self.input_proj = nn.Linear(
            in_channels, hidden_channels, 
            dtype=torch.complex64 if complex_valued else torch.float32
        )
        
        # Message passing layers
        self.mp_layers = nn.ModuleList([
            GraphMessagePassing(
                in_channels=hidden_channels,
                out_channels=hidden_channels,
                complex_valued=complex_valued
            ) for _ in range(num_layers)
        ])
        
        # Output projection
        self.output_proj = nn.Linear(
            hidden_channels, out_channels,
            dtype=torch.complex64 if complex_valued else torch.float32
        )
        
        # Layer normalization
        self.norms = nn.ModuleList([
            nn.LayerNorm(hidden_channels) for _ in range(num_layers)
        ])
        
        # Dropout
        self.dropout = nn.Dropout(dropout)
    
    def forward(self, 
               x: torch.Tensor, 
               edge_index: torch.Tensor, 
               edge_attr: torch.Tensor) -> torch.Tensor:
        """
        Forward pass for Graph Metanetwork.
        
        Args:
            x: Node features (batch_size, num_nodes, in_channels)
            edge_index: Edge indices (2, num_edges)
            edge_attr: Edge attributes (num_edges, 1)
            
        Returns:
            Updated node features
        """
        batch_size, num_nodes, _ = x.shape
        
        # Process each batch item independently
        outputs = []
        for i in range(batch_size):
            # Get features for this batch item
            node_features = x[i]
            
            # Convert to complex if needed and using complex mode
            if self.complex_valued and not torch.is_complex(node_features):
                node_features = torch.complex(node_features, torch.zeros_like(node_features))
            
            # Input projection
            h = self.input_proj(node_features)
            
            # Message passing layers
            for j, (mp_layer, norm) in enumerate(zip(self.mp_layers, self.norms)):
                # Apply message passing
                h_new = mp_layer(h, edge_index, edge_attr)
                
                # Apply normalization
                if self.complex_valued:
                    h_real = norm(h_new.real)
                    h_imag = norm(h_new.imag)
                    h_new = torch.complex(h_real, h_imag)
                else:
                    h_new = norm(h_new)
                
                # Apply dropout
                h_new = self.dropout(h_new)
                
                # Residual connection
                h = h + h_new
            
            # Output projection
            output = self.output_proj(h)
            
            outputs.append(output)
        
        # Stack outputs
        return torch.stack(outputs, dim=0)


class DiffusionEvolution(nn.Module):
    """
    Diffusion Evolution for parameter optimization.
    
    Uses latent space diffusion to evolve model parameters,
    producing diverse and optimized solutions.
    """
    
    def __init__(self,
                latent_dim: int,
                param_dim: int,
                num_steps: int = 50,
                beta_schedule: str = 'linear',
                min_beta: float = 0.1,
                max_beta: float = 0.9):
        """
        Initialize a Diffusion Evolution module.
        
        Args:
            latent_dim: Dimension of latent space
            param_dim: Dimension of parameter space
            num_steps: Number of diffusion steps
            beta_schedule: Schedule for noise scale ('linear', 'cosine', or 'quadratic')
            min_beta: Minimum noise scale
            max_beta: Maximum noise scale
        """
        super().__init__()
        self.latent_dim = latent_dim
        self.param_dim = param_dim
        self.num_steps = num_steps
        
        # Create beta schedule
        if beta_schedule == 'linear':
            self.betas = torch.linspace(min_beta, max_beta, num_steps)
        elif beta_schedule == 'cosine':
            self.betas = torch.cos(torch.linspace(0, math.pi/2, num_steps)) * (max_beta - min_beta) + min_beta
        elif beta_schedule == 'quadratic':
            self.betas = torch.linspace(min_beta**0.5, max_beta**0.5, num_steps) ** 2
        else:
            raise ValueError(f"Unsupported beta schedule: {beta_schedule}")
        
        # Precompute diffusion parameters
        self.alphas = 1.0 - self.betas
        self.alphas_cumprod = torch.cumprod(self.alphas, dim=0)
        self.alphas_cumprod_prev = F.pad(self.alphas_cumprod[:-1], (1, 0), value=1.0)
        self.sqrt_alphas_cumprod = torch.sqrt(self.alphas_cumprod)
        self.sqrt_one_minus_alphas_cumprod = torch.sqrt(1.0 - self.alphas_cumprod)
        
        # Encoder (parameter space -> latent space)
        self.encoder = nn.Sequential(
            nn.Linear(param_dim, latent_dim * 2),
            nn.SiLU(),
            nn.Linear(latent_dim * 2, latent_dim)
        )
        
        # Decoder (latent space -> parameter space)
        self.decoder = nn.Sequential(
            nn.Linear(latent_dim, latent_dim * 2),
            nn.SiLU(),
            nn.Linear(latent_dim * 2, param_dim)
        )
        
        # Denoiser (predicts noise)
        self.denoiser = nn.Sequential(
            nn.Linear(latent_dim + 1, latent_dim * 2),  # +1 for time step embedding
            nn.SiLU(),
            nn.Linear(latent_dim * 2, latent_dim * 2),
            nn.SiLU(),
            nn.Linear(latent_dim * 2, latent_dim)
        )
    
    def encode(self, params: torch.Tensor) -> torch.Tensor:
        """
        Encode parameters to latent space.
        
        Args:
            params: Parameter tensor
            
        Returns:
            Latent representation
        """
        return self.encoder(params)
    
    def decode(self, latent: torch.Tensor) -> torch.Tensor:
        """
        Decode latent representation to parameters.
        
        Args:
            latent: Latent tensor
            
        Returns:
            Parameter tensor
        """
        return self.decoder(latent)
    
    def diffuse(self, latent: torch.Tensor, t: int) -> Tuple[torch.Tensor, torch.Tensor]:
        """
        Apply diffusion for a specific time step.
        
        Args:
            latent: Latent tensor
            t: Time step
            
        Returns:
            Tuple of (noisy latent, noise)
        """
        noise = torch.randn_like(latent)
        sqrt_alphas_cumprod_t = self.sqrt_alphas_cumprod[t].view(-1, 1)
        sqrt_one_minus_alphas_cumprod_t = self.sqrt_one_minus_alphas_cumprod[t].view(-1, 1)
        
        noisy_latent = sqrt_alphas_cumprod_t * latent + sqrt_one_minus_alphas_cumprod_t * noise
        return noisy_latent, noise
    
    def denoise_step(self, noisy_latent: torch.Tensor, t: int) -> torch.Tensor:
        """
        Perform a single denoising step.
        
        Args:
            noisy_latent: Noisy latent tensor
            t: Time step
            
        Returns:
            Denoised latent tensor
        """
        # Create time embedding
        t_embed = torch.ones(noisy_latent.shape[0], 1, device=noisy_latent.device) * t / self.num_steps
        
        # Predict noise
        input_denoiser = torch.cat([noisy_latent, t_embed], dim=1)
        predicted_noise = self.denoiser(input_denoiser)
        
        # Get alpha and beta for current step
        alpha = self.alphas[t]
        beta = self.betas[t]
        
        # Compute denoised latent (reverse diffusion step)
        denoised_latent = (noisy_latent - beta * predicted_noise / torch.sqrt(1 - self.alphas_cumprod[t])) / torch.sqrt(alpha)
        
        return denoised_latent
    
    def sample(self, num_samples: int = 1, device: torch.device = None) -> torch.Tensor:
        """
        Sample parameters using diffusion.
        
        Args:
            num_samples: Number of samples to generate
            device: Device for tensors
            
        Returns:
            Sampled parameter tensor
        """
        device = device or next(self.parameters()).device
        
        # Start from random noise
        latent = torch.randn(num_samples, self.latent_dim, device=device)
        
        # Iteratively denoise
        for t in reversed(range(self.num_steps)):
            latent = self.denoise_step(latent, t)
        
        # Decode to parameter space
        params = self.decode(latent)
        
        return params
    
    def evolve(self, params: torch.Tensor, fitness_fn: Callable, num_steps: int = None) -> torch.Tensor:
        """
        Evolve parameters using diffusion and fitness selection.
        
        Args:
            params: Initial parameter tensor
            fitness_fn: Function to evaluate fitness of parameters
            num_steps: Number of diffusion steps (default: self.num_steps)
            
        Returns:
            Evolved parameter tensor
        """
        if num_steps is None:
            num_steps = self.num_steps
        
        batch_size = params.shape[0]
        device = params.device
        
        # Encode to latent space
        latent = self.encode(params)
        
        # Apply forward diffusion for half of the steps
        noisy_latent = latent
        for t in range(num_steps // 2):
            noisy_latent, _ = self.diffuse(noisy_latent, t)
        
        # Apply reverse diffusion
        evolved_latent = noisy_latent
        for t in reversed(range(num_steps // 2)):
            evolved_latent = self.denoise_step(evolved_latent, t)
        
        # Decode to parameter space
        evolved_params = self.decode(evolved_latent)
        
        # Evaluate fitness
        fitness = fitness_fn(evolved_params)
        
        # Keep only improvements
        original_fitness = fitness_fn(params)
        mask = (fitness > original_fitness).float().view(-1, 1)
        
        # Combine original and evolved parameters based on fitness
        result = mask * evolved_params + (1 - mask) * params
        
        return result


class NSGAOptimizer:
    """
    Non-dominated Sorting Genetic Algorithm (NSGA-II) for multi-objective optimization.
    
    This implements the NSGA-II algorithm for optimizing parameters with
    respect to multiple objective functions.
    """
    
    def __init__(self,
                param_dim: int,
                num_objectives: int,
                population_size: int = 100,
                mutation_rate: float = 0.1,
                crossover_rate: float = 0.7):
        """
        Initialize an NSGA-II optimizer.
        
        Args:
            param_dim: Dimension of parameter space
            num_objectives: Number of objective functions
            population_size: Size of population
            mutation_rate: Probability of mutation
            crossover_rate: Probability of crossover
        """
        self.param_dim = param_dim
        self.num_objectives = num_objectives
        self.population_size = population_size
        self.mutation_rate = mutation_rate
        self.crossover_rate = crossover_rate
    
    def _calculate_crowding_distance(self, objectives: torch.Tensor) -> torch.Tensor:
        """
        Calculate crowding distance for each individual.
        
        Args:
            objectives: Objective values for each individual
            
        Returns:
            Crowding distances
        """
        population_size, num_objectives = objectives.shape
        crowding_distance = torch.zeros(population_size, device=objectives.device)
        
        # For each objective
        for i in range(num_objectives):
            # Get values for this objective
            values = objectives[:, i]
            
            # Sort indices by this objective
            indices = torch.argsort(values)
            sorted_values = values[indices]
            
            # Set boundary points to infinity
            crowding_distance[indices[0]] = float('inf')
            crowding_distance[indices[-1]] = float('inf')
            
            # Compute crowding distance for intermediate points
            if population_size > 2:
                # Normalize by range
                value_range = sorted_values[-1] - sorted_values[0]
                if value_range > 0:
                    for j in range(1, population_size - 1):
                        crowding_distance[indices[j]] += (sorted_values[j+1] - sorted_values[j-1]) / value_range
        
        return crowding_distance
    
    def _non_dominated_sort(self, objectives: torch.Tensor) -> List[torch.Tensor]:
        """
        Perform non-dominated sorting.
        
        Args:
            objectives: Objective values for each individual
            
        Returns:
            List of arrays containing indices for each front
        """
        population_size = objectives.shape[0]
        domination_counts = torch.zeros(population_size, dtype=torch.int, device=objectives.device)
        dominated_sets = [[] for _ in range(population_size)]
        fronts = [[]]
        
        # For each individual p
        for p in range(population_size):
            # For each individual q
            for q in range(population_size):
                if p == q:
                    continue
                
                # Check if p dominates q
                p_dominates_q = torch.all(objectives[p] >= objectives[q]) and torch.any(objectives[p] > objectives[q])
                
                # Check if q dominates p
                q_dominates_p = torch.all(objectives[q] >= objectives[p]) and torch.any(objectives[q] > objectives[p])
                
                if p_dominates_q:
                    dominated_sets[p].append(q)
                elif q_dominates_p:
                    domination_counts[p] += 1
            
            # If p is not dominated by any other individual, add to first front
            if domination_counts[p] == 0:
                fronts[0].append(p)
        
        # Generate subsequent fronts
        i = 0
        while len(fronts[i]) > 0:
            next_front = []
            
            # For each individual p in current front
            for p in fronts[i]:
                # For each individual q dominated by p
                for q in dominated_sets[p]:
                    domination_counts[q] -= 1
                    
                    # If q is not dominated by any other individual, add to next front
                    if domination_counts[q] == 0:
                        next_front.append(q)
            
            i += 1
            fronts.append(next_front)
        
        # Remove empty last front
        if len(fronts[-1]) == 0:
            fronts.pop()
        
        return [torch.tensor(front, device=objectives.device) for front in fronts]
    
    def _tournament_selection(self, ranks: torch.Tensor, crowding_distances: torch.Tensor) -> int:
        """
        Perform tournament selection.
        
        Args:
            ranks: Rank for each individual
            crowding_distances: Crowding distance for each individual
            
        Returns:
            Index of selected individual
        """
        # Select two random individuals
        idx1 = torch.randint(0, len(ranks), (1,)).item()
        idx2 = torch.randint(0, len(ranks), (1,)).item()
        
        # Select the better individual
        if ranks[idx1] < ranks[idx2]:
            return idx1
        elif ranks[idx1] > ranks[idx2]:
            return idx2
        else:
            # If ranks are equal, select the one with larger crowding distance
            if crowding_distances[idx1] > crowding_distances[idx2]:
                return idx1
            else:
                return idx2
    
    def _crossover(self, parent1: torch.Tensor, parent2: torch.Tensor) -> Tuple[torch.Tensor, torch.Tensor]:
        """
        Perform crossover between two parents.
        
        Args:
            parent1: First parent
            parent2: Second parent
            
        Returns:
            Tuple of (child1, child2)
        """
        # Decide whether to perform crossover
        if torch.rand(1).item() < self.crossover_rate:
            # Simulated binary crossover (SBX)
            eta = 15  # Distribution index
            u = torch.rand(self.param_dim, device=parent1.device)
            beta = torch.where(u <= 0.5, 
                              (2 * u) ** (1 / (eta + 1)),
                              (1 / (2 * (1 - u))) ** (1 / (eta + 1)))
            
            child1 = 0.5 * ((1 + beta) * parent1 + (1 - beta) * parent2)
            child2 = 0.5 * ((1 - beta) * parent1 + (1 + beta) * parent2)
            
            return child1, child2
        else:
            return parent1.clone(), parent2.clone()
    
    def _mutate(self, individual: torch.Tensor) -> torch.Tensor:
        """
        Perform mutation on an individual.
        
        Args:
            individual: Individual to mutate
            
        Returns:
            Mutated individual
        """
        # Polynomial mutation
        eta = 20  # Distribution index
        
        # For each parameter
        for i in range(self.param_dim):
            # Decide whether to mutate this parameter
            if torch.rand(1).item() < self.mutation_rate:
                # Generate random number
                u = torch.rand(1, device=individual.device).item()
                
                # Calculate delta
                if u < 0.5:
                    delta = (2 * u) ** (1 / (eta + 1)) - 1
                else:
                    delta = 1 - (2 * (1 - u)) ** (1 / (eta + 1))
                
                # Apply mutation
                individual[i] += delta
        
        # Clip values to valid range
        individual = torch.clamp(individual, -1.0, 1.0)
        
        return individual
    
    def optimize(self, 
                initial_population: torch.Tensor, 
                objective_functions: List[Callable], 
                num_generations: int = 100) -> Tuple[torch.Tensor, torch.Tensor]:
        """
        Optimize parameters using NSGA-II.
        
        Args:
            initial_population: Initial population of parameters
            objective_functions: List of objective functions to maximize
            num_generations: Number of generations to run
            
        Returns:
            Tuple of (best_parameters, best_objectives)
        """
        assert len(objective_functions) == self.num_objectives, "Number of objective functions must match num_objectives"
        
        population = initial_population.clone()
        population_size, param_dim = population.shape
        device = population.device
        
        # Run for specified number of generations
        for generation in range(num_generations):
            # Evaluate objectives for current population
            objectives = torch.zeros(population_size, self.num_objectives, device=device)
            for i, objective_fn in enumerate(objective_functions):
                objectives[:, i] = objective_fn(population)
            
            # Perform non-dominated sorting
            fronts = self._non_dominated_sort(objectives)
            
            # Calculate crowding distance for each front
            crowding_distances = torch.zeros(population_size, device=device)
            for front in fronts:
                crowding_distances[front] = self._calculate_crowding_distance(objectives[front])
            
            # Assign ranks
            ranks = torch.zeros(population_size, dtype=torch.int, device=device)
            for i, front in enumerate(fronts):
                ranks[front] = i
            
            # Create new population
            new_population = []
            
            while len(new_population) < population_size:
                # Select parents
                parent1_idx = self._tournament_selection(ranks, crowding_distances)
                parent2_idx = self._tournament_selection(ranks, crowding_distances)
                
                # Perform crossover
                child1, child2 = self._crossover(population[parent1_idx], population[parent2_idx])
                
                # Perform mutation
                child1 = self._mutate(child1)
                child2 = self._mutate(child2)
                
                # Add children to new population
                new_population.append(child1)
                
                if len(new_population) < population_size:
                    new_population.append(child2)
            
            # Update population
            population = torch.stack(new_population)
        
        # Get final objectives
        final_objectives = torch.zeros(population_size, self.num_objectives, device=device)
        for i, objective_fn in enumerate(objective_functions):
            final_objectives[:, i] = objective_fn(population)
        
        # Get first front (Pareto-optimal solutions)
        fronts = self._non_dominated_sort(final_objectives)
        pareto_optimal = population[fronts[0]]
        pareto_objectives = final_objectives[fronts[0]]
        
        return pareto_optimal, pareto_objectives

class MultiheadAttention(nn.Module):
    """
    Multi-head attention mechanism with complex-valued weights option.
    
    A more efficient implementation of multihead attention designed for
    harmonic processing.
    """
    
    def __init__(self,
                d_model: int,
                num_heads: int = 8,
                dropout: float = 0.1,
                complex_valued: bool = True):
        """
        Initialize a multihead attention module.
        
        Args:
            d_model: Dimension of model
            num_heads: Number of attention heads
            dropout: Dropout probability
            complex_valued: Whether to use complex-valued weights
        """
        super().__init__()
        assert d_model % num_heads == 0, "d_model must be divisible by num_heads"
        
        self.d_model = d_model
        self.num_heads = num_heads
        self.head_dim = d_model // num_heads
        self.complex_valued = complex_valued
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Linear projections
        self.q_proj = nn.Linear(d_model, d_model, dtype=dtype)
        self.k_proj = nn.Linear(d_model, d_model, dtype=dtype)
        self.v_proj = nn.Linear(d_model, d_model, dtype=dtype)
        self.out_proj = nn.Linear(d_model, d_model, dtype=dtype)
        
        # Dropout
        self.dropout = nn.Dropout(dropout)
        
        # Initialize weights
        self._reset_parameters()
    
    def _reset_parameters(self):
        """Initialize weights with appropriate scaling."""
        # Xavier initialization
        gain = 1.0 / math.sqrt(2.0) if self.complex_valued else 1.0
        
        nn.init.xavier_uniform_(self.q_proj.weight, gain=gain)
        nn.init.xavier_uniform_(self.k_proj.weight, gain=gain)
        nn.init.xavier_uniform_(self.v_proj.weight, gain=gain)
        nn.init.xavier_uniform_(self.out_proj.weight, gain=gain)
        
        nn.init.zeros_(self.q_proj.bias)
        nn.init.zeros_(self.k_proj.bias)
        nn.init.zeros_(self.v_proj.bias)
        nn.init.zeros_(self.out_proj.bias)
    
    def forward(self, 
               query: torch.Tensor,
               key: Optional[torch.Tensor] = None,
               value: Optional[torch.Tensor] = None,
               attn_mask: Optional[torch.Tensor] = None) -> Tuple[torch.Tensor, torch.Tensor]:
        """
        Forward pass for multihead attention.
        
        Args:
            query: Query tensor (batch_size, seq_len_q, d_model)
            key: Key tensor (batch_size, seq_len_k, d_model)
            value: Value tensor (batch_size, seq_len_v, d_model)
            attn_mask: Attention mask
            
        Returns:
            Tuple of (output tensor, attention weights)
        """
        batch_size = query.shape[0]
        
        # Default key and value to query if not provided
        if key is None:
            key = query
        if value is None:
            value = key
        
        # Convert inputs to complex if needed and using complex mode
        if self.complex_valued and not torch.is_complex(query):
            query = torch.complex(query, torch.zeros_like(query))
            key = torch.complex(key, torch.zeros_like(key))
            value = torch.complex(value, torch.zeros_like(value))
        
        # Linear projections
        q = self.q_proj(query)
        k = self.k_proj(key)
        v = self.v_proj(value)
        
        # Reshape for multi-head attention
        q = q.view(batch_size, -1, self.num_heads, self.head_dim).transpose(1, 2)
        k = k.view(batch_size, -1, self.num_heads, self.head_dim).transpose(1, 2)
        v = v.view(batch_size, -1, self.num_heads, self.head_dim).transpose(1, 2)
        
        # Compute attention scores
        scores = torch.matmul(q, k.transpose(-2, -1)) / math.sqrt(self.head_dim)
        
        # Apply mask if provided
        if attn_mask is not None:
            scores = scores.masked_fill(attn_mask == 0, -1e9)
        
        # Apply attention weights
        if self.complex_valued:
            # For complex values, use real part for attention weights
            attn_weights = F.softmax(scores.real, dim=-1)
        else:
            attn_weights = F.softmax(scores, dim=-1)
        
        # Apply dropout to attention weights
        attn_weights = self.dropout(attn_weights)
        
        # Apply attention to values
        if self.complex_valued:
            # For complex values, convert attention weights back to complex
            attn_weights = torch.complex(attn_weights, torch.zeros_like(attn_weights))
        
        context = torch.matmul(attn_weights, v)
        
        # Reshape back
        context = context.transpose(1, 2).contiguous().view(batch_size, -1, self.d_model)
        
        # Final projection
        output = self.out_proj(context)
        
        return output, attn_weights


class TransformerEncoderLayer(nn.Module):
    """
    Transformer encoder layer with complex-valued option.
    
    Implements a single layer of the transformer encoder architecture
    with support for complex-valued computations.
    """
    
    def __init__(self,
                d_model: int,
                nhead: int = 8,
                dim_feedforward: int = 2048,
                dropout: float = 0.1,
                activation: str = 'gelu',
                complex_valued: bool = True,
                norm_first: bool = False):
        """
        Initialize a transformer encoder layer.
        
        Args:
            d_model: Dimension of model
            nhead: Number of attention heads
            dim_feedforward: Dimension of feedforward network
            dropout: Dropout probability
            activation: Activation function
            complex_valued: Whether to use complex-valued weights
            norm_first: Whether to apply normalization before attention/feedforward
        """
        super().__init__()
        self.d_model = d_model
        self.nhead = nhead
        self.dim_feedforward = dim_feedforward
        self.dropout_prob = dropout
        self.complex_valued = complex_valued
        self.norm_first = norm_first
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Self-attention
        self.self_attn = MultiheadAttention(
            d_model=d_model,
            num_heads=nhead,
            dropout=dropout,
            complex_valued=complex_valued
        )
        
        # Feedforward network
        self.linear1 = nn.Linear(d_model, dim_feedforward, dtype=dtype)
        self.dropout = nn.Dropout(dropout)
        self.linear2 = nn.Linear(dim_feedforward, d_model, dtype=dtype)
        
        # Layer normalization
        self.norm1 = nn.LayerNorm(d_model, elementwise_affine=True)
        self.norm2 = nn.LayerNorm(d_model, elementwise_affine=True)
        
        # Dropout
        self.dropout1 = nn.Dropout(dropout)
        self.dropout2 = nn.Dropout(dropout)
        
        # Activation function
        if activation == 'relu':
            self.activation = F.relu
        elif activation == 'gelu':
            self.activation = F.gelu
        elif activation == 'tanh':
            self.activation = torch.tanh
        else:
            raise ValueError(f"Unsupported activation: {activation}")
    
    def _sa_block(self, x: torch.Tensor, attn_mask: Optional[torch.Tensor] = None) -> torch.Tensor:
        """Self-attention block."""
        x_attend, _ = self.self_attn(x, attn_mask=attn_mask)
        return self.dropout1(x_attend)
    
    def _ff_block(self, x: torch.Tensor) -> torch.Tensor:
        """Feedforward block."""
        if self.complex_valued:
            # For complex values, apply activation to real and imaginary parts
            x = self.linear1(x)
            real = self.activation(x.real)
            imag = self.activation(x.imag)
            x = torch.complex(real, imag)
            x = self.dropout(x)
            x = self.linear2(x)
        else:
            x = self.linear2(self.dropout(self.activation(self.linear1(x))))
        return self.dropout2(x)
    
    def forward(self, 
               src: torch.Tensor, 
               src_mask: Optional[torch.Tensor] = None) -> torch.Tensor:
        """
        Forward pass for transformer encoder layer.
        
        Args:
            src: Source tensor (batch_size, seq_len, d_model)
            src_mask: Source mask
            
        Returns:
            Output tensor
        """
        # Convert input to complex if needed and using complex mode
        if self.complex_valued and not torch.is_complex(src):
            src = torch.complex(src, torch.zeros_like(src))
        
        # Apply layer norm first if specified
        if self.norm_first:
            # Apply self-attention with normalization
            if self.complex_valued:
                x = src + self._sa_block(self._complex_layer_norm(self.norm1, src), src_mask)
                x = x + self._ff_block(self._complex_layer_norm(self.norm2, x))
            else:
                x = src + self._sa_block(self.norm1(src), src_mask)
                x = x + self._ff_block(self.norm2(x))
        else:
            # Apply self-attention followed by normalization
            if self.complex_valued:
                x = self._complex_layer_norm(self.norm1, src + self._sa_block(src, src_mask))
                x = self._complex_layer_norm(self.norm2, x + self._ff_block(x))
            else:
                x = self.norm1(src + self._sa_block(src, src_mask))
                x = self.norm2(x + self._ff_block(x))
        
        return x
    
    def _complex_layer_norm(self, norm: nn.LayerNorm, x: torch.Tensor) -> torch.Tensor:
        """Apply layer normalization to complex tensor."""
        return torch.complex(
            norm(x.real),
            norm(x.imag)
        )

        self.complex_valued = complex_valued
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Cell states
        self.register_buffer('cell_states', torch.zeros(max_cells, dim, dtype=dtype))
        
        # Cell positions in hyperbolic space
        self.register_buffer('cell_positions', torch.zeros(max_cells, dim))
        
        # Cell activity indicators (1 = active, 0 = inactive)
        self.register_buffer('cell_active', torch.zeros(max_cells, dtype=torch.bool))
        
        # Neighborhood matrix (adjacency)
        self.register_buffer('neighborhood', torch.zeros(max_cells, max_cells))
        
        # CA Rules
        self.rule_weights = nn.Parameter(torch.randn(3, 3, dtype=dtype) * 0.1)
        
        # Initialize cells
        self._initialize_cells()
    
    def _initialize_cells(self) -> None:
        """Initialize cell positions and states."""
        # Activate initial cells
        self.cell_active[:self.initial_cells] = True
        
        # Arrange initial cells in hyperbolic space
        for i in range(self.initial_cells):
            # Use Poincaré disk model for hyperbolic space
            radius = np.random.uniform(0, 0.9)  # Keep away from boundary
            angle = 2 * np.pi * i / self.initial_cells
            
            # Convert to Cartesian coordinates
            self.cell_positions[i, 0] = radius * np.cos(angle)
            self.cell_positions[i, 1] = radius * np.sin(angle)
            
            # Random initial state
            if self.complex_valued:
                self.cell_states[i] = torch.complex(
                    torch.rand(self.dim),
                    torch.rand(self.dim)
                )
            else:
                self.cell_states[i] = torch.rand(self.dim)
        
        # Compute neighborhoods
        self._compute_neighborhoods()
    
    def _hyperbolic_distance(self, p1: torch.Tensor, p2: torch.Tensor) -> torch.Tensor:
        """
        Compute hyperbolic distance between points in Poincaré disk model.
        
        Args:
            p1: First point
            p2: Second point
            
        Returns:
            Hyperbolic distance
        """
        # Compute Euclidean distance
        euclidean_dist = torch.norm(p1 - p2)
        
        # Convert to hyperbolic distance using Poincaré disk model formula
        numer = 2 * euclidean_dist
        denom = (1 - torch.sum(p1**2)) * (1 - torch.sum(p2**2))
        
        # Handle numerical stability
        eps = 1e-6
        denom = torch.clamp(denom, min=eps)
        
        # Compute hyperbolic distance
        return torch.acosh(1 + 2 * numer**2 / denom)
    
    def _compute_neighborhoods(self) -> None:
        """Compute neighborhood relations based on hyperbolic distances."""
        # Get active cells
        active = self.cell_active.nonzero().squeeze(-1)
        num_active = active.shape[0]
        
        # Compute pairwise distances
        for i in range(num_active):
            for j in range(i+1, num_active):
                idx1 = active[i]
                idx2 = active[j]
                
                # Compute hyperbolic distance
                dist = self._hyperbolic_distance(self.cell_positions[idx1], self.cell_positions[idx2])
                
                # Set neighborhood based on distance threshold
                # Note: In hyperbolic space, the number of points within a radius grows exponentially with radius
                threshold = 1.5  # Adjust based on curvature
                if dist < threshold:
                    self.neighborhood[idx1, idx2] = 1.0
                    self.neighborhood[idx2, idx1] = 1.0
    
    def _apply_ca_rules(self, states: torch.Tensor, neighbors: torch.Tensor) -> torch.Tensor:
        """
        Apply Cellular Automata rules.
        
        Args:
            states: Cell states
            neighbors: Neighbor matrix
            
        Returns:
            New cell states
        """
        # Get active cells
        active = self.cell_active.nonzero().squeeze(-1)
        num_active = active.shape[0]
        
        # Initialize new states
        new_states = states.clone()
        
        # Apply rules
        for i in range(num_active):
            idx = active[i]
            
            # Get neighbors
            neighbor_indices = neighbors[idx].nonzero().squeeze(-1)
            
            if neighbor_indices.numel() > 0:
                # Compute neighborhood influence
                sum_neighbors = torch.sum(states[neighbor_indices], dim=0)
                avg_neighbors = sum_neighbors / neighbor_indices.numel()
                
                # Apply CA rules (simplified)
                if self.complex_valued:
                    # For complex values, apply rules separately to real and imaginary parts
                    real_influence = torch.matmul(
                        torch.stack([states[idx].real, avg_neighbors.real, torch.ones_like(states[idx].real)]),
                        self.rule_weights.real
                    )
                    imag_influence = torch.matmul(
                        torch.stack([states[idx].imag, avg_neighbors.imag, torch.ones_like(states[idx].imag)]),
                        self.rule_weights.imag
                    )
                    
                    # Combine real and imaginary parts
                    new_states[idx] = torch.complex(
                        torch.tanh(real_influence),
                        torch.tanh(imag_influence)
                    )
                else:
                    # For real values, apply rules directly
                    influence = torch.matmul(
                        torch.stack([states[idx], avg_neighbors, torch.ones_like(states[idx])]),
                        self.rule_weights
                    )
                    
                    new_states[idx] = torch.tanh(influence)
            
        return new_states
    
    def _add_new_cell(self) -> bool:
        """
        Add a new cell to the automaton.
        
        Returns:
            True if cell was added, False otherwise
        """
        # Check if we've reached maximum cells
        if torch.sum(self.cell_active) >= self.max_cells:
            return False
        
        # Find first inactive cell
        new_idx = torch.nonzero(~self.cell_active, as_tuple=True)[0][0].item()
        
        # Activate the cell
        self.cell_active[new_idx] = True
        
        # Get active cells
        active = self.cell_active.nonzero().squeeze(-1)
        num_active = active.shape[0]
        
        # Choose a random existing cell as parent
        if num_active > 1:
            parent_idx = active[torch.randint(0, num_active - 1, (1,)).item()]
            
            # Position new cell near parent in hyperbolic space
            # Move in random direction, maintaining hyperbolic constraints
            r_parent = torch.norm(self.cell_positions[parent_idx])
            theta_parent = torch.atan2(self.cell_positions[parent_idx, 1], self.cell_positions[parent_idx, 0])
            
            # Add small random offset
            dr = torch.rand(1).item() * 0.2
            dtheta = (torch.rand(1).item() - 0.5) * np.pi / 4
            
            # Calculate new position
            r_new = torch.clamp(r_parent + dr, 0, 0.95)  # Keep away from boundary
            theta_new = theta_parent + dtheta
            
            # Convert to Cartesian coordinates
            self.cell_positions[new_idx, 0] = r_new * torch.cos(theta_new)
            self.cell_positions[new_idx, 1] = r_new * torch.sin(theta_new)
            
            # Initialize state as mix of parent and random
            if self.complex_valued:
                random_state = torch.complex(torch.rand(self.dim), torch.rand(self.dim))
            else:
                random_state = torch.rand(self.dim)
            
            self.cell_states[new_idx] = 0.5 * self.cell_states[parent_idx] + 0.5 * random_state
            
            # Update neighborhoods
            self._compute_neighborhoods()
            
            return True
        
        return False
    
    def step(self, inputs: Optional[torch.Tensor] = None) -> torch.Tensor:
        """
        Perform one step of cellular automaton evolution.
        
        Args:
            inputs: Optional external inputs to cells
            
        Returns:
            Updated cell states
        """
        # Apply external inputs if provided
        if inputs is not None:
            active = self.cell_active.nonzero().squeeze(-1)
            num_active = min(active.shape[0], inputs.shape[0])
            
            # Apply inputs to active cells
            for i in range(num_active):
                idx = active[i]
                if i < inputs.shape[0]:
                    if self.complex_valued and not torch.is_complex(inputs[i]):
                        inputs_complex = torch.complex(inputs[i], torch.zeros_like(inputs[i]))
                    else:
                        inputs_complex = inputs[i]
                    
                    self.cell_states[idx] = 0.9 * self.cell_states[idx] + 0.1 * inputs_complex
        
        # Apply CA rules
        new_states = self._apply_ca_rules(self.cell_states, self.neighborhood)
        self.cell_states = new_states
        
        # Probabilistically add new cells
        if torch.rand(1).item() < 0.1:  # 10% chance per step
            self._add_new_cell()
        
        # Return states of active cells
        active = self.cell_active.nonzero().squeeze(-1)
        return self.cell_states[active]
    
    def get_adjacency_matrix(self) -> torch.Tensor:
        """
        Get adjacency matrix for active cells.
        
        Returns:
            Adjacency matrix
        """
        # Get active cells
        active = self.cell_active.nonzero().squeeze(-1)
        num_active = active.shape[0]
        
        # Extract submatrix for active cells
        adjacency = torch.zeros(num_active, num_active, device=self.neighborhood.device)
        for i in range(num_active):
            for j in range(num_active):
                adjacency[i, j] = self.neighborhood[active[i], active[j]]
        
        return adjacency


class SwarmChemistry(nn.Module):
    """
    Swarm Chemistry for emergent pattern formation.
    
    This module models digital cells as a swarm with recipe-based behaviors,
    enabling emergent pattern formation through simple local rules.
    """
    
    def __init__(self,
                num_cells: int = 100,
                num_recipes: int = 5,
                state_dim: int = 4,
                complex_valued: bool = True):
        """
        Initialize a Swarm Chemistry module.
        
        Args:
            num_cells: Number of cells in swarm
            num_recipes: Number of behavior recipes
            state_dim: Dimension of cell state
            complex_valued: Whether to use complex-valued computation
        """
        super().__init__()
        self.num_cells = num_cells
        self.num_recipes = num_recipes
        self.state_dim = state_dim
        self.complex_valued = complex_valued
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Cell states
        self.register_buffer('positions', torch.rand(num_cells, 2) * 2 - 1)  # [-1, 1]
        self.register_buffer('velocities', torch.zeros(num_cells, 2))
        self.register_buffer('cell_states', torch.zeros(num_cells, state_dim, dtype=dtype))
        
        # Assign recipes to cells
        self.register_buffer('cell_recipes', torch.randint(0, num_recipes, (num_cells,)))
        
        # Recipe parameters
        # Each recipe has: max_speed, max_accel, perception_radius, alignment, cohesion, separation
        self.recipe_params = nn.Parameter(torch.rand(num_recipes, 6))
        
        # Recipe transmission probability
        self.transmission_prob = 0.05
        
        # Initialize cell states
        self._initialize_cell_states()
    
    def _initialize_cell_states(self) -> None:
        """Initialize cell states."""
        for i in range(self.num_cells):
            if self.complex_valued:
                self.cell_states[i] = torch.complex(
                    torch.rand(self.state_dim),
                    torch.rand(self.state_dim)
                )
            else:
                self.cell_states[i] = torch.rand(self.state_dim)
    
    def _compute_swarm_forces(self) -> Tuple[torch.Tensor, torch.Tensor, torch.Tensor]:
        """
        Compute alignment, cohesion, and separation forces for swarm.
        
        Returns:
            Tuple of (alignment, cohesion, separation) forces
        """
        # Initialize forces
        alignment = torch.zeros_like(self.velocities)
        cohesion = torch.zeros_like(self.positions)
        separation = torch.zeros_like(self.positions)
        
        # Compute pairwise distances
        diffs = self.positions.unsqueeze(1) - self.positions.unsqueeze(0)  # [num_cells, num_cells, 2]
        dist_sq = torch.sum(diffs**2, dim=2)  # [num_cells, num_cells]
        
        # For each cell
        for i in range(self.num_cells):
            # Get recipe parameters
            recipe = self.cell_recipes[i].item()
            perception_radius = self.recipe_params[recipe, 2]
            
            # Find neighbors within perception radius
            neighbors = dist_sq[i] < perception_radius**2
            neighbors[i] = False  # Exclude self
            
            neighbor_count = torch.sum(neighbors)
            
            if neighbor_count > 0:
                # Alignment: average velocity of neighbors
                alignment[i] = torch.mean(self.velocities[neighbors], dim=0)
                
                # Cohesion: move toward center of mass of neighbors
                cohesion[i] = torch.mean(self.positions[neighbors], dim=0) - self.positions[i]
                
                # Separation: move away from close neighbors
                close_neighbors = dist_sq[i] < (perception_radius * 0.5)**2
                close_neighbors[i] = False  # Exclude self
                
                if torch.sum(close_neighbors) > 0:
                    # Compute repulsion direction
                    separation[i] = -torch.sum(
                        diffs[i, close_neighbors] / (torch.sqrt(dist_sq[i, close_neighbors]).unsqueeze(1) + 1e-8),
                        dim=0
                    )
        
        # Normalize forces
        alignment = self._normalize_vectors(alignment)
        cohesion = self._normalize_vectors(cohesion)
        separation = self._normalize_vectors(separation)
        
        return alignment, cohesion, separation
    
    def _normalize_vectors(self, vectors: torch.Tensor) -> torch.Tensor:
        """
        Normalize vectors to unit length.
        
        Args:
            vectors: Tensor of vectors
            
        Returns:
            Normalized vectors
        """
        norms = torch.norm(vectors, dim=1, keepdim=True)
        # Avoid division by zero
        return vectors / (norms + 1e-8)
    
    def _transmit_recipes(self) -> None:
        """Transmit recipes between cells based on proximity."""
        # Compute pairwise distances
        diffs = self.positions.unsqueeze(1) - self.positions.unsqueeze(0)
        dist_sq = torch.sum(diffs**2, dim=2)
        
        # For each cell
        for i in range(self.num_cells):
            # Get recipe parameters
            recipe = self.cell_recipes[i].item()
            perception_radius = self.recipe_params[recipe, 2]
            
            # Find neighbors within perception radius
            neighbors = dist_sq[i] < perception_radius**2
            neighbors[i] = False  # Exclude self
            
            # Transmit recipe with probability proportional to distance
            neighbor_indices = neighbors.nonzero().squeeze(-1)
            
            for j in neighbor_indices:
                # Closer neighbors have higher probability of recipe transmission
                prob = self.transmission_prob * (1 - torch.sqrt(dist_sq[i, j]) / perception_radius)
                
                if torch.rand(1).item() < prob:
                    self.cell_recipes[j] = self.cell_recipes[i]
    
    def step(self, dt: float = 0.1) -> Tuple[torch.Tensor, torch.Tensor]:
        """
        Perform one step of swarm evolution.
        
        Args:
            dt: Time step
            
        Returns:
            Tuple of (cell_states, positions)
        """
        # Compute swarm forces
        alignment, cohesion, separation = self._compute_swarm_forces()
        
        # Update velocities and positions
        for i in range(self.num_cells):
            # Get recipe parameters
            recipe = self.cell_recipes[i].item()
            max_speed = self.recipe_params[recipe, 0]
            max_accel = self.recipe_params[recipe, 1]
            align_weight = self.recipe_params[recipe, 3]
            cohesion_weight = self.recipe_params[recipe, 4]
            separation_weight = self.recipe_params[recipe, 5]
            
            # Compute acceleration
            accel = (
                align_weight * alignment[i] +
                cohesion_weight * cohesion[i] +
                separation_weight * separation[i]
            )
            
            # Limit acceleration
            accel_norm = torch.norm(accel)
            if accel_norm > max_accel:
                accel = accel * (max_accel / accel_norm)
            
            # Update velocity
            self.velocities[i] = self.velocities[i] + accel * dt
            
            # Limit velocity
            vel_norm = torch.norm(self.velocities[i])
            if vel_norm > max_speed:
                self.velocities[i] = self.velocities[i] * (max_speed / vel_norm)
            
            # Update position
            self.positions[i] = self.positions[i] + self.velocities[i] * dt
            
            # Wrap around boundaries
            self.positions[i] = torch.remainder(self.positions[i] + 1, 2) - 1
        
        # Transmit recipes between cells
        self._transmit_recipes()
        
        # Update cell states based on positions and velocities
        for i in range(self.num_cells):
            # Create positional encoding
            pos_encoding = torch.cat([
                torch.sin(self.positions[i] * 5),
                torch.cos(self.positions[i] * 5),
            ])
            
            # Update state with momentum
            if self.complex_valued:
                pos_encoding_complex = torch.complex(pos_encoding, torch.zeros_like(pos_encoding))
                state_update = 0.1 * pos_encoding_complex
                self.cell_states[i] = 0.9 * self.cell_states[i] + state_update[:self.state_dim]
            else:
                state_update = 0.1 * pos_encoding
                self.cell_states[i] = 0.9 * self.cell_states[i] + state_update[:self.state_dim]
        
        return self.cell_states, self.positions
    
    def get_recipe_distribution(self) -> torch.Tensor:
        """
        Get distribution of recipes in the swarm.
        
        Returns:
            Recipe counts
        """
        return torch.bincount(self.cell_recipes, minlength=self.num_recipes)


class SelfOrganizedCriticality(nn.Module):
    """
    Self-Organized Criticality for power-law dynamics.
    
    This module implements SOC dynamics with neuronal avalanches,
    maintaining the system at the edge of chaos (critical state).
    """
    
    def __init__(self,
                grid_size: int = 32,
                target_exponent: float = 1.5,
                complex_valued: bool = True):
        """
        Initialize a Self-Organized Criticality module.
        
        Args:
            grid_size: Size of SOC grid
            target_exponent: Target power-law exponent (typically ~1.5 for neuronal avalanches)
            complex_valued: Whether to use complex-valued computation
        """
        super().__init__()
        self.grid_size = grid_size
        self.target_exponent = target_exponent
        self.complex_valued = complex_valued
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Energy grid
        self.register_buffer('energy', torch.zeros(grid_size, grid_size, dtype=dtype))
        
        # Critical threshold
        self.threshold = nn.Parameter(torch.tensor(4.0))
        
        # Driving rate
        self.driving_rate = nn.Parameter(torch.tensor(0.01))
        
        # Dissipation rate
        self.dissipation_rate = nn.Parameter(torch.tensor(0.05))
        
        # Avalanche history
        self.avalanche_sizes = []
        self.avalanche_durations = []
        
        # Current avalanche state
        self.in_avalanche = False
        self.current_avalanche_size = 0
        self.current_avalanche_duration = 0
    
    def _add_energy(self) -> None:
        """Add energy to random sites."""
        # Add energy to random sites
        num_sites = int(self.grid_size * self.grid_size * self.driving_rate.item())
        if num_sites > 0:
            indices = torch.randint(0, self.grid_size, (num_sites, 2))
            
            for i, j in indices:
                if self.complex_valued:
                    self.energy[i, j] += torch.complex(torch.tensor(1.0), torch.tensor(0.0))
                else:
                    self.energy[i, j] += 1.0
    
    def _topple_sites(self) -> int:
        """
        Topple sites above threshold.
        
        Returns:
            Number of sites toppled
        """
        if self.complex_valued:
            above_threshold = torch.abs(self.energy) > self.threshold
        else:
            above_threshold = self.energy > self.threshold
        
        num_toppled = torch.sum(above_threshold).item()
        
        if num_toppled > 0:
            # Get indices of sites to topple
            topple_indices = above_threshold.nonzero()
            
            # For each site to topple
            for idx in topple_indices:
                i, j = idx[0].item(), idx[1].item()
                
                # Distribute energy to neighbors
                energy_to_distribute = self.energy[i, j] - self.threshold * 0.5
                self.energy[i, j] = self.threshold * 0.5
                
                # Amount each neighbor receives
                amount = energy_to_distribute / 4.0
                
                # Distribute to neighbors (with periodic boundary conditions)
                neighbors = [
                    ((i-1) % self.grid_size, j),
                    ((i+1) % self.grid_size, j),
                    (i, (j-1) % self.grid_size),
                    (i, (j+1) % self.grid_size)
                ]
                
                for ni, nj in neighbors:
                    self.energy[ni, nj] += amount
        
        return num_toppled
    
    def _dissipate_energy(self) -> None:
        """Dissipate energy from all sites."""
        self.energy *= (1.0 - self.dissipation_rate)
    
    def step(self) -> Tuple[torch.Tensor, Dict]:
        """
        Perform one step of SOC dynamics.
        
        Returns:
            Tuple of (energy grid, metrics)
        """
        # Add energy
        self._add_energy()
        
        # Topple sites
        num_toppled = self._topple_sites()
        
        # Update avalanche state
        if num_toppled > 0:
            if not self.in_avalanche:
                # Start new avalanche
                self.in_avalanche = True
                self.current_avalanche_size = num_toppled
                self.current_avalanche_duration = 1
            else:
                # Continue current avalanche
                self.current_avalanche_size += num_toppled
                self.current_avalanche_duration += 1
        elif self.in_avalanche:
            # End current avalanche
            self.in_avalanche = False
            self.avalanche_sizes.append(self.current_avalanche_size)
            self.avalanche_durations.append(self.current_avalanche_duration)
            
            # Reset current avalanche
            self.current_avalanche_size = 0
            self.current_avalanche_duration = 0
        
        # Dissipate energy
        self._dissipate_energy()
        
        # Compute metrics
        metrics = self._compute_metrics()
        
        return self.energy, metrics
    
    def _compute_metrics(self) -> Dict:
        """
        Compute SOC metrics.
        
        Returns:
            Dictionary of metrics
        """
        metrics = {
            "avalanche_exponent": self.target_exponent,  # Default value
            "mean_energy": 0.0,
            "num_active_sites": 0,
            "largest_avalanche": 0,
            "total_avalanches": len(self.avalanche_sizes)
        }
        
        # Compute mean energy
        if self.complex_valued:
            metrics["mean_energy"] = torch.mean(torch.abs(self.energy)).item()
        else:
            metrics["mean_energy"] = torch.mean(self.energy).item()
        
        # Count active sites
        if self.complex_valued:
            metrics["num_active_sites"] = torch.sum(torch.abs(self.energy) > 0.1).item()
        else:
            metrics["num_active_sites"] = torch.sum(self.energy > 0.1).item()
        
        # Get largest avalanche
        if self.avalanche_sizes:
            metrics["largest_avalanche"] = max(self.avalanche_sizes)
        
        # Compute power-law exponent if enough data
        if len(self.avalanche_sizes) > 30:
            try:
                # Convert to numpy array
                sizes = np.array(self.avalanche_sizes)
                sizes = sizes[sizes > 0]  # Remove zeros
                
                # Fit power law
                fit = powerlaw.fit(sizes)
                metrics["avalanche_exponent"] = fit[0]
            except:
                pass  # Keep default value
        
        return metrics
    
    def get_avalanche_distribution(self) -> Tuple[np.ndarray, np.ndarray]:
        """
        Get avalanche size distribution.
        
        Returns:
            Tuple of (bin_centers, counts)
        """
        if len(self.avalanche_sizes) < 10:
            return np.array([]), np.array([])
        
        # Convert to numpy array
        sizes = np.array(self.avalanche_sizes)
        sizes = sizes[sizes > 0]  # Remove zeros
        
        # Use logarithmic binning
        log_min = np.log10(np.min(sizes))
        log_max = np.log10(np.max(sizes))
        num_bins = 10
        
        bins = np.logspace(log_min, log_max, num_bins)
        counts, bin_edges = np.histogram(sizes, bins=bins, density=True)
        bin_centers = 0.5 * (bin_edges[1:] + bin_edges[:-1])
        
        return bin_centers, counts


class ReactionSystem(nn.Module):
    """
    Reaction System for local decision-making.
    
    This module implements a chemical reaction-inspired system
    for local decision-making using stochastic differential equations.
    """
    
    def __init__(self,
                num_reactants: int = 8,
                num_reactions: int = 16,
                complex_valued: bool = True):
        """
        Initialize a Reaction System.
        
        Args:
            num_reactants: Number of reactant types
            num_reactions: Number of chemical reactions
            complex_valued: Whether to use complex-valued computation
        """
        super().__init__()
        self.num_reactants = num_reactants
        self.num_reactions = num_reactions
        self.complex_valued = complex_valued
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Reactant concentrations
        self.register_buffer('concentrations', torch.ones(num_reactants, dtype=dtype) * 0.5)
        
        # Reaction rates
        self.forward_rates = nn.Parameter(torch.rand(num_reactions, dtype=dtype) * 0.1)
        self.reverse_rates = nn.Parameter(torch.rand(num_reactions, dtype=dtype) * 0.01)
        
        # Reaction stoichiometry
        # For each reaction, define reactants and products
        self.reactant_stoich = nn.Parameter(torch.zeros(num_reactions, num_reactants, dtype=dtype))
        self.product_stoich = nn.Parameter(torch.zeros(num_reactions, num_reactants, dtype=dtype))
        
        # Initialize random reactions
        self._initialize_reactions()
        
        # Noise scale for stochastic dynamics
        self.noise_scale = nn.Parameter(torch.tensor(0.05, dtype=dtype))
    
    def _initialize_reactions(self) -> None:
        """Initialize random chemical reactions."""
        for i in range(self.num_reactions):
            # Select random reactants and products
            num_inputs = torch.randint(1, 3, (1,)).item()
            num_outputs = torch.randint(1, 3, (1,)).item()
            
            # Select random reactant indices
            reactant_indices = torch.randperm(self.num_reactants)[:num_inputs]
            
            # Select random product indices
            product_indices = torch.randperm(self.num_reactants)[:num_outputs]
            
            # Set stoichiometric coefficients
            for idx in reactant_indices:
                self.reactant_stoich.data[i, idx] = 1.0
            
            for idx in product_indices:
                self.product_stoich.data[i, idx] = 1.0
    
    def _compute_reaction_rates(self, concentrations: torch.Tensor) -> torch.Tensor:
        """
        Compute reaction rates using mass action kinetics.
        
        Args:
            concentrations: Reactant concentrations
            
        Returns:
            Net reaction rates
        """
        # Forward rates
        forward_rates = torch.ones_like(self.forward_rates)
        for i in range(self.num_reactions):
            for j in range(self.num_reactants):
                if self.reactant_stoich[i, j] > 0:
                    # Apply mass action kinetics
                    power = self.reactant_stoich[i, j]
                    
                    if self.complex_valued:
                        # For complex values, use absolute value for concentration
                        forward_rates[i] *= torch.abs(concentrations[j]) ** power
                    else:
                        forward_rates[i] *= concentrations[j] ** power
        
        # Scale by rate constants
        forward_rates *= self.forward_rates
        
        # Reverse rates
        reverse_rates = torch.ones_like(self.reverse_rates)
        for i in range(self.num_reactions):
            for j in range(self.num_reactants):
                if self.product_stoich[i, j] > 0:
                    # Apply mass action kinetics
                    power = self.product_stoich[i, j]
                    
                    if self.complex_valued:
                        # For complex values, use absolute value for concentration
                        reverse_rates[i] *= torch.abs(concentrations[j]) ** power
                    else:
                        reverse_rates[i] *= concentrations[j] ** power
        
        # Scale by rate constants
        reverse_rates *= self.reverse_rates
        
        # Net rates
        net_rates = forward_rates - reverse_rates
        
        return net_rates
    
    def step(self, dt: float = 0.1, inputs: Optional[torch.Tensor] = None) -> Tuple[torch.Tensor, Dict]:
        """
        Perform one step of reaction system dynamics.
        
        Args:
            dt: Time step
            inputs: Optional external inputs
            
        Returns:
            Tuple of (concentrations, metrics)
        """
        # Apply external inputs if provided
        if inputs is not None:
            if len(inputs) == self.num_reactants:
                if self.complex_valued and not torch.is_complex(inputs):
                    inputs_complex = torch.complex(inputs, torch.zeros_like(inputs))
                else:
                    inputs_complex = inputs
                
                self.concentrations = 0.9 * self.concentrations + 0.1 * inputs_complex
        
        # Compute reaction rates
        net_rates = self._compute_reaction_rates(self.concentrations)
        
        # Compute concentration changes
        concentration_changes = torch.zeros_like(self.concentrations)
        
        for i in range(self.num_reactions):
            # Consumption of reactants
            concentration_changes -= self.reactant_stoich[i] * net_rates[i]
            
            # Production of products
            concentration_changes += self.product_stoich[i] * net_rates[i]
        
        # Add stochastic noise (scaled by concentrations)
        if self.complex_valued:
            noise_real = torch.randn_like(concentration_changes.real) * self.noise_scale.real * torch.sqrt(torch.abs(self.concentrations))
            noise_imag = torch.randn_like(concentration_changes.imag) * self.noise_scale.imag * torch.sqrt(torch.abs(self.concentrations))
            noise = torch.complex(noise_real, noise_imag)
        else:
            noise = torch.randn_like(concentration_changes) * self.noise_scale * torch.sqrt(torch.abs(self.concentrations))
        
        # Update concentrations using Euler-Maruyama method
        self.concentrations = torch.clamp(self.concentrations + concentration_changes * dt + noise * torch.sqrt(dt), min=0.01)
        
        # Compute metrics
        metrics = {
            "mean_concentration": torch.mean(torch.abs(self.concentrations) if self.complex_valued else self.concentrations).item(),
            "max_concentration": torch.max(torch.abs(self.concentrations) if self.complex_valued else self.concentrations).item(),
            "min_concentration": torch.min(torch.abs(self.concentrations) if self.complex_valued else self.concentrations).item(),
            "active_reactions": torch.sum(torch.abs(net_rates) > 0.01).item()
        }
        
        return self.concentrations, metrics

class SpeechProcessor(nn.Module):
    """
    Speech processing using WhisperX for transcription and alignment.
    
    This module provides audio processing capabilities with phoneme-level
    alignment for harmonic encoding of speech signals.
    """
    
    def __init__(self,
                config,
                model_name: str = "large-v2",
                device: Optional[str] = None,
                complex_valued: bool = True):
        """
        Initialize a Speech Processor.
        
        Args:
            config: Configuration object
            model_name: WhisperX model to use
            device: Device to run on (default: from config)
            complex_valued: Whether to use complex-valued computation
        """
        super().__init__()
        self.config = config
        self.model_name = model_name
        self.device = device or config.device
        self.complex_valued = complex_valued
        
        # Load WhisperX model if available
        if WHISPERX_AVAILABLE:
            logger.info(f"Loading WhisperX model: {model_name}")
            try:
                self.whisperx_model = whisperx.load_model(model_name, self.device)
                logger.info("WhisperX model loaded successfully")
            except Exception as e:
                logger.error(f"Error loading WhisperX model: {e}")
                self.whisperx_model = MockWhisperX(model_name)
        else:
            logger.warning("Using mock WhisperX model")
            self.whisperx_model = MockWhisperX(model_name)
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Phoneme embedding
        self.phoneme_embedding = nn.Embedding(50, config.d_model, dtype=dtype)  # 50 is approx number of phonemes in English
        
        # Mel spectrogram processing
        self.mel_processor = nn.Sequential(
            nn.Conv1d(128, 256, kernel_size=3, stride=1, padding=1, dtype=dtype),
            nn.LayerNorm(256) if not complex_valued else nn.Identity(),
            nn.GELU() if not complex_valued else nn.Identity(),
            nn.Conv1d(256, config.d_model, kernel_size=1, dtype=dtype)
        )
        
        # Phoneme-to-harmonic mapper
        self.phoneme_to_harmonic = nn.Linear(config.d_model, config.d_model, dtype=dtype)
        
        # Transformer encoder for sequence processing
        self.transformer = nn.TransformerEncoder(
            nn.TransformerEncoderLayer(
                d_model=config.d_model,
                nhead=8,
                dim_feedforward=config.d_model * 4,
                dropout=0.1,
                batch_first=True
            ),
            num_layers=4
        )
        
        # Time embedding
        self.time_embedding = nn.Parameter(torch.randn(1, 1000, config.d_model, dtype=dtype) * 0.02)
        
        # Register phoneme mapping
        self.register_buffer('phoneme_map', self._create_phoneme_map())
        
        # Register frequency map for harmonic encoding
        self.register_buffer('frequency_map', self._create_frequency_map())
    
    def _create_phoneme_map(self) -> Dict[str, int]:
        """
        Create mapping from phonemes to indices.
        
        Returns:
            Dictionary mapping phoneme symbols to indices
        """
        # ARPABET phoneme set
        phonemes = [
            'AA', 'AE', 'AH', 'AO', 'AW', 'AY', 'B', 'CH', 'D', 'DH', 'EH', 'ER', 'EY',
            'F', 'G', 'HH', 'IH', 'IY', 'JH', 'K', 'L', 'M', 'N', 'NG', 'OW', 'OY', 'P',
            'R', 'S', 'SH', 'T', 'TH', 'UH', 'UW', 'V', 'W', 'Y', 'Z', 'ZH',
            'AA0', 'AA1', 'AA2', 'AE0', 'AE1', 'AE2', 'AH0', 'AH1', 'AH2',
            'SIL', 'SP', '<UNK>'
        ]
        
        # Create mapping
        phoneme_map = {p: i for i, p in enumerate(phonemes)}
        
        return phoneme_map
    
    def _create_frequency_map(self) -> torch.Tensor:
        """
        Create frequency map for harmonic encoding.
        
        Returns:
            Frequency map tensor
        """
        # Create logarithmically spaced frequencies (20 Hz to 20 kHz)
        num_freqs = self.config.d_model // 2
        frequencies = torch.logspace(np.log10(20), np.log10(20000), num_freqs)
        
        return frequencies
    
    def process_audio(self, audio: torch.Tensor) -> Dict[str, Any]:
        """
        Process audio using WhisperX.
        
        Args:
            audio: Audio tensor (batch_size, samples) or (samples,)
            
        Returns:
            Dictionary of processed outputs
        """
        # Ensure audio is on CPU as numpy array
        if isinstance(audio, torch.Tensor):
            audio_np = audio.detach().cpu().numpy()
        else:
            audio_np = audio
        
        # Transcribe audio
        result = self.whisperx_model.transcribe(audio_np, batch_size=16)
        
        # Get language
        language = result.get("language", "en")
        
        # Perform alignment to get word and phoneme timings
        try:
            alignment = self.whisperx_model.align(result["segments"], audio_np, language=language)
            aligned_segments = alignment.get("segments", result["segments"])
        except Exception as e:
            logger.warning(f"Alignment failed: {e}")
            aligned_segments = result["segments"]
        
        # Extract relevant features
        features = {
            "text": " ".join([segment["text"] for segment in aligned_segments]),
            "segments": aligned_segments,
            "language": language,
        }
        
        # Process mel spectrograms if available
        mel_specs = []
        for segment in aligned_segments:
            if "mel_spectrogram" in segment:
                mel_specs.append(torch.tensor(segment["mel_spectrogram"]))
        
        if mel_specs:
            # Combine mel spectrograms
            mel_tensor = torch.cat(mel_specs, dim=1)
            
            # Process through mel processor
            if self.complex_valued and not torch.is_complex(mel_tensor):
                mel_tensor = torch.complex(mel_tensor, torch.zeros_like(mel_tensor))
            
            processed_mel = self.mel_processor(mel_tensor)
            features["processed_mel"] = processed_mel
        
        # Extract and process phonemes if available
        phoneme_embeddings = []
        phoneme_timings = []
        
        for segment in aligned_segments:
            if "phonemes" in segment:
                for phoneme_info in segment["phonemes"]:
                    # Get phoneme and convert to index
                    phoneme = phoneme_info["phoneme"]
                    phoneme_idx = self.phoneme_map.get(phoneme, self.phoneme_map["<UNK>"])
                    
                    # Get timing
                    start = phoneme_info["start"]
                    end = phoneme_info["end"]
                    duration = end - start
                    
                    # Create embedding
                    phoneme_idx_tensor = torch.tensor([phoneme_idx], device=self.device)
                    embedding = self.phoneme_embedding(phoneme_idx_tensor)
                    
                    phoneme_embeddings.append(embedding.squeeze(0))
                    phoneme_timings.append((start, end, duration))
        
        if phoneme_embeddings:
            features["phoneme_embeddings"] = torch.stack(phoneme_embeddings)
            features["phoneme_timings"] = phoneme_timings
        
        return features
    
    def encode_as_harmonic(self, processed_features: Dict[str, Any]) -> torch.Tensor:
        """
        Encode processed features as harmonic representation.
        
        Args:
            processed_features: Dictionary of processed features from process_audio
            
        Returns:
            Harmonic encoding tensor
        """
        encodings = []
        
        # Use phoneme embeddings if available
        if "phoneme_embeddings" in processed_features:
            phoneme_embeds = processed_features["phoneme_embeddings"]
            timings = processed_features["phoneme_timings"]
            
            # Map to harmonic space
            harmonic_phonemes = self.phoneme_to_harmonic(phoneme_embeds)
            
            # Create time-dependent encoding
            max_time = max(end for _, end, _ in timings)
            num_steps = min(1000, int(max_time * 100))  # 100 steps per second
            
            # Initialize time series
            time_series = torch.zeros(num_steps, self.config.d_model, device=self.device)
            
            # Fill with phoneme activations
            for embed, (start, end, _) in zip(harmonic_phonemes, timings):
                start_idx = min(int(start * 100), num_steps - 1)
                end_idx = min(int(end * 100), num_steps - 1)
                
                if end_idx > start_idx:
                    time_series[start_idx:end_idx] += embed
            
            # Add harmonic oscillations
            t = torch.linspace(0, max_time, num_steps, device=self.device).unsqueeze(1)
            freqs = self.frequency_map[:self.config.d_model//2].unsqueeze(0)
            
            if self.complex_valued:
                # Create complex exponentials (e^(i*ω*t))
                oscillations = torch.exp(1j * 2 * math.pi * freqs * t)
                
                # Combine real and imaginary parts
                harmonic_basis = torch.cat([oscillations.real, oscillations.imag], dim=1)
            else:
                # Use real oscillations (cos(ω*t))
                harmonic_basis = torch.cos(2 * math.pi * freqs * t)
                
                # Add sine components
                harmonic_basis_sin = torch.sin(2 * math.pi * freqs * t)
                harmonic_basis = torch.cat([harmonic_basis, harmonic_basis_sin], dim=1)
            
            # Modulate basis with time series
            harmonic_encoding = time_series * harmonic_basis
            
            encodings.append(harmonic_encoding)
        
        # Use processed mel if available
        if "processed_mel" in processed_features:
            mel = processed_features["processed_mel"]
            
            # Ensure correct shape (batch_dim, seq_len, features)
            if mel.dim() == 2:
                mel = mel.permute(1, 0).unsqueeze(0)
            
            # Get appropriate time embedding
            seq_len = mel.shape[1]
            if seq_len <= self.time_embedding.shape[1]:
                time_embed = self.time_embedding[:, :seq_len]
            else:
                # If sequence is longer than embedding, repeat
                repeat_times = (seq_len // self.time_embedding.shape[1]) + 1
                time_embed = self.time_embedding.repeat(1, repeat_times, 1)[:, :seq_len]
            
            # Add time embedding
            mel_with_time = mel + time_embed
            
            # Process through transformer
            mel_encoding = self.transformer(mel_with_time)
            
            encodings.append(mel_encoding.squeeze(0))
        
        # If no encodings, return empty tensor
        if not encodings:
            return torch.zeros(1, self.config.d_model, device=self.device)
        
        # Combine encodings (average if multiple)
        if len(encodings) > 1:
            # Resample to same length
            max_len = max(enc.shape[0] for enc in encodings)
            resampled = []
            
            for enc in encodings:
                if enc.shape[0] == max_len:
                    resampled.append(enc)
                else:
                    # Use interpolation to resample
                    indices = torch.linspace(0, enc.shape[0] - 1, max_len, device=self.device)
                    indices_floor = indices.floor().long()
                    indices_ceil = indices.ceil().long().clamp(max=enc.shape[0] - 1)
                    weights_ceil = indices - indices_floor
                    weights_floor = 1 - weights_ceil
                    
                    resampled_enc = weights_floor.unsqueeze(1) * enc[indices_floor] + weights_ceil.unsqueeze(1) * enc[indices_ceil]
                    resampled.append(resampled_enc)
            
            combined = torch.stack(resampled).mean(dim=0)
        else:
            combined = encodings[0]
        
        return combined


class HarmonicSpeechEncoder(nn.Module):

        """

        Args:
            config: Configuration object
            speech_processor: Speech processor module (or None to create new one)
            complex_valued: Whether to use complex-valued computation
        """
        super().__init__()
        self.config = config
        self.complex_valued = complex_valued
        
        # Create or use speech processor
        self.speech_processor = speech_processor or SpeechProcessor(
            config=config,
            model_name="large-v2",
            complex_valued=complex_valued
        )
        
        # Choose dtype based on complex_valued flag
        dtype = torch.complex64 if complex_valued else torch.float32
        
        # Harmonic projection
        self.harmonic_projection = nn.Linear(config.d_model, config.d_model, dtype=dtype)
        
        # Quantum properties
        self.phase_modulation = nn.Parameter(torch.randn(config.d_model // 2, dtype=dtype) * 0.01)
        self.entanglement_matrix = nn.Parameter(torch.eye(config.d_model, dtype=dtype) * 0.1)

        self.activation = nn.Tanh()

        """

            audio: Audio tensor
            
        Returns:
            Harmonic encoding tensor

        processed_features = self.speech_processor.process_audio(audio)

        harmonic_encoding = self.speech_processor.encode_as_harmonic(processed_features)
        
        # Project to final harmonic space
        projected = self.harmonic_projection(harmonic_encoding)

        if self.complex_valued:
            # Modulate with phase
            phase_mod = self.phase_modulation.unsqueeze(0)
            modulated = projected * torch.exp(1j * phase_mod)

            entangled = torch.matmul(modulated, self.entanglement_matrix)

            output = self.activation(entangled.real)
        else:
            # Apply entanglement
            entangled = torch.matmul(projected, self.entanglement_matrix)
            
            # Apply activation
            output = self.activation(entangled)
