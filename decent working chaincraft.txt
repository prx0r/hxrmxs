import React, { useState } from 'react';
import { supabase } from '../lib/supabase';

const Auth = ({ theme, showToast }) => {
  const [mode, setMode] = useState('signin');
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [loading, setLoading] = useState(false);
  const [debugInfo, setDebugInfo] = useState('Ready for authentication');

  const logError = (context, error, additionalData = {}) => {
    const errorLog = {
      timestamp: new Date().toISOString(),
      context,
      error: error.message || error,
      stack: error.stack,
      userAgent: navigator.userAgent,
      url: window.location.href,
      ...additionalData
    };
    console.error('AUTH_ERROR:', errorLog);
    setDebugInfo(`ERROR: ${context} - ${error.message}`);
    return errorLog;
  };

  const updateDebug = (message) => {
    console.log('AUTH_DEBUG:', message);
    setDebugInfo(message);
  };

  const handleAuth = async (e) => {
    e.preventDefault();
    if (!email || !password) {
      showToast({ message: 'Email and password required', type: 'error' });
      return;
    }
    
    setLoading(true);
    updateDebug(`Starting ${mode}...`);
    
    try {
      let result;
      if (mode === 'signin') {
        updateDebug('Attempting sign in...');
        result = await supabase.auth.signInWithPassword({ email, password });
      } else {
        updateDebug('Attempting sign up...');
        result = await supabase.auth.signUp({ 
          email, 
          password,
          options: {
            emailRedirectTo: window.location.origin
          }
        });
      }
      
      if (result.error) {
        logError('AUTH_FAILURE', result.error, { mode, email });
        throw result.error;
      }
      
      updateDebug('Authentication successful');
      if (mode === 'signup' && !result.data.session) {
        showToast({ message: 'Check email for confirmation', type: 'info' });
      } else {
        showToast({ message: 'Welcome back!', type: 'success' });
      }
    } catch (error) {
      logError('AUTH_EXCEPTION', error, { mode, email });
      showToast({ message: error.message, type: 'error' });
    } finally {
      setLoading(false);
    }
  };

  const handleKeyDown = (e) => {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault();
      handleAuth(e);
    }
  };

  return (
    <div style={{ 
      minHeight: '100vh', 
      display: 'flex', 
      alignItems: 'center', 
      justifyContent: 'center', 
      backgroundColor: theme.colors.bgPrimary,
      ...theme.styles 
    }}>
      <div style={{ 
        backgroundColor: theme.colors.bgSecondary, 
        padding: '40px', 
        borderRadius: '8px', 
        border: `1px solid ${theme.colors.border}`, 
        width: '400px',
        ...theme.styles 
      }}>
        <h1 style={{ 
          textAlign: 'center', 
          marginBottom: '30px', 
          color: theme.colors.accent,
          fontFamily: theme.fonts.system,
          textShadow: theme.styles?.textShadow || 'none'
        }}>
          no-more-fucking-bullshit
        </h1>
        
        {/* Debug Panel - Always visible during development */}
        <details style={{ 
          marginBottom: '20px', 
          fontSize: '12px', 
          color: theme.colors.textSecondary,
          fontFamily: theme.fonts.system 
        }}>
          <summary style={{ cursor: 'pointer', marginBottom: '8px' }}>
            Debug Info ({debugInfo.includes('ERROR') ? '❌' : '✅'})
          </summary>
          <pre style={{ 
            whiteSpace: 'pre-wrap', 
            maxHeight: '200px', 
            overflow: 'auto',
            backgroundColor: theme.colors.bgPrimary,
            padding: '8px',
            borderRadius: '4px',
            border: `1px solid ${theme.colors.border}`,
            fontSize: '11px'
          }}>
            {debugInfo}
          </pre>
        </details>

        <form onSubmit={handleAuth}>
          <input
            type="email"
            placeholder="Email"
            value={email}
            onChange={(e) => setEmail(e.target.value)}
            onKeyDown={handleKeyDown}
            style={{ 
              width: '100%', 
              padding: '12px', 
              marginBottom: '16px', 
              border: `1px solid ${theme.colors.border}`, 
              borderRadius: '4px', 
              backgroundColor: theme.colors.bgPrimary, 
              color: theme.colors.textPrimary,
              fontFamily: theme.fonts.system,
              fontSize: '14px',
              boxSizing: 'border-box'
            }}
          />
          <input
            type="password"
            placeholder="Password"
            value={password}
            onChange={(e) => setPassword(e.target.value)}
            onKeyDown={handleKeyDown}
            style={{ 
              width: '100%', 
              padding: '12px', 
              marginBottom: '20px', 
              border: `1px solid ${theme.colors.border}`, 
              borderRadius: '4px', 
              backgroundColor: theme.colors.bgPrimary, 
              color: theme.colors.textPrimary,
              fontFamily: theme.fonts.system,
              fontSize: '14px',
              boxSizing: 'border-box'
            }}
          />
          <button
            type="submit"
            disabled={loading || !email || !password}
            style={{ 
              width: '100%', 
              padding: '12px', 
              backgroundColor: loading ? theme.colors.textSecondary : theme.colors.buttonPrimary, 
              color: '#fff', 
              border: 'none', 
              borderRadius: '4px', 
              cursor: loading ? 'not-allowed' : 'pointer', 
              marginBottom: '16px',
              fontFamily: theme.fonts.system,
              fontSize: '14px',
              fontWeight: '600'
            }}
          >
            {loading ? 'Loading...' : (mode === 'signin' ? 'Sign In' : 'Sign Up')}
          </button>
        </form>
        
        <button
          onClick={() => {
            setMode(mode === 'signin' ? 'signup' : 'signin');
            updateDebug(`Switched to ${mode === 'signin' ? 'signup' : 'signin'} mode`);
          }}
          style={{ 
            width: '100%', 
            padding: '8px', 
            backgroundColor: 'transparent', 
            color: theme.colors.textSecondary, 
            border: 'none', 
            cursor: 'pointer',
            fontFamily: theme.fonts.system,
            fontSize: '14px'
          }}
        >
          {mode === 'signin' ? 'Need an account? Sign up' : 'Have an account? Sign in'}
        </button>
      </div>
    </div>
  );
};

// History Modal
export const HistoryModal = ({ history, onClose, onDelete, onExport, theme, systemFontSize = 14 }) => {
  const [searchQuery, setSearchQuery] = useState('');
  const [expandedId, setExpandedId] = useState(null);
  const [selectedForExport, setSelectedForExport] = useState(new Set());

  const filteredHistory = history.filter(item => {
    const lowerCaseQuery = searchQuery.toLowerCase();
    const inputData = item.input_data || item.input;
    const queryText = typeof inputData === 'object' ? inputData.query : inputData;
    const inputMatch = queryText && typeof queryText === 'string' && queryText.toLowerCase().includes(lowerCaseQuery);
    const outputMatch = item.output_data && item.output_data.toLowerCase().includes(lowerCaseQuery);
    return inputMatch || outputMatch;
  }).sort((a, b) => new Date(b.created_at || b.timestamp) - new Date(a.created_at || a.timestamp));

  const toggleSelection = (id) => {
    const newSelection = new Set(selectedForExport);
    if (newSelection.has(id)) newSelection.delete(id); 
    else newSelection.add(id);
    setSelectedForExport(newSelection);
  };

  const toggleSelectAll = () => {
    if (selectedForExport.size === filteredHistory.length) setSelectedForExport(new Set());
    else setSelectedForExport(new Set(filteredHistory.map(item => item.id)));
  };

  const handleItemClick = (id) => setExpandedId(prev => (prev === id ? null : id));

  const handleDeleteClick = () => {
    if (selectedForExport.size > 0) {
      onDelete(Array.from(selectedForExport));
      setSelectedForExport(new Set());
    }
  };

  return (
    <div style={{ position: 'fixed', top: 0, left: 0, right: 0, bottom: 0, backgroundColor: 'rgba(0,0,0,0.8)', display: 'flex', alignItems: 'center', justifyContent: 'center', zIndex: 1000 }}>
      <div style={{ backgroundColor: theme.colors.bgSecondary, padding: '20px', borderRadius: '6px', width: '90%', maxWidth: '800px', height: '80vh', border: `1px solid ${theme.colors.border}`, display: 'flex', flexDirection: 'column' }}>
        <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: '16px', flexShrink: 0 }}>
          <h2 style={{ margin: 0, fontSize: '14px', fontWeight: '600', color: theme.colors.accent, fontFamily: 'inherit' }}>Analysis History</h2>
          <button onClick={onClose} style={{ background: 'none', border: 'none', fontSize: '16px', cursor: 'pointer', color: theme.colors.textSecondary, fontFamily: 'inherit' }}>✕</button>
        </div>
        
        <div style={{ marginBottom: '16px', display: 'flex', gap: '12px', flexShrink: 0, alignItems: 'center' }}>
          <input 
            type="text" 
            placeholder="search_history();" 
            value={searchQuery} 
            onChange={(e) => setSearchQuery(e.target.value)} 
            style={{ flex: 1, padding: '8px 12px', border: `1px solid ${theme.colors.border}`, borderRadius: '3px', fontSize: `${systemFontSize * 0.9}px`, backgroundColor: theme.colors.bgPrimary, color: theme.colors.textPrimary, fontFamily: 'inherit' }} 
          />
          <button onClick={toggleSelectAll} style={{ padding: '8px 12px', fontSize: `${systemFontSize * 0.9}px`, backgroundColor: theme.colors.buttonSecondary, color: theme.colors.textPrimary, border: `1px solid ${theme.colors.border}`, borderRadius: '3px', cursor: 'pointer', fontFamily: 'inherit' }}>
            {selectedForExport.size === filteredHistory.length ? 'Deselect All' : 'Select All'}
          </button>
          <button onClick={() => onExport(selectedForExport)} disabled={selectedForExport.size === 0} style={{ padding: '8px 12px', fontSize: `${systemFontSize * 0.9}px`, backgroundColor: theme.colors.buttonSecondary, color: theme.colors.textPrimary, border: `1px solid ${theme.colors.border}`, borderRadius: '3px', cursor: selectedForExport.size > 0 ? 'pointer' : 'not-allowed', fontFamily: 'inherit', whiteSpace: 'nowrap', opacity: selectedForExport.size > 0 ? 1 : 0.5 }}>
            Export JSON
          </button>
          <button onClick={handleDeleteClick} disabled={selectedForExport.size === 0} style={{ padding: '8px', backgroundColor: theme.colors.error, color: 'white', border: `1px solid ${theme.colors.border}`, borderRadius: '3px', cursor: selectedForExport.size > 0 ? 'pointer' : 'not-allowed', opacity: selectedForExport.size > 0 ? 1 : 0.5 }}>
            <Trash2 size={16} />
          </button>
        </div>

        <div style={{ flex: 1, overflowY: 'auto', border: `1px solid ${theme.colors.border}`, borderRadius: '3px', padding: '10px' }}>
          {filteredHistory.length > 0 ? (
            filteredHistory.map(item => {
              const inputData = item.input_data || item.input;
              const displayInput = typeof inputData === 'object' ? inputData.query : inputData;
              const timestamp = item.created_at || item.timestamp;
              
              return (
                <div key={item.id} style={{ marginBottom: '12px', padding: '12px', backgroundColor: theme.colors.bgPrimary, borderRadius: '4px', borderLeft: `3px solid ${theme.colors.accent}`, cursor: 'pointer' }} onClick={() => handleItemClick(item.id)}>
                  <div style={{display: 'flex', alignItems: 'center', gap: '12px'}}>
                    <input 
                      type="checkbox" 
                      checked={selectedForExport.has(item.id)} 
                      onChange={() => toggleSelection(item.id)} 
                      onClick={(e) => e.stopPropagation()} 
                      style={{cursor: 'pointer'}} 
                    />
                    <div style={{flex: 1}}>
                      <div style={{ fontSize: `${systemFontSize * 0.8}px`, color: theme.colors.textSecondary, marginBottom: '8px' }}>
                        {new Date(timestamp).toLocaleString()}
                      </div>
                      <p style={{ margin: '0 0 8px 0', fontSize: `${systemFontSize * 0.9}px`, fontWeight: '600', color: theme.colors.textPrimary, whiteSpace: 'pre-wrap', maxHeight: '60px', overflow: 'hidden', textOverflow: 'ellipsis' }}>
                        {displayInput}
                      </p>
                      <p style={{ margin: 0, fontSize: `${systemFontSize * 0.85}px`, color: theme.colors.textSecondary, whiteSpace: 'pre-wrap', maxHeight: '40px', overflow: 'hidden', textOverflow: 'ellipsis', fontStyle: 'italic' }}>
                        {(item.output_data || item.output || '').split('\n')[0]}
                      </p>
                    </div>
                  </div>
                  {expandedId === item.id && (
                    <div style={{marginTop: '12px', paddingTop: '12px', borderTop: `1px solid ${theme.colors.border}`}}>
                      <FormattedOutput text={item.output_data || item.output} theme={theme} fontStyle={{fontFamily: 'inherit', fontSize: `${systemFontSize}px`}} />
                    </div>
                  )}
                </div>
              );
            })
          ) : (
            <div style={{ color: theme.colors.textSecondary, fontSize: `${systemFontSize * 0.9}px` }}>No history matching search</div>
          )}
        </div>
      </div>
    </div>
  );
};

// Model Selector
export const ModelSelector = ({ agentKey, allModels, currentModel, onModelChange, editorState = {}, onStateChange, favoriteModels, theme }) => {
  const { search = '', sort = 'trending', freeToggle = false } = editorState;

  const formatModelDisplayText = (model) => {
    const name = `${favoriteModels.has(model.id) ? '★ ' : ''}${model.name || model.id}`;
    const cost = model.costPerMToken > 0 ? `${model.costPerMToken?.toFixed(2) || '?'}/M` : ' ';
    const context = model.contextLength || '?';
    return `${name.padEnd(60)}${cost.padEnd(15)}${context}`;
  };

  const filteredAndSortedModels = useMemo(() => {
    let models = [...allModels];
    if (freeToggle) models = models.filter(m => !m.isPaid);
    if (search) models = models.filter(model => (model.name || model.id).toLowerCase().includes(search.toLowerCase()));
    
    models.sort((a, b) => {
      switch (sort) {
        case 'a-z': return (a.name || a.id).localeCompare(b.name || b.id);
        case 'cost':
          if (a.costPerMToken === 0 && b.costPerMToken > 0) return -1;
          if (b.costPerMToken === 0 && a.costPerMToken > 0) return 1;
          return (a.costPerMToken || Infinity) - (b.costPerMToken || Infinity);
        case 'context':
          const parseContext = (ctx) => {
            if (!ctx) return 0;
            const ctxStr = String(ctx).toUpperCase();
            const match = ctxStr.match(/(\d+(?:\.\d+)?)(K|M)?/);
            if (!match) return 0;
            const num = parseFloat(match[1]);
            const unit = match[2];
            if (unit === 'M') return num * 1000000;
            if (unit === 'K') return num * 1000;
            return num;
          };
          return parseContext(b.contextLength) - parseContext(a.contextLength);
        case 'trending': 
        default: 
          return (a.ranking || 999) - (b.ranking || 999);
      }
    });
    
    return models.sort((a, b) => {
      const aIsFav = favoriteModels.has(a.id);
      const bIsFav = favoriteModels.has(b.id);
      if (aIsFav && !bIsFav) return -1;
      if (!aIsFav && bIsFav) return 1;
      return 0;
    });
  }, [allModels, search, sort, favoriteModels, freeToggle]);

  return (
    <div style={{ marginBottom: '12px' }}>
      <input 
        type="text" 
        placeholder="Search models..." 
        value={search} 
        onChange={(e) => onStateChange(agentKey, 'search', e.target.value)} 
        style={{ width: '100%', padding: '6px 8px', marginBottom: '8px', border: `1px solid ${theme.colors.border}`, borderRadius: '4px', backgroundColor: theme.colors.bgPrimary, color: theme.colors.textPrimary, fontSize: '12px', outline: 'none', fontFamily: 'inherit' }} 
      />
      <div style={{ display: 'flex', justifyContent: 'space-between', marginBottom: '8px', alignItems: 'center' }}>
        <div style={{ fontSize: '12px', display: 'flex', alignItems: 'center', gap: '8px' }}>
          {['trending', 'a-z', 'cost', 'context'].map(sortKey => (
            <button 
              key={sortKey} 
              onClick={() => onStateChange(agentKey, 'sort', sortKey)} 
              style={{ background: 'none', border: 'none', padding: '4px 0', color: sort === sortKey ? theme.colors.accent : theme.colors.textSecondary, fontSize: '12px', cursor: 'pointer', fontWeight: sort === sortKey ? '600' : '400', textDecoration: 'none', fontFamily: 'inherit', textTransform: 'lowercase' }}
            >
              {sortKey}
            </button>
          ))}
          <button 
            onClick={() => onStateChange(agentKey, 'freeToggle', !freeToggle)} 
            style={{ background: 'none', border: 'none', padding: '4px 0', color: freeToggle ? theme.colors.success : theme.colors.textSecondary, fontSize: '12px', cursor: 'pointer', fontWeight: freeToggle ? '600' : '400', textDecoration: 'none', fontFamily: 'inherit', textTransform: 'lowercase', marginLeft: '8px' }}
          >
            free
          </button>
        </div>
      </div>
      <select 
        value={currentModel} 
        onChange={onModelChange} 
        style={{ width: '100%', padding: '8px', border: `1px solid ${theme.colors.border}`, borderRadius: '4px', backgroundColor: theme.colors.bgPrimary, color: theme.colors.textPrimary, fontSize: '12px', fontFamily: '"JetBrains Mono", "SF Mono", Monaco, Menlo, "Courier New", monospace', outline: 'none' }}
      >
        {filteredAndSortedModels.map(model => 
          <option key={model.id} value={model.id} title={model.description}>
            {formatModelDisplayText(model)}
          </option>
        )}
      </select>
    </div>
  );
};

export const Toast = ({ toast, onRemove, theme }) => {
  const [exiting, setExiting] = useState(false);

  useEffect(() => {
    const timer = setTimeout(() => {
      setExiting(true);
      setTimeout(() => onRemove(toast.id), 300);
    }, toast.duration || 4000);

    return () => clearTimeout(timer);
  }, [toast.id, onRemove, toast.duration]);

  const handleClose = () => {
    setExiting(true);
    setTimeout(() => onRemove(toast.id), 300);
  };

  const typeStyles = {
    success: { icon: <CheckCircle />, color: theme.colors.success },
    error: { icon: <XCircle />, color: theme.colors.error },
    warning: { icon: <AlertTriangle />, color: theme.colors.warning },
    info: { icon: <Info />, color: theme.colors.accent },
  };

  const { icon, color } = typeStyles[toast.type] || typeStyles.info;

  return (
    <div
      style={{
        display: 'flex',
        alignItems: 'center',
        padding: '12px 16px',
        borderRadius: '8px',
        backgroundColor: theme.colors.bgSecondary,
        border: `1px solid ${theme.colors.border}`,
        boxShadow: '0 4px 12px rgba(0,0,0,0.15)',
        color: theme.colors.textPrimary,
        fontFamily: theme.fonts.system,
        minWidth: '250px',
        maxWidth: '350px',
        transition: 'all 0.3s ease-in-out',
        animation: exiting ? 'toast-exit 0.3s ease-out forwards' : 'toast-enter 0.3s ease-in-out forwards',
      }}
    >
      <div style={{ color: color, marginRight: '12px', flexShrink: 0 }}>{icon}</div>
      <div style={{ flex: 1, fontSize: '14px' }}>{toast.message}</div>
      <button onClick={handleClose} style={{ background: 'none', border: 'none', color: theme.colors.textSecondary, cursor: 'pointer', marginLeft: '12px' }}>
        <X size={18} />
      </button>
    </div>
  );
};

// Formatted Output Component
export const FormattedOutput = ({ text, theme, fontStyle }) => {
  if (!theme || !theme.colors) {
    return <div style={{ color: '#7d8590', fontFamily: 'inherit', fontSize: '16px' }}>{/* output */}</div>;
  }
  if (!text) {
    return <div style={{ ...fontStyle, color: theme.colors.textSecondary || '#7d8590' }}>{/* output */}</div>;
  }

  const parseLine = (line, index) => {
    const getColors = (theme) => {
      switch(theme.name) {
        case 'Dark': return { headers: '#79c0ff', subheaders: '#ffa657', emphasis: '#f0f6fc', body: '#e6edf3', muted: '#7d8590' };
        case 'Matrix': return { headers: '#00ff41', subheaders: '#33ff66', emphasis: '#66ff99', body: '#00ff41', muted: '#008F11' };
        case 'Paper': return { headers: '#b58900', subheaders: '#cb4b16', emphasis: '#2aa198', body: '#586e75', muted: '#93a1a1' };
        case 'Neon': return { headers: '#ff79c6', subheaders: '#8be9fd', emphasis: '#50fa7b', body: '#f8f8f2', muted: '#6272a4' };
        default: return { headers: theme.colors.accent || '#79c0ff', subheaders: theme.colors.warning || '#ffa657', emphasis: theme.colors.textPrimary || '#f0f6fc', body: theme.colors.textPrimary || '#e6edf3', muted: theme.colors.textSecondary || '#7d8590' };
      }
    };

    const colors = getColors(theme);
    if (line.startsWith('#### ')) return <h4 key={index} style={{ margin: '16px 0 8px 0', fontSize: '1em', color: colors.muted, fontFamily: 'inherit', fontWeight: '600', textShadow: theme.name === 'Matrix' ? 'none' : undefined }}>{line.substring(5)}</h4>;
    if (line.startsWith('### ')) return <h3 key={index} style={{ margin: '18px 0 10px 0', fontSize: '1.1em', color: colors.subheaders, fontFamily: 'inherit', fontWeight: '600', textShadow: theme.name === 'Matrix' ? 'none' : undefined }}>{line.substring(4)}</h3>;
    if (line.startsWith('## ')) return <h2 key={index} style={{ margin: '20px 0 12px 0', fontSize: '1.25em', color: colors.headers, fontFamily: 'inherit', fontWeight: '600', textShadow: theme.name === 'Matrix' ? 'none' : undefined }}>{line.substring(3)}</h2>;
    if (line.startsWith('# ')) return <h1 key={index} style={{ margin: '24px 0 16px 0', fontSize: '1.5em', color: colors.headers, fontFamily: 'inherit', fontWeight: '600', textShadow: theme.name === 'Matrix' ? 'none' : undefined }}>{line.substring(2)}</h1>;
    if (line.match(/^[A-Z\s]{8,}:?\s*$/)) return <div key={index} style={{ margin: '16px 0 8px 0', fontSize: '1em', color: colors.subheaders, fontFamily: 'inherit', fontWeight: '600', letterSpacing: '1px', textShadow: theme.name === 'Matrix' ? 'none' : undefined }}>{line}</div>;
    if (line.match(/^[\s]*[-•]\s/)) return <div key={index} style={{ margin: '4px 0', color: colors.body, fontFamily: 'inherit', paddingLeft: '16px', textShadow: theme.name === 'Matrix' ? 'none' : undefined }}>{line}</div>;
    if (line.match(/^[\s]*\d+\.\s/)) return <div key={index} style={{ margin: '6px 0', color: colors.emphasis, fontFamily: 'inherit', fontWeight: '500', textShadow: theme.name === 'Matrix' ? 'none' : undefined }}>{line}</div>;
    const cleanLine = line.replace(/\*\*/g, '').replace(/\*/g, '');
    if (!cleanLine.trim()) return <div key={index} style={{ height: '8px' }}></div>;
    return <p key={index} style={{ margin: '8px 0', fontFamily: 'inherit', lineHeight: '1.5', color: colors.body, textShadow: theme.name === 'Matrix' ? 'none' : undefined }}>{cleanLine}</p>;
  };

  return <div style={{ ...fontStyle, lineHeight: '1.6' }}>{text.split('\n').map((line, i) => parseLine(line, i))}</div>;
};

// Result Card Component
export const ResultCard = ({ title, responses, color, theme, fontStyle }) => {
  if (!responses || responses.length === 0) return null;
  const agentName = title.replace('agent_', '');
  const agentColorFinal = theme.name === 'Matrix' ? theme.colors.textPrimary : color;
  
  return (
    <div style={{ marginBottom: '24px', backgroundColor: theme.colors.bgSecondary, border: `1px solid ${theme.colors.border}`, borderRadius: '6px', padding: '16px' }}>
      <h3 style={{ fontSize: '16px', fontWeight: '600', color: agentColorFinal, marginBottom: '12px', fontFamily: 'inherit', textTransform: 'lowercase', letterSpacing: '0.5px' }}>agent_{agentName}</h3>
      {responses.map((content, index) => (
        <div key={index} style={{ marginBottom: index === responses.length - 1 ? 0 : '20px' }}>

          <FormattedOutput text={content} theme={theme} fontStyle={fontStyle} />
        </div>
      ))}
    </div>
  );
};

// Cursor Blink Component
export const CursorBlink = ({ theme }) => {
  const [visible, setVisible] = useState(true);
  useEffect(() => {
    const interval = setInterval(() => setVisible(v => !v), 500);
    return () => clearInterval(interval);
  }, []);
  return <span style={{ opacity: visible ? 1 : 0, color: theme.colors.accent }}>|</span>;
};

import { themes, generateRandomTheme, getAgentColors, getButtonStyle, INITIAL_FONTS } from './utils/themeSystem';

export default function App() {
  // Toast system
  const [toasts, setToasts] = useState([]);
  const toastIdRef = useRef(0);

  const showToast = useCallback(({ message, type = 'info', duration = 4000 }) => {
    const id = toastIdRef.current++;
    setToasts(prev => [...prev, { id, message, type, duration }]);
  }, []);

  const removeToast = (id) => {
    setToasts(prev => prev.filter(t => t.id !== id));
  };

  // Data management
  const dataManager = useDataManager(showToast);
  const { 
    user, profile, chains, history, loading: dataLoading, error: dataError,
    updateProfile, saveChain, deleteChain, saveAnalysis, deleteHistoryItems, logout
  } = dataManager;

  // Chain sharing
  const { shareChain, importFromUrl, exportChainsAsFile, importChainsFromFile } = useChainSharing(showToast);

  // Core application state
  const [query, setQuery] = useState('');
  const [context, setContext] = useState('');
  const [selectedChain, setSelectedChain] = useState('nobullshit');
  const [selectedDepth, setSelectedDepth] = useState('Quick');
  const [viewMode, setViewMode] = useState('default');

  // Analysis state
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState('');
  const [currentStatus, setCurrentStatus] = useState('ready');
  const [agentResults, setAgentResults] = useState({});
  const [finalOutput, setFinalOutput] = useState('');
  const [progress, setProgress] = useState({ current: 0, total: 1 });

  // UI state
  const [dialogueTab, setDialogueTab] = useState('input');
  const [editMode, setEditMode] = useState(false);
  const [editChain, setEditChain] = useState(null);
  const [showSettings, setShowSettings] = useState(false);
  const [showHistory, setShowHistory] = useState(false);
  const [showTutorial, setShowTutorial] = useState(false);
  const [showImport, setShowImport] = useState(false);
  const [importString, setImportString] = useState('');
  const [copiedState, setCopiedState] = useState(false);

  // Model and cost state
  const [availableModels, setAvailableModels] = useState([]);
  const [isFetchingModels, setIsFetchingModels] = useState(false);
  const [estimatedCost, setEstimatedCost] = useState(0);
  const [lastAnalysisCost, setLastAnalysisCost] = useState(0);

  // Refinement state
  const [refinementPrompt, setRefinementPrompt] = useState('');
  const [isRefining, setIsRefining] = useState(false);

  // Theme and font state
  const [dynamicThemes, setDynamicThemes] = useState({});
  const [availableFonts, setAvailableFonts] = useState(INITIAL_FONTS);
  const [customFontInput, setCustomFontInput] = useState('');
  const [favoriteModels, setFavoriteModels] = useState(new Set());
  const [editorAgentState, setEditorAgentState] = useState({
    agent_a: { search: '', sort: 'trending', freeToggle: false },
    agent_b: { search: '', sort: 'trending', freeToggle: false },
    agent_c: { search: '', sort: 'trending', freeToggle: false },
  });

  // Refs
  const queryInputRef = useRef(null);
  const refinementPromptRef = useRef(null);
  const customFontInputRef = useRef(null);
  const settingsModalRef = useRef(null);
  const scrollContainerRef = useRef(null);
  const lenisRef = useRef(null);
  const patternExecutorRef = useRef(null);

  // Drag state for smooth scrolling
  const dragState = useRef({ 
    isDragging: false, 
    initialY: 0,
    initialScrollTop: 0,
    lastY: 0,
    velocity: 0,
    momentumId: null
  });

  // Theme management
  const allThemes = { ...themes, ...dynamicThemes };
  const currentTheme = allThemes[profile?.theme || 'dark'] || themes.dark;

  // Load saved preferences on profile load
  useEffect(() => {
    if (profile) {
      setAvailableFonts(profile.available_fonts || INITIAL_FONTS);
      setFavoriteModels(new Set(profile.favorite_models || []));
      setDynamicThemes(profile.dynamic_themes || {});
      setEditorAgentState(profile.editor_agent_state || editorAgentState);
    }
  }, [profile]);

  // Load models when API key is available
  useEffect(() => {
    if (profile?.api_key) {
      setIsFetchingModels(true);
      MODEL_DATA_MANAGER.fetchAllModels(profile.api_key)
        .then(setAvailableModels)
        .catch(error => {
          console.error("Error loading model data:", error);
          setError("Failed to load models. Check API key/network.");
        })
        .finally(() => setIsFetchingModels(false));
    }
  }, [profile?.api_key]);

  // Import chain from URL on load
  useEffect(() => {
    const importedChain = importFromUrl();
    if (importedChain) {
      const chainWithId = { ...importedChain, id: `imported_${Date.now()}` };
      if (user) {
        saveChain(chainWithId).then(() => {
          setSelectedChain(chainWithId.id);
          showToast({ message: `Imported "${chainWithId.name}" successfully!`, type: 'success' });
        });
      } else {
        // Handle guest import - would need local state management
        showToast({ message: 'Sign in to import chains', type: 'warning' });
      }
    }
  }, [user, importFromUrl, saveChain, showToast]);

  // Cost estimation
  useEffect(() => {
    const chain = chains[selectedChain];
    if (!chain || availableModels.length === 0 || !query) {
      setEstimatedCost(0);
      return;
    }

    const promptText = query + context;
    const promptTokens = estimateTokens(promptText);

    const getModelCost = (modelId, tokens) => {
      const modelData = availableModels.find(m => m.id === modelId);
      if (!modelData || !modelData.costPerMToken) return 0;
      const promptCost = (tokens / 1_000_000) * modelData.costPerMToken;
      const completionTokens = Math.ceil(tokens * 1.5);
      const completionCost = (completionTokens / 1_000_000) * modelData.costPerMToken;
      return promptCost + completionCost;
    };

    let totalCost = 0;
    const pattern = chain.pattern || 'a-b-c';
    
    const steps = pattern.split(/[-+]/);
    steps.forEach(step => {
      const agents = step.split('');
      agents.forEach(agentChar => {
        const agentKey = { a: 'agent_a', b: 'agent_b', c: 'agent_c' }[agentChar];
        if (agentKey && chain.agents[agentKey]) {
          totalCost += getModelCost(chain.agents[agentKey].model, promptTokens);
        }
      });
    });

    setEstimatedCost(totalCost);
  }, [query, context, selectedChain, chains, availableModels]);

  // Status prefixes for loading states
  const statusPrefixes = [
    'Unleashing analysis...',
    'Engaging logic cores...',
    'Forging insights...',
    'Executing directive...',
    'Synthesizing data streams...',
    'Ripping apart assumptions...',
    'Interrogating the context...',
    'Cutting through the noise...',
    'Assembling truth fragments...',
    'Activating agent...'
  ];

  // Main analysis function
  const analyzeStream = useCallback(async () => {
    if (!query.trim() || !profile?.api_key) return;
    const chain = chains[selectedChain];
    if (!chain) {
      setError("Could not find the selected chain.");
      return;
    }

    setLoading(true);
    setError('');
    setDialogueTab('process');
    setAgentResults({});
    setFinalOutput('');
    setProgress({ current: 0, total: getStepCount(chain.pattern) });
    setCurrentStatus('initializing...');

    // Create pattern executor
    const executor = new PatternExecutor(
      profile.api_key,
      availableModels,
      (agentKey, response, roundNum, stepIndex, totalSteps) => {
        // Update agent results in real-time
        setAgentResults(prev => ({
          ...prev,
          [agentKey]: {
            ...prev[agentKey],
            [roundNum - 1]: response
          }
        }));
        setProgress({ current: stepIndex, total: totalSteps });
        
        const getRandomPrefix = () => statusPrefixes[Math.floor(Math.random() * statusPrefixes.length)];
        const agentName = chain.agents[agentKey]?.name || agentKey;
        setCurrentStatus(`${getRandomPrefix()} [${agentName}]`);
      },
      (errorMessage) => {
        setError(errorMessage);
      }
    );

    patternExecutorRef.current = executor;

    try {
      const result = await executor.execute(chain, query, context);
      
      if (result) {
        setFinalOutput(result.output);
        setLastAnalysisCost(result.cost);
        setDialogueTab('output');
        setCurrentStatus('complete');
        setProgress(prev => ({ ...prev, current: prev.total }));

        // Save to history
        if (user) {
          await saveAnalysis(
            { query, context },
            result.output,
            selectedChain,
            result.cost
          );
        }
      }
    } catch (error) {
      if (error.name !== 'AbortError') {
        setError(error.message);
        setCurrentStatus('error');
      }
    } finally {
      setLoading(false);
      patternExecutorRef.current = null;
    }
  }, [query, context, profile?.api_key, chains, selectedChain, availableModels, user, saveAnalysis, statusPrefixes]);

  // Refinement function
  const refineSynthesis = useCallback(async () => {
    if (!refinementPrompt.trim() || !profile?.api_key || !finalOutput) return;

    setIsRefining(true);
    setError('');

    const refinementSystemPrompt = `You are an editor. Your task is to refine the following text based on a specific instruction.\n\nINSTRUCTION: "${refinementPrompt}"\n\nORIGINAL TEXT:\n"""\n${finalOutput}\n"""\n\nBased on the instruction, provide only the refined text. Format your response using simple Markdown: use ## for headings and ** for bold text. Do not use any other formatting.`;

    try {
      const response = await fetch("https://openrouter.ai/api/v1/chat/completions", {
        method: 'POST',
        headers: {
          "Authorization": `Bearer ${profile.api_key}`,
          "Content-Type": "application/json",
          "HTTP-Referer": window.location.href,
          "X-Title": "NoBS Analysis"
        },
        body: JSON.stringify({
          model: "deepseek/deepseek-r1-0528:free",
          messages: [{"role": "user", "content": refinementSystemPrompt}],
          max_tokens: 4096,
          temperature: 0.7
        })
      });

      if (!response.ok) {
        const errData = await response.json();
        throw new Error(errData.error.message || 'Refinement API call failed');
      }

      const data = await response.json();
      const newSynthesis = data.choices[0].message.content;
      
      setFinalOutput(newSynthesis);
      showToast({ message: 'Output refined successfully!', type: 'success' });
    } catch (e) {
      setError(`Refinement failed: ${e.message}`);
    } finally {
      setIsRefining(false);
    }
  }, [refinementPrompt, finalOutput, profile?.api_key, showToast]);

  // Handler functions
  const handleQueryChange = (newValue) => {
    const commandRegex = /^c\/(.*?)\n/;
    const match = newValue.match(commandRegex);
    if (match) {
      const chainNameQuery = match[1].trim().toLowerCase();
      const restOfQuery = newValue.replace(commandRegex, '');
      const foundChain = Object.values(chains).find(
        (chain) => chain.name.toLowerCase() === chainNameQuery
      );
      if (foundChain) {
        setSelectedChain(foundChain.id);
        setQuery(restOfQuery);
        showToast({ message: `Chain set to: ${foundChain.name}`, type: 'info' });
      } else {
        setQuery(newValue);
        showToast({ message: `Chain "${match[1].trim()}" not found.`, type: 'warning' });
      }
    } else {
      setQuery(newValue);
    }
  };

  const handleClear = () => {
    if (patternExecutorRef.current) {
      patternExecutorRef.current.abort();
    }
    setQuery('');
    setContext('');
    setAgentResults({});
    setFinalOutput('');
    setError('');
    setLastAnalysisCost(0);
    setCurrentStatus('ready');
    setDialogueTab('input');
    setLoading(false);
    setProgress({ current: 0, total: 1 });
  };

  const handleCopy = () => {
    let textToCopy = '';
    switch (dialogueTab) {
      case 'input':
        textToCopy = `Query:\n${query}`;
        if (context.trim()) {
          textToCopy += `\n\nContext:\n${context.trim()}`;
        }
        break;
      case 'process':
        const processText = Object.entries(agentResults).map(([agentKey, rounds]) => {
          const roundTexts = Object.entries(rounds).map(([roundIndex, content]) => 
            `--- ${agentKey} Round ${parseInt(roundIndex) + 1} ---\n${content}`
          ).join('\n\n');
          return roundTexts;
        }).filter(t => t).join('\n\n');
        textToCopy = processText;
        break;
      case 'output':
        textToCopy = finalOutput || '';
        break;
      default:
        return;
    }

    if (!textToCopy.trim()) return;

    navigator.clipboard.writeText(textToCopy.trim()).then(() => {
      setCopiedState(true);
      setTimeout(() => setCopiedState(false), 2000);
    }).catch(err => {
      console.error('Failed to copy text: ', err);
    });
  };

  const handleThemeChange = (newThemeKey) => {
    if (user) {
      updateProfile({ theme: newThemeKey });
    }
  };

  const handleRandomizeTheme = () => {
    const newTheme = generateRandomTheme(availableFonts, currentTheme);
    const newThemeKey = `random_${Date.now()}`;
    const newDynamicThemes = { ...dynamicThemes, [newThemeKey]: newTheme };
    setDynamicThemes(newDynamicThemes);
    
    if (user) {
      updateProfile({ 
        theme: newThemeKey,
        dynamic_themes: newDynamicThemes,
        system_font: newTheme.fonts.system,
        output_font: newTheme.fonts.output
      });
    }
  };

  const handleAddNewChain = () => {
    const newChain = {
      id: `custom_${Date.now()}`,
      name: 'new_chain',
      description: 'A new custom analysis chain.',
      pattern: 'a-b-c',
      category: 'Quick',
      user_id: user?.id,
      is_default: false,
      is_public: false,
      created_at: new Date().toISOString(),
      agents: {
        agent_a: { name: "Agent A", model: "deepseek/deepseek-r1-0528:free", prompt: "You are Agent A. Your goal is to...", rounds: {} },
        agent_b: { name: "Agent B", model: "deepseek/deepseek-r1-0528:free", prompt: "You are Agent B. Your goal is to...", rounds: {} },
        agent_c: { name: "Agent C", model: "deepseek/deepseek-r1-0528:free", prompt: "You are Agent C. Your goal is to...", rounds: {} }
      }
    };
    
    if (user) {
      saveChain(newChain).then(() => {
        setSelectedChain(newChain.id);
        setEditChain(newChain);
        setEditMode(true);
        showToast({ message: `New chain "${newChain.name}" created!`, type: 'success' });
      });
    }
  };

  const saveEditedChain = async () => {
    if (!editChain || !user) return;

    try {
      const chainToSave = { ...editChain, user_id: user.id };
      await saveChain(chainToSave);
      setSelectedChain(chainToSave.id);
      setEditMode(false);
      setEditChain(null);
      showToast({ message: 'Chain saved successfully!', type: 'success' });
    } catch (error) {
      showToast({ message: `Failed to save chain: ${error.message}`, type: 'error' });
    }
  };

  const handleExportHistory = (selectedIds) => {
    const itemsToExport = history
      .filter(item => selectedIds.has(item.id))
      .map(({ created_at, input_data, output_data }) => ({ 
        timestamp: created_at, 
        input: input_data, 
        output: output_data 
      }));

    const dataStr = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(itemsToExport, null, 2));
    const downloadAnchorNode = document.createElement('a');
    downloadAnchorNode.setAttribute("href", dataStr);
    downloadAnchorNode.setAttribute("download", `analysis_history_${Date.now()}.json`);
    document.body.appendChild(downloadAnchorNode);
    downloadAnchorNode.click();
    document.body.removeChild(downloadAnchorNode);
  };

  // Filtered chains for display
  const filteredChains = useMemo(() => {
    return Object.values(chains)
      .filter(chain => {
        const stepCount = getStepCount(chain.pattern);
        const category = stepCount === 1 ? 'Quick' : stepCount <= 3 ? 'Quick' : stepCount <= 5 ? 'Standard' : 'Deep';
        return category === selectedDepth;
      })
      .filter(chain => viewMode === 'default' ? chain.is_default : !chain.is_default)
      .sort((a, b) => a.name.localeCompare(b.name));
  }, [chains, selectedDepth, viewMode]);

  // Get current chain and agent colors
  const currentChain = chains[selectedChain];
  const agentMap = getAgentColors(currentTheme);

  // Font and theme styles
  const outputFontStyle = {
    fontFamily: `${profile?.output_font || currentTheme.fonts.output}, serif`,
    fontSize: `${profile?.output_font_size || 14}px`,
  };

  const mainContainerStyle = {
    minHeight: '100vh',
    backgroundColor: currentTheme.colors.bgPrimary,
    color: currentTheme.colors.textPrimary,
    fontFamily: `${profile?.system_font || currentTheme.fonts.system}, monospace`,
    fontSize: `${profile?.system_font_size || 13}px`,
    padding: '0',
    margin: '0',
    ...(currentTheme.styles?.backgroundImage && { 
      backgroundImage: currentTheme.styles.backgroundImage, 
      backgroundSize: currentTheme.styles.backgroundSize, 
      backgroundPosition: currentTheme.styles.backgroundPosition 
    }),
  };

  // Show auth screen if not loading and no user
  if (!dataLoading && !user) {
    return <Auth theme={currentTheme} showToast={showToast} />;
  }

  // Show loading screen during data load
  if (dataLoading) {
    return (
      <div style={mainContainerStyle}>
        <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'center', minHeight: '100vh', flexDirection: 'column', gap: '20px' }}>
          <div style={{ color: currentTheme.colors.accent, fontSize: '24px', fontWeight: '600' }}>
            no-more-fucking-bullshit
          </div>
          <div style={{ color: currentTheme.colors.textSecondary, fontSize: '14px' }}>
            {dataManager.authDebug}
          </div>
          <FluidMultiBarProgress loading={true} currentTheme={currentTheme} />
        </div>
      </div>
    );
  }

  return (
    <div style={mainContainerStyle}>
      <ToastContainer toasts={toasts} onRemove={removeToast} theme={currentTheme} />
      
      {/* Global Styles */}
      <style>{`
        @keyframes toast-enter {
          from { opacity: 0; transform: translateX(100%); }
          to { opacity: 1; transform: translateX(0); }
        }
        @keyframes toast-exit {
          from { opacity: 1; transform: translateX(0); }
          to { opacity: 0; transform: translateX(100%); }
        }
        @keyframes neon-gradient-move {
          0% { background-position: 0% 50%; }
          50% { background-position: 100% 50%; }
          100% { background-position: 0% 50%; }
        }
        @keyframes matrix-bit-flicker {
          0% { opacity: 0.8; }
          50% { opacity: 1; }
          100% { opacity: 0.8; }
        }
        @keyframes dark-bar-pulse {
          0% { opacity: 0.8; }
          50% { opacity: 1; }
          100% { opacity: 0.8; }
        }
        
        textarea::-webkit-scrollbar { display: none; }
        ::-webkit-scrollbar { width: 8px; height: 8px; }
        ::-webkit-scrollbar-track { background: ${currentTheme.colors.bgSecondary}; }
        ::-webkit-scrollbar-thumb { background: ${currentTheme.colors.border}; border-radius: 4px; }
        ::-webkit-scrollbar-thumb:hover { background: ${currentTheme.colors.accent}; }
        
        .grabbable { cursor: grab; user-select: none; }
        .grabbable.is-grabbing { cursor: grabbing; }
        .smooth-scroll { -webkit-overflow-scrolling: touch; }
      `}</style>

      {/* Google Fonts */}
      <link rel="preconnect" href="https://fonts.googleapis.com" />
      <link rel="preconnect" href="https://fonts.gstatic.com" crossOrigin="true" />
      {availableFonts.length > 0 && (
        <link href={`https://fonts.googleapis.com/css2?family=${availableFonts.map(f => f.replace(/ /g, '+')).join('&family=')}&display=swap`} rel="stylesheet" />
      )}

      {/* Header */}
      <div style={{
        backgroundColor: currentTheme.colors.bgSecondary,
        borderBottom: `1px solid ${currentTheme.colors.border}`,
        padding: '12px 20px',
        display: 'flex',
        alignItems: 'center',
        gap: '12px',
        position: 'sticky',
        top: 0,
        zIndex: 100,
        ...currentTheme.styles,
      }}>
        <div style={{ fontSize: '1em', color: currentTheme.colors.textPrimary, fontWeight: '600', textShadow: currentTheme.styles?.textShadow || 'none' }}>
          no-more-fucking-bullshit
        </div>
        <div style={{ marginLeft: 'auto', display: 'flex', alignItems: 'center', gap: '16px' }}>
          <div style={{ fontSize: '0.8em', color: currentTheme.colors.textSecondary, display: 'flex', gap: '16px' }}>
            <span>usage: ${(profile?.total_spend || 0).toFixed(4)}</span>
            <span>last run: <span style={{color: currentTheme.colors.warning}}>${lastAnalysisCost.toFixed(4)}</span></span>
          </div>
          <button onClick={() => setShowTutorial(true)} style={{ background: 'none', border: 'none', color: currentTheme.colors.textSecondary, cursor: 'pointer', padding: '0', display: 'flex', alignItems: 'center' }} title="Quick Tour">
            <HelpCircle size={20} />
          </button>
          <button onClick={logout} style={{ background: 'none', border: 'none', color: currentTheme.colors.textSecondary, cursor: 'pointer', padding: '4px 8px', fontSize: '12px' }}>
            logout
          </button>
        </div>
      </div>

      {/* Main Content */}
      <div style={{ maxWidth: '1200px', margin: '0 auto', padding: '20px' }}>
        {/* Main Panel */}
        <div style={{
          backgroundColor: currentTheme.name === 'Paper' ? 'transparent' : currentTheme.colors.bgSecondary,
          border: currentTheme.name === 'Paper' ? 'none' : `1px solid ${currentTheme.colors.border}`,
          borderRadius: currentTheme.name === 'Paper' ? '0' : '6px',
          boxShadow: currentTheme.name === 'Paper' ? '0 2px 6px rgba(0,0,0,0.02)' : '0 4px 12px rgba(0,0,0,0.15)',
          overflow: 'hidden',
          marginBottom: '20px',
          position: 'relative',
          ...currentTheme.styles,
        }}>
          {/* Tab Header */}
          <div style={{
            backgroundColor: currentTheme.name === 'Paper' ? 'transparent' : currentTheme.colors.bgSecondary,
            borderBottom: `1px solid ${currentTheme.colors.border}`,
            display: 'flex',
            justifyContent: 'space-between',
            alignItems: 'center'
          }}>
            <div style={{ display: 'flex' }}>
              {['input', 'process', 'output'].map(tab => (
                <button
                  key={tab}
                  onClick={() => setDialogueTab(tab)}
                  style={{
                    padding: '12px 16px', fontSize: '0.9em', fontFamily: 'inherit', border: 'none',
                    backgroundColor: 'transparent', color: dialogueTab === tab ? currentTheme.colors.accent : currentTheme.colors.textSecondary,
                    cursor: 'pointer', borderBottom: `2px solid ${dialogueTab === tab ? currentTheme.colors.accent : 'transparent'}`,
                    fontWeight: '600', textShadow: currentTheme.styles?.textShadow && dialogueTab === tab ? currentTheme.styles.textShadow : 'none'
                  }}
                >
                  {tab}
                </button>
              ))}
            </div>
            <button
              onClick={handleCopy}
              style={{
                marginRight: '12px', padding: '6px 12px', fontSize: '0.8em',
                backgroundColor: currentTheme.colors.buttonSecondary, color: currentTheme.colors.textPrimary,
                border: `1px solid ${currentTheme.colors.border}`, borderRadius: '3px',
                cursor: 'pointer', fontFamily: 'inherit',
              }}
            >
              {copiedState ? 'Copied!' : 'Copy'}
            </button>
          </div>

          {/* Tab Content */}
          <div style={{ padding: '20px', minHeight: '250px', maxHeight: '80vh', display: 'flex', flexDirection: 'column' }}>
            {dialogueTab === 'input' && (
              <div style={{ position: 'relative', height: '100%', width: '100%' }}>
                <textarea
                  ref={queryInputRef}
                  placeholder={!profile?.api_key ? "configure api key in settings" : "c/[chain name]\nYour prompt here..."}
                  value={query}
                  onChange={(e) => handleQueryChange(e.target.value)}
                  disabled={!profile?.api_key}
                  style={{
                    ...outputFontStyle, width: '100%', height: '100%', border: 'none', resize: 'none', outline: 'none',
                    backgroundColor: 'transparent', color: !profile?.api_key ? currentTheme.colors.textSecondary : currentTheme.colors.textPrimary,
                    lineHeight: '1.5', opacity: !profile?.api_key ? 0.6 : 1, padding: '12px', paddingBottom: '124px'
                  }}
                />
                <div style={{ position: 'absolute', bottom: '-30px', left: '12px', width: 'calc(100% - 800px)', height: '100px' }}>
                  <div style={{ position: 'relative', width: '320px', height: '120px', margin: '0', padding: '0' }}>
                    <textarea
                      placeholder="context"
                      value={context}
                      onChange={(e) => setContext(e.target.value)}
                      style={{
                        ...outputFontStyle,
                        width: '320px',

export const useChainSharing = (showToast) => {
  const encodeChain = useCallback((chain) => {
    try {
      const shareData = {
        name: chain.name,
        description: chain.description,
        pattern: chain.pattern,
        agents: chain.agents,
        category: chain.category
      };
      
      const encoded = btoa(encodeURIComponent(JSON.stringify(shareData)));
      return `${window.location.origin}${window.location.pathname}#chain=${encoded}`;
    } catch (error) {
      console.error('Chain encoding failed:', error);
      showToast({ message: 'Failed to generate share URL', type: 'error' });
      return null;
    }
  }, [showToast]);

  const decodeChain = useCallback((hashFragment) => {
    try {
      if (!hashFragment.startsWith('#chain=')) return null;
      
      const encoded = hashFragment.substring(7);
      const decoded = JSON.parse(decodeURIComponent(atob(encoded)));
      
      return {
        ...decoded,
        id: `imported_${Date.now()}`,
        user_id: null,
        is_public: false,
        is_default: false,
        created_at: new Date().toISOString()
      };
    } catch (error) {
      console.error('Chain decoding failed:', error);
      showToast({ message: 'Invalid share URL format', type: 'error' });
      return null;
    }
  }, [showToast]);

  const shareChain = useCallback(async (chain) => {
    const shareUrl = encodeChain(chain);
    if (!shareUrl) return;

    try {
      await navigator.clipboard.writeText(shareUrl);
      showToast({ message: 'Share URL copied to clipboard!', type: 'success' });
    } catch (error) {
      // Fallback for older browsers
      const textArea = document.createElement('textarea');
      textArea.value = shareUrl;
      document.body.appendChild(textArea);
      textArea.select();
      document.execCommand('copy');
      document.body.removeChild(textArea);
      showToast({ message: 'Share URL copied!', type: 'success' });
    }
  }, [encodeChain, showToast]);

  const importFromUrl = useCallback(() => {
    const chain = decodeChain(window.location.hash);
    if (chain) {
      window.location.hash = ''; // Clear the hash
      return chain;
    }
    return null;
  }, [decodeChain]);

  const exportChainsAsFile = useCallback((chains) => {
    try {
      const customChains = Object.fromEntries(
        Object.entries(chains).filter(([key]) => key.startsWith('custom_'))
      );
      
      if (Object.keys(customChains).length === 0) {
        showToast({ message: "No custom chains to export.", type: 'warning' });
        return;
      }

      const dataStr = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(customChains, null, 2));
      const downloadAnchorNode = document.createElement('a');
      downloadAnchorNode.setAttribute("href", dataStr);
      downloadAnchorNode.setAttribute("download", `nobs_chains_backup_${Date.now()}.json`);
      document.body.appendChild(downloadAnchorNode);
      downloadAnchorNode.click();
      downloadAnchorNode.remove();
      showToast({ message: "Custom chains exported!", type: 'success' });
    } catch (error) {
      console.error('Export failed:', error);
      showToast({ message: `Export failed: ${error.message}`, type: 'error' });
    }
  }, [showToast]);

  const importChainsFromFile = useCallback(() => {
    return new Promise((resolve, reject) => {
      const input = document.createElement('input');
      input.type = 'file';
      input.accept = '.json';
      input.onchange = (e) => {
        const file = e.target.files[0];
        if (!file) {
          resolve(null);
          return;
        }
        
        const reader = new FileReader();
        reader.onload = (event) => {
          try {
            const importedChains = JSON.parse(event.target.result);
            if (typeof importedChains !== 'object' || importedChains === null) {
              throw new Error("Invalid JSON format.");
            }
            resolve(importedChains);
            showToast({ message: `Successfully imported ${Object.keys(importedChains).length} chains.`, type: 'success' });
          } catch (error) {
            console.error("Import failed:", error);
            showToast({ message: `Import failed: ${error.message}`, type: 'error' });
            reject(error);
          }
        };
        reader.readAsText(file);
      };
      input.click();
    });
  }, [showToast]);

  return { 
    shareChain, 
    importFromUrl, 
    encodeChain, 
    decodeChain,
    exportChainsAsFile,
    importChainsFromFile
  };
};

 import { useState, useEffect, useCallback } from 'react';
import { supabase, dbHelpers } from '../lib/supabase';

const useDataManager = (showToast) => {
  const [dataState, setDataState] = useState({
    user: null,
    profile: null,
    chains: DEFAULT_CHAINS,
    history: [],
    loading: true,
    error: null,
    authDebug: 'Initializing...'
  });

  const logDataError = (operation, error, context = {}) => {
    const errorLog = {
      timestamp: new Date().toISOString(),
      operation,
      error: error.message || error,
      context,
      userId: dataState.user?.id
    };
    console.error('DATA_ERROR:', errorLog);
    return errorLog;
  };

  const updateDebug = (message) => {
    setDataState(prev => ({ ...prev, authDebug: message }));
    console.log('DATA_DEBUG:', message);
  };

  const loadUserData = useCallback(async (user) => {
    if (!user) {
      updateDebug('No user - resetting to guest state');
      setDataState(prev => ({ 
        ...prev, 
        user: null, 
        profile: null, 
        chains: DEFAULT_CHAINS, 
        history: [], 
        loading: false,
        authDebug: 'Guest mode'
      }));
      return;
    }

    try {
      updateDebug(`Loading data for user: ${user.id}`);
      
      // Load all data in parallel - the key architectural principle
      const [profile, commands, history] = await Promise.all([
        dbHelpers.getOrCreateProfile(user.id),
        dbHelpers.getUserChains(user.id),
        dbHelpers.getAnalysisHistory(user.id)
      ]);

      updateDebug('Data loaded successfully');

      // Transform commands array to chains object
      const userChains = commands.reduce((acc, cmd) => {
        acc[cmd.id] = cmd;
        return acc;
      }, {});

      // Merge with default chains
      const allChains = { ...DEFAULT_CHAINS, ...userChains };

      setDataState({
        user,
        profile,
        chains: allChains,
        history,
        loading: false,
        error: null,
        authDebug: `Loaded: ${Object.keys(allChains).length} chains, ${history.length} history items`
      });

    } catch (error) {
      logDataError('DATA_LOAD_EXCEPTION', error, { userId: user.id });
      updateDebug(`Data load failed: ${error.message}`);
      setDataState(prev => ({ 
        ...prev, 
        loading: false, 
        error: error.message,
        authDebug: `Error: ${error.message}`
      }));
      showToast({ message: `Data load failed: ${error.message}`, type: 'error' });
    }
  }, [showToast]);

  // Core mutation functions - update local state after Supabase success
  const updateProfile = useCallback(async (updates) => {
    if (!dataState.user) return;

    try {
      const updatedProfile = await dbHelpers.updateProfile(dataState.user.id, updates);
      setDataState(prev => ({ 
        ...prev, 
        profile: { ...prev.profile, ...updatedProfile } 
      }));
      return updatedProfile;
    } catch (error) {
      logDataError('PROFILE_UPDATE', error, { updates });
      showToast({ message: `Profile update failed: ${error.message}`, type: 'error' });
      throw error;
    }
  }, [dataState.user, showToast]);

  const saveChain = useCallback(async (chain) => {
    if (!dataState.user) return;

    try {
      const chainData = {
        ...chain,
        user_id: dataState.user.id,
        is_default: false,
        is_public: false
      };

      const savedChain = await dbHelpers.saveChain(chainData);
      
      // Update local state
      setDataState(prev => ({
        ...prev,
        chains: { ...prev.chains, [savedChain.id]: savedChain }
      }));

      return savedChain;
    } catch (error) {
      logDataError('CHAIN_SAVE', error, { chainId: chain.id });
      showToast({ message: `Chain save failed: ${error.message}`, type: 'error' });
      throw error;
    }
  }, [dataState.user, showToast]);

  const deleteChain = useCallback(async (chainId) => {
    if (!dataState.user) return;

    try {
      await dbHelpers.deleteChain(chainId);
      
      // Update local state
      setDataState(prev => {
        const newChains = { ...prev.chains };
        delete newChains[chainId];
        return { ...prev, chains: newChains };
      });
    } catch (error) {
      logDataError('CHAIN_DELETE', error, { chainId });
      showToast({ message: `Chain delete failed: ${error.message}`, type: 'error' });
      throw error;
    }
  }, [dataState.user, showToast]);

  const saveAnalysis = useCallback(async (inputData, outputData, chainUsed, cost = 0) => {
    if (!dataState.user) return;

    try {
      const savedAnalysis = await dbHelpers.saveAnalysis(
        dataState.user.id, 
        inputData, 
        outputData, 
        chainUsed, 
        cost
      );

      // Update local state
      setDataState(prev => ({
        ...prev,
        history: [savedAnalysis, ...prev.history]
      }));

      // Update total spend in profile
      if (cost > 0) {
        await updateProfile({ 
          total_spend: (dataState.profile?.total_spend || 0) + cost 
        });
      }

      return savedAnalysis;
    } catch (error) {
      logDataError('ANALYSIS_SAVE', error, { chainUsed, cost });
      showToast({ message: `Analysis save failed: ${error.message}`, type: 'error' });
      throw error;
    }
  }, [dataState.user, dataState.profile, updateProfile, showToast]);

  const deleteHistoryItems = useCallback(async (historyIds) => {
    if (!dataState.user) return;

    try {
      await dbHelpers.deleteHistoryItems(dataState.user.id, historyIds);
      
      // Update local state
      setDataState(prev => ({
        ...prev,
        history: prev.history.filter(item => !historyIds.includes(item.id))
      }));
    } catch (error) {
      logDataError('HISTORY_DELETE', error, { historyIds });
      showToast({ message: `History delete failed: ${error.message}`, type: 'error' });
      throw error;
    }
  }, [dataState.user, showToast]);

  // Auth state management
  useEffect(() => {
    let mounted = true;
    
    const handleAuthChange = (session) => {
      if (!mounted) return;
      updateDebug(session ? `Auth change: ${session.user.email}` : 'Auth change: signed out');
      loadUserData(session?.user || null);
    };

    // Get initial session
    supabase.auth.getSession().then(({ data: { session }, error }) => {
      if (error) {
        updateDebug(`Session error: ${error.message}`);
        console.error('Session error:', error);
      }
      handleAuthChange(session);
    });

    // Listen for auth changes
    const { data: { subscription } } = supabase.auth.onAuthStateChange((_event, session) => {
      handleAuthChange(session);
    });

    return () => {
      mounted = false;
      subscription?.unsubscribe();
    };
  }, [loadUserData]);

  return {
    ...dataState,
    // Mutation functions
    updateProfile,
    saveChain,
    deleteChain,
    saveAnalysis,
    deleteHistoryItems,
    // Utility
    logout: () => supabase.auth.signOut()
  };
};

} export class PatternExecutor {
  constructor(apiKey, availableModels, updateCallback, errorCallback) {
    this.apiKey = apiKey;
    this.availableModels = availableModels;
    this.updateCallback = updateCallback;
    this.errorCallback = errorCallback;
    this.abortController = null;
  }

  parsePattern(pattern) {
    const steps = pattern.split('-');
    return steps.map(step => ({
      agents: step.includes('+') ? step.split('+') : [step],
      isParallel: step.includes('+')
    }));
  }

  calculateRoundNumber(pattern, stepIndex, agentChar) {
    const beforeSteps = pattern.split('-').slice(0, stepIndex + 1).join('-');
    return (beforeSteps.match(new RegExp(agentChar, 'g')) || []).length;
  }

  async executeAgent(agentKey, config, context, roundNum, stepIndex, totalSteps) {
    const promptAddon = config.rounds?.[roundNum]?.prompt_addon || '';
    const systemPrompt = `${config.prompt}\n\n${promptAddon}`;
    
    try {
      const response = await fetch("https://openrouter.ai/api/v1/chat/completions", {
        method: 'POST',
        headers: { 
          "Authorization": `Bearer ${this.apiKey}`, 
          "Content-Type": "application/json",
          "HTTP-Referer": window.location.href,
          "X-Title": "NoBS Analysis"
        },
        body: JSON.stringify({
          model: config.model,
          messages: [
            { role: 'system', content: systemPrompt }, 
            { role: 'user', content: context }
          ],
          stream: true,
          temperature: 0.7,
          max_tokens: 4000
        }),
        signal: this.abortController.signal,
      });

      if (!response.ok) {
        const errorText = await response.text();
        throw new Error(`API Error (${response.status}): ${errorText}`);
      }
      
      const reader = response.body.getReader();
      const decoder = new TextDecoder();
      let fullResponse = "";
      
      while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        
        const chunk = decoder.decode(value);
        const lines = chunk.split('\n');
        
        for (const line of lines) {
          if (line.startsWith('data: ')) {
            const data = line.substring(6);
            if (data === '[DONE]') continue;
            
            try {
              const parsed = JSON.parse(data);
              const delta = parsed.choices[0]?.delta?.content || "";
              if (delta) {
                fullResponse += delta;
                this.updateCallback(agentKey, fullResponse, roundNum, stepIndex, totalSteps);
              }
            } catch (e) {
              console.warn('Stream parse error:', e, data);
            }
          }
        }
      }

      return {
        agentKey,
        response: fullResponse,
        cost: this.calculateCost(config.model, context, fullResponse),
        roundNum
      };

    } catch (error) {
      if (error.name === 'AbortError') throw error;
      this.errorCallback(`Agent ${agentKey} failed: ${error.message}`);
      throw error;
    }
  }

  calculateCost(modelId, prompt, completion) {
    const model = this.availableModels.find(m => m.id === modelId);
    if (!model?.costPerMToken) return 0;
    
    const promptTokens = Math.ceil(prompt.length / 4);
    const completionTokens = Math.ceil(completion.length / 4);
    
    // Use costPerMToken which is already the combined rate
    return ((promptTokens + completionTokens) / 1000000) * model.costPerMToken;
  }

  async execute(chain, query, context = '') {
    this.abortController = new AbortController();
    const fullContext = `USER QUERY: ${query}\n\nADDITIONAL CONTEXT: ${context || 'None'}`;
    const parsedSteps = this.parsePattern(chain.pattern);
    let accumulatedContext = fullContext;
    let totalCost = 0;
    let stepResults = {};

    try {
      for (let stepIndex = 0; stepIndex < parsedSteps.length; stepIndex++) {
        const step = parsedSteps[stepIndex];
        
        if (step.isParallel) {
          // Parallel execution
          const promises = step.agents.map(agentChar => {
            const agentKey = `agent_${agentChar}`;
            const config = chain.agents[agentKey];
            if (!config) return null;
            
            const roundNum = this.calculateRoundNumber(chain.pattern, stepIndex, agentChar);
            return this.executeAgent(agentKey, config, accumulatedContext, roundNum, stepIndex, parsedSteps.length);
          });

          const results = await Promise.all(promises.filter(Boolean));
          let stepOutput = "";
          
          results.forEach(result => {
            stepOutput += `\n\n--- ${result.agentKey.toUpperCase()} ---\n${result.response}`;
            totalCost += result.cost;
            
            // Store result for local state updates
            if (!stepResults[result.agentKey]) stepResults[result.agentKey] = [];
            stepResults[result.agentKey][result.roundNum - 1] = result.response;
          });
          
          accumulatedContext += stepOutput;
        } else {
          // Sequential execution
          const agentChar = step.agents[0];
          const agentKey = `agent_${agentChar}`;
          const config = chain.agents[agentKey];
          
          if (config) {
            const roundNum = this.calculateRoundNumber(chain.pattern, stepIndex, agentChar);
            const result = await this.executeAgent(agentKey, config, accumulatedContext, roundNum, stepIndex, parsedSteps.length);
            
            accumulatedContext += `\n\n--- ${result.agentKey.toUpperCase()} ---\n${result.response}`;
            totalCost += result.cost;
            
            // Store result for local state updates
            if (!stepResults[result.agentKey]) stepResults[result.agentKey] = [];
            stepResults[result.agentKey][result.roundNum - 1] = result.response;
          }
        }
      }

      // Determine final output (last agent's last response)
      const lastStepAgents = parsedSteps[parsedSteps.length - 1].agents;
      const lastAgentChar = lastStepAgents[lastStepAgents.length - 1];
      const lastAgentKey = `agent_${lastAgentChar}`;
      const finalOutput = stepResults[lastAgentKey]?.[stepResults[lastAgentKey].length - 1] || '';

      return { 
        output: finalOutput, 
        cost: totalCost,
        stepResults: stepResults 
      };
    } catch (error) {
      if (error.name === 'AbortError') {
        console.log('Analysis aborted by user');
        return null;
      }
      throw error;
    }
  }

  abort() {
    if (this.abortController) {
      this.abortController.abort();
      this.abortController = null;
    }
  }
}

export const MODEL_DATA_MANAGER = {
  async fetchAllModels(apiKey) {
    const response = await fetch('https://openrouter.ai/api/v1/models', {
      headers: { 'Authorization': `Bearer ${apiKey}` }
    });
    if (!response.ok) throw new Error('Failed to fetch the model list from OpenRouter API.');
    const { data } = await response.json();
    return data.map((model, index) => ({
      id: model.id, 
      name: model.name, 
      description: model.description, 
      ranking: index + 1,
      costPerMToken: model.pricing ? ((parseFloat(model.pricing.prompt || 0) + parseFloat(model.pricing.completion || 0)) / 2 * 1000000) : 0,
      contextLength: model.context_length ? `${Math.round(model.context_length / 1000)}K` : '?',
      isPaid: model.pricing && parseFloat(model.pricing.prompt || 0) > 0,
      pricing: model.pricing // Keep original pricing for accurate calculations
    }));
  }
