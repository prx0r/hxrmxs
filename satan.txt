class EntityNormalizer:
    """Advanced entity normalization using medical NLP models"""
    
    def __init__(self):
        try:
            # Load scientific NLP model for medical entity processing
            self.nlp = spacy.load("en_core_sci_sm")
            logger.info("Successfully loaded scientific NLP model")
        except OSError:
            logger.warning("Scientific NLP model not found, falling back to basic model")
            self.nlp = spacy.load("en_core_web_sm")
        
        # Entity type mappings for standardization
        self.entity_type_mappings = {
            'drug': 'MEDICATION',
            'medication': 'MEDICATION',
            'compound': 'MEDICATION',
            'disease': 'CONDITION',
            'disorder': 'CONDITION',
            'condition': 'CONDITION',
            'symptom': 'SYMPTOM',
            'side_effect': 'SIDE_EFFECT',
            'treatment': 'TREATMENT',
            'therapy': 'TREATMENT'
        }
    
    def normalize_entity_name(self, name: str) -> str:
        """Normalize entity names for consistency"""
        if not name:
            return ""
        
        # Basic normalization
        normalized = name.strip().lower()
        
        # Use NLP model for additional normalization
        doc = self.nlp(normalized)
        
        # Extract lemmatized tokens, excluding stop words and punctuation
        tokens = [token.lemma_ for token in doc 
                 if not token.is_stop and not token.is_punct and token.text.strip()]
        
        return " ".join(tokens) if tokens else normalized
    
    def classify_entity_type(self, entity_name: str, context: str = "") -> str:
        """Classify entity type based on name and context"""
        entity_lower = entity_name.lower()
        context_lower = context.lower()
        
        # Use NLP model for entity recognition
        doc = self.nlp(f"{entity_name} {context}")
        
        # Check for medical entities
        for ent in doc.ents:
            if ent.label_ in ['CHEMICAL', 'DRUG']:
                return 'MEDICATION'
            elif ent.label_ in ['DISEASE', 'DISORDER']:
                return 'CONDITION'
        
        # Fallback to keyword matching
        for keyword, entity_type in self.entity_type_mappings.items():
            if keyword in entity_lower or keyword in context_lower:
                return entity_type
        
        return 'UNKNOWN'
    
    def extract_confidence_score(self, source_data: Dict[str, Any]) -> float:
        """Extract confidence score from source data"""
        # Look for various confidence indicators
        confidence_fields = ['confidence', 'score', 'probability', 'certainty']
        
        for field in confidence_fields:
            if field in source_data:
                try:
                    return float(source_data[field])
                except (ValueError, TypeError):
                    continue
        
        # Default confidence based on source type
        source_type = source_data.get('source_type', '').lower()
        if 'pubmed' in source_type:
            return 0.9
        elif 'drugbank' in source_type:
            return 0.95
        elif 'manual' in source_type or 'expert' in source_type:
            return 1.0
        
        return 0.7  # Default moderate confidence

class KnowledgeGraphParser:
    """Parses external knowledge graph data into standardized format"""
    
    def __init__(self):
        self.normalizer = EntityNormalizer()
    
    async def parse_adhd_kg(self, file_path: Path) -> List[Dict[str, Any]]:
        """Parse ADHD-KG RDF data (.n3 format)"""
        try:
            logger.info(f"Parsing ADHD-KG file: {file_path}")
            
            # Load RDF graph
            g = rdflib.Graph()
            g.parse(file_path, format="n3")
            
            entities = []
            
            # Extract triples from the graph
            for subject, predicate, obj in g:
                # Convert RDF terms to strings
                subject_str = str(subject)
                predicate_str = str(predicate)
                object_str = str(obj)
                
                # Extract entity names from URIs
                subject_name = self._extract_name_from_uri(subject_str)
                object_name = self._extract_name_from_uri(object_str)
                relationship = self._extract_name_from_uri(predicate_str)
                
                if subject_name and object_name and relationship:
                    entity = {
                        'entity_id': f"adhd_kg_{hashlib.md5(f'{subject_name}_{relationship}_{object_name}'.encode()).hexdigest()[:16]}",
                        'source_kg': 'ADHD-KG',
                        'source_link': str(subject),
                        'entity_type': self.normalizer.classify_entity_type(subject_name, 'adhd'),
                        'entity_name': self.normalizer.normalize_entity_name(subject_name),
                        'relationship_type': relationship.upper().replace('-', '_'),
                        'related_entity_name': self.normalizer.normalize_entity_name(object_name),
                        'source_confidence': 0.85,
                        'metadata': {
                            'original_subject': subject_str,
                            'original_predicate': predicate_str,
                            'original_object': object_str
                        }
                    }
                    entities.append(entity)
            
            logger.info(f"Successfully parsed {len(entities)} entities from ADHD-KG")
            return entities
            
        except Exception as e:
            logger.error(f"Error parsing ADHD-KG file {file_path}: {str(e)}")
            raise KnowledgeIngestionError(f"Failed to parse ADHD-KG: {str(e)}")
    
    async def parse_drugbank(self, file_path: Path) -> List[Dict[str, Any]]:
        """Parse DrugBank XML data"""
        try:
            logger.info(f"Parsing DrugBank file: {file_path}")
            
            tree = ET.parse(file_path)
            root = tree.getroot()
            
            entities = []
            
            # Navigate through DrugBank XML structure
            for drug in root.findall('.//{http://www.drugbank.ca}drug'):
                drug_id = drug.get('id', '')
                
                # Extract drug name
                name_elem = drug.find('.//{http://www.drugbank.ca}name')
                drug_name = name_elem.text if name_elem is not None else ''
                
                if not drug_name:
                    continue
                
                # Extract drug interactions
                interactions = drug.findall('.//{http://www.drugbank.ca}drug-interaction')
                for interaction in interactions:
                    interaction_drug = interaction.find('.//{http://www.drugbank.ca}name')
                    description = interaction.find('.//{http://www.drugbank.ca}description')
                    
                    if interaction_drug is not None and description is not None:
                        entity = {
                            'entity_id': f"drugbank_{drug_id}_{hashlib.md5(interaction_drug.text.encode()).hexdigest()[:16]}",
                            'source_kg': 'DrugBank',
                            'source_link': f"https://go.drugbank.com/drugs/{drug_id}",
                            'entity_type': 'MEDICATION',
                            'entity_name': self.normalizer.normalize_entity_name(drug_name),
                            'relationship_type': 'INTERACTS_WITH',
                            'related_entity_name': self.normalizer.normalize_entity_name(interaction_drug.text),
                            'source_confidence': 0.95,
                            'metadata': {
                                'drug_id': drug_id,
                                'interaction_description': description.text,
                                'interaction_type': 'drug_interaction'
                            }
                        }
                        entities.append(entity)
                
                # Extract side effects
                side_effects = drug.findall('.//{http://www.drugbank.ca}adverse-reaction')
                for side_effect in side_effects:
                    if side_effect.text:
                        entity = {
                            'entity_id': f"drugbank_{drug_id}_side_effect_{hashlib.md5(side_effect.text.encode()).hexdigest()[:16]}",
                            'source_kg': 'DrugBank',
                            'source_link': f"https://go.drugbank.com/drugs/{drug_id}",
                            'entity_type': 'MEDICATION',
                            'entity_name': self.normalizer.normalize_entity_name(drug_name),
                            'relationship_type': 'CAUSES_SIDE_EFFECT',
                            'related_entity_name': self.normalizer.normalize_entity_name(side_effect.text),
                            'source_confidence': 0.9,
                            'metadata': {
                                'drug_id': drug_id,
                                'side_effect_type': 'adverse_reaction'
                            }
                        }
                        entities.append(entity)
            
            logger.info(f"Successfully parsed {len(entities)} entities from DrugBank")
            return entities
            
        except Exception as e:
            logger.error(f"Error parsing DrugBank file {file_path}: {str(e)}")
            raise KnowledgeIngestionError(f"Failed to parse DrugBank: {str(e)}")
    
    async def parse_pubmed(self, query: str, max_results: int = 50) -> List[Dict[str, Any]]:
        """Fetch PubMed abstracts via API"""
        try:
            logger.info(f"Fetching PubMed data for query: {query}")
            
            base_url = "https://eutils.ncbi.nlm.nih.gov/entrez/eutils/"
            
            async with aiohttp.ClientSession() as session:
                # Search for PMIDs
                search_url = f"{base_url}esearch.fcgi"
                search_params = {
                    'db': 'pubmed',
                    'term': query,
                    'retmax': max_results,
                    'retmode': 'json'
                }
                
                async with session.get(search_url, params=search_params) as response:
                    if response.status != 200:
                        raise KnowledgeIngestionError(f"PubMed search failed: {response.status}")
                    
                    search_data = await response.json()
                    pmids = search_data.get('esearchresult', {}).get('idlist', [])
                
                if not pmids:
                    logger.warning(f"No PubMed results found for query: {query}")
                    return []
                
                # Fetch abstracts
                fetch_url = f"{base_url}efetch.fcgi"
                fetch_params = {
                    'db': 'pubmed',
                    'id': ','.join(pmids),
                    'retmode': 'xml'
                }
                
                async with session.get(fetch_url, params=fetch_params) as response:
                    if response.status != 200:
                        raise KnowledgeIngestionError(f"PubMed fetch failed: {response.status}")
                    
                    xml_data = await response.text()
            
            # Parse XML response
            root = ET.fromstring(xml_data)
            entities = []
            
            for article in root.findall('.//PubmedArticle'):
                pmid_elem = article.find('.//PMID')
                title_elem = article.find('.//ArticleTitle')
                abstract_elem = article.find('.//AbstractText')
                
                if pmid_elem is not None and title_elem is not None:
                    pmid = pmid_elem.text
                    title = title_elem.text or ''
                    abstract = abstract_elem.text if abstract_elem is not None else ''
                    
                    # Extract entities from title and abstract using NLP
                    text = f"{title} {abstract}"
                    doc = self.normalizer.nlp(text)
                    
                    for ent in doc.ents:
                        if ent.label_ in ['CHEMICAL', 'DRUG', 'DISEASE', 'DISORDER']:
                            entity = {
                                'entity_id': f"pubmed_{pmid}_{ent.label_}_{hashlib.md5(ent.text.encode()).hexdigest()[:16]}",
                                'source_kg': 'PubMed',
                                'source_link': f"https://pubmed.ncbi.nlm.nih.gov/{pmid}/",
                                'entity_type': self.normalizer.classify_entity_type(ent.text, ent.label_),
                                'entity_name': self.normalizer.normalize_entity_name(ent.text),
                                'relationship_type': 'MENTIONED_IN',
                                'related_entity_name': title,
                                'source_confidence': 0.75,
                                'metadata': {
                                    'pmid': pmid,
                                    'title': title,
                                    'entity_label': ent.label_,
                                    'context': ent.sent.text if ent.sent else ''
                                }
                            }
                            entities.append(entity)
            
            logger.info(f"Successfully parsed {len(entities)} entities from PubMed")
            return entities
            
        except Exception as e:
            logger.error(f"Error fetching PubMed data: {str(e)}")
            raise KnowledgeIngestionError(f"Failed to fetch PubMed data: {str(e)}")
    
    def _extract_name_from_uri(self, uri: str) -> str:
        """Extract entity name from URI"""
        if not uri:
            return ""
        
        # Extract the last part of the URI
        if '#' in uri:
            name = uri.split('#')[-1]
        elif '/' in uri:
            name = uri.split('/')[-1]
        else:
            name = uri
        
        # Clean up the name
        name = name.replace('_', ' ').replace('-', ' ')
        return name.strip()

class DeduplicationEngine:
    """Handles deduplication of knowledge entities"""
    
    def __init__(self):
        self.processed_hashes: Set[str] = set()
    
    def generate_content_hash(self, entity: Dict[str, Any]) -> str:
        """Generate SHA-256 hash for content deduplication"""
        # Create a normalized representation for hashing
        content_parts = [
            entity.get('entity_name', '').lower().strip(),
            entity.get('entity_type', '').lower().strip(),
            entity.get('relationship_type', '').lower().strip(),
            entity.get('related_entity_name', '').lower().strip(),
            entity.get('source_kg', '').lower().strip()
        ]
        
        content_string = '|'.join(content_parts)
        return hashlib.sha256(content_string.encode('utf-8')).hexdigest()
    
    def is_duplicate(self, entity: Dict[str, Any]) -> bool:
        """Check if entity is a duplicate based on content hash"""
        content_hash = self.generate_content_hash(entity)
        
        if content_hash in self.processed_hashes:
            return True
        
        self.processed_hashes.add(content_hash)
        return False
    
    async def deduplicate_against_db(self, 
                                   entities: List[Dict[str, Any]], 
                                   session: AsyncSession) -> List[Dict[str, Any]]:
        """Remove duplicates against existing database entries"""
        if not entities:
            return []
        
        # Get existing content hashes from database
        existing_hashes_query = select(ExternalKnowledge.content_hash)
        result = await session.execute(existing_hashes_query)
        existing_hashes = {row[0] for row in result.fetchall()}
        
        # Filter out duplicates
        unique_entities = []
        for entity in entities:
            content_hash = self.generate_content_hash(entity)
            entity['content_hash'] = content_hash
            
            if content_hash not in existing_hashes and not self.is_duplicate(entity):
                unique_entities.append(entity)
            else:
                logger.debug(f"Skipping duplicate entity: {entity.get('entity_name', 'Unknown')}")
        
        logger.info(f"Filtered {len(entities) - len(unique_entities)} duplicates, {len(unique_entities)} unique entities remain")
        return unique_entities

class KnowledgeIngestionEngine:
    """Main engine for knowledge graph ingestion with comprehensive ETL pipeline"""
    
    def __init__(self):
        self.parser = KnowledgeGraphParser()
        self.deduplication_engine = DeduplicationEngine()
    
    async def ingest_from_file(self, file_path: Path, source_type: str) -> Dict[str, Any]:
        """Ingest knowledge from a file source"""
        try:
            logger.info(f"Starting ingestion from {file_path} (type: {source_type})")
            
            # Parse based on source type
            if source_type.lower() == 'adhd_kg':
                entities = await self.parser.parse_adhd_kg(file_path)
            elif source_type.lower() == 'drugbank':
                entities = await self.parser.parse_drugbank(file_path)
            else:
                raise ValidationError(f"Unsupported source type: {source_type}")
            
            return await self._process_and_store_entities(entities)
            
        except Exception as e:
            logger.error(f"Failed to ingest from file {file_path}: {str(e)}")
            raise
    
    async def ingest_from_pubmed(self, query: str, max_results: int = 50) -> Dict[str, Any]:
        """Ingest knowledge from PubMed API"""
        try:
            logger.info(f"Starting PubMed ingestion for query: {query}")
            
            entities = await self.parser.parse_pubmed(query, max_results)
            return await self._process_and_store_entities(entities)
            
        except Exception as e:
            logger.error(f"Failed to ingest from PubMed: {str(e)}")
            raise
    
    async def _process_and_store_entities(self, entities: List[Dict[str, Any]]) -> Dict[str, Any]:
        """Process and store entities in database"""
        if not entities:
            return {'total_processed': 0, 'new_entities': 0, 'duplicates_filtered': 0}
        
        async with get_async_session() as session:
            try:
                # Deduplicate against existing database
                unique_entities = await self.deduplication_engine.deduplicate_against_db(entities, session)
                
                if not unique_entities:
                    logger.info("No new entities to store after deduplication")
                    return {
                        'total_processed': len(entities),
                        'new_entities': 0,
                        'duplicates_filtered': len(entities)
                    }
                
                # Batch insert using PostgreSQL's ON CONFLICT
                stmt = insert(ExternalKnowledge).values(unique_entities)
                stmt = stmt.on_conflict_do_nothing(index_elements=['entity_id'])
                
                result = await session.execute(stmt)
                await session.commit()
                
                inserted_count = result.rowcount
                
                logger.info(f"Successfully inserted {inserted_count} new entities")
                
                return {
                    'total_processed': len(entities),
                    'new_entities': inserted_count,
                    'duplicates_filtered': len(entities) - len(unique_entities)
                }
                
            except Exception as e:
                await session.rollback()
                logger.error(f"Database error during entity storage: {str(e)}")
                raise KnowledgeIngestionError(f"Failed to store entities: {str(e)}")
    
    async def validate_entity(self, entity: Dict[str, Any]) -> bool:
        """Validate entity data before storage"""
        required_fields = [
            'entity_id', 'source_kg', 'entity_type', 
            'entity_name', 'relationship_type', 'related_entity_name'
        ]
        
        # Check required fields
        for field in required_fields:
            if not entity.get(field):
                logger.warning(f"Entity missing required field '{field}': {entity}")
                return False
        
        # Validate data types and ranges
        if 'source_confidence' in entity:
            confidence = entity['source_confidence']
            if not isinstance(confidence, (int, float)) or not (0 <= confidence <= 1):
                logger.warning(f"Invalid confidence score: {confidence}")
                return False
        
        return True
    
    async def get_ingestion_stats(self) -> Dict[str, Any]:
        """Get statistics about ingested knowledge"""
        async with get_async_session() as session:
            try:
                # Total entities
                total_query = select(ExternalKnowledge).count()
                total_result = await session.execute(total_query)
                total_entities = total_result.scalar()
                
                # Entities by source
                source_query = select(
                    ExternalKnowledge.source_kg,
                    ExternalKnowledge.id.count().label('count')
                ).group_by(ExternalKnowledge.source_kg)
                
                source_result = await session.execute(source_query)
                source_stats = {row[0]: row[1] for row in source_result.fetchall()}
                
                # Entities by type
                type_query = select(
                    ExternalKnowledge.entity_type,
                    ExternalKnowledge.id.count().label('count')
                ).group_by(ExternalKnowledge.entity_type)
                
                type_result = await session.execute(type_query)
                type_stats = {row[0]: row[1] for row in type_result.fetchall()}
                
                return {
                    'total_entities': total_entities,
                    'entities_by_source': source_stats,
                    'entities_by_type': type_stats,
                    'last_updated': None  # Could add timestamp tracking
                }
                
            except Exception as e:
                logger.error(f"Error getting ingestion stats: {str(e)}")
                raise

async def run_ingestion_batch(config: Dict[str, Any]):
    """Run a batch ingestion job based on configuration"""
    engine = KnowledgeIngestionEngine()
    results = []
    
    try:
        # Process file sources
        for file_config in config.get('files', []):
            file_path = Path(file_config['path'])
            source_type = file_config['type']
            
            if file_path.exists():
                result = await engine.ingest_from_file(file_path, source_type)
                result['source'] = str(file_path)
                result['type'] = source_type
                results.append(result)
            else:
                logger.error(f"File not found: {file_path}")
        
        # Process PubMed queries
        for pubmed_config in config.get('pubmed_queries', []):
            query = pubmed_config['query']
            max_results = pubmed_config.get('max_results', 50)
            
            result = await engine.ingest_from_pubmed(query, max_results)
            result['source'] = f"PubMed: {query}"
            result['type'] = 'pubmed'
            results.append(result)
        
        # Log summary
        total_new = sum(r['new_entities'] for r in results)
        total_processed = sum(r['total_processed'] for r in results)
        
        logger.info(f"Batch ingestion complete: {total_new}/{total_processed} new entities stored")
        
        return {
            'batch_results': results,
            'summary': {
                'total_new_entities': total_new,
                'total_processed': total_processed,
                'sources_processed': len(results)
            }
        }
        
    except Exception as e:
        logger.error(f"Batch ingestion failed: {str(e)}")
        raise

class FactualQueryDetector:
    """Detects factual queries that require external knowledge retrieval"""
    
    def __init__(self):
        # Medical and factual query patterns
        self.factual_patterns = {
            'medication_query': [
                r'\b(medication|drug|medicine|pill|tablet|dosage|side effect)\b',
                r'\b(prescription|pharma|therapeutic|treatment)\b',
                r'\b(mg|ml|dose|daily|twice|morning|evening)\b'
            ],
            'condition_query': [
                r'\b(condition|disorder|disease|syndrome|diagnosis)\b',
                r'\b(symptoms|treatment|therapy|cure|prognosis)\b',
                r'\b(adhd|anxiety|depression|bipolar|ptsd|ocd)\b'
            ],
            'research_query': [
                r'\b(study|research|evidence|clinical trial|meta-analysis)\b',
                r'\b(published|journal|findings|results|statistics)\b',
                r'\b(effective|efficacy|safety|contraindication)\b'
            ],
            'interaction_query': [
                r'\b(interact|combination|together|mix|concurrent)\b',
                r'\b(contraindicate|avoid|warning|caution)\b'
            ]
        }
        
        # Question indicators
        self.question_patterns = [
            r'\bwhat\s+(is|are|about|regarding)\b',
            r'\bhow\s+(does|do|can|should)\b',
            r'\bwhy\s+(is|are|does|do)\b',
            r'\bcan\s+i\b',
            r'\bis\s+it\b',
            r'\bshould\s+i\b'
        ]
    
    async def detect_factual_query(self, user_input: str) -> Optional[Dict[str, Any]]:
        """Detect if user input contains factual queries needing external knowledge"""
        try:
            input_lower = user_input.lower()
            detected_categories = []
            detected_terms = []
            confidence_score = 0.0
            
            # Check for factual patterns
            for category, patterns in self.factual_patterns.items():
                category_matches = 0
                for pattern in patterns:
                    matches = re.findall(pattern, input_lower)
                    if matches:
                        category_matches += len(matches)
                        detected_terms.extend(matches)
                
                if category_matches > 0:
                    detected_categories.append(category)
                    confidence_score += min(category_matches * 0.2, 0.8)
            
            # Check for question patterns
            question_score = 0.0
            for pattern in self.question_patterns:
                if re.search(pattern, input_lower):
                    question_score += 0.3
            
            confidence_score += min(question_score, 0.5)
            confidence_score = min(confidence_score, 1.0)
            
            # Return detection result if confidence is above threshold
            if confidence_score >= 0.3 or detected_categories:
                return {
                    'is_factual_query': True,
                    'detected_categories': detected_categories,
                    'detected_terms': list(set(detected_terms)),
                    'confidence_score': confidence_score,
                    'requires_external_knowledge': confidence_score >= 0.5
                }
            
            return None
            
        except Exception as e:
            logger.error(f"Error in factual query detection: {str(e)}")
            return None
    
    def extract_entities(self, user_input: str) -> List[str]:
        """Extract potential entities for knowledge graph lookup"""
        # Simple entity extraction - could be enhanced with NLP
        entities = []
        
        # Extract medication names (simple pattern)
        med_patterns = [
            r'\b[A-Z][a-z]+(?:xr|xl|er|sr|cr)?\b',  # Capitalized words (potential drug names)
            r'\b\d+\s*mg\b',  # Dosages
        ]
        
        for pattern in med_patterns:
            matches = re.findall(pattern, user_input)
            entities.extend(matches)
        
        return list(set(entities))

class ExternalKnowledgeRetriever:
    """Retrieves relevant external knowledge from the Super-Graph"""
    
    def __init__(self):
        self.max_results = 10
    
    async def query_knowledge_graph(self, 
                                  user_input: str, 
                                  detected_terms: List[str],
                                  categories: List[str]) -> List[Dict[str, Any]]:
        """Query the Super-Graph for relevant external knowledge"""
        try:
            async with get_async_session() as session:
                # Build search query based on detected terms and categories
                search_terms = self._prepare_search_terms(user_input, detected_terms)
                
                # Create base query
                query = select(ExternalKnowledge)
                
                # Add filters based on detected categories
                conditions = []
                
                for term in search_terms:
                    term_conditions = or_(
                        ExternalKnowledge.entity_name.ilike(f'%{term}%'),
                        ExternalKnowledge.related_entity_name.ilike(f'%{term}%')
                    )
                    conditions.append(term_conditions)
                
                # Add category-specific filters
                if 'medication_query' in categories:
                    conditions.append(ExternalKnowledge.entity_type == 'MEDICATION')
                if 'condition_query' in categories:
                    conditions.append(ExternalKnowledge.entity_type.in_(['CONDITION', 'SYMPTOM']))
                
                if conditions:
                    query = query.where(or_(*conditions))
                
                # Order by confidence and limit results
                query = query.order_by(
                    ExternalKnowledge.source_confidence.desc(),
                    ExternalKnowledge.created_at.desc()
                ).limit(self.max_results)
                
                result = await session.execute(query)
                knowledge_entries = result.scalars().all()
                
                # Format results for prompt inclusion
                formatted_results = []
                for entry in knowledge_entries:
                    formatted_results.append({
                        'entity_name': entry.entity_name,
                        'entity_type': entry.entity_type,
                        'relationship_type': entry.relationship_type,
                        'related_entity': entry.related_entity_name,
                        'source': entry.source_kg,
                        'confidence': entry.source_confidence,
                        'source_link': entry.source_link,
                        'metadata': entry.metadata
                    })
                
                logger.info(f"Retrieved {len(formatted_results)} knowledge entries for query")
                return formatted_results
                
        except Exception as e:
            logger.error(f"Error querying knowledge graph: {str(e)}")
            return []
    
    def _prepare_search_terms(self, user_input: str, detected_terms: List[str]) -> List[str]:
        """Prepare and clean search terms for knowledge graph lookup"""
        # Combine detected terms with extracted entities
        all_terms = set(detected_terms)
        
        # Extract additional terms from user input
        # Remove common stop words and focus on meaningful terms
        stop_words = {'i', 'me', 'my', 'am', 'is', 'are', 'the', 'a', 'an', 'and', 'or', 'but'}
        
        words = re.findall(r'\b\w+\b', user_input.lower())
        meaningful_words = [word for word in words if len(word) > 2 and word not in stop_words]
        
        all_terms.update(meaningful_words[:5])  # Limit to avoid overly broad searches
        
        return list(all_terms)

class ReasoningEngine:
    """
    Advanced reasoning engine that integrates internal therapeutic knowledge (TCL)
    with external factual knowledge (Super-Graph) for comprehensive responses
    """
    
    def __init__(self, 
                 model_manager: ModelManager,
                 tcl_service: TherapeuticComponentLibraryService,
                 memory_service: MemoryService,
                 profile_service: ProfileService):
        
        self.model_manager = model_manager
        self.tcl_service = tcl_service
        self.memory_service = memory_service
        self.profile_service = profile_service
        
        # Initialize detectors and retrievers
        self.factual_detector = FactualQueryDetector()
        self.knowledge_retriever = ExternalKnowledgeRetriever()
        
        # Initialize template environment
        template_path = settings.PROMPT_TEMPLATES_PATH
        self.jinja_env = Environment(loader=FileSystemLoader(template_path))
        
        logger.info("ReasoningEngine initialized successfully")
    
    async def process_user_input(self, 
                               user_input: str, 
                               user_id: str,
                               session_id: str,
                               context: Optional[Dict[str, Any]] = None) -> Dict[str, Any]:
        """
        Process user input with integrated reasoning over both internal TCL
        and external Super-Graph knowledge
        """
        try:
            start_time = datetime.now()
            
            # 1. Analyze user input for factual queries
            factual_query = await self.factual_detector.detect_factual_query(user_input)
            logger.debug(f"Factual query detection: {factual_query}")
            
            # 2. Perform Dual Retrieval
            external_knowledge = []
            if factual_query and factual_query.get('requires_external_knowledge'):
                external_knowledge = await self.knowledge_retriever.query_knowledge_graph(
                    user_input,
                    factual_query.get('detected_terms', []),
                    factual_query.get('detected_categories', [])
                )
            
            # Search internal therapeutic components
            therapeutic_components = await self.tcl_service.search_components(
                user_input, user_id
            )
            
            # 3. Get user context
            user_profile = await self.profile_service.get_user_profile(user_id)
            session_context = await self.memory_service.get_session_context(session_id)
            
            # 4. Build deliberative prompt
            prompt = await self._build_response_prompt(
                user_input=user_input,
                components=therapeutic_components,
                external_knowledge=external_knowledge,
                user_profile=user_profile,
                session_context=session_context,
                factual_query=factual_query
            )
            
            # 5. Generate response
            response_text = await self.model_manager.generate_response(
                prompt=prompt,
                user_id=user_id,
                max_tokens=settings.MAX_RESPONSE_TOKENS
            )
            
            # 6. Calculate processing metrics
            processing_time = (datetime.now() - start_time).total_seconds() * 1000
            
            # 7. Log interaction
            await self._log_interaction(
                user_id=user_id,
                session_id=session_id,
                user_input=user_input,
                ai_response=response_text,
                components_used=therapeutic_components,
                external_knowledge_used=external_knowledge,
                processing_time_ms=int(processing_time),
                factual_query=factual_query
            )
            
            return {
                'response_text': response_text,
                'components_used': [comp['component_id'] for comp in therapeutic_components],
                'external_knowledge_used': len(external_knowledge),
                'factual_query_detected': factual_query is not None,
                'processing_time_ms': int(processing_time),
                'confidence_score': self._calculate_response_confidence(
                    therapeutic_components, external_knowledge, factual_query
                )
            }
            
        except Exception as e:
            logger.error(f"Error processing user input: {str(e)}")
            raise ReasoningError(f"Failed to process user input: {str(e)}")
    
    async def _build_response_prompt(self, 
                                   user_input: str,
                                   components: List[Dict[str, Any]],
                                   external_knowledge: List[Dict[str, Any]],
                                   user_profile: Dict[str, Any],
                                   session_context: Dict[str, Any],
                                   factual_query: Optional[Dict[str, Any]]) -> str:
        """Build the deliberative prompt with both internal and external knowledge"""
        
        try:
            # Load the deliberative prompt template
            template = self.jinja_env.get_template("deliberative_prompt.txt")
            
            # Prepare context data
            context_data = {
                'user_input': user_input,
                'components': components,
                'external_knowledge': external_knowledge,
                'user_profile': user_profile,
                'session_context': session_context,
                'factual_query': factual_query,
                'has_external_knowledge': len(external_knowledge) > 0,
                'has_therapeutic_components': len(components) > 0,
                'timestamp': datetime.now().isoformat()
            }
            
            # Render the prompt
            prompt = template.render(**context_data)
            
            logger.debug(f"Generated prompt with {len(components)} components and {len(external_knowledge)} knowledge entries")
            return prompt
            
        except Exception as e:
            logger.error(f"Error building response prompt: {str(e)}")
            raise ReasoningError(f"Failed to build prompt: {str(e)}")
    
    def _calculate_response_confidence(self, 
                                     components: List[Dict[str, Any]],
                                     external_knowledge: List[Dict[str, Any]],
                                     factual_query: Optional[Dict[str, Any]]) -> float:
        """Calculate confidence score for the response"""
        
        confidence = 0.5  # Base confidence
        
        # Boost confidence based on available components
        if components:
            confidence += min(len(components) * 0.1, 0.3)
        
        # Boost confidence if factual query was detected and answered
        if factual_query and external_knowledge:
            confidence += factual_query.get('confidence_score', 0) * 0.2
        
        # Boost confidence based on external knowledge quality
        if external_knowledge:
            avg_source_confidence = sum(
                entry.get('confidence', 0.5) for entry in external_knowledge
            ) / len(external_knowledge)
            confidence += avg_source_confidence * 0.2
        
        return min(confidence, 1.0)
    
    async def _log_interaction(self,

                             processing_time_ms: int,
                             factual_query: Optional[Dict[str, Any]]):
        """Log the interaction for analysis and improvement"""
        
        try:
            async with get_async_session() as session:
                # Get current user profile snapshot
                profile_query = select(UserProfile).where(UserProfile.user_id == user_id)
                profile_result = await session.execute(profile_query)
                user_profile = profile_result.scalar_one_or_none()
                
                personality_snapshot = None
                if user_profile:
                    personality_snapshot = {
                        'openness': user_profile.openness,
                        'conscientiousness': user_profile.conscientiousness,
                        'extraversion': user_profile.extraversion,
                        'agreeableness': user_profile.agreeableness,
                        'neuroticism': user_profile.neuroticism,
                        'attributes': user_profile.attributes,
                        'current_tier': user_profile.current_tier
                    }
                
                # Create interaction log entry
                interaction = InteractionLog(
                    session_id=session_id,
                    user_id=user_id,
                    user_input=user_input,
                    ai_response=ai_response,
                    components_used=[comp['component_id'] for comp in components_used],
                    personality_snapshot=personality_snapshot,
                    external_knowledge_used=[
                        {
                            'entity_name': entry['entity_name'],
                            'source': entry['source'],
                            'confidence': entry['confidence']
                        }
                        for entry in external_knowledge_used
                    ],
                    response_time_ms=processing_time_ms,
                    confidence_score=self._calculate_response_confidence(
                        components_used, external_knowledge_used, factual_query
                    ),
                    safety_flags=[]  # Will be populated by safety governor
                )
                
                session.add(interaction)
                await session.commit()
                
                logger.debug(f"Logged interaction for user {user_id}")
                
        except Exception as e:
            logger.error(f"Error logging interaction: {str(e)}")
            # Don't raise exception as this shouldn't break the main flow
    
    async def get_reasoning_analytics(self, 
                                    user_id: Optional[str] = None,
                                    start_date: Optional[datetime] = None,
                                    end_date: Optional[datetime] = None) -> Dict[str, Any]:
        """Get analytics about reasoning engine performance"""
        
        try:
            async with get_async_session() as session:
                # Base query
                query = select(InteractionLog)
                
                # Apply filters
                if user_id:
                    query = query.where(InteractionLog.user_id == user_id)
                if start_date:
                    query = query.where(InteractionLog.created_at >= start_date)
                if end_date:
                    query = query.where(InteractionLog.created_at <= end_date)
                
                # Get interactions
                result = await session.execute(query)
                interactions = result.scalars().all()
                
                if not interactions:
                    return {'message': 'No interactions found for the specified criteria'}
                
                # Calculate analytics
                total_interactions = len(interactions)
                avg_response_time = sum(i.response_time_ms or 0 for i in interactions) / total_interactions
                avg_confidence = sum(i.confidence_score or 0 for i in interactions) / total_interactions
                
                # Component usage statistics
                component_usage = {}
                external_knowledge_usage = 0
                
                for interaction in interactions:
                    # Count component usage
                    if interaction.components_used:
                        for comp_id in interaction.components_used:
                            component_usage[comp_id] = component_usage.get(comp_id, 0) + 1
                    
                    # Count external knowledge usage
                    if interaction.external_knowledge_used:
                        external_knowledge_usage += 1
                
                # Top components
                top_components = sorted(component_usage.items(), key=lambda x: x[1], reverse=True)[:10]
                
                return {
                    'total_interactions': total_interactions,
                    'average_response_time_ms': round(avg_response_time, 2),
                    'average_confidence_score': round(avg_confidence, 3),
                    'external_knowledge_usage_rate': round(external_knowledge_usage / total_interactions, 3),
                    'top_components': top_components,
                    'date_range': {
                        'start': start_date.isoformat() if start_date else None,
                        'end': end_date.isoformat() if end_date else None
                    }
                }
                
        except Exception as e:
            logger.error(f"Error getting reasoning analytics: {str(e)}")
            raise

class DeliberativePromptBuilder:
    """Helper class for building complex deliberative prompts"""
    
    def __init__(self, template_env: Environment):
        self.template_env = template_env
    
    def build_factual_context_section(self, external_knowledge: List[Dict[str, Any]]) -> str:
        """Build the factual context section of the prompt"""
        if not external_knowledge:
            return "No specific factual information was found in the knowledge base for this query."
        
        context_parts = []
        context_parts.append("The following factual information is available from verified sources:")
        
        for i, entry in enumerate(external_knowledge[:5], 1):  # Limit to top 5 results
            source_info = f"Source: {entry['source']}"
            if entry.get('source_link'):
                source_info += f" ({entry['source_link']})"
            
            fact_text = f"{i}. {entry['entity_name']} {entry['relationship_type'].replace('_', ' ').lower()} {entry['related_entity']}"
            if entry.get('confidence'):
                fact_text += f" (Confidence: {entry['confidence']:.2f})"
            
            context_parts.append(fact_text)
            context_parts.append(f"   {source_info}")
        
        return "\n".join(context_parts)
    
    def build_therapeutic_context_section(self, components: List[Dict[str, Any]]) -> str:
        """Build the therapeutic tools section of the prompt"""
        if not components:
            return "No specific therapeutic components were identified for this input."
        
        context_parts = []
        context_parts.append("Based on the user input, the following therapeutic skills and tools are available:")
        
        for i, comp in enumerate(components[:3], 1):  # Limit to top 3 components
            comp_text = f"{i}. **{comp['component_id']}** ({comp['pillar']} - {comp['theory']})"
            context_parts.append(comp_text)
            
            if comp.get('skill_definition'):
                context_parts.append(f"   Definition: {comp['skill_definition'].get('summary', 'Not available')}")
            
            if comp.get('execution_process'):
                context_parts.append(f"   Process: {comp['execution_process'].get('overview', 'Not available')}")
        
        return "\n".join(context_parts)
    
    def build_user_context_section(self, user_profile: Dict[str, Any], session_context: Dict[str, Any]) -> str:
        """Build the user context section of the prompt"""
        context_parts = []
        
        if user_profile:
            context_parts.append("User Profile Summary:")
            context_parts.append(f"- Current Tier: {user_profile.get('current_tier', 1)}")
            
            # Big Five summary
            personality = []
            traits = ['openness', 'conscientiousness', 'extraversion', 'agreeableness', 'neuroticism']
            for trait in traits:
                score = user_profile.get(trait, 0.5)
                level = "high" if score > 0.7 else "low" if score < 0.3 else "moderate"
                personality.append(f"{trait}: {level}")
            
            context_parts.append(f"- Personality: {', '.join(personality)}")
            
            # Additional attributes
            if user_profile.get('attributes'):
                attrs = user_profile['attributes']
                if attrs:
                    attr_summary = []
                    for key, value in attrs.items():
                        if isinstance(value, (int, float)):
                            attr_summary.append(f"{key}: {value}")
                    if attr_summary:
                        context_parts.append(f"- Assessment Scores: {', '.join(attr_summary)}")
        
        if session_context:
            session_summary = session_context.get('summary', 'New session')
            context_parts.append(f"Session Context: {session_summary}")
        
        return "\n".join(context_parts) if context_parts else "Limited user context available."

class BeliefGraphManager:
    """
    Core manager for user's symbolic belief graph (Self-Graph).
    Provides CRUD operations with user authority preservation.
    """
    
    def __init__(self, user_id: str):
        self.user_id = user_id
        self.graph = nx.MultiDiGraph()  # In-memory graph representation
        self._loaded = False
        
    async def load_user_graph(self) -> None:
        """Load user's belief graph from database into memory"""
        try:
            async with get_async_session() as session:
                # Load belief nodes
                nodes_query = select(BeliefNode).where(
                    BeliefNode.user_id == self.user_id
                ).options(
                    selectinload(BeliefNode.source_relationships),
                    selectinload(BeliefNode.target_relationships)
                )
                
                nodes_result = await session.execute(nodes_query)
                nodes = nodes_result.scalars().all()
                
                # Add nodes to graph
                for node in nodes:
                    self.graph.add_node(
                        str(node.node_id),
                        content=node.content,
                        node_type=node.node_type,
                        certainty_level=node.certainty_level,
                        emotional_valence=node.emotional_valence,
                        user_validated=node.user_validated,
                        ai_suggested=node.ai_suggested,
                        created_at=node.created_at,
                        updated_at=node.updated_at
                    )
                
                # Load relationships
                relationships_query = select(BeliefRelationship).join(
                    BeliefNode, BeliefRelationship.source_node_id == BeliefNode.node_id
                ).where(BeliefNode.user_id == self.user_id)
                
                rel_result = await session.execute(relationships_query)
                relationships = rel_result.scalars().all()
                
                # Add edges to graph
                for rel in relationships:
                    self.graph.add_edge(
                        str(rel.source_node_id),
                        str(rel.target_node_id),
                        relationship_id=str(rel.relationship_id),
                        relationship_type=rel.relationship_type,
                        strength=rel.strength,
                        user_validated=rel.user_validated,
                        ai_inferred=rel.ai_inferred,
                        created_at=rel.created_at
                    )
                
                self._loaded = True
                logger.info(f"Loaded belief graph for user {self.user_id}: {len(self.graph.nodes)} nodes, {len(self.graph.edges)} edges")
                
        except Exception as e:
            logger.error(f"Error loading belief graph for user {self.user_id}: {str(e)}")
            raise BeliefGraphError(f"Failed to load belief graph: {str(e)}")
    
    async def add_belief_node(self, 
                            content: str,
                            node_type: str,
                            certainty_level: float = 0.5,
                            emotional_valence: float = 0.0,
                            ai_suggested: bool = False) -> str:
        """Add a new belief node to the graph"""
        try:
            if not self._loaded:
                await self.load_user_graph()
            
            node_id = str(uuid4())
            
            # Validate inputs
            if not content.strip():
                raise ValidationError("Belief content cannot be empty")
            
            if not (0 <= certainty_level <= 1):
                raise ValidationError("Certainty level must be between 0 and 1")
            
            if not (-1 <= emotional_valence <= 1):
                raise ValidationError("Emotional valence must be between -1 and 1")
            
            # Add to in-memory graph
            self.graph.add_node(
                node_id,
                content=content,
                node_type=node_type,
                certainty_level=certainty_level,
                emotional_valence=emotional_valence,
                user_validated=not ai_suggested,
                ai_suggested=ai_suggested,
                created_at=datetime.now(),
                updated_at=datetime.now()
            )
            
            # Persist to database
            async with get_async_session() as session:
                db_node = BeliefNode(
                    node_id=node_id,
                    user_id=self.user_id,
                    content=content,
                    node_type=node_type,
                    certainty_level=certainty_level,
                    emotional_valence=emotional_valence,
                    user_validated=not ai_suggested,
                    ai_suggested=ai_suggested
                )
                
                session.add(db_node)
                await session.commit()
            
            logger.info(f"Added belief node {node_id} for user {self.user_id}")
            return node_id
            
        except Exception as e:
            logger.error(f"Error adding belief node: {str(e)}")
            raise BeliefGraphError(f"Failed to add belief node: {str(e)}")
    
    async def add_belief_relationship(self,
                                    source_node_id: str,
                                    target_node_id: str,
                                    relationship_type: str,
                                    strength: float = 0.5,
                                    ai_inferred: bool = False) -> str:
        """Add a relationship between two belief nodes"""
        try:
            if not self._loaded:
                await self.load_user_graph()
            
            # Validate nodes exist
            if source_node_id not in self.graph:
                raise ValidationError(f"Source node {source_node_id} does not exist")
            if target_node_id not in self.graph:
                raise ValidationError(f"Target node {target_node_id} does not exist")
            
            # Prevent self-references
            if source_node_id == target_node_id:
                raise ValidationError("Cannot create relationship from node to itself")
            
            # Validate strength
            if not (0 <= strength <= 1):
                raise ValidationError("Relationship strength must be between 0 and 1")
            
            relationship_id = str(uuid4())
            
            # Add to in-memory graph
            self.graph.add_edge(
                source_node_id,
                target_node_id,
                relationship_id=relationship_id,
                relationship_type=relationship_type,
                strength=strength,
                user_validated=not ai_inferred,
                ai_inferred=ai_inferred,
                created_at=datetime.now()
            )
            
            # Persist to database
            async with get_async_session() as session:
                db_relationship = BeliefRelationship(
                    relationship_id=relationship_id,
                    source_node_id=source_node_id,
                    target_node_id=target_node_id,
                    relationship_type=relationship_type,
                    strength=strength,
                    user_validated=not ai_inferred,
                    ai_inferred=ai_inferred
                )
                
                session.add(db_relationship)
                await session.commit()
            
            logger.info(f"Added relationship {relationship_id} between {source_node_id} and {target_node_id}")
            return relationship_id
            
        except Exception as e:
            logger.error(f"Error adding relationship: {str(e)}")
            raise BeliefGraphError(f"Failed to add relationship: {str(e)}")
    
    async def update_node_validation(self, node_id: str, user_validated: bool) -> None:
        """Update user validation status of a belief node"""
        try:
            if not self._loaded:
                await self.load_user_graph()
            
            if node_id not in self.graph:
                raise ValidationError(f"Node {node_id} does not exist")
            
            # Update in-memory graph
            self.graph.nodes[node_id]['user_validated'] = user_validated
            self.graph.nodes[node_id]['updated_at'] = datetime.now()
            
            # Update database
            async with get_async_session() as session:
                update_query = select(BeliefNode).where(
                    and_(
                        BeliefNode.node_id == node_id,
                        BeliefNode.user_id == self.user_id
                    )
                )
                result = await session.execute(update_query)
                node = result.scalar_one_or_none()
                
                if node:
                    node.user_validated = user_validated
                    node.updated_at = datetime.now()
                    await session.commit()
                else:
                    raise ValidationError(f"Node {node_id} not found in database")
            
            logger.info(f"Updated validation for node {node_id} to {user_validated}")
            
        except Exception as e:
            logger.error(f"Error updating node validation: {str(e)}")
            raise BeliefGraphError(f"Failed to update node validation: {str(e)}")
    
    async def remove_node(self, node_id: str) -> None:
        """Remove a belief node and all its relationships"""
        try:
            if not self._loaded:
                await self.load_user_graph()
            
            if node_id not in self.graph:
                raise ValidationError(f"Node {node_id} does not exist")
            
            # Remove from in-memory graph
            self.graph.remove_node(node_id)
            
            # Remove from database (cascades to relationships)
            async with get_async_session() as session:
                # Delete relationships first
                rel_delete_query = delete(BeliefRelationship).where(
                    or_(
                        BeliefRelationship.source_node_id == node_id,
                        BeliefRelationship.target_node_id == node_id
                    )
                )
                await session.execute(rel_delete_query)
                
                # Delete node
                node_delete_query = delete(BeliefNode).where(
                    and_(
                        BeliefNode.node_id == node_id,
                        BeliefNode.user_id == self.user_id
                    )
                )
                await session.execute(node_delete_query)
                await session.commit()
            
            logger.info(f"Removed node {node_id} and its relationships")
            
        except Exception as e:
            logger.error(f"Error removing node: {str(e)}")
            raise BeliefGraphError(f"Failed to remove node: {str(e)}")
    
    def get_node_neighbors(self, node_id: str) -> Dict[str, List[str]]:
        """Get neighboring nodes of a given node"""
        if node_id not in self.graph:
            return {'predecessors': [], 'successors': []}
        
        return {
            'predecessors': list(self.graph.predecessors(node_id)),
            'successors': list(self.graph.successors(node_id))
        }
    
    def get_graph_summary(self) -> Dict[str, Any]:
        """Get summary statistics of the belief graph"""
        if not self._loaded:
            return {'error': 'Graph not loaded'}
        
        node_types = {}
        validated_nodes = 0
        ai_suggested_nodes = 0
        
        for node_id, data in self.graph.nodes(data=True):
            node_type = data.get('node_type', 'unknown')
            node_types[node_type] = node_types.get(node_type, 0) + 1
            
            if data.get('user_validated', False):
                validated_nodes += 1
            if data.get('ai_suggested', False):
                ai_suggested_nodes += 1
        
        relationship_types = {}
        for u, v, data in self.graph.edges(data=True):
            rel_type = data.get('relationship_type', 'unknown')
            relationship_types[rel_type] = relationship_types.get(rel_type, 0) + 1
        
        return {
            'total_nodes': len(self.graph.nodes),
            'total_relationships': len(self.graph.edges),
            'node_types': node_types,
            'relationship_types': relationship_types,
            'validated_nodes': validated_nodes,
            'ai_suggested_nodes': ai_suggested_nodes,
            'graph_density': nx.density(self.graph),
            'connected_components': nx.number_weakly_connected_components(self.graph)
        }

class OntologicalSandbox:
    """
    Implements the Just-in-Time Ontological Stabilizer
    Allows safe exploration of belief modifications
    """
    
    def __init__(self, belief_manager: BeliefGraphManager):
        self.belief_manager = belief_manager
        self.sandbox_graph = None
        self.proposed_changes = []
    
    async def create_sandbox(self) -> None:
        """Create a sandbox copy of the current belief graph"""
        try:
            if not self.belief_manager._loaded:
                await self.belief_manager.load_user_graph()
            
            # Create deep copy of the graph
            self.sandbox_graph = self.belief_manager.graph.copy()
            self.proposed_changes = []
            
            logger.info(f"Created belief sandbox for user {self.belief_manager.user_id}")
            
        except Exception as e:
            logger.error(f"Error creating sandbox: {str(e)}")
            raise BeliefGraphError(f"Failed to create sandbox: {str(e)}")
    
    def apply_change_to_sandbox(self, change: Dict[str, Any]) -> None:
        """Apply a proposed change to the sandbox graph"""
        try:
            change_type = change.get('type')
            
            if change_type == 'add_node':
                node_id = change['node_id']
                self.sandbox_graph.add_node(node_id, **change['attributes'])
                
            elif change_type == 'modify_node':
                node_id = change['node_id']
                if node_id in self.sandbox_graph:
                    self.sandbox_graph.nodes[node_id].update(change['attributes'])
                
            elif change_type == 'remove_node':
                node_id = change['node_id']
                if node_id in self.sandbox_graph:
                    self.sandbox_graph.remove_node(node_id)
                    
            elif change_type == 'add_relationship':
                self.sandbox_graph.add_edge(
                    change['source_node_id'],
                    change['target_node_id'],
                    **change['attributes']
                )
                
            elif change_type == 'remove_relationship':
                if self.sandbox_graph.has_edge(change['source_node_id'], change['target_node_id']):
                    self.sandbox_graph.remove_edge(
                        change['source_node_id'],
                        change['target_node_id']
                    )
            
            self.proposed_changes.append(change)
            
        except Exception as e:
            logger.error(f"Error applying sandbox change: {str(e)}")
            raise BeliefGraphError(f"Failed to apply sandbox change: {str(e)}")
    
    async def analyze_sandbox_implications(self) -> Dict[str, Any]:
        """Analyze the implications of proposed changes"""
        try:
            if not self.sandbox_graph:
                raise BeliefGraphError("Sandbox not created")
            
            # Detect dissonances in the sandbox
            dissonance_detector = DissonanceDetectionEngine(self)
            new_dissonances = await dissonance_detector.detect_all_patterns()
            
            # Compare with original graph dissonances
            original_detector = DissonanceDetectionEngine(self.belief_manager)
            original_dissonances = await original_detector.detect_all_patterns()
            
            # Calculate impact metrics
            impact_analysis = {
                'new_dissonances': len(new_dissonances) - len(original_dissonances),
                'resolved_dissonances': [],
                'created_dissonances': [],
                'graph_metrics_change': self._calculate_metrics_change(),
                'stability_score': self._calculate_stability_score(new_dissonances)
            }
            
            # Identify specific changes in dissonance patterns
            original_patterns = {d.pattern_type + str(sorted(d.involved_nodes)) for d in original_dissonances}
            new_patterns = {d.pattern_type + str(sorted(d.involved_nodes)) for d in new_dissonances}
            
            impact_analysis['resolved_dissonances'] = list(original_patterns - new_patterns)
            impact_analysis['created_dissonances'] = list(new_patterns - original_patterns)
            
            return impact_analysis
            
        except Exception as e:
            logger.error(f"Error analyzing sandbox implications: {str(e)}")
            raise BeliefGraphError(f"Failed to analyze implications: {str(e)}")
    
    def _calculate_metrics_change(self) -> Dict[str, float]:
        """Calculate how graph metrics change with proposed modifications"""
        original_metrics = {
            'density': nx.density(self.belief_manager.graph),
            'nodes': len(self.belief_manager.graph.nodes),
            'edges': len(self.belief_manager.graph.edges)
        }
        
        sandbox_metrics = {
            'density': nx.density(self.sandbox_graph),
            'nodes': len(self.sandbox_graph.nodes),
            'edges': len(self.sandbox_graph.edges)
        }
        
        return {
            'density_change': sandbox_metrics['density'] - original_metrics['density'],
            'nodes_change': sandbox_metrics['nodes'] - original_metrics['nodes'],
            'edges_change': sandbox_metrics['edges'] - original_metrics['edges']
        }
    
    def _calculate_stability_score(self, dissonances: List[DissonancePattern]) -> float:
        """Calculate overall stability score for the sandbox graph"""
        if not dissonances:
            return 1.0
        
        # Weight dissonances by strength and type
        total_dissonance = 0
        weights = {
            'logical_contradiction': 1.0,
            'value_behavior_mismatch': 0.8,
            'identity_evidence_discord': 0.7,
            'temporal_inconsistency': 0.6
        }
        
        for dissonance in dissonances:
            weight = weights.get(dissonance.pattern_type, 0.5)
            total_dissonance += dissonance.strength * weight
        
        # Normalize by graph size
        graph_size = max(len(self.sandbox_graph.nodes), 1)
        normalized_dissonance = total_dissonance / graph_size
        
        # Convert to stability score (inverse of dissonance)
        stability_score = max(0, 1 - normalized_dissonance)
        return stability_score
    
    async def generate_user_summary(self) -> str:
        """Generate a user-friendly summary of proposed changes"""
        try:
            if not self.proposed_changes:
                return "No changes proposed."
            
            implications = await self.analyze_sandbox_implications()
            
            summary_parts = []
            
            # Summarize proposed changes
            summary_parts.append("**Proposed Changes:**")
            for change in self.proposed_changes[:3]:  # Limit to first 3 changes
                if change['type'] == 'add_node':
                    summary_parts.append(f"- Add belief: \"{change['attributes']['content']}\"")
                elif change['type'] == 'modify_node':
                    summary_parts.append(f"- Modify existing belief")
                elif change['type'] == 'add_relationship':
                    summary_parts.append(f"- Connect beliefs with relationship: {change['attributes']['relationship_type']}")
            
            # Summarize implications
            summary_parts.append("\n**Potential Impact:**")
            
            if implications['new_dissonances'] > 0:
                summary_parts.append(f"- May create {implications['new_dissonances']} new internal conflicts")
            elif implications['new_dissonances'] < 0:
                summary_parts.append(f"- May resolve {abs(implications['new_dissonances'])} existing conflicts")
            else:
                summary_parts.append("- Should not create new internal conflicts")
            
            stability_score = implications['stability_score']
            if stability_score > 0.8:
                summary_parts.append("- Overall impact appears positive and stabilizing")
            elif stability_score > 0.6:
                summary_parts.append("- Impact appears manageable with some adjustment needed")
            else:
                summary_parts.append("- May require careful consideration and gradual implementation")
            
            return "\n".join(summary_parts)
            
        except Exception as e:
            logger.error(f"Error generating user summary: {str(e)}")
            return "Unable to analyze the impact of proposed changes."

class DissonanceDetectionEngine:
    """
    Advanced engine for detecting cognitive dissonance patterns in belief graphs
    """
    
    def __init__(self, graph_source):
        """Initialize with either BeliefGraphManager or OntologicalSandbox"""
        if isinstance(graph_source, BeliefGraphManager):
            self.graph = graph_source.graph
            self.user_id = graph_source.user_id
        elif isinstance(graph_source, OntologicalSandbox):
            self.graph = graph_source.sandbox_graph
            self.user_id = graph_source.belief_manager.user_id
        else:
            raise ValueError("Invalid graph source type")
    
    async def detect_all_patterns(self) -> List[DissonancePattern]:
        """Detect all types of dissonance patterns"""
        try:
            all_patterns = []
            
            # Detect different types of dissonance
            all_patterns.extend(await self._detect_logical_contradictions())
            all_patterns.extend(await self._detect_value_behavior_mismatches())
            all_patterns.extend(await self._detect_identity_evidence_discord())
            all_patterns.extend(await self._detect_temporal_inconsistencies())
            
            # Sort by strength (highest first)
            all_patterns.sort(key=lambda x: x.strength, reverse=True)
            
            logger.info(f"Detected {len(all_patterns)} dissonance patterns for user {self.user_id}")
            return all_patterns
            
        except Exception as e:
            logger.error(f"Error detecting dissonance patterns: {str(e)}")
            return []
    
    async def _detect_logical_contradictions(self) -> List[DissonancePattern]:
        """Detect direct logical contradictions in beliefs"""
        patterns = []
        
        try:
            # Find nodes connected by CONTRADICTS relationships
            for u, v, data in self.graph.edges(data=True):
                if data.get('relationship_type') == RelationshipType.CONTRADICTS.value:
                    strength = data.get('strength', 0.5)
                    
                    node_u_data = self.graph.nodes[u]
                    node_v_data = self.graph.nodes[v]
                    
                    pattern = DissonancePattern(
                        pattern_type='logical_contradiction',
                        involved_nodes=[u, v],
                        strength=strength,
                        description=f"Direct contradiction between '{node_u_data['content']}' and '{node_v_data['content']}'",
                        resolution_suggestions=[
                            "Consider which belief is more strongly supported by evidence",
                            "Look for ways to reconcile or find common ground",
                            "Examine the context where each belief applies"
                        ]
                    )
                    patterns.append(pattern)
            
            # Detect indirect contradictions through inference chains
            patterns.extend(await self._detect_indirect_contradictions())
            
        except Exception as e:
            logger.error(f"Error detecting logical contradictions: {str(e)}")
        
        return patterns
    
    async def _detect_indirect_contradictions(self) -> List[DissonancePattern]:
        """Detect contradictions through chains of reasoning"""
        patterns = []
        
        try:
            # Find chains like: A CAUSES B, B CONTRADICTS C, C SUPPORTS A
            # This creates an indirect logical inconsistency
            
            for node in self.graph.nodes():
                # Get all paths of length 3 starting from this node
                try:
                    paths = list(nx.all_simple_paths(self.graph, node, node, cutoff=3))
                    
                    for path in paths:
                        if len(path) == 4:  # Full cycle
                            # Analyze the relationship types in the cycle
                            cycle_relationships = []
                            for i in range(len(path) - 1):
                                edge_data = self.graph.get_edge_data(path[i], path[i + 1])
                                if edge_data:
                                    cycle_relationships.append(edge_data[0].get('relationship_type', ''))
                            
                            # Check for contradictory cycles
                            if self._is_contradictory_cycle(cycle_relationships):
                                avg_strength = 0.6  # Default for indirect contradictions
                                
                                pattern = DissonancePattern(
                                    pattern_type='logical_contradiction',
                                    involved_nodes=path[:-1],  # Remove duplicate last node
                                    strength=avg_strength,
                                    description=f"Indirect contradiction through belief chain involving {len(path)-1} beliefs",
                                    resolution_suggestions=[
                                        "Examine the logical chain for weak links",
                                        "Consider whether some relationships are contextual",
                                        "Look for missing qualifications or conditions"
                                    ]
                                )
                                patterns.append(pattern)
                
                except nx.NetworkXNoPath:
                    continue  # No cycles from this node
                    
        except Exception as e:
            logger.error(f"Error detecting indirect contradictions: {str(e)}")
        
        return patterns
    
    def _is_contradictory_cycle(self, relationships: List[str]) -> bool:
        """Check if a cycle of relationships creates a logical contradiction"""
        # Simplified logic - could be made more sophisticated
        contradiction_indicators = [
            RelationshipType.CONTRADICTS.value,
            RelationshipType.CONFLICTS_WITH.value
        ]
        
        support_indicators = [
            RelationshipType.SUPPORTS.value,
            RelationshipType.CAUSES.value,
            RelationshipType.DERIVED_FROM.value
        ]
        
        has_contradiction = any(rel in contradiction_indicators for rel in relationships)
        has_support = any(rel in support_indicators for rel in relationships)
        
        # A cycle with both support and contradiction relationships is problematic
        return has_contradiction and has_support
    
    async def _detect_value_behavior_mismatches(self) -> List[DissonancePattern]:
        """Detect mismatches between stated values and reported behaviors"""
        patterns = []
        
        try:
            # Find value nodes
            value_nodes = [
                node_id for node_id, data in self.graph.nodes(data=True)
                if data.get('node_type') == BeliefNodeType.VALUE_SYSTEM.value
            ]
            
            # Find behavior nodes
            behavior_nodes = [
                node_id for node_id, data in self.graph.nodes(data=True)
                if data.get('node_type') == BeliefNodeType.BEHAVIORAL_PATTERN.value
            ]
            
            # Check for conflicts between values and behaviors
            for value_node in value_nodes:
                for behavior_node in behavior_nodes:
                    # Check if there's a CONFLICTS_WITH relationship
                    if self.graph.has_edge(value_node, behavior_node):
                        edge_data = self.graph.get_edge_data(value_node, behavior_node)[0]
                        if edge_data.get('relationship_type') == RelationshipType.CONFLICTS_WITH.value:
                            
                            strength = edge_data.get('strength', 0.5)
                            value_content = self.graph.nodes[value_node]['content']
                            behavior_content = self.graph.nodes[behavior_node]['content']
                            
                            pattern = DissonancePattern(
                                pattern_type='value_behavior_mismatch',
                                involved_nodes=[value_node, behavior_node],
                                strength=strength,
                                description=f"Mismatch between value '{value_content}' and behavior '{behavior_content}'",
                                resolution_suggestions=[
                                    "Consider why this behavior conflicts with your values",
                                    "Explore barriers to value-aligned behavior",
                                    "Look for small steps toward better alignment",
                                    "Consider if the value needs refinement or the behavior needs change"
                                ]
                            )
                            patterns.append(pattern)
            
        except Exception as e:
            logger.error(f"Error detecting value-behavior mismatches: {str(e)}")
        
        return patterns
    
    async def _detect_identity_evidence_discord(self) -> List[DissonancePattern]:
        """Detect mismatches between identity beliefs and supporting evidence"""
        patterns = []
        
        try:
            # Find identity component nodes
            identity_nodes = [
                node_id for node_id, data in self.graph.nodes(data=True)
                if data.get('node_type') == BeliefNodeType.IDENTITY_COMPONENT.value
            ]
            
            for identity_node in identity_nodes:
                # Find evidence nodes connected to this identity component
                evidence_nodes = []
                
                # Look for nodes that provide evidence (SUPPORTS relationship)
                for predecessor in self.graph.predecessors(identity_node):
                    edge_data = self.graph.get_edge_data(predecessor, identity_node)[0]
                    if edge_data.get('relationship_type') == RelationshipType.SUPPORTS.value:
                        evidence_nodes.append(predecessor)
                
                # Look for contradictory evidence
                contradictory_evidence = []
                for predecessor in self.graph.predecessors(identity_node):
                    edge_data = self.graph.get_edge_data(predecessor, identity_node)[0]
                    if edge_data.get('relationship_type') == RelationshipType.CONTRADICTS.value:
                        contradictory_evidence.append(predecessor)
                
                # If there's contradictory evidence, create a dissonance pattern
                if contradictory_evidence:
                    avg_strength = sum(
                        self.graph.get_edge_data(node, identity_node)[0].get('strength', 0.5)
                        for node in contradictory_evidence
                    ) / len(contradictory_evidence)
                    
                    identity_content = self.graph.nodes[identity_node]['content']
                    
                    pattern = DissonancePattern(
                        pattern_type='identity_evidence_discord',
                        involved_nodes=[identity_node] + contradictory_evidence,
                        strength=avg_strength,
                        description=f"Identity belief '{identity_content}' conflicts with personal evidence",
                        resolution_suggestions=[
                            "Examine whether the evidence is being interpreted fairly",
                            "Consider if the identity belief is overly rigid",
                            "Look for additional evidence that supports your identity",
                            "Consider how this identity serves you"
                        ]
                    )
                    patterns.append(pattern)
            
        except Exception as e:
            logger.error(f"Error detecting identity-evidence discord: {str(e)}")
        
        return patterns
    
    async def _detect_temporal_inconsistencies(self) -> List[DissonancePattern]:
        """Detect beliefs that contradict across different time contexts"""
        patterns = []
        
        try:
            # This is a simplified version - could be enhanced with temporal reasoning
            # Look for beliefs that explicitly mention time and might conflict
            
            temporal_keywords = ['always', 'never', 'sometimes', 'used to', 'will', 'past', 'future']
            
            temporal_nodes = []
            for node_id, data in self.graph.nodes(data=True):
                content = data.get('content', '').lower()
                if any(keyword in content for keyword in temporal_keywords):
                    temporal_nodes.append(node_id)
            
            # Check for contradictions between temporal beliefs
            for i, node1 in enumerate(temporal_nodes):
                for node2 in temporal_nodes[i+1:]:
                    if self.graph.has_edge(node1, node2):
                        edge_data = self.graph.get_edge_data(node1, node2)[0]
                        if edge_data.get('relationship_type') == RelationshipType.CONTRADICTS.value:
                            
                            content1 = self.graph.nodes[node1]['content']
                            content2 = self.graph.nodes[node2]['content']
                            strength = edge_data.get('strength', 0.5)
                            
                            pattern = DissonancePattern(
                                pattern_type='temporal_inconsistency',
                                involved_nodes=[node1, node2],
                                strength=strength,
                                description=f"Temporal inconsistency between '{content1}' and '{content2}'",
                                resolution_suggestions=[
                                    "Consider how your beliefs have evolved over time",
                                    "Examine whether both beliefs can be true in different contexts",
                                    "Look for growth and change as natural processes"
                                ]
                            )
                            patterns.append(pattern)
            
        except Exception as e:
            logger.error(f"Error detecting temporal inconsistencies: {str(e)}")
        
        return patterns

class BeliefGraphService:
    """
    High-level service for managing belief graphs with AI integration
    """
    
    def __init__(self, model_manager: ModelManager):
        self.model_manager = model_manager
        self._user_managers = {}  # Cache of user managers
    
    async def get_belief_manager(self, user_id: str) -> BeliefGraphManager:
        """Get or create a belief graph manager for a user"""
        if user_id not in self._user_managers:
            manager = BeliefGraphManager(user_id)
            await manager.load_user_graph()
            self._user_managers[user_id] = manager
        
        return self._user_managers[user_id]
    
    async def suggest_belief_from_statement(self, user_id: str, statement: str) -> Optional[Dict[str, Any]]:
        """Analyze a user statement and suggest a belief node"""
        try:
            # Use AI to analyze the statement for belief content
            analysis_prompt = f"""
            Analyze this statement for implicit beliefs, values, or identity components:
            
            Statement: "{statement}"
            
            If this statement contains an implicit belief, value, or identity component, extract it and classify it.
            
            Respond with JSON in this format:
            {{
                "has_belief": true/false,
                "belief_content": "extracted belief statement",
                "belief_type": "core_belief" | "value_system" | "identity_component" | "behavioral_pattern" | "thought_pattern",
                "certainty_level": 0.0-1.0,
                "emotional_valence": -1.0 to 1.0,
                "explanation": "why this is considered a belief"
            }}
            """
            
            response = await self.model_manager.generate_response(
                prompt=analysis_prompt,
                user_id=user_id,
                max_tokens=300
            )
            
            # Parse AI response
            try:
                belief_analysis = json.loads(response)
                if belief_analysis.get('has_belief', False):
                    return belief_analysis
            except json.JSONDecodeError:
                logger.warning(f"Failed to parse belief analysis response: {response}")
            
            return None
            
        except Exception as e:
            logger.error(f"Error suggesting belief from statement: {str(e)}")
            return None
    
    async def interactive_belief_exploration(self, user_id: str, proposed_change: Dict[str, Any]) -> Dict[str, Any]:
        """
        Interactive exploration of belief changes with user feedback
        This implements the "Just-in-Time Ontological Stabilizer"
        """
        try:
            # Get user's belief manager
            belief_manager = await self.get_belief_manager(user_id)
            
            # Create sandbox for exploration
            sandbox = OntologicalSandbox(belief_manager)
            await sandbox.create_sandbox()
            
            # Apply proposed change to sandbox
            sandbox.apply_change_to_sandbox(proposed_change)
            
            # Analyze implications
            implications = await sandbox.analyze_sandbox_implications()
            
            # Generate user-friendly summary
            summary = await sandbox.generate_user_summary()
            
            return {
                'sandbox_created': True,
                'change_summary': summary,
                'impact_analysis': implications,
                'stability_score': implications['stability_score'],
                'recommendation': self._generate_recommendation(implications)
            }
            
        except Exception as e:
            logger.error(f"Error in interactive belief exploration: {str(e)}")
            raise BeliefGraphError(f"Failed to explore belief change: {str(e)}")
    
    def _generate_recommendation(self, implications: Dict[str, Any]) -> str:
        """Generate a recommendation based on change implications"""
        stability_score = implications['stability_score']
        new_dissonances = implications['new_dissonances']
        
        if stability_score > 0.8 and new_dissonances <= 0:
            return "This change appears beneficial and is recommended."
        elif stability_score > 0.6:
            return "This change has mixed implications. Consider implementing gradually."
        elif new_dissonances > 2:
            return "This change may create significant internal conflicts. Consider exploring alternatives."
        else:
            return "This change requires careful consideration. Would you like to explore modifications?"
    
    async def get_user_dissonance_report(self, user_id: str) -> Dict[str, Any]:
        """Generate a comprehensive dissonance report for a user"""
        try:
            belief_manager = await self.get_belief_manager(user_id)
            dissonance_detector = DissonanceDetectionEngine(belief_manager)
            
            patterns = await dissonance_detector.detect_all_patterns()
            
            # Categorize patterns
            pattern_categories = {}
            for pattern in patterns:
                category = pattern.pattern_type
                if category not in pattern_categories:
                    pattern_categories[category] = []
                pattern_categories[category].append(pattern)
            
            # Generate summary
            graph_summary = belief_manager.get_graph_summary()
            
            return {
                'total_dissonances': len(patterns),
                'pattern_categories': {
                    cat: len(patterns) for cat, patterns in pattern_categories.items()
                },
                'high_priority_patterns': [
                    {
                        'type': p.pattern_type,
                        'strength': p.strength,
                        'description': p.description,
                        'suggestions': p.resolution_suggestions
                    }
                    for p in patterns[:3]  # Top 3 highest strength
                ],
                'graph_summary': graph_summary,
                'overall_stability': self._calculate_overall_stability(patterns, graph_summary)
            }
            
        except Exception as e:
            logger.error(f"Error generating dissonance report: {str(e)}")
            raise BeliefGraphError(f"Failed to generate dissonance report: {str(e)}")
    
    def _calculate_overall_stability(self, patterns: List[DissonancePattern], graph_summary: Dict[str, Any]) -> float:
        """Calculate overall belief system stability"""
        if not patterns:
            return 1.0
        
        total_nodes = graph_summary.get('total_nodes', 1)
        total_dissonance = sum(p.strength for p in patterns)
        
        # Normalize by graph size
        normalized_dissonance = total_dissonance / total_nodes
        
        # Convert to stability score
        stability = max(0, 1 - normalized_dissonance)
        return round(stability, 3)

class DialogueTurn:
    """Represents a single turn in a therapeutic dialogue"""
    
    def __init__(self, speaker: str, content: str, turn_number: int, metadata: Optional[Dict[str, Any]] = None):
        self.speaker = speaker  # 'client' or 'therapist'
        self.content = content
        self.turn_number = turn_number
        self.metadata = metadata or {}
        self.timestamp = datetime.now()

class GoldenDialogue:
    """Represents a complete expert therapeutic dialogue"""
    
    def __init__(self, dialogue_id: str, turns: List[DialogueTurn], metadata: Optional[Dict[str, Any]] = None):
        self.dialogue_id = dialogue_id
        self.turns = turns
        self.metadata = metadata or {}
        self.created_at = datetime.now()
    
    def get_client_turns(self) -> List[DialogueTurn]:
        """Get all client turns from the dialogue"""
        return [turn for turn in self.turns if turn.speaker == 'client']
    
    def get_therapist_turns(self) -> List[DialogueTurn]:
        """Get all therapist turns from the dialogue"""
        return [turn for turn in self.turns if turn.speaker == 'therapist']
    
    def get_turn_pairs(self) -> List[Tuple[DialogueTurn, DialogueTurn]]:
        """Get client-therapist turn pairs for simulation"""
        pairs = []
        client_turns = self.get_client_turns()
        therapist_turns = self.get_therapist_turns()
        
        # Pair consecutive turns
        for i in range(min(len(client_turns), len(therapist_turns))):
            pairs.append((client_turns[i], therapist_turns[i]))
        
        return pairs

        self.client = httpx.AsyncClient(timeout=30.0)
        
        # Performance tracking
        self.validation_stats = {
            'total_validations': 0,
            'critiques_generated': 0,
            'training_examples_created': 0
        }
    
    async def validate_against_golden_dialogue(self, 
                                             golden_dialogue: GoldenDialogue,
                                             user_id: str) -> Dict[str, Any]:
        """Validate AI performance against a complete golden dialogue"""
        try:
            logger.info(f"Starting validation against dialogue {golden_dialogue.dialogue_id}")
            
            # Simulate the conversation
            simulation_results = await self.simulate_conversation(golden_dialogue, user_id)
            
            # Generate critiques for each turn
            critiques = []
            training_examples = []
            
            for turn_result in simulation_results['turns']:
                if turn_result['ai_response'] and turn_result['expert_response']:
                    critique = await self.critique_response(
                        client_input=turn_result['client_input'],
                        ai_response=turn_result['ai_response'],
                        expert_response=turn_result['expert_response']
                    )
                    
                    if critique:
                        critiques.append(critique)
                        
                        # Create training example if critique shows significant improvement needed
                        if critique.improvement_score > 0.3:  # Threshold for creating training examples
                            training_example = await self.create_training_example(critique)
                            if training_example:
                                training_examples.append(training_example)
            
            # Store training examples
            if training_examples:
                await self.store_training_examples(training_examples)
            
            # Calculate overall performance metrics
            performance_metrics = self.calculate_performance_metrics(critiques, simulation_results)
            
            self.validation_stats['total_validations'] += 1
            self.validation_stats['critiques_generated'] += len(critiques)
            self.validation_stats['training_examples_created'] += len(training_examples)
            
            logger.info(f"Validation complete: {len(critiques)} critiques, {len(training_examples)} training examples")
            
            return {
                'dialogue_id': golden_dialogue.dialogue_id,
                'simulation_results': simulation_results,
                'critiques': [self._serialize_critique(c) for c in critiques],
                'training_examples_created': len(training_examples),
                'performance_metrics': performance_metrics,
                'validation_timestamp': datetime.now().isoformat()
            }
            
        except Exception as e:
            logger.error(f"Error validating against golden dialogue: {str(e)}")
            raise GoldenStandardError(f"Validation failed: {str(e)}")
    
    async def simulate_conversation(self, 
                                  golden_dialogue: GoldenDialogue,
                                  user_id: str) -> Dict[str, Any]:
        """Simulate a conversation with HelloKoa API using golden dialogue"""
        try:
            turn_results = []
            session_id = f"golden_sim_{golden_dialogue.dialogue_id}_{datetime.now().timestamp()}"
            
            # Get turn pairs (client input -> therapist response)
            turn_pairs = golden_dialogue.get_turn_pairs()
            
            for i, (client_turn, expert_turn) in enumerate(turn_pairs):
                try:
                    # Get AI response using reasoning engine
                    ai_result = await self.reasoning_engine.process_user_input(
                        user_input=client_turn.content,
                        user_id=user_id,
                        session_id=session_id
                    )
                    
                    ai_response = ai_result.get('response_text', '')
                    
                    turn_result = {
                        'turn_number': i + 1,
                        'client_input': client_turn.content,
                        'ai_response': ai_response,
                        'expert_response': expert_turn.content,
                        'ai_processing_time_ms': ai_result.get('processing_time_ms', 0),
                        'ai_confidence': ai_result.get('confidence_score', 0),
                        'components_used': ai_result.get('components_used', [])
                    }
                    
                    turn_results.append(turn_result)
                    
                    # Add small delay between turns to simulate natural conversation
                    await asyncio.sleep(0.1)
                    
                except Exception as e:
                    logger.warning(f"Error simulating turn {i+1}: {str(e)}")
                    turn_result = {
                        'turn_number': i + 1,
                        'client_input': client_turn.content,
                        'ai_response': None,
                        'expert_response': expert_turn.content,
                        'error': str(e)
                    }
                    turn_results.append(turn_result)
            
            return {
                'session_id': session_id,
                'total_turns': len(turn_pairs),
                'successful_turns': len([r for r in turn_results if r.get('ai_response')]),
                'turns': turn_results,
                'simulation_duration': len(turn_pairs) * 0.1  # Approximate duration
            }
            
        except Exception as e:
            logger.error(f"Error simulating conversation: {str(e)}")
            raise GoldenStandardError(f"Conversation simulation failed: {str(e)}")
    
    async def critique_response(self, 
                              client_input: str,
                              ai_response: str,
                              expert_response: str) -> Optional[PerformanceCritique]:
        """Use LLM to critique AI response against expert response"""
        try:
            # Load critique template
            template = self.jinja_env.get_template("critique_and_correct_prompt.txt")
            
            # Build critique prompt
            prompt = template.render(
                client_input=client_input,
                ai_response=ai_response,
                expert_response=expert_response
            )
            
            # Generate critique using model manager
            critique_response = await self.model_manager.generate_response(
                prompt=prompt,
                user_id="golden_standard_validator",
                max_tokens=500,
                temperature=0.1  # Low temperature for consistent analysis
            )
            
            # Parse the critique response
            try:
                critique_data = json.loads(critique_response)
                
                # Validate required fields
                required_fields = ['critique', 'identified_skill_name', 'dora_category']
                if not all(field in critique_data for field in required_fields):
                    logger.warning(f"Incomplete critique response: {critique_response}")
                    return None
                
                # Calculate improvement score based on critique severity
                improvement_score = self._calculate_improvement_score(critique_data['critique'])
                
                # Extract specific issues
                specific_issues = self._extract_specific_issues(critique_data['critique'])
                
                critique = PerformanceCritique(
                    client_input=client_input,
                    ai_response=ai_response,
                    expert_response=expert_response,
                    critique_text=critique_data['critique'],
                    identified_skill=critique_data['identified_skill_name'],
                    dora_category=critique_data['dora_category'],
                    improvement_score=improvement_score,
                    specific_issues=specific_issues
                )
                
                return critique
                
            except json.JSONDecodeError as e:
                logger.warning(f"Failed to parse critique JSON: {critique_response}")
                return None
            
        except Exception as e:
            logger.error(f"Error generating critique: {str(e)}")
            return None
    
    def _calculate_improvement_score(self, critique_text: str) -> float:
        """Calculate improvement score based on critique severity"""
        critique_lower = critique_text.lower()
        
        # Severity indicators
        critical_indicators = ['major', 'significant', 'serious', 'critical', 'harmful']
        moderate_indicators = ['should', 'could', 'missing', 'lacks', 'insufficient']
        minor_indicators = ['slightly', 'minor', 'small', 'little']
        
        critical_count = sum(1 for indicator in critical_indicators if indicator in critique_lower)
        moderate_count = sum(1 for indicator in moderate_indicators if indicator in critique_lower)
        minor_count = sum(1 for indicator in minor_indicators if indicator in critique_lower)
        
        # Calculate weighted score
        if critical_count > 0:
            return 0.8 + min(critical_count * 0.05, 0.2)
        elif moderate_count > 0:
            return 0.4 + min(moderate_count * 0.1, 0.4)
        elif minor_count > 0:
            return 0.1 + min(minor_count * 0.05, 0.2)
        else:
            return 0.2  # Default moderate improvement needed
    
    def _extract_specific_issues(self, critique_text: str) -> List[str]:
        """Extract specific issues from critique text"""
        # Simple extraction based on sentence patterns
        import re
        
        sentences = re.split(r'[.!?]+', critique_text)
        issues = []
        
        issue_patterns = [
            r'should\s+([^.!?]+)',
            r'lacks\s+([^.!?]+)',
            r'missing\s+([^.!?]+)',
            r'fails\s+to\s+([^.!?]+)',
            r'does\s+not\s+([^.!?]+)'
        ]
        
        for sentence in sentences:
            for pattern in issue_patterns:
                matches = re.findall(pattern, sentence, re.IGNORECASE)
                issues.extend(matches)
        
        return issues[:5]  # Limit to top 5 issues
    
    async def create_training_example(self, critique: PerformanceCritique) -> Optional[Dict[str, Any]]:
        """Create a training example from performance critique"""
        try:
            # Calculate weight multiplier based on improvement score
            weight_multiplier = 1.0 + critique.improvement_score  # Range: 1.0 to 2.0
            
            training_example = {
                'user_input': critique.client_input,
                'ai_response': critique.ai_response,
                'expert_response': critique.expert_response,
                'critique': critique.critique_text,
                'identified_skill_name': critique.identified_skill,
                'dora_category': critique.dora_category,
                'weight_multiplier': weight_multiplier,
                'validation_score': 1.0 - critique.improvement_score,  # Higher score = less improvement needed
                'source_dialogue_id': f"golden_validation_{datetime.now().timestamp()}"
            }
            
            return training_example
            
        except Exception as e:
            logger.error(f"Error creating training example: {str(e)}")
            return None

    def calculate_performance_metrics(self, 
                                    critiques: List[PerformanceCritique],
                                    simulation_results: Dict[str, Any]) -> Dict[str, Any]:
        """Calculate overall performance metrics from critiques"""
        if not critiques:
            return {
                'overall_score': 1.0,
                'total_critiques': 0,
                'average_improvement_needed': 0.0,
                'skill_performance': {},
                'category_performance': {}
            }
        
        # Calculate overall performance
        total_improvement_needed = sum(c.improvement_score for c in critiques)
        average_improvement_needed = total_improvement_needed / len(critiques)
        overall_score = max(0, 1.0 - average_improvement_needed)
        
        # Performance by skill
        skill_performance = {}
        skill_groups = {}
        for critique in critiques:
            skill = critique.identified_skill
            if skill not in skill_groups:
                skill_groups[skill] = []
            skill_groups[skill].append(critique.improvement_score)
        
        for skill, scores in skill_groups.items():
            avg_score = sum(scores) / len(scores)
            skill_performance[skill] = {
                'average_improvement_needed': avg_score,
                'performance_score': max(0, 1.0 - avg_score),
                'critique_count': len(scores)
            }
        
        # Performance by DORA category
        category_performance = {}
        category_groups = {}
        for critique in critiques:
            category = critique.dora_category
            if category not in category_groups:
                category_groups[category] = []
            category_groups[category].append(critique.improvement_score)
        
        for category, scores in category_groups.items():
            avg_score = sum(scores) / len(scores)
            category_performance[category] = {
                'average_improvement_needed': avg_score,
                'performance_score': max(0, 1.0 - avg_score),
                'critique_count': len(scores)
            }
        
        return {
            'overall_score': round(overall_score, 3),
            'total_critiques': len(critiques),
            'average_improvement_needed': round(average_improvement_needed, 3),
            'successful_turn_rate': simulation_results['successful_turns'] / simulation_results['total_turns'],
            'skill_performance': skill_performance,
            'category_performance': category_performance
        }

    async def load_golden_dialogue_from_file(self, file_path: Path) -> GoldenDialogue:
        """Load a golden dialogue from a JSON file"""
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                data = json.load(f)
            
            # Parse turns
            turns = []
            for turn_data in data.get('turns', []):
                turn = DialogueTurn(
                    speaker=turn_data['speaker'],
                    content=turn_data['content'],
                    turn_number=turn_data.get('turn_number', len(turns) + 1),
                    metadata=turn_data.get('metadata', {})
                )
                turns.append(turn)
            
            dialogue = GoldenDialogue(
                dialogue_id=data.get('dialogue_id', str(file_path.stem)),
                turns=turns,
                metadata=data.get('metadata', {})
            )
            
            logger.info(f"Loaded golden dialogue {dialogue.dialogue_id} with {len(turns)} turns")
            return dialogue
            
        except Exception as e:
            logger.error(f"Error loading golden dialogue from {file_path}: {str(e)}")
            raise GoldenStandardError(f"Failed to load dialogue: {str(e)}")
    
    async def batch_validate_dialogues(self, 
                                     dialogue_directory: Path,
                                     user_id: str) -> Dict[str, Any]:
        """Validate against multiple golden dialogues in a directory"""
        try:
            dialogue_files = list(dialogue_directory.glob('*.json'))
            
            if not dialogue_files:
                raise GoldenStandardError(f"No dialogue files found in {dialogue_directory}")
            
            logger.info(f"Starting batch validation of {len(dialogue_files)} dialogues")
            
            validation_results = []
            total_training_examples = 0
            
            for file_path in dialogue_files:
                try:
                    # Load dialogue
                    dialogue = await self.load_golden_dialogue_from_file(file_path)
                    
                    # Validate
                    result = await self.validate_against_golden_dialogue(dialogue, user_id)
                    validation_results.append(result)
                    total_training_examples += result['training_examples_created']
                    
                    logger.info(f"Validated {dialogue.dialogue_id}: Score {result['performance_metrics']['overall_score']}")
                    
                except Exception as e:
                    logger.error(f"Error validating {file_path}: {str(e)}")
                    validation_results.append({
                        'dialogue_id': str(file_path.stem),
                        'error': str(e),
                        'validation_timestamp': datetime.now().isoformat()
                    })
            
            # Calculate batch metrics
            successful_validations = [r for r in validation_results if 'error' not in r]
            
            if successful_validations:
                avg_score = sum(r['performance_metrics']['overall_score'] for r in successful_validations) / len(successful_validations)
                total_critiques = sum(len(r['critiques']) for r in successful_validations)
            else:
                avg_score = 0.0
                total_critiques = 0
            
            batch_summary = {
                'total_dialogues': len(dialogue_files),
                'successful_validations': len(successful_validations),
                'failed_validations': len(dialogue_files) - len(successful_validations),
                'average_performance_score': round(avg_score, 3),
                'total_critiques_generated': total_critiques,
                'total_training_examples_created': total_training_examples,
                'validation_timestamp': datetime.now().isoformat()
            }
            
            logger.info(f"Batch validation complete: {batch_summary}")
            
            return {
                'batch_summary': batch_summary,
                'individual_results': validation_results,
                'validation_stats': self.validation_stats
            }
            
        except Exception as e:
            logger.error(f"Error in batch validation: {str(e)}")
            raise GoldenStandardError(f"Batch validation failed: {str(e)}")
    
    async def get_training_examples_for_retraining(self, 
                                                 limit: Optional[int] = None,
                                                 min_weight: float = 1.2) -> List[Dict[str, Any]]:
        """Get training examples for model retraining"""
        try:
            async with get_async_session() as session:
                query = select(TrainingExample).where(
                    and_(
                        TrainingExample.weight_multiplier >= min_weight,
                        TrainingExample.used_in_training == False
                    )
                ).order_by(desc(TrainingExample.weight_multiplier))
                
                if limit:
                    query = query.limit(limit)
                
                result = await session.execute(query)
                examples = result.scalars().all()
                
                # Format for training
                training_data = []
                for example in examples:
                    training_data.append({
                        'example_id': str(example.example_id),
                        'input': example.user_input,
                        'target_output': example.expert_response,
                        'negative_output': example.ai_response,
                        'critique': example.critique,
                        'skill': example.identified_skill_name,
                        'category': example.dora_category,
                        'weight': example.weight_multiplier,
                        'validation_score': example.validation_score
                    })
                
                logger.info(f"Retrieved {len(training_data)} training examples for retraining")
                return training_data
                
        except Exception as e:
            logger.error(f"Error getting training examples: {str(e)}")
            raise GoldenStandardError(f"Failed to get training examples: {str(e)}")
    
    async def mark_examples_as_used(self, example_ids: List[str]) -> None:
        """Mark training examples as used in training"""
        try:
            async with get_async_session() as session:
                # Update examples to mark as used
                for example_id in example_ids:
                    query = select(TrainingExample).where(TrainingExample.example_id == example_id)
                    result = await session.execute(query)
                    example = result.scalar_one_or_none()
                    
                    if example:
                        example.used_in_training = True
                
                await session.commit()
                logger.info(f"Marked {len(example_ids)} training examples as used")
                
        except Exception as e:
            logger.error(f"Error marking examples as used: {str(e)}")
            raise GoldenStandardError(f"Failed to mark examples as used: {str(e)}")

# Exception utilities
def format_exception_details(exception: TCISBaseException) -> Dict[str, Any]:
    """Format exception details for logging or API responses"""
    return {
        "error_type": type(exception).__name__,
        "message": exception.message,
        "error_code": exception.error_code,
        "details": exception.details,
        "timestamp": datetime.now().isoformat()
    }

def create_error_response(exception: TCISBaseException, include_details: bool = False) -> Dict[str, Any]:
    """Create standardized error response for API"""
    response = {
        "error": True,
        "error_type": type(exception).__name__,
        "message": exception.message
    }
    
    if exception.error_code:
        response["error_code"] = exception.error_code
    
    if include_details and exception.details:
        response["details"] = exception.details
    
    return response

# Error handler decorator
def handle_tcis_exceptions(func):
    """Decorator to handle TCIS exceptions and log them"""
    import functools
    
    @functools.wraps(func)
    async def wrapper(*args, **kwargs):
        try:
            return await func(*args, **kwargs)
        except TCISBaseException as e:
            logger = logging.getLogger(func.__module__)
            logger.error(f"TCIS Exception in {func.__name__}: {format_exception_details(e)}")
            raise
        except Exception as e:
            logger = logging.getLogger(func.__module__)
            logger.error(f"Unexpected exception in {func.__name__}: {str(e)}", exc_info=True)
            # Convert to TCIS exception
            raise TCISBaseException(
                message=f"Unexpected error in {func.__name__}",
                error_code="UNEXPECTED_ERROR",
                details={"original_error": str(e)}
            )
    
    return wrapper

async def get_async_session() -> AsyncGenerator[AsyncSession, None]:
    """
    Dependency to get async database session
    Use with 'async with get_async_session() as session:'
    """
    async with AsyncSessionLocal() as session:
        try:
            yield session
        except Exception as e:
            await session.rollback()
            logger.error(f"Database session error: {str(e)}")
            raise DatabaseError(f"Database operation failed: {str(e)}")
        finally:
            await session.close()

@asynccontextmanager
async def get_db_session():
    """Context manager for database sessions"""
    async with AsyncSessionLocal() as session:
        try:
            yield session
            await session.commit()
        except Exception as e:
            await session.rollback()
            logger.error(f"Database transaction failed: {str(e)}")
            raise
        finally:
            await session.close()

async def create_tables():

    from app.models.core import Base

class DatabaseManager:
    """Database manager for handling connections and operations"""
    
    def __init__(self):
        self.engine = engine
        self.session_factory = AsyncSessionLocal
    
    async def initialize(self):
        """Initialize database (create tables if needed)"""
        try:
            await check_database_connection()
            await create_tables()
            logger.info("Database initialized successfully")
        except Exception as e:
            logger.error(f"Database initialization failed: {str(e)}")
            raise
    
    async def close(self):
        """Close database connections"""
        await self.engine.dispose()
        logger.info("Database connections closed")
    
    async def health_check(self) -> Dict[str, Any]:
        """Perform database health check"""
        try:
            start_time = datetime.now()
            async with self.session_factory() as session:
                await session.execute("SELECT 1")
            end_time = datetime.now()
            
            response_time = (end_time - start_time).total_seconds() * 1000
            
            return {
                "status": "healthy",
                "response_time_ms": round(response_time, 2),
                "connection_pool_size": self.engine.pool.size(),
                "checked_out_connections": self.engine.pool.checkedout(),
                "timestamp": end_time.isoformat()
            }
        except Exception as e:
            return {
                "status": "unhealthy",
                "error": str(e),
                "timestamp": datetime.now().isoformat()
            }
