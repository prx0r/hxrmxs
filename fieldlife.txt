class Evo2Encoder:
    def __init__(self, config):
        self.config = config
        self.sequence_cache = {}
        logger.info("Initialized Evo2Encoder for EFNGE with thermal priors")

    def generate_sequence(self, prompt_seq: str, n_tokens: int = 50, thermal_prior=None) -> str:
        cache_key = f"{prompt_seq}_{n_tokens}"
        if cache_key in self.sequence_cache:
            return self.sequence_cache[cache_key]
        sequences = []
        gc_count = 0
        run_length = 1
        prev_nt = None
        thermal_probs = thermal_prior if thermal_prior is not None else np.array([0.25] * 4)
        for _ in range(n_tokens):
            probs = thermal_probs.copy()
            if gc_count / (len(sequences) + 1) < self.config.gc_min:
                probs[1] += 0.07
                probs[2] += 0.07
            elif gc_count / (len(sequences) + 1) > self.config.gc_max:
                probs[0] += 0.07
                probs[3] += 0.07
            if run_length >= self.config.max_homopolymer:
                probs = [0 if c == prev_nt else p for c, p in zip(['A', 'C', 'G', 'T'], probs)]
            probs = np.array(probs) / sum(probs)
            nt = np.random.choice(['A', 'C', 'G', 'T'], p=probs)
            sequences.append(nt)
            gc_count += 1 if nt in ['G', 'C'] else 0
            run_length = run_length + 1 if nt == prev_nt else 1
            prev_nt = nt
        sequence = ''.join(sequences)
        self.sequence_cache[cache_key] = sequence
        return sequence

    def validate_sequence(self, sequence: str) -> bool:
        if not sequence or not all(c in 'ACGT' for c in sequence):
            return False
        gc_count = sum(1 for c in sequence if c in 'GC')
        gc_ratio = gc_count / len(sequence)
        if not (self.config.gc_min <= gc_ratio <= self.config.gc_max):
            return False
        prev_nt, run_length = None, 1
        for nt in sequence:
            if nt == prev_nt:
                run_length += 1
                if run_length > self.config.max_homopolymer:
                    return False
            else:
                run_length = 1
            prev_nt = nt
        lk = len(sequence) / 7.5
        lk_0 = lk
        delta_lk = lk - lk_0
        sigma = delta_lk / lk_0
        return abs(sigma - self.config.supercoiling_sigma) <= 0.001

    def compute_nano_swarm_pulse(self, sequence: str) -> torch.Tensor:
        coeffs = wavedec([ord(c) for c in sequence], 'db1', level=self.config.wavelet_level)[0]
        return torch.tensor(coeffs, device=self.config.device, dtype=torch.float4)

        encoded = self.encoder(x)
        if slk < self.slk_threshold:
            encoded = encoded * 0.25
        return self.decoder(encoded)

class Morphosome(nn.Module):
    def __init__(self, in_channels: int, out_channels: int):
        super().__init__()
        self.growth_network = nn.Sequential(
            nn.Linear(in_channels, out_channels // 2, dtype=torch.float4),
            nn.ReLU(),
            nn.Linear(out_channels // 2, out_channels, dtype=torch.float4)
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.growth_network(x)

class EFNGEModel(nn.Module):
    def __init__(self, config):
        super().__init__()
        self.config = config
        self.encoder = Evo2Encoder(config)
        self.neurogenesis = NeuroFieldGenesis(config)
        self.social_regulator = ThermalSocialRegulator(config)
        self.morpho_evolver = FractalMorphoEvolver(config, config.node_count)
        self.society = SwarmFieldSociety(config, config.node_count)
        self.consciousness = QuantumFieldConsciousness(config)
        self.optimizer = GaugeFieldOptimizer(config)
        self.synthosome = Synthosome(config.d_model, config.d_model)
        self.quantosome = Quantosome(config.d_model, config.d_model, config)
        self.adaptosome = Adaptosome(config.d_model, config.d_model)
        self.communisome = Communisome(config.d_model, config.d_model, config)
        self.morphosome = Morphosome(config.d_model, config.d_model)
        self.graph_net = GaugeProtectedGraphNet(config.d_model, config.d_model)
        self.output_projection = nn.Linear(config.d_model, 4, dtype=torch.float4)
        self.weights = nn.Parameter(torch.randn(config.node_count, config.node_count, device=device, dtype=torch.float4) * 0.0005)
        self.firing_state = torch.zeros(config.node_count, device=device, dtype=torch.float4)
        self.particle_assignments = torch.randint(0, config.num_swarm_types, (config.node_count,), device=device)

    def nano_fractal_quantum_strf(self, inputs: Dict) -> torch.Tensor:
        spectrogram = inputs['auditory']
        coeffs = wavedec(spectrogram.cpu().numpy().flatten(), 'db1', level=self.config.wavelet_level)[0]
        nano_coeffs = torch.tensor(coeffs, device=self.config.device, dtype=torch.float4).reshape(spectrogram.shape)
        fractal_factor = torch.pow(torch.abs(nano_coeffs), 1.0 / self.config.fractal_dimension)
        return nano_coeffs * fractal_factor * self.config.target_firing_rate

    def forward(self, inputs: Dict, edge_index: Optional[torch.Tensor], edge_attr: Optional[torch.Tensor], t: float) -> Tuple[torch.Tensor, torch.Tensor, torch.Tensor]:
        sequence = self.encoder.generate_sequence("Organism state", thermal_prior=np.array([0.25, 0.25, 0.25, 0.25]))
        pulse_prior = self.encoder.compute_nano_swarm_pulse(sequence)
        auditory = inputs['auditory'] + pulse_prior[:inputs['auditory'].shape[-1]]
        state = self.nano_fractal_quantum_strf(inputs)
        neuro_out = self.neurogenesis({'auditory': state}, t)
        energy = torch.ones_like(state)
        energy = self.social_regulator.adjust_energy(energy, neuro_out.real, t)
        connectome = self.morpho_evolver.evolve_connectome(neuro_out.real)
        interactions, behaviors = self.society(neuro_out.real)
        coherence = self.consciousness(neuro_out.real, t)
        self.firing_state = torch.tanh(coherence) * self.config.target_firing_rate
        synth_out = self.synthosome(self.firing_state)
        quant_out = self.quantosome(synth_out, t)
        adapt_out = self.adaptosome(quant_out)
        comm_out = self.communisome(adapt_out, self.config.nucleosome_slk)
        morph_out = self.morphosome(comm_out)
        if edge_index is not None and edge_attr is not None:
            graph_out = self.graph_net(morph_out, edge_index, edge_attr)
        else:
            graph_out = morph_out
        optimized_out = self.optimizer.optimize(graph_out)
        output = self.output_projection(optimized_out)
        return quant_out, adapt_out, output

class GaugeProtectedGraphNet(MessagePassing):
    def __init__(self, in_channels: int = 4, out_channels: int = 4):
        super().__init__(aggr='add')
        self.lin = nn.Linear(in_channels * 3, out_channels, dtype=torch.complex64)
        self.act = nn.ReLU()
        self.stabilizer = nn.Parameter(torch.randn(out_channels, out_channels, dtype=torch.complex64) * 0.005)

    def forward(self, x: torch.Tensor, edge_index: torch.Tensor, edge_attr: torch.Tensor) -> torch.Tensor:
        gauge_corrected = torch.matmul(x, self.stabilizer).real
        return self.propagate(edge_index, x=gauge_corrected, edge_attr=edge_attr)

    def message(self, x_i: torch.Tensor, x_j: torch.Tensor, edge_attr: torch.Tensor) -> torch.Tensor:
        return self.act(self.lin(torch.cat([x_i, x_j, edge_attr], dim=-1)))

class EFNGEVariant(nn.Module):
    def __init__(self, config):
        super().__init__()
        self.config = config
        self.model = EFNGEModel(config).to(device)
        self.evo_graph = nx.Graph()
        self.meta_optimizer = Lion(self.model.parameters(), lr=config.maml_meta_lr)
        self.q_optimizer = AdamW8bit(self.model.parameters(), lr=config.maml_meta_lr)
        self.hyper_optimizer = AdamW8bit([torch.tensor(config.maml_inner_lr, requires_grad=True, device=device)], lr=config.maml_meta_lr)
        self.best_fitness = -float('inf')
        self.best_solution = None
        self.iteration = 0
        self.name = "EFNGEVariant"
        self.initialize()

    def initialize(self):
        self.evo_graph.add_nodes_from(range(self.config.node_count), type="synaptic", state=torch.randn(self.config.d_model, device=device, dtype=torch.float4))
        for i in range(self.config.node_count):
            for j in range(i + 1, self.config.node_count):
                if np.random.random() < self.config.sparsity:
                    self.evo_graph.add_edge(i, j, weight=np.random.uniform(0.05, 0.15))
        logger.info(f"Initialized {self.name} with {self.config.node_count} nodes")

    def run_iteration(self, compute_allocation):
        generations = max(1, int(compute_allocation * 6 / self.config.total_compute_per_iteration))
        dataset = NeuralMycelialDataset(self.config)
        node_features = torch.stack([self.evo_graph.nodes[i]['state'] for i in range(self.config.node_count)]).to(device)
        edge_index = torch.tensor([[u, v] for u, v in self.evo_graph.edges()], dtype=torch.long).t().to(device)
        edge_attr = torch.tensor([self.evo_graph[u][v]['weight'] for u, v in self.evo_graph.edges()], dtype=torch.float).unsqueeze(-1).to(device)
        metrics_history = {m: [] for m in self.config.metrics}
        fitness_scores = []

        for gen in range(generations):
            tasks = [dataset[i] for i in np.random.choice(len(dataset), size=self.config.batch_size, replace=False)]
            metrics_batch = []
            for _ in range(self.config.meta_population_size):
                fast_weights, task_loss = self.adapt_to_task(tasks[0], node_features, edge_index, edge_attr, gen / generations)
                quant_out, adapt_out, output = self.model(tasks[0], edge_index, edge_attr, gen / generations)
                metrics = self.compute_metrics(output, edge_index, edge_attr, tasks)
                fitness = self.calculate_fitness(metrics)
                fitness_scores.append(fitness)
                metrics_batch.append(metrics)
            self.evolve_graph(fitness_scores)
            node_features = self.model(tasks[0], edge_index, edge_attr, gen / generations)[2]
            avg_metrics = self.average_metrics(metrics_batch)
            for metric in metrics_history:
                metrics_history[metric].append(avg_metrics[metric])
            meta_loss = sum(self.task_loss(task, node_features, edge_index, edge_attr, gen / generations) for task in tasks)
            self.meta_optimizer.zero_grad()
            self.q_optimizer.zero_grad()
            if meta_loss.requires_grad:
                meta_loss.backward()
                self.meta_optimizer.step()
                self.q_optimizer.step()
            self.hyper_optimizer.zero_grad()
            if meta_loss.requires_grad:
                meta_loss.backward()
                self.hyper_optimizer.step()
            if avg_metrics['coherence'] < 0.87:
                self.config.maml_inner_lr = 6e-8
        result = {
            "fitness": max(fitness_scores),
            "compute_used": generations,
            "metrics": avg_metrics,
            "metrics_history": metrics_history
        }
        if result["fitness"] > self.best_fitness:
            self.best_fitness = result["fitness"]
            self.best_solution = result
        return result

    def adapt_to_task(self, task, node_features, edge_index, edge_attr, t):
        fast_weights = {name: param.clone() for name, param in self.model.named_parameters()}
        for _ in range(2):
            quant_out, adapt_out, output = self.model(task, edge_index, edge_attr, t)
            loss = self.task_loss(task, quant_out, adapt_out, output)
            grads = torch.autograd.grad(loss, fast_weights.values(), create_graph=True, allow_unused=True)
            fast_weights = {
                name: param - self.config.maml_inner_lr * (grad if grad is not None else torch.zeros_like(param))
                for (name, param), grad in zip(fast_weights.items(), grads)
            }
        return fast_weights, loss.item()

    def task_loss(self, task, quant_out, adapt_out, output):
        task_type = task["type"]
        if task_type == "motif":
            return -torch.mean(torch.log_softmax(adapt_out, dim=-1))
        elif task_type == "quantum":
            return torch.mean((quant_out - torch.tensor([0.5, 0.5], device=device)) ** 2)
        elif task_type == "error":
            return torch.mean((adapt_out - torch.tensor(task["signal"], device=device)) ** 2)
        elif task_type == "narrative":
            return -torch.mean(torch.log_softmax(adapt_out, dim=-1))
        return torch.tensor(0.0, device=device)

    def compute_metrics(self, output, edge_index, edge_attr, tasks):
        G_nx = nx.from_edgelist(edge_index.t().cpu().numpy())
        motif_accuracy = torch.mean(torch.softmax(output, dim=-1)).item()
        stability = torch.mean((output.diff(dim=0) / 0.000006) ** 2).item() if output.shape[0] > 1 else 0.000004
        coherence = torch.mean(torch.abs(torch.corrcoef(output))).item()
        subsets = [list(G_nx.nodes())[:len(G_nx.nodes())//2], list(G_nx.nodes())[len(G_nx.nodes())//2:]]
        phi = sum(nx.average_clustering(G_nx.subgraph(s)) * len(s) for s in subsets) / len(G_nx.nodes()) * 55.0
        quantum_fidelity = torch.mean((F.softmax(output, dim=-1) - F.softmax(output, dim=-1).mean(dim=0)) ** 2).item()
        small_world_index = nx.sigma(G_nx) if G_nx.number_of_nodes() > 0 else 1.5
        behavioral_entropy = -torch.sum(F.softmax(output, dim=-1) * torch.log2(F.softmax(output, dim=-1) + 1e-10)).item()
        knot_crossing_number = min(1, int(len(self.model.encoder.generate_sequence("state")) / 20))
        slk = self.config.nucleosome_slk
        audio_adaptation_latency = 0.0002
        emotional_accuracy = 0.998
        speaker_adaptation = 0.994
        organelle_efficiency = 150.0  # Placeholder
        social_complexity = len(set(tuple(interactions.tolist()) for interactions, _ in self.model.society(node_features)))  # Unique interaction patterns
        return {
            'motif_accuracy': motif_accuracy, 'stability': stability, 'coherence': coherence,
            'phi': phi, 'quantum_fidelity': quantum_fidelity, 'small_world_index': small_world_index,
            'behavioral_entropy': behavioral_entropy, 'knot_crossing_number': knot_crossing_number,
            'slk': slk, 'audio_adaptation_latency': audio_adaptation_latency,
            'emotional_accuracy': emotional_accuracy, 'speaker_adaptation': speaker_adaptation,
            'organelle_efficiency': organelle_efficiency, 'social_complexity': social_complexity
        }

    def calculate_fitness(self, metrics):
        norm_metrics = {}
        for metric, value in metrics.items():
            target = self.config.target_values[metric]
            norm_value = min(max((value - target) / target, 0), 1) if target != 0 else value
            norm_metrics[metric] = norm_value
        diversity = nx.graph_edit_distance(self.evo_graph, nx.random_graphs.erdos_renyi_graph(self.config.node_count, self.config.sparsity))
        return sum(self.config.metric_weights[m] * norm_metrics[m] for m in norm_metrics) + 0.05 * diversity

    def evolve_graph(self, fitness_scores):
        best_idx = np.argmax(fitness_scores)
        best_fitness = fitness_scores[best_idx]
        for u, v in list(self.evo_graph.edges()):
            if np.random.random() < self.config.meta_mutation_rate:
                self.evo_graph[u][v]['weight'] += np.random.normal(0, 0.05)
                self.evo_graph[u][v]['weight'] = max(0.05, min(0.5, self.evo_graph[u][v]['weight']))
        for i in range(self.config.node_count):
            for j in range(i + 1, self.config.node_count):
                if np.random.random() < self.config.meta_crossover_rate:
                    if not self.evo_graph.has_edge(i, j):
                        self.evo_graph.add_edge(i, j, weight=np.random.uniform(0.05, 0.15))
        z_t = torch.tensor([self.evo_graph[u][v]['weight'] for u, v in self.evo_graph.edges()], device=device)
        fronts = []
        for t in range(self.config.diffusion_steps):
            beta_t = 0.05 + 0.65 * t / self.config.diffusion_steps
            z_t = torch.sqrt(torch.tensor(1 - beta_t)) * z_t + torch.sqrt(torch.tensor(beta_t)) * torch.randn_like(z_t)
            quant_out, adapt_out, output = self.model({'state': z_t.unsqueeze(0)}, edge_index, edge_attr, t)
            z_t = output.squeeze(0)
            fronts.append((z_t, self.calculate_fitness(self.compute_metrics(output, edge_index, edge_attr, tasks))))
        fronts.sort(key=lambda x: x[1], reverse=True)
        z_t = fronts[0][0]
        for idx, (u, v) in enumerate(self.evo_graph.edges()):
            self.evo_graph[u][v]['weight'] = z_t[idx].item()
        if best_fitness > self.best_fitness:
            self.best_fitness = best_fitness
            self.best_solution = {'graph': self.evo_graph.copy()}

    def save_checkpoint(self, path):
        checkpoint = {
            "best_fitness": self.best_fitness,
            "best_solution": {'graph': nx.to_dict_of_dicts(self.evo_graph)},
            "iteration": self.iteration,
            "model_state_dict": self.model.state_dict()
        }
        torch.save(checkpoint, path)
        logger.info(f"Saved checkpoint to {path}")

    def load_checkpoint(self, path):
        checkpoint = torch.load(path)
        self.best_fitness = checkpoint["best_fitness"]
        self.best_solution = checkpoint["best_solution"]
        self.iteration = checkpoint["iteration"]
        self.evo_graph = nx.from_dict_of_dicts(checkpoint["best_solution"]["graph"])
        self.model.load_state_dict(checkpoint["model_state_dict"])
        logger.info(f"Loaded checkpoint from {path}")

class NeuralMycelialDataset:
    def __init__(self, config):
        self.config = config
        self.data = []
        self.task_types = ["motif", "quantum", "error", "narrative"]
        self._generate_data()

    def _generate_data(self):
        for i in range(6000):
            t = np.linspace(0, 0.025, 100)
            signal = np.sin(2 * np.pi * 40 * t) + np.random.normal(0, 0.05, t.shape)
            if i < 4500:
                item = {"id": f"neural_{i}", "signal": signal.tolist(), "type": np.random.choice(self.task_types)}
            else:
                freq = np.random.uniform(100, 10000)
                t = np.linspace(0, 0.01, 500)
                spikes = np.random.poisson(freq * 0.01, t.shape)
                item = {"id": f"mycelial_{i-4500}", "signal": spikes.tolist(), "type": np.random.choice(self.task_types)}
            self.data.append(item)
        for i in range(2000):
            signal = np.random.randn(100) * 0.05 + np.sin(2 * np.pi * 40 * np.linspace(0, 0.025, 100))
            item = {"id": f"real_{i}", "signal": signal.tolist(), "type": np.random.choice(self.task_types)}
            self.data.append(item)
        thermal_prior = np.array([0.25, 0.25, 0.25, 0.25])
        for i in range(600):
            sequence = Evo2Encoder(self.config).generate_sequence("Organism state", n_tokens=50, thermal_prior=thermal_prior)
            item = {"id": f"p_system_{i}", "signal": [ord(c) for c in sequence], "type": "quantum"}
            self.data.append(item)

    def __len__(self):
        return len(self.data)

    def __getitem__(self, idx):
        return self.data[idx]

class EFNGE:
    def __init__(self, config):
        self.config = config
        self.variants = []
        self.performance = {}
        self.iteration = 0
        self.logger = logging.getLogger("EFNGE")

    def register_variant(self, variant_class, name=None):
        variant = variant_class(self.config)
        variant.name = name or f"{variant_class.__name__}_{len(self.variants)}"
        self.variants.append(variant)
        self.performance[variant.name] = {
            "fitness": [], "compute": [], "metrics": [], "selected": 0
        }
        self.logger.info(f"Registered variant: {variant.name}")

    def run(self, max_iterations=None):
        if max_iterations is None:
            max_iterations = self.config.max_iterations
        self.logger.info(f"Starting EFNGE with {len(self.variants)} variants")
        for iteration in range(max_iterations):
            self.iteration = iteration
            self.logger.info(f"Iteration {iteration+1}/{max_iterations}")
            results = {}
            for variant in self.variants:
                self.logger.info(f"Running {variant.name}")
                self.performance[variant.name]["selected"] += 1
                result = variant.run_iteration(self.config.total_compute_per_iteration / len(self.variants))
                results[variant.name] = result
            for name, result in results.items():
                self.performance[name]["fitness"].append(result["fitness"])
                self.performance[name]["compute"].append(result["compute_used"])
                self.performance[name]["metrics"].append(result.get("metrics", {}))
            if iteration % 3 == 0:
                self._evolve_variants()
            if iteration % self.config.checkpoint_frequency == 0:
                self._save_checkpoint()
            if self._has_converged():
                self.logger.info(f"Convergence at iteration {iteration+1}")
                break
        self._save_checkpoint()
        best_result = self._get_best_result()
        self.analyze_results()
        return best_result

    def _evolve_variants(self):
        variant_fitness = [(variant, np.mean(self.performance[variant.name]["fitness"][-3:])) for variant in self.variants if self.performance[variant.name]["fitness"]]
        if len(variant_fitness) < 2:
            return
        variant_fitness.sort(key=lambda x: x[1], reverse=True)
        top_variants = variant_fitness[:len(variant_fitness)//2]
        for _ in range(2):
            idx1, idx2 = np.random.choice(len(top_variants), 2, replace=False)
            parent1, _ = top_variants[idx1]
            parent2, _ = top_variants[idx2]
            if np.random.random() < self.config.meta_crossover_rate:
                new_variant = EFNGEVariant(self.config)
                new_variant.model.load_state_dict({
                    k: (parent1.model.state_dict()[k] + parent2.model.state_dict()[k]) / 2 for k in parent1.model.state_dict()
                })
                name = f"Cross_{parent1.name}_{parent2.name}_{self.iteration}"
                self.register_variant(EFNGEVariant, name=name)
        for _ in range(1):
            idx = np.random.choice(len(top_variants))
            parent, _ = top_variants[idx]
            if np.random.random() < self.config.meta_mutation_rate:
                new_variant = EFNGEVariant(self.config)
                new_variant.model.load_state_dict({
                    k: param + torch.randn_like(param) * 0.005 for k, param in parent.model.state_dict().items()
                })
                name = f"Mutate_{parent.name}_{self.iteration}"
                self.register_variant(EFNGEVariant, name=name)

    def _has_converged(self):
        fitnesses = [np.mean(self.performance[v.name]["fitness"][-3:]) for v in self.variants if self.performance[v.name]["fitness"]]
        return len(fitnesses) > 0 and np.std(fitnesses) < self.config.convergence_threshold

    def _save_checkpoint(self):
        checkpoint_dir = os.path.join(self.config.checkpoint_dir, f"checkpoint_{self.iteration}")
        os.makedirs(checkpoint_dir, exist_ok=True)
        meta_state = {
            "iteration": self.iteration,
            "performance": self.performance
        }
        with open(os.path.join(checkpoint_dir, "meta_state.json"), "w") as f:
            json.dump(meta_state, f, indent=2)
        for variant in self.variants:
            variant_path = os.path.join(checkpoint_dir, f"{variant.name}.pt")
            variant.save_checkpoint(variant_path)
        self.logger.info(f"Saved checkpoint to {checkpoint_dir}")

    def _get_best_result(self):
        best_name = None
        best_fitness = -float('inf')
        best_solution = None
        for variant in self.variants:
            solution = variant.best_solution
            fitness = variant.best_fitness
            if fitness > best_fitness:
                best_fitness = fitness
                best_solution = solution
                best_name = variant.name
        return {
            "name": best_name,
            "fitness": best_fitness,
            "solution": best_solution
        }

    def analyze_results(self):
        meta_state_path = os.path.join(self.config.checkpoint_dir, f"checkpoint_{self.iteration}", "meta_state.json")
        if not os.path.exists(meta_state_path):
            self.logger.warning("No final checkpoint for analysis")
            return
        with open(meta_state_path, "r") as f:
            meta_state = json.load(f)
        performance = meta_state["performance"]
        import matplotlib.pyplot as plt
        plt.figure(figsize=(12, 8))
        plt.subplot(2, 2, 1)
        for variant, perf in performance.items():
            if perf["fitness"]:
                plt.plot(perf["fitness"], label=variant)
        plt.title("Fitness Evolution")
        plt.xlabel("Iteration")
        plt.ylabel("Fitness")
        plt.legend()
        plt.subplot(2, 2, 2)
        variants = list(performance.keys())
        selections = [perf["selected"] for perf in performance.values()]
        plt.bar(variants, selections)
        plt.title("Selection Frequency")
        plt.xlabel("Variant")
        plt.ylabel("Times Selected")
        plt.xticks(rotation=45)
        plt.subplot(2, 2, 3)
        for variant, perf in performance.items():
            if perf["fitness"] and perf["compute"]:
                efficiency = [f/c for f, c in zip(perf["fitness"], perf["compute"])]
                plt.plot(efficiency, label=variant)
        plt.title("Compute Efficiency")
        plt.xlabel("Iteration")
        plt.ylabel("Fitness / Compute")
        plt.legend()
        best_variant = max(performance.items(), key=lambda x: max(x[1]["fitness"]) if x[1]["fitness"] else -float('inf'))[0]
        plt.subplot(2, 2, 4)
        metrics = performance[best_variant]["metrics"]
        if metrics:
            for metric in set().union(*[m.keys() for m in metrics]):
                values = [m.get(metric, 0) for m in metrics]
                plt.plot(values, label=metric)
            plt.title(f"Metrics for {best_variant}")
            plt.xlabel("Iteration")
            plt.ylabel("Value")
            plt.legend()
        plt.tight_layout()
        plt.savefig(os.path.join(self.config.output_dir, "performance_analysis.png"))
        plt.close()
        self.logger.info(f"Analysis saved to {self.config.output_dir}/performance_analysis.png")
