class HarmonicAudioProcessor:
    def __init__(self, config):
        self.config = config
        self.model = whisperx.load_model("tiny", device=config.device, compute_type="float32")
        self.faction_freqs = torch.tensor([f["freq"] for f in config.factions], dtype=dtype, device=config.device)
        self.weights = nn.Parameter(torch.ones(len(config.factions), dtype=dtype, device=config.device))
        self.sigma = nn.Parameter(torch.tensor(0.1, dtype=dtype, device=config.device))

    def process(self, audio: np.ndarray, t: float) -> Tuple[torch.Tensor, str, float]:
        result = self.model.transcribe(audio, batch_size=16)
        text = " ".join(seg["text"] for seg in result["segments"]) if result["segments"] else ""
        stft = librosa.stft(audio, n_fft=self.config.n_fft)
        magnitudes = np.abs(stft)
        freqs = torch.linspace(200, 4000, self.config.d_model // 2, device=self.config.device)
        times = torch.linspace(0, 0.025, self.config.d_model // 2, device=self.config.device)
        strf = torch.exp(-((times[:, None] - 0.0125) ** 2) / (2 * 0.005 ** 2)) * torch.cos(2 * np.pi * freqs[None, :] * times[:, None])
        audio_tensor = torch.tensor(magnitudes.flatten()[:self.config.d_model // 2], dtype=dtype, device=self.config.device)
        strf_features = strf @ audio_tensor * torch.randn(self.config.d_model // 2, device=self.config.device)
        fractal_factor = torch.pow(torch.abs(strf_features), 1.0 / self.config.fractal_dimension)
        pulse = torch.tanh(
            self.config.G * torch.sin(t) +
            self.config.h * torch.cos(t * self.config.cosmic_frequency) +
            self.config.c * torch.sin(t / self.config.cosmic_frequency)
        )
        cosmic_mod = torch.sin(2 * np.pi * self.config.cosmic_frequency * t) * pulse
        strf_features = strf_features * fractal_factor * cosmic_mod
        t_vec = torch.linspace(0, len(audio)/self.config.sample_rate, self.config.d_model // 2, device=self.config.device)
        modulation = torch.sum(torch.stack([self.weights[i] * torch.cos(2 * np.pi * f * t_vec) for i, f in enumerate(self.faction_freqs)]), dim=0)
        ohc = strf_features * modulation
        pitch = librosa.yin(audio, fmin=50, fmax=200).mean()
        amplitude = np.abs(audio).mean()
        centroid = librosa.feature.spectral_centroid(y=audio, sr=self.config.sample_rate).mean()
        vibe = 0.4 * (pitch / 200) + 0.3 * (amplitude / np.max(np.abs(audio))) + 0.3 * (centroid / 4000)
        return ohc[:self.config.d_model], text, vibe

class ResonantTriadModule(nn.Module):
    def __init__(self, config):
        super().__init__()
        self.config = config
        self.S = torch.ones(1, dtype=dtype, device=config.device)
        self.P = torch.ones(1, dtype=dtype, device=config.device)
        self.B = torch.ones(1, dtype=dtype, device=config.device)
        self.gamma = nn.Parameter(torch.tensor(0.1, dtype=dtype, device=config.device))
        self.eta = nn.Parameter(torch.tensor(0.1, dtype=dtype, device=config.device))
        self.delta = nn.Parameter(torch.tensor(0.05, dtype=dtype, device=config.device))
        self.r = torch.tensor(0.1, dtype=dtype, device=config.device)
        self.d = torch.tensor(0.05, dtype=dtype, device=config.device)
        self.beta = torch.tensor(0.1, dtype=dtype, device=config.device)
        self.kappa = torch.tensor(0.05, dtype=dtype, device=config.device)
        self.alpha = torch.tensor(0.1, dtype=dtype, device=config.device)
        self.Ks = torch.tensor(1.0, dtype=dtype, device=config.device)
        self.Kp = torch.tensor(1.0, dtype=dtype, device=config.device)
        self.sigma = nn.Parameter(torch.tensor(0.1, dtype=dtype, device=config.device))

    def update_triad(self, S, P, B, t, dt, vibe):
        pulse = torch.tanh(
            self.config.G * torch.sin(t) +
            self.config.h * torch.cos(t * self.config.cosmic_frequency) +
            self.config.c * torch.sin(t / self.config.cosmic_frequency)
        )
        freq = self.config.factions[int(t*10)%len(self.config.factions)]['freq']
        modulation = torch.cos(2 * np.pi * freq * t) * pulse
        E = torch.tensor(1.0, dtype=dtype, device=self.config.device)
        R = torch.tensor(1.0, dtype=dtype, device=self.config.device)
        Qr = torch.tensor(0.95, dtype=dtype, device=self.config.device)
        H = (S * E * R) / (1 + S)
        dS_dt = self.gamma * S * (1 - S/self.Ks) + self.eta * Qr * R * S / (1 + S) - self.delta * S
        dP_dt = self.r * P * (1 - P/self.Kp) + 0.01 * torch.randn(1, dtype=dtype, device=self.config.device) - self.d * P
        dB_dt = self.beta * H - self.kappa * B
        synergy = self.alpha * S * P * B
        S_new = S + dt * (dS_dt + synergy) * modulation
        P_new = P + dt * (dP_dt + synergy) * modulation
        B_new = B + dt * (dB_dt + synergy) * modulation
        return S_new, P_new, B_new

    def forward(self, t, dt, vibe, phi):
        S_new, P_new, B_new = self.update_triad(self.S, self.P, self.B, t, dt, vibe)
        self.S.data = S_new
        self.P.data = P_new
        self.B.data = B_new
        Psi = (self.S + self.P + self.B) / 3
        return {'Psi': Psi.item()}

class GameTheoryModule(nn.Module):
    def __init__(self, config):
        super().__init__()
        self.config = config
        self.strategy_network = nn.Linear(config.d_model, len(config.strategy_types), dtype=dtype)
        self.payoff_matrix = nn.Parameter(torch.tensor([
            [0, -2, 1, 0, 2], [2, 0, -1, 1, 0], [-1, 1, 0, -2, 1], [0, -1, 2, 0, -1], [-2, 0, -1, 1, 0]
        ], dtype=dtype, device=device))
        self.opponent_history = []
        self.position_network = nn.Linear(config.d_model, 1, dtype=dtype)

    def compute_nash_equilibrium(self, opponent_probs: torch.Tensor) -> torch.Tensor:
        expected_payoffs = torch.matmul(self.payoff_matrix, opponent_probs)
        probs = torch.softmax(-expected_payoffs, dim=0)
        return probs / probs.sum()

    def apply_iesds(self, payoff_matrix: torch.Tensor) -> torch.Tensor:
        dominated = torch.ones(payoff_matrix.shape[0], dtype=torch.bool, device=device)
        for i in range(payoff_matrix.shape[0]):
            for j in range(i + 1, payoff_matrix.shape[0]):
                if torch.all(payoff_matrix[i] > payoff_matrix[j]):
                    dominated[j] = False
        return dominated

    def hotelling_position(self, inputs: torch.Tensor) -> float:
        return torch.sigmoid(self.position_network(inputs)).item() * 100

    def strategize(self, inputs: torch.Tensor, opponent_dna: str, phi: float, feedback_score: float) -> Tuple[torch.Tensor, float]:
        opponent_moves = [m for m in self.opponent_history if m['dna'] == opponent_dna]
        opponent_probs = torch.ones(len(self.config.strategy_types), device=device) / len(self.config.strategy_types)
        if opponent_moves:
            counts = torch.zeros(len(self.config.strategy_types), device=device)
            for move in opponent_moves:
                counts[move['strategy']] += 1
            opponent_probs = counts / counts.sum()
        strategy_probs = self.compute_nash_equilibrium(opponent_probs)
        strategy_output = self.strategy_network(inputs) * strategy_probs
        valid_strategies = self.apply_iesds(self.payoff_matrix)
        strategy_output = strategy_output * valid_strategies.float()
        position = self.hotelling_position(inputs)
        chosen_strategy = torch.argmax(strategy_output).item()
        self.opponent_history.append({'dna': opponent_dna, 'strategy': chosen_strategy})
        return strategy_output, position

class PersonaModule(nn.Module):
    def __init__(self, config):
        super().__init__()
        self.config = config
        self.emotion_network = nn.Linear(config.d_model, len(config.emotional_dimensions), dtype=dtype)
        self.lstm = nn.LSTM(
            input_size=config.d_model + len(config.emotional_dimensions),
            hidden_size=64,
            num_layers=2,
            batch_first=True,
            dtype=dtype
        )
        self.emotional_state = nn.Parameter(torch.randn(config.N, len(config.emotional_dimensions), dtype=dtype) * 0.01)
        self.persona_evolution_rate = 0.05
        self.sigma = nn.Parameter(torch.tensor(0.1, dtype=dtype, device=config.device))

    def update_persona(self, dna: str, interaction_data: Dict, vibe: float, strategy_probs: torch.Tensor, agent_idx: int, phi: float, t: float):
        audio_features = interaction_data.get('audio_features', torch.zeros(self.config.d_model, device=self.config.device))
        emotion_input = torch.cat([audio_features, torch.tensor([vibe], dtype=dtype, device=self.config.device)])
        emotion_output = self.emotion_network(emotion_input)
        lstm_input = torch.cat([emotion_input, emotion_output]).unsqueeze(0).unsqueeze(0)
        lstm_output, _ = self.lstm(lstm_input)
        memory_influence = lstm_output.squeeze().mean()
        pulse = torch.tanh(
            self.config.G * torch.sin(t) +
            self.config.h * torch.cos(t * self.config.cosmic_frequency) +
            self.config.c * torch.sin(t / self.config.cosmic_frequency)
        )
        cosmic_mod = torch.sin(2 * np.pi * self.config.cosmic_frequency * t) * pulse
        self.emotional_state.data[agent_idx] = self.emotional_state[agent_idx] + self.persona_evolution_rate * (emotion_output + memory_influence + 0.05 * cosmic_mod)
        logger.info(f"Updated persona for agent {agent_idx}: Emotional state {self.emotional_state[agent_idx].mean().item():.4f}")

    def express_persona(self, inputs: torch.Tensor, t: float, agent_idx: int) -> torch.Tensor:
        emotion_output = self.emotion_network(inputs)
        persona_output = emotion_output * self.emotional_state[agent_idx].sum()
        pulse = torch.tanh(
            self.config.G * torch.sin(t) +
            self.config.h * torch.cos(t * self.config.cosmic_frequency) +
            self.config.c * torch.sin(t / self.config.cosmic_frequency)
        )
        temporal_modulation = torch.cos(2 * np.pi * 40 * t) * pulse
        return torch.softmax(persona_output + 0.1 * temporal_modulation, dim=-1)

class VAE(nn.Module):
    def __init__(self, input_dim=64, latent_dim=32):
        super().__init__()
        self.encoder = nn.Sequential(
            nn.Linear(input_dim, 128, dtype=dtype),
            nn.ReLU(),
            nn.Linear(128, latent_dim * 2, dtype=dtype)
        )
        self.decoder = nn.Sequential(
            nn.Linear(latent_dim, 128, dtype=dtype),
            nn.ReLU(),
            nn.Linear(128, input_dim, dtype=dtype),
            nn.Sigmoid()
        )

    def forward(self, x):
        mu_logvar = self.encoder(x)
        mu, logvar = mu_logvar[:, :mu_logvar.shape[1]//2], mu_logvar[:, mu_logvar.shape[1]//2:]
        z = mu + torch.exp(0.5 * logvar) * torch.randn_like(mu).to(x.device)
        recon = self.decoder(z)
        return z, recon, mu, logvar

class QGAN(nn.Module):
    def __init__(self, config):
        super().__init__()
        self.config = config
        self.generator = nn.Sequential(
            nn.Linear(32, 128, dtype=dtype),
            nn.ReLU(),
            nn.Linear(128, config.d_model, dtype=dtype)
        ).to(config.device)
        self.discriminator = nn.Sequential(
            nn.Linear(config.d_model, 128, dtype=dtype),
            nn.ReLU(),
            nn.Linear(128, 1, dtype=dtype),
            nn.Sigmoid()
        ).to(config.device)

    def forward(self, z):
        fake_data = self.generator(z)
        fake_prob = self.discriminator(fake_data)
        return fake_data, fake_prob

    def train_step(self, real_data, optimizer_g, optimizer_d):
        batch_size = real_data.shape[0]
        z = torch.randn(batch_size, 32, dtype=dtype, device=self.config.device)
        fake_data, fake_prob = self.forward(z)
        real_prob = self.discriminator(real_data)
        loss_d = -torch.mean(torch.log(real_prob + 1e-10) + torch.log(1 - fake_prob + 1e-10))
        loss_g = -torch.mean(torch.log(fake_prob + 1e-10))
        optimizer_d.zero_grad()
        loss_d.backward(retain_graph=True)
        optimizer_d.step()
        optimizer_g.zero_grad()
        loss_g.backward()
        optimizer_g.step()
        return loss_d.item(), loss_g.item()

class ChatbotPSystem:
    def __init__(self, config):
        self.config = config
        self.membranes = ['A1', 'Belt', 'Parabelt']
        self.rules = [
            {'rule': 'audio -> features | HarmonicAudioProcessor', 'update': 'audio'},
            {'rule': 'features -> Psi | ResonantTriadModule', 'update': 'triad'},
            {'rule': 'features -> spikes | MetaSentienceController', 'update': 'neural'},
            {'rule': 'features -> strategy | GameTheoryModule', 'update': 'game'},
            {'rule': 'features -> persona | PersonaModule', 'update': 'persona'},
            {'rule': 'features -> memory | FractalMemoryLattice', 'update': 'memory'},
            {'rule': 'response -> feedback | FeedbackAssessor', 'update': 'feedback'}
        ]
        self.sigma = nn.Parameter(torch.tensor(0.1, dtype=dtype, device=config.device))

    def apply_rules(self, chatbot, audio, text, user_response, user_rating, t, dt):
        pulse = torch.tanh(
            self.config.G * torch.sin(t) +
            self.config.h * torch.cos(t * self.config.cosmic_frequency) +
            self.config.c * torch.sin(t / self.config.cosmic_frequency)
        )
        freq = self.config.factions[int(t*10)%len(self.config.factions)]['freq']
        modulation = torch.cos(2 * np.pi * freq * t) * pulse
        updates = {}
        for rule in self.rules:
            update_type = rule['update']
            if update_type == 'audio':
                updates['audio'] = chatbot.audio_processor.process(audio, t) if audio is not None else (torch.zeros(chatbot.config.d_model, device=chatbot.config.device), "", 0.5)
            elif update_type == 'triad':
                audio_features, _, vibe = updates['audio']
                updates['triad'] = chatbot.triad(t, dt, vibe, chatbot.meta_controller(audio_features.view(chatbot.config.N, -1), t, dt)['phi'])
            elif update_type == 'neural':
                audio_features, _, _ = updates['audio']
                updates['neural'] = chatbot.meta_controller(audio_features.view(chatbot.config.N, -1), t, dt)
            elif update_type == 'game':
                audio_features, _, _ = updates['audio']
                updates['game'] = chatbot.game_theory.strategize(
                    audio_features, chatbot.user_dna, updates['neural']['phi'], updates.get('feedback', {'feedback_score': 0.5})['feedback_score']
                )
            elif update_type == 'persona':
                audio_features, _, vibe = updates['audio']
                chatbot.persona.update_persona(
                    chatbot.user_dna, {'audio_features': audio_features, 'strategy_probs': updates['game'][0], 'feedback_score': updates.get('feedback', {'feedback_score': 0.5})['feedback_score']},
                    vibe, updates['game'][0], 0, updates['neural']['phi'], t
                )
                updates['persona'] = chatbot.persona.express_persona(audio_features, t, 0)
            elif update_type == 'memory':
                audio_features, transcribed_text, vibe = updates['audio']
                response_text, _, _, response_embedding = chatbot.generate_response(
                    transcribed_text or text, audio_features, vibe, t, updates['neural']
                )
                interaction = {'text': transcribed_text or text, 'response': response_text}
                metrics = {'phi': updates['neural']['phi'], 'coherence': updates['neural'].get('coherence', 0.5)}
                updates['memory'] = chatbot.memory_lattice.add_memory(
                    interaction, metrics, t, updates.get('feedback', {'feedback_score': 0.5})['feedback_score']
                )
            elif update_type == 'feedback':
                audio_features, transcribed_text, vibe = updates['audio']
                response_text, _, _, response_embedding = chatbot.generate_response(
                    transcribed_text or text, audio_features, vibe, t, updates['neural']
                )
                user_response_embedding = torch.mean(
                    chatbot.language_model(**chatbot.tokenizer(user_response or response_text, return_tensors="pt").to(self.config.device), output_hidden_states=True).hidden_states[-1], dim=1
                )
                state = {'vibe': vibe, 'coherence': updates['neural'].get('coherence', 0.5), 'phi': updates['neural']['phi']}
                action = torch.argmax(updates['game'][0]).item()
                updates['feedback'] = chatbot.feedback_assessor.assess(
                    response_embedding, user_response_embedding, user_rating or chatbot.collect_user_rating(), state, action, state
                )
        return updates

    def collect_user_rating(self):

    def generate_response(self, text: str, audio_features: torch.Tensor, vibe: float, t: float, meta_metrics: Dict) -> Tuple[str, np.ndarray, Dict, torch.Tensor]:

        self.persona.update_persona(user_dna, {'audio_features': audio_features, 'strategy_probs': strategy_probs, 'feedback_score': meta_metrics['feedback_score']}, thermal_state, strategy_probs, 0, meta_metrics['phi'], t)

        recent_interactions = self.memory["interactions"][-5:]

                keywords.extend([t for t in tokens if len(t) > 3])
            common_keyword = max(set(keywords), key=keywords.count, default="dialogue") if keywords else "dialogue"

            faction = self.config.factions[faction_idx]

            response_text = f"With {theme}, I sense {emotion} in our {common_keyword}: {response_text}"

    def forward(self, audio: Optional[np.ndarray] = None, text: Optional[str] = None, user_response: Optional[str] = None, user_rating: Optional[int] = None) -> Dict:
        updates = self.p_system.apply_rules(self, audio, text, user_response, user_rating, self.t, self.config.dt)
        audio_features, transcribed_text, vibe = updates['audio']
        input_text = text if text else transcribed_text
        meta_metrics = updates['neural']
        triad_metrics = updates['triad']
        response_text, response_audio, art, response_embedding = self.generate_response(input_text, audio_features, vibe, self.t, meta_metrics)
        phi = self.memory_lattice.compute_phi()
        d_f = self.memory_lattice.compute_fractal_dimension(audio_features.unsqueeze(0))
        user_rating = user_rating or self.collect_user_rating()
        feedback_score = 0.6 * vibe + 0.4 * (user_rating / 5.0)
        state = {'vibe': vibe, 'coherence': meta_metrics['coherence'], 'phi': phi}
        action = torch.argmax(updates['game'][0]).item()
        next_state = state
        user_response_embedding = torch.mean(self.language_model(**self.tokenizer(user_response or response_text, return_tensors="pt").to(self.config.device), output_hidden_states=True).hidden_states[-1], dim=1)
        feedback_loss, reward = updates['feedback']
        metrics = {
            'phi': phi, 'd_f': d_f, 'coherence': meta_metrics['coherence'], 'vibe': vibe,
            'Psi': triad_metrics['Psi'], 'timestamp': self.t, 'feedback_score': feedback_score, 'reward': reward
        }
        interaction = {
            'text': input_text, 'response': response_text, 'valence': vibe,
            'timestamp': self.t, 'feedback_score': feedback_score
        }
        self.memory_lattice.add_memory(interaction, metrics, self.t, feedback_score)
        self.memory["interactions"].append(interaction)
        if len(self.memory["interactions"]) > 1000:
            self.memory["interactions"].pop(0)
        self.save_memory()
        for plugin in self.plugins.values():
            plugin.update(self, metrics)
        self.t += self.config.dt
        return {
            "text": input_text, "response": response_text, "transcribed_text": transcribed_text,
            "audio": response_audio, "metrics": metrics, "art": art, "insight": response_text
        }

    def train_step(self, audio: np.ndarray, text: str, target_response: str, user_rating: int) -> float:
        self.optimizer.zero_grad()
        self.feedback_optimizer.zero_grad()
        audio_features, transcribed_text, vibe = self.audio_processor.process(audio, self.t)
        meta_metrics = self.meta_controller(audio_features.view(self.config.N, -1), self.t, self.config.dt)
        triad_metrics = self.triad(self.t, self.config.dt, vibe, meta_metrics['phi'])
        inputs = self.tokenizer(text, return_tensors="pt", padding=True, truncation=True, max_length=512).to(self.config.device)
        outputs = self.language_model(**inputs, output_hidden_states=True)
        hidden_states = outputs.hidden_states[-1]
        text_embedding = torch.mean(hidden_states, dim=1)
        target_inputs = self.tokenizer(target_response, return_tensors="pt", padding=True, truncation=True, max_length=512).to(self.config.device)
        target_ids = target_inputs["input_ids"]
        z = torch.randn(1, 32, dtype=dtype, device=self.config.device)
        fake_data, _ = self.qgan(z)
        response_logits = self.lm_head(text_embedding + fake_data) * (1 + meta_metrics['quantum_state'].sum())
        loss_lm = F.cross_entropy(response_logits.view(-1, self.tokenizer.vocab_size), target_ids.view(-1))
        z_vae, recon, mu, logvar = self.vae(audio_features.unsqueeze(0))
        loss_vae = F.binary_cross_entropy(recon.squeeze(0), audio_features, reduction='sum') - 0.5 * torch.sum(1 + logvar - mu.pow(2) - logvar.exp())
        loss_qgan_d, loss_qgan_g = self.qgan.train_step(audio_features.unsqueeze(0), self.optimizer_g, self.optimizer_d)
        feedback_score = 0.6 * vibe + 0.4 * (user_rating / 5.0)
        state = {'vibe': vibe, 'coherence': meta_metrics['coherence'], 'phi': meta_metrics['phi']}
        action = torch.argmax(self.game_theory.strategize(audio_features, '', meta_metrics['phi'], feedback_score)[0]).item()
        response_embedding = text_embedding
        user_response_embedding = torch.mean(self.language_model(**self.tokenizer(target_response, return_tensors="pt").to(self.config.device), output_hidden_states=True).hidden_states[-1], dim=1)
        feedback_loss, reward = self.feedback_assessor.assess(response_embedding, user_response_embedding, user_rating, state, action, state)
        total_loss = loss_lm + 0.1 * loss_vae + 0.05 * (loss_qgan_d + loss_qgan_g) + 0.2 * feedback_loss
        total_loss.backward()
        self.optimizer.step()
        self.feedback_optimizer.step()
        self.t += self.config.dt
        return total_loss.item()

    def save_checkpoint(self) -> str:
        checkpoint_path = os.path.join(self.config.checkpoint_dir, f"chatbot_{int(self.t*1000)}.pt")
        torch.save({
            "model_state_dict": self.state_dict(),
            "optimizer_state_dict": self.optimizer.state_dict(),
            "t": self.t,
        }, checkpoint_path)
        logger.info(f"Saved checkpoint to {checkpoint_path}")
        return checkpoint_path

    def load_checkpoint(self, checkpoint_path: str):

import torch
import torch.nn as nn
import numpy as np
import xgboost as xgb
from typing import Dict
import logging

logger = logging.getLogger("FeedbackAssessor")

class FeedbackAssessor:
    def __init__(self, config):
        self.config = config
        self.lstm = nn.LSTM(config.d_model, 128, batch_first=True).to(config.device)
        self.lstm_output = nn.Linear(128, config.d_model).to(config.device)
        self.xgb = xgb.XGBClassifier(n_estimators=50, max_depth=3)
        self.q_table = {}
        self.rating_history = []
        self.weights = nn.Parameter(torch.tensor([0.6, 0.4], dtype=dtype, device=config.device))
        self.sigma = nn.Parameter(torch.tensor(0.1, dtype=dtype, device=config.device))

    def predict_response(self, system_output, history):
        inputs = torch.stack([h['response_embedding'] for h in history[-5:]] + [system_output], dim=1)
        lstm_out, _ = self.lstm(inputs)
        predicted = self.lstm_output(lstm_out[:, -1, :])
        return predicted

    def train_lstm(self, system_output, user_response, optimizer):
        predicted = self.predict_response(system_output, self.rating_history)
        loss = nn.MSELoss()(predicted, user_response)
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
        return loss.item()

    def train_xgb(self, features, rating):
        X = np.array([[f['vibe'], f['coherence'], f['phi']] for f in features])
        y = np.array([rating - 1])
        self.xgb.fit(X, y)
        return self.xgb.predict_proba(X)[0]

    def update_q_table(self, state, action, reward, next_state):
        state_key = tuple(state.values())
        if state_key not in self.q_table:
            self.q_table[state_key] = torch.zeros(5, device=self.config.device)
        self.q_table[state_key][action] += 0.1 * (reward + 0.9 * torch.max(self.q_table.get(tuple(next_state.values()), torch.zeros(5, device=self.config.device))) - self.q_table[state_key][action])

    def compute_loss(self, system_output, user_response, rating, state, action, next_state, optimizer):
        mse_loss = self.train_lstm(system_output, user_response, optimizer)
        xgb_probs = self.train_xgb(self.rating_history, rating)
        ce_loss = -np.log(xgb_probs[rating-1] + 1e-10)
        reward = rating / 5.0 + user_response.norm().item() / 100.0
        self.update_q_table(state, action, reward, next_state)
        phi = state['phi']
        w_1 = 0.4 if rating > 3 else 0.2
        w_2 = 0.3 * (phi / self.config.target_phi)
        w_3 = 0.3 * (1 - state['coherence'])
        q_loss = 1 - self.q_table.get(tuple(state.values()), torch.zeros(5, device=self.config.device))[action]
        return w_1 * mse_loss + w_2 * ce_loss + w_3 * q_loss, reward

    def assess(self, system_output, user_response, rating, state, action, next_state):
        optimizer = torch.optim.Adam([p for p in self.lstm.parameters()] + [p for p in self.lstm_output.parameters()], lr=0.001)
        loss, reward = self.compute_loss(system_output, user_response, rating, state, action, next_state, optimizer)
        self.rating_history.append({'response_embedding': system_output, 'vibe': state['vibe'], 'coherence': state['coherence'], 'phi': state['phi']})
        logger.info(f"Feedback assessed: Loss={loss:.4f}, Reward={reward:.4f}")
        return loss, reward
