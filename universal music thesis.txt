## I. Theoretical Framework

### A. Tymoczko's Five Features of Tonality as Translation Pillars

**1. Conjunct Melodic Motion**
```
Mathematical Mapping: M₁(v) = (1 - chaos_level) × speed_normalization
Where: speed_normalization = min(movement_speed/2.0, 1.0)
```
- **Rationale**: Visual continuity correlates with melodic smoothness
- **Range**: [0,1] where 1 = pure stepwise motion
- **Application**: Controls interval size in melodic generation

**2. Harmonic Consistency** 
```
Mathematical Mapping: M₂(v) = geometric_alignment
Direct mapping from visual geometric coherence
```
- **Rationale**: Visual geometric alignment reflects harmonic consonance
- **Range**: [0,1] where 1 = perfect consonance (major triads)
- **Application**: Determines chord quality and dissonance level

**3. Voice-Leading Efficiency**
```
Mathematical Mapping: M₃(v) = (geometric_alignment + intensity_stability)/2
Where: intensity_stability = 1 - |intensity - 0.5| × 2
```
- **Rationale**: Smooth visual transitions require efficient voice leading
- **Application**: Minimizes voice-leading distances between chords

**4. Pitch Circulation**
```
Mathematical Mapping: M₄(v) = hue_to_fifths_transformation(color_hue)
Circle of Fifths: hue° → note = fifths_wheel[⌊hue/30⌋ × 30]
```
- **Rationale**: Color wheel naturally maps to circle of fifths
- **Application**: Determines tonal centers and modulation paths

**5. Chord Spacing**
```
Mathematical Mapping: M₅(v) = density_to_voicing_function(particle_density)
```
- **Rationale**: Visual density correlates with harmonic density
- **Application**: Controls chord voicing spread and register distribution

```
For k pulses distributed over n steps:
1. Initialize array of length n with all false
2. For each step i:
   - bucket += k
   - if bucket ≥ n:
     - bucket -= n
     - array[i] = true (pulse)
```

// Add Web MIDI API integration for real-time playback
class MIDIOutputEngine {
    constructor() {
        this.midiOutput = null;
        this.initializeMIDI();
    }
    
    async initializeMIDI() {
        if (navigator.requestMIDIAccess) {
            const midiAccess = await navigator.requestMIDIAccess();
            const outputs = Array.from(midiAccess.outputs.values());
            this.midiOutput = outputs[0]; // Use first available output
        }
    }
    
    playScore(musicalScore) {
        if (!this.midiOutput) return;
        
        // Convert score to MIDI and play
        const midiEvents = this.convertScoreToMIDI(musicalScore);
        this.sendMIDIEvents(midiEvents);
    }
    
    convertScoreToMIDI(score) {
        const events = [];
        let time = 0;
        
        // Play chord
        const chord = score.harmony.chord;
        const chordNotes = this.getChordNotes(chord);
        
        chordNotes.forEach(note => {
            events.push({
                type: 'noteOn',
                note: note,
                velocity: this.dynamicsToVelocity(score.instrumentation.dynamics),
                time: time
            });
        });
        
        // Play melody/improvisation
        score.melody.improvisation.forEach((note, index) => {
            events.push({
                type: 'noteOn',
                note: note.note,
                velocity: note.velocity,
                time: time + (index * 0.25) // Quarter note spacing
            });
        });
        
        return events;
    }
    
    sendMIDIEvents(events) {
        events.forEach(event => {
            setTimeout(() => {
                if (event.type === 'noteOn') {
                    this.midiOutput.send([0x90, event.note, event.velocity]);
                    // Schedule note off
                    setTimeout(() => {
                        this.midiOutput.send([0x80, event.note, 0]);
                    }, 500); // Half second notes
                }
            }, event.time * 1000);
        });
    }
}

class AdvancedTrainingDataCollector {
    constructor() {
        this.sessions = [];
        this.currentSession = null;
    }
    
    startSession(sessionName) {
        this.currentSession = {
            name: sessionName,
            startTime: performance.now(),
            data: [],
            metadata: {
                browser: navigator.userAgent,
                screenSize: `${window.innerWidth}x${window.innerHeight}`,
                timestamp: new Date().toISOString()
            }
        };
    }
    
    recordDataPoint(visualState, musicalScore, userFeedback = null) {
        if (!this.currentSession) return;
        
        const dataPoint = {
            timestamp: performance.now() - this.currentSession.startTime,
            
            // Input: Visual features
            visual: {
                geometric_alignment: visualState.geometric_alignment,
                movement_speed: visualState.movement_speed,
                chaos_level: visualState.chaos_level,
                intensity: visualState.intensity,

                particle_density: visualState.particle_density,
                scene_index: visualState.scene_index
            },
            
            // Output: Musical features  
            musical: {
                // Harmony
                chord_root: musicalScore.harmony.chord.root,
                chord_quality: musicalScore.harmony.chord.quality,
                chord_extensions: musicalScore.harmony.chord.extensions,
                
                // Rhythm
                tempo: musicalScore.metadata.tempo,
                time_signature: musicalScore.metadata.timeSignature,
                rhythm_pattern: musicalScore.rhythm.pattern,
                
                // Melody
                improvisation_notes: musicalScore.melody.improvisation.map(n => ({
                    note: n.note,
                    duration: n.duration,
                    velocity: n.velocity
                })),
                
                // Instrumentation
                primary_instrument: musicalScore.instrumentation.instruments.primary[0],
                dynamics: musicalScore.instrumentation.dynamics,
                style: musicalScore.metadata.style,
                
                // Form
                phrase_length: musicalScore.form.phrase_length,
                cadence_type: musicalScore.form.cadence_type
            },
            
            // Mathematical features for validation
            mathematical: {
                tymoczko_features: {
                    melodic_conjunction: musicalScore.tymoczko?.melodicConjunction || 0,
                    harmonic_consistency: musicalScore.tymoczko?.harmonicConsistency || 0,
                    voice_leading_efficiency: musicalScore.tymoczko?.voiceLeadingEfficiency || 0,
                    pitch_circulation: musicalScore.tymoczko?.pitchCirculation || 0,
                    chord_spacing: musicalScore.tymoczko?.chordSpacing || 0
                },
                consonance_score: this.calculateConsonanceScore(musicalScore.harmony.chord),
                voice_leading_distance: this.calculateVoiceLeadingDistance(musicalScore),
                rhythmic_complexity: this.calculateRhythmicComplexity(musicalScore.rhythm)
            },
            
            // User feedback (if available)
            feedback: userFeedback
        };
        
        this.currentSession.data.push(dataPoint);
    }
    
    endSession() {
        if (!this.currentSession) return null;
        
        this.currentSession.endTime = performance.now();
        this.currentSession.duration = this.currentSession.endTime - this.currentSession.startTime;
        this.sessions.push(this.currentSession);
        
        const sessionData = this.currentSession;
        this.currentSession = null;
        
        return sessionData;
    }
    
    exportForMagenta() {
        return {
            format: "visual_music_training_data_v1.0",
            generator: "UniversalVisualMusicEngine",
            export_timestamp: new Date().toISOString(),
            sessions: this.sessions,
            
            // Aggregate statistics
            statistics: {
                total_sessions: this.sessions.length,
                total_data_points: this.sessions.reduce((sum, s) => sum + s.data.length, 0),
                avg_session_duration: this.sessions.reduce((sum, s) => sum + s.duration, 0) / this.sessions.length,
                style_distribution: this.calculateStyleDistribution(),
                key_distribution: this.calculateKeyDistribution(),
                tempo_range: this.calculateTempoRange()
            }
        };
    }
    
    calculateConsonanceScore(chord) {
        // Implementation of consonance calculation based on harmonic ratios
        const ratios = this.getChordRatios(chord);
        return ratios.reduce((sum, ratio) => {
            const [num, den] = ratio;
            return sum + (1 / (num + den - this.gcd(num, den)));
        }, 0) / ratios.length;
    }
    
    calculateStyleDistribution() {
        const styles = {};
        this.sessions.forEach(session => {
            session.data.forEach(point => {
                const style = point.musical.style;
                styles[style] = (styles[style] || 0) + 1;
            });
        });
        return styles;
    }
}

function downloadJSON(data, filename) {
    const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' });
    downloadBlob(blob, filename);
}

function downloadMIDI(data, filename) {
    const blob = new Blob([data], { type: 'audio/midi' });
    downloadBlob(blob, filename);
}

function downloadXML(data, filename) {
    const blob = new Blob([data], { type: 'application/xml' });
    downloadBlob(blob, filename);
}

function downloadBlob(blob, filename) {
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = filename;
    a.click();
    URL.revokeObjectURL(url);
}

function startTrainingSession() {
    const sessionName = prompt('Enter session name:') || `Session_${Date.now()}`;
    trainingCollector.startSession(sessionName);
    console.log(`Training session '${sessionName}' started`);
}
