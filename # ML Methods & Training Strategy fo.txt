## 🧠 Core ML Architecture Decision Matrix

| Component | Recommended Approach | Alternative | Reasoning |
|-----------|---------------------|-------------|-----------|
| **Base Model** | Gemma 3 4B + LoRAs | Fine-tuned smaller models | Best balance of capability and VRAM efficiency |
| **Visual Understanding** | Built-in multimodal | Separate vision encoder | Unified processing, lower memory |
| **Music Generation** | Custom VAE + TensorFlow.js | MusicVAE | Specialized for our UGF format |

## 🎵 MusicVAE vs Custom Audio ML

### **Option A: Leverage MusicVAE/Magenta.js**

**Pros:**
- Pre-trained on massive music datasets
- Proven architecture for musical creativity
- Direct browser integration
- MIDI output compatibility

**Cons:**
- Not optimized for our UGF audio mapping format
- Limited real-time parameter control
- Focused on note sequences, not parameter mappings

### **Option B: Custom Audio-Visual VAE (Recommended)**

```python
# Custom VAE for UGF Audio Mappings
class UGFAudioMappingVAE(nn.Module):
    def __init__(self):
        super().__init__()
        # Encoder: Audio features → Latent space
        self.audio_encoder = nn.Sequential(
            nn.Linear(256, 128),  # Audio features
            nn.ReLU(),
            nn.Linear(128, 64),
            nn.ReLU(),
            nn.Linear(64, 32)     # Latent dim
        )
        
        # Decoder: Latent space → UGF mappings
        self.mapping_decoder = nn.Sequential(
            nn.Linear(32, 64),
            nn.ReLU(),
            nn.Linear(64, 128),
            nn.ReLU(),
            nn.Linear(128, 256)   # UGF mapping parameters
        )
        
    def forward(self, audio_features):
        latent = self.audio_encoder(audio_features)
        mappings = self.mapping_decoder(latent)
        return mappings, latent
```

**Why Custom is Better:**
- Direct UGF format output
- Real-time parameter optimization
- Visual-audio cross-modal training
- Specialized for generative art (not just music)

### **Proximal Policy Optimization (PPO) for Creative Improvement**

```python
class CreativePPOAgent:
    def __init__(self, base_model):
        self.base_model = base_model
        self.policy_network = CreativePolicyNetwork()
        self.value_network = ValueEstimationNetwork()
        
    def update_policy(self, feedback_batch):
        """Update creative policy based on feedback"""
        
        # Calculate advantages
        advantages = self.calculate_advantages(feedback_batch)
        
        # PPO loss
        policy_loss = self.compute_policy_loss(feedback_batch, advantages)
        value_loss = self.compute_value_loss(feedback_batch)
        
        # Combined loss
        total_loss = policy_loss + 0.5 * value_loss
        
        # Gradient update
        self.optimizer.zero_grad()
        total_loss.backward()
        torch.nn.utils.clip_grad_norm_(self.parameters(), 0.5)
        self.optimizer.step()
        
        return {
            "policy_loss": policy_loss.item(),
            "value_loss": value_loss.item(),
            "total_loss": total_loss.item()
        }
    
    def generate_improved_ugf(self, prompt, previous_feedback=None):
        """Generate UGF schema with policy improvements"""
        
        # Encode current state
        state = self.encode_creative_state(prompt, previous_feedback)
        
        # Sample action from improved policy
        action = self.policy_network.sample_action(state)
        
        # Convert action to UGF parameters
        ugf_parameters = self.action_to_ugf_parameters(action)
        
        # Generate schema
        ugf_schema = self.base_model.generate_ugf(prompt, ugf_parameters)
        
        return ugf_schema, action
```

        # 2. Load most common LoRAs (500MB each)

        for lora_name in common_loras:
            lora = await self.load_lora(lora_name)
            self.models["visual_loras"][lora_name] = lora
            yield f"lora_loaded_{lora_name}"
        
        # 3. Load additional LoRAs on demand
        yield "ready_for_inference"
    
    async def generate_ugf_optimized(self, prompt, style_preference=None):
        """Memory-optimized UGF generation"""
        
        # Select appropriate LoRA
        lora = self.select_optimal_lora(prompt, style_preference)
        
        # Load if not in cache
        if lora not in self.model_cache:
            await self.model_cache.load_lora(lora)

            ugf_schema = self.models["base"].generate_with_lora(
                prompt, 
                lora, 
                max_tokens=512,
                temperature=0.8
            )

        mappings = []
        
        for i, feature in enumerate(style_config["features"]):
            mapping = style_config["mappings"][i % len(style_config["mappings"])]
            
            mapping_def = {

    def _generate_mapping_function(self, feature: str) -> str:
        """Generate realistic mapping functions"""
        import random
        
        functions = {
            "rms": [
                "base_value * (1 + audio_feature * 2.0)",
                "base_value + audio_feature * 0.5",
                "mix(base_value, base_value * 2.0, audio_feature)"
            ],

                "if(audio_feature > 0.8, base_value * 1.5, base_value)",
                "base_value + step(0.7, audio_feature) * 0.3"
            ]
        }
        
        available_functions = functions.get(feature, ["base_value * (1 + audio_feature)"])
        return random.choice(available_functions)

class UGFLoRATrainer:
    """Handles LoRA training for UGF models"""
    
    def __init__(self, config: UGFTrainingConfig):
        self.config = config
        self.model = None
        self.tokenizer = None
        self.peft_model = None
        
        # Setup logging
        logging.basicConfig(level=logging.INFO)
        self.logger = logging.getLogger(__name__)
    
    def setup_model(self):
        """Initialize base model and tokenizer"""
        self.logger.info(f"Loading base model: {self.config.base_model}")
        
        # Load model with 4-bit quantization for 6GB VRAM
        from transformers import BitsAndBytesConfig
        
        quantization_config = BitsAndBytesConfig(
            load_in_4bit=True,
            bnb_4bit_compute_dtype=torch.float16,
            bnb_4bit_use_double_quant=True,
            bnb_4bit_quant_type="nf4"
        )
        
        self.model = AutoModelForCausalLM.from_pretrained(
            self.config.base_model,
            quantization_config=quantization_config,
            device_map="auto",
            torch_dtype=torch.float16,
            trust_remote_code=True
        )
        
        self.tokenizer = AutoTokenizer.from_pretrained(self.config.base_model)
        if self.tokenizer.pad_token is None:
            self.tokenizer.pad_token = self.tokenizer.eos_token
        
        self.logger.info("Base model loaded successfully")
    
    def create_lora_config(self, task_type: str = "CAUSAL_LM") -> LoraConfig:
        """Create LoRA configuration"""
        return LoraConfig(
            task_type=getattr(TaskType, task_type),
            r=self.config.lora_rank,
            lora_alpha=self.config.lora_alpha,
            lora_dropout=self.config.lora_dropout,
            target_modules=self.config.target_modules,
            bias="none",
            inference_mode=False
        )
    
    def prepare_dataset(self, examples: List[Dict]) -> Dataset:
        """Prepare dataset for training"""
        
        def format_example(example):
            """Format example for instruction tuning"""
            instruction = example["input"]
            response = example["output"]
            
            # Format as conversation
            formatted = f"<|user|>\n{instruction}\n<|assistant|>\n{response}<|end|>"
            
            return {"text": formatted}
        
        # Format all examples
        formatted_examples = [format_example(ex) for ex in examples]
        
        # Create dataset
        dataset = Dataset.from_list(formatted_examples)
        
        # Tokenize
        def tokenize_function(examples):
            return self.tokenizer(
                examples["text"],
                truncation=True,
                padding=True,
                max_length=self.config.max_seq_length,
                return_tensors="pt"
            )
        
        tokenized_dataset = dataset.map(tokenize_function, batched=True)
        return tokenized_dataset
    
    async def train_visual_style_lora(self, style_name: str, num_examples: int = 1000):
        """Train a LoRA for specific visual style"""
        
        self.logger.info(f"Training visual style LoRA: {style_name}")
        
        # Generate training data
        data_generator = UGFDatasetGenerator()
        examples = data_generator.generate_visual_style_dataset(style_name, num_examples)
        
        # Prepare dataset
        dataset = self.prepare_dataset(examples)
        
        # Create LoRA model
        lora_config = self.create_lora_config()
        self.peft_model = get_peft_model(self.model, lora_config)
        
        # Training arguments
        training_args = TrainingArguments(
            output_dir=f"{self.config.output_dir}/{style_name}",
            per_device_train_batch_size=self.config.training_batch_size,
            gradient_accumulation_steps=self.config.gradient_accumulation_steps,
            learning_rate=self.config.learning_rate,
            num_train_epochs=self.config.num_train_epochs,
            max_steps=self.config.max_steps,
            warmup_steps=self.config.warmup_steps,
            logging_steps=10,
            save_steps=100,
            fp16=True,
            dataloader_drop_last=True,
            report_to=None  # Disable wandb for now
        )
        
        # Create trainer
        from trl import SFTTrainer

        self.peft_model.save_pretrained(lora_path)
        
        self.logger.info(f"LoRA saved to {lora_path}")

class UGFFeedbackCollector:
    """Collects and processes feedback for continuous learning"""
    
    def __init__(self):
        self.feedback_buffer = []
        self.aesthetic_evaluator = AestheticEvaluator()
        self.performance_monitor = PerformanceMonitor()
    
    def collect_artwork_feedback(self, ugf_schema: Dict, user_interaction: Dict) -> Dict:
        """Collect comprehensive feedback on generated artwork"""
        
        feedback = {
            "timestamp": time.time(),
            "ugf_schema": ugf_schema,
            "user_interaction": user_interaction,
            
            # Automated evaluations
            "aesthetic_score": self.aesthetic_evaluator.evaluate(ugf_schema),
            "technical_quality": self.evaluate_technical_quality(ugf_schema),
            "performance_metrics": self.performance_monitor.get_metrics(),
            
            # User engagement metrics
            "engagement_score": self.calculate_engagement_score(user_interaction),
            "interaction_entropy": self.calculate_interaction_entropy(user_interaction),
            
            # Creative parameter analysis
            "parameter_effectiveness": self.evaluate_parameter_effectiveness(ugf_schema, user_interaction)
        }
        
        self.feedback_buffer.append(feedback)
        return feedback
    
    def calculate_engagement_score(self, user_interaction: Dict) -> float:
        """Calculate user engagement score from interaction data"""
        
        score = 0.0
        
        # Time spent interacting
        interaction_time = user_interaction.get("duration", 0)
        score += min(interaction_time / 300.0, 1.0) * 0.3  # Up to 5 minutes
        
        # Mouse movement activity
        mouse_activity = user_interaction.get("mouse_movement_distance", 0)
        score += min(mouse_activity / 10000.0, 1.0) * 0.2
        
        # Audio interaction
        if user_interaction.get("audio_enabled", False):
            score += 0.2
        
        # Parameter adjustments
        param_changes = user_interaction.get("parameter_changes", 0)
        score += min(param_changes / 10.0, 1.0) * 0.3
        
        return min(score, 1.0)
    
    def calculate_interaction_entropy(self, user_interaction: Dict) -> float:
        """Calculate entropy of user interactions (higher = more exploratory)"""
        
        actions = user_interaction.get("actions", [])
        if not actions:
            return 0.0
        
        # Count action types
        action_counts = {}
        for action in actions:
            action_type = action.get("type", "unknown")
            action_counts[action_type] = action_counts.get(action_type, 0) + 1
        
        # Calculate entropy
        total_actions = len(actions)
        entropy = 0.0
        
        for count in action_counts.values():
            p = count / total_actions
            if p > 0:
                entropy -= p * np.log2(p)
        
        return entropy
    
    def evaluate_parameter_effectiveness(self, ugf_schema: Dict, user_interaction: Dict) -> Dict:
        """Evaluate how effective different parameters were"""
        
        creative_dna = ugf_schema.get("creative_dna", {})
        engagement = self.calculate_engagement_score(user_interaction)
        
        # Evaluate each parameter's contribution to engagement
        effectiveness = {}
        
        for param, value in creative_dna.items():
            # Heuristic: higher engagement with more extreme values suggests effectiveness
            param_extremeness = abs(value - 0.5) * 2  # 0 to 1
            effectiveness[param] = param_extremeness * engagement
        
        return effectiveness

    def evaluate(self, ugf_schema: Dict) -> float:

        # to evaluate visual aesthetics, color harmony, composition, etc.
        
        creative_dna = ugf_schema.get("creative_dna", {})

        score = 0.0
        
        # Complexity balance (not too simple, not too chaotic)
        complexity = creative_dna.get("complexity", 0.5)
        complexity_score = 1.0 - abs(complexity - 0.7) * 2  # Prefer ~0.7 complexity
        score += complexity_score * 0.3
        
        # Energy appropriateness
        energy = creative_dna.get("energy", 0.5)
        energy_score = 1.0 - abs(energy - 0.6) * 2  # Prefer moderate-high energy
        score += energy_score * 0.2
        
        # System diversity
        system_diversity = len(set(s.get("type", "") for s in systems)) / 3.0
        score += system_diversity * 0.3
        
        # Chaos control (some chaos is good, too much is bad)
        chaos = creative_dna.get("chaos", 0.3)
        chaos_score = 1.0 - max(0, chaos - 0.6) * 2  # Penalize excessive chaos
        score += chaos_score * 0.2
        
        return max(0.0, min(1.0, score))

class ContinuousLearningManager:
    """Manages continuous learning from user feedback"""
    
    def __init__(self, trainer: UGFLoRATrainer):
        self.trainer = trainer
        self.feedback_collector = UGFFeedbackCollector()
        self.learning_schedule = ContinuousLearningSchedule()
        self.is_running = False
    
    async def start_continuous_learning(self):
        """Start the continuous learning loop"""
        
        self.is_running = True
        self.logger = logging.getLogger(__name__)
        self.logger.info("Starting continuous learning system...")
        
        while self.is_running:
            try:
                # Collect recent feedback
                recent_feedback = self.get_recent_feedback()
                
                if len(recent_feedback) >= 32:  # Minimum batch size
                    # Analyze feedback for learning opportunities
                    learning_analysis = self.analyze_learning_opportunities(recent_feedback)
                    
                    # Execute learning updates
                    await self.execute_learning_updates(learning_analysis)
                    
                    # Update learning schedule
                    self.learning_schedule.update_schedule(learning_analysis)
                
                # Wait before next iteration
                await asyncio.sleep(self.learning_schedule.get_next_interval())
                
            except Exception as e:
                self.logger.error(f"Error in continuous learning loop: {e}")
                await asyncio.sleep(60)  # Wait before retrying
    
    def get_recent_feedback(self) -> List[Dict]:
        """Get recent feedback for learning"""
        
        current_time = time.time()
        recent_feedback = []
        
        for feedback in self.feedback_collector.feedback_buffer:
            if current_time - feedback["timestamp"] < 3600:  # Last hour
                recent_feedback.append(feedback)
        
        return recent_feedback
    
    def analyze_learning_opportunities(self, feedback_batch: List[Dict]) -> Dict:
        """Analyze feedback to identify learning opportunities"""
        
        analysis = {
            "needs_visual_update": False,
            "needs_audio_update": False,
            "underperforming_styles": [],
            "high_performing_patterns": [],
            "performance_issues": []
        }
        
        # Analyze aesthetic scores by style
        style_scores = {}
        for feedback in feedback_batch:
            ugf_schema = feedback["ugf_schema"]
            aesthetic_score = feedback["aesthetic_score"]
            
            # Extract style indicators from schema
            style = self.extract_style_from_schema(ugf_schema)
            if style not in style_scores:
                style_scores[style] = []
            style_scores[style].append(aesthetic_score)
        
        # Identify underperforming styles
        for style, scores in style_scores.items():
            avg_score = np.mean(scores)
            if avg_score < 0.6:  # Threshold for update
                analysis["underperforming_styles"].append(style)
                analysis["needs_visual_update"] = True
        
        # Check for performance issues
        performance_scores = [f["performance_metrics"]["fps"] for f in feedback_batch]
        if np.mean(performance_scores) < 45:  # FPS threshold
            analysis["performance_issues"].append("low_fps")
        
        return analysis
    
    async def execute_learning_updates(self, analysis: Dict):
        """Execute learning updates based on analysis"""
        
        if analysis["needs_visual_update"]:
            for style in analysis["underperforming_styles"]:

            await self.optimize_for_performance()
