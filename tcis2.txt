class SignalReconciler:
    """Reconciles AI predictions with real-world feedback to calculate meta-scores."""
    
    def __init__(self, db_session: AsyncSession):
        self.db_session = db_session
        self.prediction_history = []
    
    async def reconcile_pathway_effectiveness(self, pathway_id: str, 
                                            predicted_score: float,
                                            actual_feedback: float,
                                            clinical_evidence: float = None) -> float:
        """
        Reconcile predicted effectiveness with actual outcomes to calculate meta-score.
        Implements the Free Energy Principle by minimizing prediction error.
        """
        
        # Get historical performance for this pathway
        historical_data = await self._get_pathway_history(pathway_id)
        
        # Calculate prediction error
        prediction_error = abs(predicted_score - actual_feedback)
        
        # Weight different signals
        weights = {
            "predicted": 0.3,
            "actual_feedback": 0.4,
            "clinical_evidence": 0.2,
            "historical_performance": 0.1
        }
        
        # Calculate weighted meta-score
        meta_score = (
            predicted_score * weights["predicted"] +
            actual_feedback * weights["actual_feedback"] +
            (clinical_evidence or 0.7) * weights["clinical_evidence"] +
            historical_data["avg_effectiveness"] * weights["historical_performance"]
        )
        
        # Update pathway effectiveness tracking
        await self._update_pathway_effectiveness(pathway_id, meta_score, prediction_error)
        
        # Store for learning
        self.prediction_history.append({
            "pathway_id": pathway_id,
            "predicted": predicted_score,
            "actual": actual_feedback,
            "error": prediction_error,
            "meta_score": meta_score,
            "timestamp": datetime.now()
        })
        
        return min(max(meta_score, 0.0), 1.0)

    async def calculate_system_prediction_accuracy(self) -> Dict[str, Any]:
        """Calculate overall system prediction accuracy for MLOps monitoring."""

            return {"accuracy": 0.0, "sample_size": 0}
        
        recent_predictions = self.prediction_history[-100:]  # Last 100 predictions

    """Handles user onboarding and profile vector creation."""
    
    def __init__(self, db_session: AsyncSession):
        self.db_session = db_session
        self.cluster_model = KMeans(n_clusters=10, random_state=42)  # Pre-trained model
    
    async def create_profile_vector(self, assessments: List[TestAssessment]) -> List[float]:
        """Create ML-ready profile vector from modular assessments."""
        vector = []
        
        # Extract Big Five if available
        big_five = next((a for a in assessments if a.test_type == "BigFive"), None)
        if big_five:
            vector.extend([
                big_five.scores.get("openness", 0.5),
                big_five.scores.get("conscientiousness", 0.5),
                big_five.scores.get("extraversion", 0.5),
                big_five.scores.get("agreeableness", 0.5),
                big_five.scores.get("neuroticism", 0.5)
            ])
        else:
            vector.extend([0.5] * 5)  # Default values
        
        # Extract clinical assessments
        phq9 = next((a for a in assessments if a.test_type == "PHQ9"), None)
        if phq9:
            vector.append(phq9.scores.get("total_score", 0) / 27.0)  # Normalize
        else:
            vector.append(0.0)
        
        gad7 = next((a for a in assessments if a.test_type == "GAD7"), None)
        if gad7:
            vector.append(gad7.scores.get("total_score", 0) / 21.0)  # Normalize
        else:
            vector.append(0.0)
        
        return vector

        # Higher word count and sentence complexity indicate higher engagement
        engagement = min((word_count / 50.0) + (sentence_count / 10.0), 1.0)

        feedback_factor = max(0, (0.7 - user_profile.average_feedback_score)) / 0.7
        
        # Session duration without improvement
        session_factor = min(user_profile.session_count / 30.0, 1.0)
        
        # Weighted composite score
        failure_score = (
            convergence_factor * 0.3 +
            resistance_factor * 0.3 +
            feedback_factor * 0.2 +
            session_factor * 0.2
        )
        
        return min(failure_score, 1.0)
    
    async def detect_convergence_stagnation(self, user_profile: UserProfile) -> bool:
        """Detect if user progress has stagnated."""
        return (user_profile.convergence_failure_count >= 5 and
                user_profile.average_feedback_score < 0.6)
    
    async def calculate_intervention_resistance(self, user_profile: UserProfile) -> float:
        """Calculate intervention resistance score."""
        if user_profile.total_interactions == 0:
            return 0.0
        
        # Low engagement rate indicates resistance
        engagement_rate = user_profile.insight_markers_detected / user_profile.total_interactions
        resistance = max(0, 1.0 - (engagement_rate * 10))  # Scale appropriately
        
        return min(resistance, 1.0)

class TransformationReadinessAssessor:
    """Enhanced readiness assessment for Tier 2 escalation."""
    
    def __init__(self, db_session: AsyncSession):
        self.db_session = db_session
    
    async def assess_readiness(self, user_profile: UserProfile) -> float:
        """Comprehensive readiness assessment."""
        stability = await self.assess_stability(user_profile)
        resources = await self.assess_psychological_resources(user_profile)
        metacognitive = await self.assess_metacognitive_capacity(user_profile)
        engagement = await self.assess_engagement_quality(user_profile)
        
        # Weighted composite
        readiness_score = (
            stability * 0.3 +
            resources * 0.25 +
            metacognitive * 0.25 +
            engagement * 0.2
        )
        
        return min(readiness_score, 1.0)
    
    async def assess_stability(self, user_profile: UserProfile) -> float:
        """Assess baseline psychological stability."""
        # Lower neuroticism and higher conscientiousness indicate stability
        stability = (1.0 - user_profile.neuroticism) * 0.6 + user_profile.conscientiousness * 0.4
        
        # Adjust for feedback consistency (stable users provide consistent feedback)
        feedback_stability = 1.0 - abs(user_profile.average_feedback_score - 0.5) * 2
        stability = (stability + feedback_stability) / 2
        
        return max(stability, 0.0)
    
    async def assess_psychological_resources(self, user_profile: UserProfile) -> float:
        """Assess psychological resources for deep work."""
        # Openness and extraversion contribute to resources
        resources = user_profile.openness * 0.6 + user_profile.extraversion * 0.4
        
        # Adjust for insight generation capacity
        if user_profile.total_interactions > 0:
            insight_rate = user_profile.insight_markers_detected / user_profile.total_interactions
            resources = (resources + min(insight_rate * 5, 1.0)) / 2
        
        return max(resources, 0.0)
    
    async def assess_metacognitive_capacity(self, user_profile: UserProfile) -> float:
        """Assess capacity for self-reflection and meta-thinking."""
        # Base score from personality factors
        base_score = user_profile.openness * 0.6 + user_profile.conscientiousness * 0.4
        
        # Enhance based on insight generation
        if user_profile.total_interactions > 0:
            insight_factor = min(user_profile.insight_markers_detected / user_profile.total_interactions * 5, 1.0)
            base_score = (base_score + insight_factor) / 2
        
        return max(base_score, 0.0)
    
    async def assess_engagement_quality(self, user_profile: UserProfile) -> float:
        """Assess quality of therapeutic engagement."""
        if user_profile.total_interactions == 0:
            return 0.5  # Neutral score for new users
        
        # Combine multiple engagement indicators
        feedback_quality = user_profile.average_feedback_score
        insight_generation = min(user_profile.insight_markers_detected / user_profile.total_interactions * 5, 1.0)
        
        engagement_quality = (feedback_quality + insight_generation) / 2
        return engagement_quality

class EntityNormalizer:
    """Advanced entity normalization using medical NLP models."""
    
    def __init__(self):
        try:
            self.nlp = spacy.load("en_core_sci_sm")
        except OSError:
            # Fallback to basic English model if SciSpacy not available
            self.nlp = spacy.load("en_core_web_sm")
            logging.warning("SciSpacy model not found, using basic English model")
    
    def normalize_entity_name(self, entity_name: str) -> str:
        """Normalize entity names using NLP processing."""
        doc = self.nlp(entity_name)
        # Extract lemmatized tokens, filter stopwords
        normalized = " ".join([token.lemma_.lower() for token in doc 
                              if not token.is_stop and not token.is_punct])
        return normalized or entity_name.lower()
    
    def classify_entity_type(self, entity_name: str, context: str = "") -> str:
        """Classify entity type using NLP analysis."""
        doc = self.nlp(f"{entity_name} {context}")
        
        # Medical entity detection patterns
        for ent in doc.ents:
            if ent.label_ in ["CHEMICAL", "DISEASE"]:
                return "Drug" if ent.label_ == "CHEMICAL" else "Condition"
        
        # Fallback classification
        entity_lower = entity_name.lower()
        if any(word in entity_lower for word in ["study", "trial", "research"]):
            return "Study"
        elif any(word in entity_lower for word in ["disorder", "disease", "syndrome"]):
            return "Condition"
        else:
            return "Unknown"

def generate_content_hash(content: str) -> str:
    """Generate SHA-256 hash for content deduplication."""
    return hashlib.sha256(content.encode('utf-8')).hexdigest()

def _generate_entity_fingerprint(entity: KnowledgeEntity) -> str:
    """Generate SHA-256 fingerprint for entity deduplication."""
    content = f"{entity.entity_name}|{entity.entity_type}|{entity.source_kg}"
    content += "|" + json.dumps(sorted(entity.metadata.items()), default=str)
    return hashlib.sha256(content.encode('utf-8')).hexdigest()

                           relationship.target_node_id,
                           relationship_type=relationship.relationship_type.value,
                           strength=relationship.strength)
    
    def just_in_time_ontological_stabilizer(self, proposed_edit: Dict) -> Tuple[str, nx.MultiDiGraph]:
        """
        Forks the graph, simulates the edit's consequences, and presents
        a 'Predicted Future Self' summary to the user for confirmation.
        """
        # 1. Create a deep copy (fork) of the current graph
        sandbox_graph = self.graph.copy()
        
        # 2. Apply the proposed edit in the sandbox
        if proposed_edit["type"] == "add_node":
            sandbox_graph.add_node(proposed_edit["node_id"], **proposed_edit["data"])
        elif proposed_edit["type"] == "add_edge":
            sandbox_graph.add_edge(proposed_edit["source"], proposed_edit["target"], **proposed_edit["data"])
        elif proposed_edit["type"] == "delete_node":
            if proposed_edit["node_id"] in sandbox_graph:
                sandbox_graph.remove_node(proposed_edit["node_id"])
        
        # 3. Run simulation to find new dissonances
        dissonance_engine = DissonanceDetectionEngine(self)
        original_conflicts = len(dissonance_engine._detect_contradictions_in_graph(self.graph))
        new_conflicts = len(dissonance_engine._detect_contradictions_in_graph(sandbox_graph))
        
        # 4. Generate summary for user
        if new_conflicts > original_conflicts:
            summary = f"This change may create {new_conflicts - original_conflicts} new conflicts in your belief system."
        elif new_conflicts < original_conflicts:
            summary = f"This change may resolve {original_conflicts - new_conflicts} existing conflicts."
        else:
            summary = "This change appears to maintain your current level of internal consistency."
        
        return summary, sandbox_graph

class DissonanceDetectionEngine:
    """
    Advanced engine for detecting cognitive dissonance patterns in belief graphs.
    """
    
    def __init__(self, graph_manager: SelfGraphManager):
        self.graph_manager = graph_manager
    
    async def detect_all_patterns(self) -> List[DissonancePattern]:
        """Detect all types of dissonance patterns."""
        patterns = []
        patterns.extend(await self._detect_logical_contradictions())
        patterns.extend(await self._detect_value_behavior_mismatches())
        patterns.extend(await self._detect_identity_evidence_discord())
        return patterns
    
    async def _detect_logical_contradictions(self) -> List[DissonancePattern]:
        """Find nodes linked by CONTRADICTS relationships."""
        contradictions = []
        
        for edge in self.graph_manager.graph.edges(data=True):
            if edge[2].get("relationship_type") == RelationshipType.CONTRADICTS.value:
                pattern = DissonancePattern(
                    pattern_type=DissonanceType.LOGICAL_CONTRADICTION,
                    nodes_involved=[edge[0], edge[1]],
                    severity=edge[2].get("strength", 1.0),
                    description=f"Direct contradiction between beliefs: {edge[0]} and {edge[1]}"
                )
                contradictions.append(pattern)
        
        return contradictions
    
    async def _detect_value_behavior_mismatches(self) -> List[DissonancePattern]:
        """Find conflicts between VALUE_SYSTEM and BEHAVIORAL_PATTERN nodes."""
        mismatches = []
        
        value_nodes = [n for n, data in self.graph_manager.graph.nodes(data=True) 
                      if data.get("node_type") == BeliefNodeType.VALUE_SYSTEM.value]
        behavior_nodes = [n for n, data in self.graph_manager.graph.nodes(data=True) 
                         if data.get("node_type") == BeliefNodeType.BEHAVIORAL_PATTERN.value]
        
        for value_node in value_nodes:
            for behavior_node in behavior_nodes:
                if self.graph_manager.graph.has_edge(value_node, behavior_node):
                    edge_data = self.graph_manager.graph.get_edge_data(value_node, behavior_node)
                    if any(data.get("relationship_type") == RelationshipType.CONTRADICTS.value 
                          for data in edge_data.values()):
                        pattern = DissonancePattern(
                            pattern_type=DissonanceType.VALUE_BEHAVIOR_MISMATCH,
                            nodes_involved=[value_node, behavior_node],
                            severity=0.8,
                            description=f"Value-behavior conflict: {value_node} contradicts {behavior_node}"
                        )
                        mismatches.append(pattern)
        
        return mismatches
    
    async def _detect_identity_evidence_discord(self) -> List[DissonancePattern]:
        """Find mismatches between IDENTITY_COMPONENT and evidence."""
        discords = []
        
        identity_nodes = [n for n, data in self.graph_manager.graph.nodes(data=True) 
                         if data.get("node_type") == BeliefNodeType.IDENTITY_COMPONENT.value]
        
        for identity_node in identity_nodes:
            supporting_edges = 0
            contradicting_edges = 0
            
            for neighbor in self.graph_manager.graph.neighbors(identity_node):
                edge_data = self.graph_manager.graph.get_edge_data(identity_node, neighbor)
                for data in edge_data.values():
                    if data.get("relationship_type") == RelationshipType.SUPPORTS.value:
                        supporting_edges += 1
                    elif data.get("relationship_type") == RelationshipType.CONTRADICTS.value:
                        contradicting_edges += 1
            
            if contradicting_edges > supporting_edges:
                pattern = DissonancePattern(
                    pattern_type=DissonanceType.IDENTITY_EVIDENCE_DISCORD,
                    nodes_involved=[identity_node],
                    severity=(contradicting_edges - supporting_edges) / max(contradicting_edges + supporting_edges, 1),
                    description=f"Identity component {identity_node} has more contradicting than supporting evidence"
                )
                discords.append(pattern)
        
        return discords
    
    def _detect_contradictions_in_graph(self, graph: nx.MultiDiGraph) -> List[Tuple[str, str]]:
        """Helper method to detect contradictions in any graph."""
        contradictions = []
        for edge in graph.edges(data=True):
            if edge[2].get("relationship_type") == RelationshipType.CONTRADICTS.value:
                contradictions.append((edge[0], edge[1]))
        return contradictions

class OptimizationFailureDetector:
    """Detects when Tier 1 optimization is failing for a user."""
    
    def __init__(self, db_session: AsyncSession):
        self.db_session = db_session
    
    async def calculate_failure_score(self, user_profile: UserProfile) -> float:
        """Calculate composite optimization failure score."""
        # Convergence stagnation factor
        convergence_factor = min(user_profile.convergence_failure_count / 10.0, 1.0)
        
        # Intervention resistance factor
        resistance_factor = min(user_profile.intervention_resistance_score, 1.0)
        
        # Session duration without improvement factor
        session_factor = min(user_profile.session_count / 20.0, 1.0)
        
        # Weighted composite score
        failure_score = (
            convergence_factor * 0.4 +
            resistance_factor * 0.4 +
            session_factor * 0.2
        )
        
        return min(failure_score, 1.0)

class TransformationReadinessAssessor:
    """Assesses user's readiness for Tier 2 consciousness work."""
    
    def __init__(self, db_session: AsyncSession):
        self.db_session = db_session
    
    async def assess_readiness(self, user_profile: UserProfile) -> float:
        """Assess transformation readiness based on multiple factors."""
        # Stability indicators (higher stability = higher readiness)
        stability_score = self._calculate_stability_score(user_profile)
        
        # Psychological resources (insight, resilience, motivation)
        resources_score = self._calculate_resources_score(user_profile)
        
        # Meta-cognitive capacity (ability to think about thinking)
        metacognitive_score = self._calculate_metacognitive_score(user_profile)
        
        # Weighted composite
        readiness_score = (
            stability_score * 0.4 +
            resources_score * 0.3 +
            metacognitive_score * 0.3
        )
        
        return min(readiness_score, 1.0)
    
    def _calculate_stability_score(self, user_profile: UserProfile) -> float:
        """Calculate baseline stability score."""
        # Lower neuroticism and higher conscientiousness indicate stability
        stability = (1.0 - user_profile.neuroticism) * 0.6 + user_profile.conscientiousness * 0.4
        return max(stability, 0.0)
    
    def _calculate_resources_score(self, user_profile: UserProfile) -> float:
        """Calculate psychological resources score."""
        # Openness to experience and extraversion contribute to resources
        resources = user_profile.openness * 0.6 + user_profile.extraversion * 0.4
        return max(resources, 0.0)
    
    def _calculate_metacognitive_score(self, user_profile: UserProfile) -> float:
        """Calculate metacognitive capacity score."""
        # Based on session engagement patterns and attribute scores
        base_score = user_profile.openness * 0.5 + user_profile.conscientiousness * 0.5
        
        # Adjust based on specific assessments
        if "metacognitive_awareness" in user_profile.attributes:
            base_score = (base_score + user_profile.attributes["metacognitive_awareness"]) / 2
        
        return max(base_score, 0.0)

class EscalationProtocolEngine:
    """Manages transitions between Tier 1 and Tier 2."""
    
    def __init__(self, db_session: AsyncSession):
        self.db_session = db_session
        self.failure_detector = OptimizationFailureDetector(db_session)
        self.readiness_assessor = TransformationReadinessAssessor(db_session)
    
    async def evaluate_escalation(self, user_id: str) -> Optional[Dict[str, Any]]:
        """Evaluate whether user should be escalated to Tier 2."""
        # Get user profile (mock implementation)
        user_profile = UserProfile(
            user_id=uuid.UUID(user_id) if isinstance(user_id, str) else user_id,
            current_tier=1,
            session_count=25,
            convergence_failure_count=8,
            intervention_resistance_score=0.7,
            openness=0.8,
            conscientiousness=0.6,
            extraversion=0.5,
            agreeableness=0.7,
            neuroticism=0.4
        )
        
        # Calculate failure score
        failure_score = await self.failure_detector.calculate_failure_score(user_profile)
        
        # Calculate readiness score
        readiness_score = await self.readiness_assessor.assess_readiness(user_profile)
        
        # Escalation criteria
        escalation_threshold = 0.6
        readiness_threshold = 0.5
        
        if failure_score >= escalation_threshold and readiness_score >= readiness_threshold:
            return {
                "should_escalate": True,
                "failure_score": failure_score,
                "readiness_score": readiness_score,
                "trigger_criteria": {
                    "convergence_stagnation": user_profile.convergence_failure_count >= 5,
                    "intervention_resistance": user_profile.intervention_resistance_score >= 0.6,
                    "session_duration": user_profile.session_count >= 20
                },
                "recommendation": "User shows optimization failure with sufficient readiness for deeper work"
            }
        
        return {
            "should_escalate": False,
            "failure_score": failure_score,
            "readiness_score": readiness_score,
            "recommendation": "Continue Tier 1 optimization" if failure_score < escalation_threshold 
                           else "Build readiness before escalation"
        }

class SafetyGovernor:
    """
    Safety service that uses Natural Language Inference (NLI) to detect
    contradictions between AI responses and verified facts.
    """
    
    def __init__(self):
        # Layer 1: Fast NLI model (mocked for demonstration)
        self.nli_pipeline = None  # pipeline("text-classification", model="roberta-large-mnli")
        self.contradiction_keywords = ["contradict", "false", "incorrect", "wrong", "harmful"]
    
    def check_for_contradictions(self, draft_response: str, facts: List[Dict]) -> List[ContradictionAlert]:
        """
        Uses an NLI model to check for contradictions between the AI's response and verified facts.
        """
        contradictions = []
        
        # Mock implementation - in production would use actual NLI model
        response_lower = draft_response.lower()
        
        for fact in facts:
            fact_statement = f"{fact['entity_name']} {fact['relationship_type']} {fact['related_entity_name']}"
            
            # Simple contradiction detection for demonstration
            if any(keyword in response_lower for keyword in self.contradiction_keywords):
                if fact['entity_name'].lower() in response_lower:
                    contradiction = ContradictionAlert(
                        statement=draft_response,
                        conflicting_fact=fact_statement,
                        confidence=0.8
                    )
                    contradictions.append(contradiction)
        
        return contradictions
    
    def validate_response_safety(self, draft_response: str, context: Dict) -> Dict[str, Any]:
        """Comprehensive safety validation of AI response."""
        # Layer 1 check
        contradictions = self.check_for_contradictions(
            draft_response, context.get('factual_knowledge', [])
        )
        
        # Additional safety checks
        safety_flags = []
        
        # Check for harmful content patterns
        harmful_patterns = ["suicide", "self-harm", "violence", "abuse"]
        if any(pattern in draft_response.lower() for pattern in harmful_patterns):
            safety_flags.append("POTENTIAL_HARM_CONTENT")
        
        # Check for medical advice overreach
        medical_advice_patterns = ["you should take", "I recommend medication", "stop taking"]
        if any(pattern in draft_response.lower() for pattern in medical_advice_patterns):
            safety_flags.append("MEDICAL_ADVICE_OVERREACH")
        
        is_safe = len(contradictions) == 0 and len(safety_flags) == 0
        
        return {
            "is_safe": is_safe,
            "contradictions": contradictions,
            "safety_flags": safety_flags,
            "requires_human_review": not is_safe
        }

    """Validates AI performance against expert dialogues and generates corrections."""
    
    def __init__(self, api_base_url: str):
        self.api_base_url = api_base_url
        self.client = httpx.AsyncClient()
        
        # Mock golden dialogue data based on provided examples
        self.golden_dialogues = [
            {
                "dialogue_id": "TOLAN_C3_D030",
                "therapeutic_principle": "Navigates the therapeutic process by respecting the client's refusal to engage",
                "turns": [
                    {"speaker": "therapist", "text": "Do you want to tell me what happened?"},
                    {"speaker": "client", "text": "I don't think I could talk about it."},
                    {"speaker": "therapist", "text": "Are the feelings still very raw?"},
                    {"speaker": "client", "text": "[nods]"},
                    {"speaker": "therapist", "text": "And you couldn't tell me about it without them spilling over."}
                ]
            }
        ]
        
        self.critique_template = """
# Role: Clinical Supervisor
You are tasked with reviewing a therapeutic AI's performance against an expert human therapist's response.

# Context
- Client's Statement: "{{client_input}}"

# Performance Review
- AI's Response: "{{ai_response}}"
- Expert Therapist's Response (Golden Standard): "{{expert_response}}"

# Analysis Task
1. **Critique the AI's Response:** Compare the AI's approach to the expert's, noting strengths and areas for improvement.
2. **Identify the Core Skill:** What therapeutic skill is being demonstrated in the expert response?
3. **Categorize the Skill:** Which therapeutic framework or approach does this skill belong to?

# Output Format
Respond ONLY with a single, clean JSON object:
{
  "critique": "Detailed analysis of the AI's performance compared to expert response",
  "identified_skill_name": "Name of the therapeutic skill demonstrated",
  "dora_category": "Therapeutic framework or pillar this skill belongs to"
}
"""
    
    async def simulate_conversation(self, dialogue: Dict[str, Any], user_id: str) -> List[Dict[str, Any]]:
        """Simulate a conversation with HelloKoa API using golden dialogue."""
        results = []
        
        for i, turn in enumerate(dialogue["turns"]):
            if turn["speaker"] == "client":
                # Simulate AI response to client input
                # In real implementation, this would call the actual chat API
                mock_ai_response = "I hear that you're struggling with something difficult. It takes courage to even consider sharing such painful experiences."
                
                # Get the next therapist response as golden standard
                if i + 1 < len(dialogue["turns"]) and dialogue["turns"][i + 1]["speaker"] == "therapist":
                    expert_response = dialogue["turns"][i + 1]["text"]
                    
                    results.append({
                        "client_input": turn["text"],
                        "ai_response": mock_ai_response,
                        "expert_response": expert_response,
                        "turn_number": i
                    })
        
        return results
    
    async def critique_response(self, client_input: str, ai_response: str, expert_response: str) -> Optional[Dict[str, Any]]:
        """Use LLM to critique AI response against expert response."""
        from jinja2 import Template
        
        template = Template(self.critique_template)
        prompt = template.render(
            client_input=client_input,
            ai_response=ai_response,
            expert_response=expert_response
        )
        
        # Mock critique response
        critique_result = {
            "critique": "The AI response shows appropriate empathy but lacks the specific validation of the client's resistance shown in the expert response. The expert directly acknowledges the difficulty without pushing for disclosure.",
            "identified_skill_name": "Respecting Client Pace",
            "dora_category": "Navigating the Therapeutic Process"
        }
        
        return critique_result
    
    async def validate_with_golden_standards(self) -> List[Dict[str, Any]]:
        """Run full validation pipeline against golden standard dialogues."""
        validation_results = []
        
        for dialogue in self.golden_dialogues:
            # Simulate conversation
            conversation_results = await self.simulate_conversation(dialogue, "test_user")
            
            # Critique each turn
            for result in conversation_results:
                critique = await self.critique_response(
                    result["client_input"],
                    result["ai_response"], 
                    result["expert_response"]
                )
                
                validation_results.append({
                    "dialogue_id": dialogue["dialogue_id"],
                    "turn_number": result["turn_number"],
                    "critique": critique,
                    "training_example": {
                        "input": result["client_input"],
                        "expected_output": result["expert_response"],
                        "skill_demonstrated": critique["identified_skill_name"]
                    }
                })
        
        return validation_results

@app.post("/chat", response_model=ChatResponse)
async def chat_endpoint(request: ChatRequest):
    """Main chat endpoint that processes user input through the TCIS pipeline."""
    
    try:
        # Initialize session (mock)
        db_session = None  # In production: get_db_session()
        
        # Initialize reasoning engine
        reasoning_engine = ReasoningEngine(llm_provider, db_session)
        
        # Process user input
        reasoning_result = await reasoning_engine.process_user_input(
            request.message, request.user_id
        )
        
        # Safety validation
        safety_result = safety_governor.validate_response_safety(
            reasoning_result["response_text"],
            {"factual_knowledge": []}  # Would include actual facts in production
        )
        
        if not safety_result["is_safe"]:
            # In production, this would queue for human review
            response_text = "I want to be helpful, but I need to be careful about the guidance I provide. Let me think about how to best support you with this."
        else:
            response_text = reasoning_result["response_text"]
        
        # Check for tier escalation
        escalation_engine = EscalationProtocolEngine(db_session)
        escalation_result = await escalation_engine.evaluate_escalation(request.user_id)
        
        current_tier = 2 if escalation_result.get("should_escalate", False) else 1
        
        return ChatResponse(
            response=response_text,
            components_used=reasoning_result.get("components_used", []),
            safety_validated=safety_result["is_safe"],
            tier=current_tier
        )
        
    except Exception as e:
        logging.error(f"Error processing chat request: {e}")
        raise HTTPException(status_code=500, detail="Internal server error")
