class ProfileService:
    """Manages user profile tests and cluster prediction for personalized therapy."""
    
    def __init__(self, db_manager: DatabaseManager, cache_manager: CacheManager, embedding_manager: EmbeddingManager):
        self.db_manager = db_manager
        self.cache_manager = cache_manager

        self.cache_namespace = "user_profiles"

    def _validate_test_data(self, test_data: Dict[str, Any]) -> bool:
        """Validate test data structure."""
        required_keys = {"test_type", "scores"}
        if not all(key in test_data for key in required_keys):
            return False
        
        if not isinstance(test_data["scores"], dict):
            return False
        
        # Validate specific test types
        test_type = test_data["test_type"]
        
        if test_type == "big_five":
            required_scores = {"openness", "conscientiousness", "extraversion", "agreeableness", "neuroticism"}
            if not required_scores.issubset(test_data["scores"].keys()):
                return False
            # Validate score ranges (0.0 to 1.0)
            for score in test_data["scores"].values():
                if not isinstance(score, (int, float)) or not 0.0 <= score <= 1.0:
                    return False
                    
        elif test_type == "adhd":
            required_scores = {"symptom_score"}
            if not required_scores.issubset(test_data["scores"].keys()):
                return False
            # Validate ADHD symptom score (0.0 to 1.0)
            if not 0.0 <= test_data["scores"]["symptom_score"] <= 1.0:
                return False
                
        elif test_type == "anxiety":
            required_scores = {"anxiety_level"}
            if not required_scores.issubset(test_data["scores"].keys()):
                return False
            if not 0.0 <= test_data["scores"]["anxiety_level"] <= 1.0:
                return False
        
        return True

    async def update_profile(self, user_id: str, test_data: Dict[str, Any]) -> Dict[str, Any]:
        """Update user profile with new test data."""
        try:
            if not self._validate_test_data(test_data):
                raise ValidationError("Invalid test data schema or values")

            fingerprint = generate_fingerprint(test_data)
            test_type = test_data["test_type"]
            
            async with self.db_manager.get_session() as session:
                # Get or create user profile
                stmt = select(UserProfile).where(UserProfile.user_id == user_id)
                result = await session.execute(stmt)
                profile = result.scalar()
                
                if not profile:
                    profile = UserProfile(
                        user_id=user_id,
                        attributes={}
                    )
                    session.add(profile)
                
                # Check for duplicate test data
                attributes = profile.attributes or {}
                if test_type in attributes:
                    existing_fingerprint = attributes[test_type].get("fingerprint")
                    if existing_fingerprint == fingerprint:
                        logger.debug(f"Duplicate test {test_type} for user {user_id} skipped")
                        return attributes[test_type]
                
                # Update test data
                attributes[test_type] = {
                    "scores": test_data["scores"],
                    "fingerprint": fingerprint,
                    "timestamp": datetime.utcnow().isoformat(),
                    "metadata": test_data.get("metadata", {})
                }
                
                # Update Big Five scores in main columns if applicable
                if test_type == "big_five":
                    scores = test_data["scores"]
                    profile.openness = scores.get("openness", profile.openness)
                    profile.conscientiousness = scores.get("conscientiousness", profile.conscientiousness)
                    profile.extraversion = scores.get("extraversion", profile.extraversion)
                    profile.agreeableness = scores.get("agreeableness", profile.agreeableness)
                    profile.neuroticism = scores.get("neuroticism", profile.neuroticism)
                
                profile.attributes = attributes
                profile.last_updated = datetime.utcnow()
                
                await session.commit()
                
                # Update cache
                profile_dict = {
                    "user_id": profile.user_id,
                    "openness": profile.openness,
                    "conscientiousness": profile.conscientiousness,
                    "extraversion": profile.extraversion,
                    "agreeableness": profile.agreeableness,
                    "neuroticism": profile.neuroticism,
                    "attributes": profile.attributes,
                    "last_updated": profile.last_updated.isoformat(),
                    "session_count": profile.session_count,
                    "total_interactions": profile.total_interactions
                }
                
                await self.cache_manager.set(
                    self.cache_namespace, 
                    user_id, 
                    profile_dict, 
                    ttl=self.cache_ttl
                )
                
                logger.info(f"Updated profile for user {user_id} with test {test_type}")
                return attributes[test_type]

        except Exception as e:
            logger.error(f"Profile update failed for user {user_id}: {e}")
            raise DatabaseError(f"Failed to update profile: {e}")

    async def get_profile(self, user_id: str) -> Optional[Dict[str, Any]]:
        """Get user profile with caching."""
        try:
            # Try cache first
            cached_profile = await self.cache_manager.get(self.cache_namespace, user_id)
            if cached_profile:
                return cached_profile
            
            # Fetch from database
            async with self.db_manager.get_session() as session:
                stmt = select(UserProfile).where(UserProfile.user_id == user_id)
                result = await session.execute(stmt)
                profile = result.scalar()
                
                if not profile:
                    return None
                
                profile_dict = {
                    "user_id": profile.user_id,
                    "openness": profile.openness,
                    "conscientiousness": profile.conscientiousness,
                    "extraversion": profile.extraversion,
                    "agreeableness": profile.agreeableness,
                    "neuroticism": profile.neuroticism,
                    "attributes": profile.attributes or {},
                    "last_updated": profile.last_updated.isoformat() if profile.last_updated else None,
                    "session_count": profile.session_count,
                    "total_interactions": profile.total_interactions
                }
                
                # Cache for future requests
                await self.cache_manager.set(
                    self.cache_namespace, 
                    user_id, 
                    profile_dict, 
                    ttl=self.cache_ttl
                )
                
                return profile_dict

        except Exception as e:
            logger.error(f"Failed to get profile for user {user_id}: {e}")
            return None

    async def predict_cluster(self, user_id: str, user_input: Optional[str] = None) -> Optional[int]:
        """Predict user cluster based on profile and optional input context."""
        try:
            profile = await self.get_profile(user_id)
            if not profile:
                logger.debug(f"No profile found for user {user_id}, using default cluster")
                return 0  # Default cluster
            
            # Load cluster model if available
            cluster_model_path = Path("data/models/component_clusters.pkl")
            if not cluster_model_path.exists():
                logger.warning("Cluster model not found, using default cluster")
                return 0
            
            with open(cluster_model_path, "rb") as f:
                clusters_data = pickle.load(f)
                centroids = np.array(clusters_data["centroids"])
                feature_names = clusters_data.get("feature_names", [])
            
            # Build feature vector
            features = [
                profile.get("openness", 0.5),
                profile.get("conscientiousness", 0.5),
                profile.get("extraversion", 0.5),
                profile.get("agreeableness", 0.5),
                profile.get("neuroticism", 0.5)
            ]
            
            # Add modular test scores
            attributes = profile.get("attributes", {})
            
            # ADHD symptom score
            if "adhd" in attributes:
                adhd_score = attributes["adhd"]["scores"].get("symptom_score", 0.5)
                features.append(adhd_score)
            else:
                features.append(0.5)  # Default value
            
            # Anxiety level
            if "anxiety" in attributes:
                anxiety_score = attributes["anxiety"]["scores"].get("anxiety_level", 0.5)
                features.append(anxiety_score)
            else:
                features.append(0.5)  # Default value
            
            # Add input embedding features if available
            if user_input and self.embedding_manager:
                try:
                    input_embedding = await self.embedding_manager.get_embedding(user_input)
                    if input_embedding is not None:
                        # Use first 10 dimensions of embedding as features
                        embedding_features = input_embedding[:10].tolist()
                        features.extend(embedding_features)
                    else:
                        # Pad with zeros if embedding failed
                        features.extend([0.0] * 10)
                except Exception as e:
                    logger.warning(f"Failed to get input embedding: {e}")
                    features.extend([0.0] * 10)
            else:
                # Pad with zeros if no input provided
                features.extend([0.0] * 10)
            
            # Ensure feature vector matches expected dimensions
            features = np.array(features)
            if len(features) != centroids.shape[1]:
                logger.warning(f"Feature dimension mismatch: {len(features)} vs {centroids.shape[1]}")
                # Pad or truncate as needed
                if len(features) < centroids.shape[1]:
                    features = np.pad(features, (0, centroids.shape[1] - len(features)))
                else:
                    features = features[:centroids.shape[1]]
            
            # Find closest cluster
            distances = np.linalg.norm(centroids - features, axis=1)
            cluster = int(np.argmin(distances))
            
            logger.debug(f"Predicted cluster {cluster} for user {user_id} (distance: {distances[cluster]:.3f})")
            return cluster

        except Exception as e:
            logger.error(f"Cluster prediction failed for user {user_id}: {e}")
            return 0  # Default cluster on error

    async def get_cluster_characteristics(self, cluster_id: int) -> Optional[Dict[str, Any]]:

            with open(cluster_model_path, "rb") as f:

                cluster_info = clusters_data.get("cluster_info", {})
                
                if str(cluster_id) in cluster_info:
                    return cluster_info[str(cluster_id)]
            
            return None

        except Exception as e:
            logger.error(f"Failed to get cluster characteristics for {cluster_id}: {e}")
            return None

You are correct; the system is explicitly designed to track the user's therapeutic journey. This is not an afterthought but a core function enabled by the architecture.

The Mechanism: Every user interaction is captured as a rich datapoint in the InteractionLog table. This log doesn't just store the text; it records the components_used, the dora_category applied, the user's personality_snapshot at that moment, and the user_feedback received.
The "Goldmine" Tools: The analytical scripts from our plan are designed to mine this data:
analyze_efficacy.py  creates the "Consumer Map" you envisioned. It directly analyzes which therapeutic pathways and components are most effective for different user profiles (e.g., users with high neuroticism) by correlating them with feedback scores.

build_knowledge_graph.py  visually maps these user journeys, creating a network where the most frequently traveled and highly-rated "neural pathways" become visible. This allows us to see how users organically connect different therapeutic ideas.

2. The Anti-Hallucination Engine: Your "Objective Science Bank"
This is the crucial advantage of the Super-Graph approach. You are right that we are not just retrieving from an uncontrolled external source; we are building our own proprietary, standardized "science bank."

The Mechanism: The standardize_and_ingest_external_kg.py script  is the engine for this. It acts as an ETL (Extract, Transform, Load) pipeline that pulls data from verifiable sources like PubMed, DrugBank, and the expert-curated ADHD-KG.




Organized and Verifiable: This data is then standardized into our own ExternalKnowledge database model. This model is designed for traceability, with columns like source_kg, source_link, and entity_name.

Building User Trust: When the ReasoningEngine uses this information, it can present it to the user with full transparency, as envisioned in the manifesto. For example, a response could be phrased as: "Based on information from DrugBank, a known side effect is.... It's completely understandable to feel worried about that. Let's talk about what that brings up for you." This directly combats hallucination by grounding factual statements in verifiable sources.

3. Intelligent Query Optimization: Context-Aware Retrieval
Your point about optimizing the query based on the current therapeutic pathway is exactly how the system is designed to function. The AI doesn't just perform a generic web search; its consultation of the "science bank" is guided by the therapeutic context.

The Mechanism: The updated ReasoningEngine  performs a "Dual-Retrieval." Before it queries the ExternalKnowledge table, it first analyzes the user's input and memory to understand their immediate therapeutic need.

How it Works in Practice:
A user on an anxiety-focused pathway asks, "My doctor mentioned Adderall, what is it?"
The ReasoningEngine recognizes the keyword "Adderall" and the user's context of "anxiety."
Its query to the Super-Graph is not just for "Adderall." It is an intelligent, multi-faceted query for:
Factual data on "Adderall" from the ExternalKnowledge table.

def generate_fingerprint(entity_name: str, relationship: str, related_name: str) -> str:
    """Generate a fingerprint for deduplication."""
    content = f"{entity_name}:{relationship}:{related_name}"
    return hashlib.sha256(content.encode()).hexdigest()[:16]

def validate_entity_schema(entity: Dict[str, Any]) -> bool:
    """Validate entity schema before ingestion."""
    required_keys = {"entity_id", "source_kg", "entity_type", "entity_name"}
    return all(key in entity for key in required_keys)

class KGParser:
    """Parses external KG data into a standardized format."""
    
    def __init__(self):
        try:
            self.nlp = spacy.load("en_core_sci_sm")
        except OSError:
            logger.warning("SciSpacy model not found, using basic spacy model")
            self.nlp = spacy.load("en_core_web_sm")
        
        self.namespace_map = {
            "mesh": "http://id.nlm.nih.gov/mesh/",
            "adhd": "http://example.com/adhd/",
            "rdf": "http://www.w3.org/1999/02/22-rdf-syntax-ns#",
            "rdfs": "http://www.w3.org/2000/01/rdf-schema#"
        }

    async def parse_adhd_kg(self, file_path: Path) -> List[Dict[str, Any]]:
        """Parse ADHD-KG RDF data (.n3 format)."""
        logger.info(f"Parsing ADHD-KG from {file_path}")
        
        try:
            graph = rdflib.Graph()
            graph.parse(file_path, format="n3")
            
            entities = []
            processed_subjects = set()
            
            for subj, pred, obj in graph:
                subj_str = str(subj)
                pred_str = str(pred)
                obj_str = str(obj)
                
                # Skip if we've already processed this subject
                if subj_str in processed_subjects:
                    continue
                
                # Extract label for the subject
                label = None
                for _, _, l in graph.triples((subj, rdflib.RDFS.label, None)):
                    label = str(l)
                    break
                
                if not label:
                    # Use the last part of the URI as label if no rdfs:label
                    label = subj_str.split("/")[-1].replace("_", " ").replace("-", " ")
                
                # Use NLP to determine entity type
                doc = self.nlp(label)
                entity_type = "Concept"  # Default
                for ent in doc.ents:
                    if ent.label_ in ["DISEASE", "CONDITION"]:
                        entity_type = "Disease"
                        break
                    elif ent.label_ in ["DRUG", "CHEMICAL"]:
                        entity_type = "Drug"
                        break
                
                # Extract relationship and related entity
                relationship = pred_str.split("/")[-1].replace("_", " ")
                related_name = obj_str.split("/")[-1].replace("_", " ") if "/" in obj_str else str(obj_str)
                
                entity_id = generate_id(subj_str)
                fingerprint = generate_fingerprint(label, relationship, related_name)
                
                entity = {
                    "entity_id": entity_id,
                    "source_kg": "ADHD-KG",
                    "entity_type": entity_type,
                    "entity_name": label,
                    "relationship_type": relationship,
                    "related_entity_name": related_name,
                    "source_link": subj_str,
                    "metadata": {
                        "mesh_term": subj_str.split("/")[-1] if "mesh" in subj_str else None,
                        "predicate_uri": pred_str,
                        "object_uri": obj_str if "/" in obj_str else None
                    },
                    "fingerprint": fingerprint
                }
                
                if validate_entity_schema(entity):
                    entities.append(entity)
                    processed_subjects.add(subj_str)
            
            logger.info(f"Parsed {len(entities)} entities from ADHD-KG")
            return entities
            
        except Exception as e:
            logger.error(f"Failed to parse ADHD-KG: {e}")
            return []

    async def parse_drugbank(self, file_path: Path) -> List[Dict[str, Any]]:
        """Parse DrugBank XML data."""
        logger.info(f"Parsing DrugBank from {file_path}")
        
        try:
            tree = ET.parse(file_path)
            root = tree.getroot()
            ns = {"db": "http://www.drugbank.ca"}
            
            entities = []
            
            for drug in root.findall("db:drug", ns):
                try:
                    name_elem = drug.find("db:name", ns)
                    if name_elem is None:
                        continue
                    
                    drug_name = name_elem.text
                    drug_id = generate_id(drug_name)
                    
                    # Get DrugBank ID
                    drugbank_id_elem = drug.find("db:drugbank-id", ns)
                    drugbank_id = drugbank_id_elem.text if drugbank_id_elem is not None else None
                    
                    # Parse side effects
                    side_effects = drug.findall(".//db:side-effect", ns)
                    for side_effect in side_effects:
                        try:
                            effect_name_elem = side_effect.find("db:name", ns)
                            if effect_name_elem is None:
                                continue
                            
                            effect_name = effect_name_elem.text
                            fingerprint = generate_fingerprint(drug_name, "has_side_effect", effect_name)
                            
                            entity = {
                                "entity_id": f"{drug_id}_se_{generate_id(effect_name)}",
                                "source_kg": "DrugBank",
                                "entity_type": "Drug",
                                "entity_name": drug_name,
                                "relationship_type": "has_side_effect",
                                "related_entity_name": effect_name,
                                "source_link": f"https://go.drugbank.com/drugs/{drugbank_id}" if drugbank_id else None,
                                "metadata": {
                                    "drugbank_id": drugbank_id,
                                    "side_effect_category": side_effect.find("db:category", ns).text if side_effect.find("db:category", ns) is not None else None
                                },
                                "fingerprint": fingerprint
                            }
                            
                            if validate_entity_schema(entity):
                                entities.append(entity)
                                
                        except Exception as e:
                            logger.warning(f"Failed to parse side effect for {drug_name}: {e}")
                            continue
                    
                    # Parse indications
                    indications = drug.findall(".//db:indication", ns)
                    for indication in indications:
                        try:
                            indication_text = indication.text
                            if not indication_text:
                                continue
                            
                            fingerprint = generate_fingerprint(drug_name, "treats", indication_text)
                            
                            entity = {
                                "entity_id": f"{drug_id}_ind_{generate_id(indication_text)}",
                                "source_kg": "DrugBank",
                                "entity_type": "Drug",
                                "entity_name": drug_name,
                                "relationship_type": "treats",
                                "related_entity_name": indication_text,
                                "source_link": f"https://go.drugbank.com/drugs/{drugbank_id}" if drugbank_id else None,
                                "metadata": {
                                    "drugbank_id": drugbank_id,
                                    "indication_full_text": indication_text
                                },
                                "fingerprint": fingerprint
                            }
                            
                            if validate_entity_schema(entity):
                                entities.append(entity)
                                
                        except Exception as e:
                            logger.warning(f"Failed to parse indication for {drug_name}: {e}")
                            continue
                            
                except Exception as e:
                    logger.warning(f"Failed to parse drug: {e}")
                    continue
            
            logger.info(f"Parsed {len(entities)} entities from DrugBank")
            return entities
            
        except Exception as e:
            logger.error(f"Failed to parse DrugBank: {e}")
            return []

    async def parse_pubmed(self, query: str, max_results: int = 50) -> List[Dict[str, Any]]:
        """Fetch PubMed abstracts via API."""
        logger.info(f"Fetching PubMed data for query: {query}")
        
        try:
            entities = []
            
            async with aiohttp.ClientSession() as session:
                # Search for PMIDs
                search_url = f"https://eutils.ncbi.nlm.nih.gov/entrez/eutils/esearch.fcgi"
                search_params = {
                    "db": "pubmed",
                    "term": query,
                    "retmax": max_results,
                    "retmode": "json"
                }
                
                async with session.get(search_url, params=search_params) as response:
                    search_result = await response.json()
                    pmids = search_result.get("esearchresult", {}).get("idlist", [])
                
                if not pmids:
                    logger.warning(f"No PMIDs found for query: {query}")
                    return entities
                
                # Fetch article details
                fetch_url = f"https://eutils.ncbi.nlm.nih.gov/entrez/eutils/efetch.fcgi"
                pmid_str = ",".join(pmids)
                fetch_params = {
                    "db": "pubmed",
                    "id": pmid_str,
                    "retmode": "xml"
                }
                
                async with session.get(fetch_url, params=fetch_params) as response:
                    xml_content = await response.text()
                    root = ET.fromstring(xml_content)
                    
                    for article in root.findall(".//PubmedArticle"):
                        try:
                            pmid_elem = article.find(".//PMID")
                            title_elem = article.find(".//ArticleTitle")
                            abstract_elem = article.find(".//AbstractText")
                            
                            if pmid_elem is None or title_elem is None:
                                continue
                            
                            pmid = pmid_elem.text
                            title = title_elem.text
                            abstract = abstract_elem.text if abstract_elem is not None else ""
                            
                            # Use NLP to extract key terms
                            doc = self.nlp(f"{title} {abstract}"[:1000])  # Limit text length
                            key_terms = [ent.text for ent in doc.ents if ent.label_ in ["DISEASE", "DRUG", "CHEMICAL"]]
                            
                            entity_id = generate_id(pmid)
                            fingerprint = generate_fingerprint(title, "studies", query)
                            
                            entity = {
                                "entity_id": entity_id,
                                "source_kg": "PubMed",
                                "entity_type": "Study",
                                "entity_name": title,
                                "relationship_type": "studies",
                                "related_entity_name": query,
                                "source_link": f"https://pubmed.ncbi.nlm.nih.gov/{pmid}/",
                                "metadata": {
                                    "pmid": pmid,
                                    "abstract": abstract[:500],  # Truncate abstract
                                    "key_terms": key_terms[:10],  # Limit key terms
                                    "query": query
                                },
                                "fingerprint": fingerprint
                            }
                            
                            if validate_entity_schema(entity):
                                entities.append(entity)
                                
                        except Exception as e:
                            logger.warning(f"Failed to parse PubMed article: {e}")
                            continue
            
            logger.info(f"Parsed {len(entities)} entities from PubMed")
            return entities
            
        except Exception as e:
            logger.error(f"Failed to parse PubMed: {e}")
            return []

async def standardize_and_ingest(source: str, input_path: Optional[Path], output_path: Path, query: Optional[str] = None):
    """Standardize and ingest external KG data into the Super-Graph."""
    print(f"🔄 Standardizing and ingesting knowledge from {source}...")

    try:
        config = get_config()
        db_manager = DatabaseManager()
        await db_manager.initialize()
        cache_manager = CacheManager(db_manager)
        await cache_manager.initialize()
        
        parser = KGParser()

        # Parse based on source type
        if source == "adhd_kg" and input_path:
            entities = await parser.parse_adhd_kg(input_path)
        elif source == "drugbank" and input_path:
            entities = await parser.parse_drugbank(input_path)
        elif source == "pubmed" and query:
            entities = await parser.parse_pubmed(query)
        else:
            raise ValueError(f"Invalid source/parameters: {source}, input_path={input_path}, query={query}")

        if not entities:
            print("⚠️ No entities parsed from source")
            return

        # Ingest into database
        created_count = 0
        duplicate_count = 0
        
        async with db_manager.get_session() as session:
            for entity in entities:
                try:
                    # Check for existing entity
                    existing = await session.execute(
                        select(ExternalKnowledge).where(ExternalKnowledge.fingerprint == entity["fingerprint"])
                    )
                    if existing.scalar():
                        duplicate_count += 1
                        continue
                    
                    # Create new entity
                    knowledge = ExternalKnowledge(
                        entity_id=entity["entity_id"],
                        source_kg=entity["source_kg"],
                        entity_type=entity["entity_type"],
                        entity_name=entity["entity_name"],
                        relationship_type=entity.get("relationship_type"),
                        related_entity_name=entity.get("related_entity_name"),
                        source_link=entity.get("source_link"),
                        metadata=entity.get("metadata", {}),
                        fingerprint=entity["fingerprint"]
                    )
                    session.add(knowledge)
                    created_count += 1
                    
                    # Create knowledge relation if applicable
                    if entity.get("related_entity_name") and entity.get("relationship_type"):
                        relation = KnowledgeRelation(
                            source_id=knowledge.id,
                            target_id=generate_id(entity["related_entity_name"]),
                            target_type="external_knowledge",
                            predicate=entity["relationship_type"],
                            confidence=0.9,
                            metadata={"auto_generated": True}
                        )
                        session.add(relation)
                    
                except Exception as e:
                    logger.warning(f"Failed to create entity {entity.get('entity_name', 'unknown')}: {e}")
                    continue
            
            await session.commit()

        # Save standardized output
        output_data = {
            "source": source,
            "total_entities": len(entities),
            "created_count": created_count,
            "duplicate_count": duplicate_count,
            "entities": entities
        }
        
        async with aiofiles.open(output_path, 'w', encoding='utf-8') as f:
            await f.write(json.dumps(output_data, indent=2, ensure_ascii=False))

        print(f"✅ Ingested {created_count} new entities ({duplicate_count} duplicates skipped)")
        print(f"📄 Standardized data saved to {output_path}")

    except Exception as e:
        logger.error(f"Knowledge ingestion failed: {e}")
        print(f"❌ Ingestion failed: {e}")
        sys.exit(1)
    finally:
        await db_manager.close()

    async def _detect_factual_query(self, user_input: str) -> Optional[Dict[str, Any]]:
        """Detect if user input contains factual queries that need external knowledge."""
        # Common patterns that suggest factual queries
        factual_keywords = [
            "what is", "what are", "how does", "side effects", "symptoms", "causes",
            "treatment", "medication", "drug", "studies", "research", "evidence",
            "statistics", "prevalence", "diagnosis", "therapy", "effectiveness"
        ]
        
        user_lower = user_input.lower()
        
        # Check for factual keywords
        has_factual_keywords = any(keyword in user_lower for keyword in factual_keywords)
        
        # Check for drug names or medical terms (simple heuristic)
        medical_indicators = [
            "adhd", "adderall", "ritalin", "anxiety", "depression", "therapy",
            "counseling", "psychiatrist", "psychologist", "medication", "diagnosis"
        ]
        
        has_medical_terms = any(term in user_lower for term in medical_indicators)
        
        if has_factual_keywords or has_medical_terms:
            return {
                "is_factual": True,
                "confidence": 0.8 if has_factual_keywords else 0.6,
                "detected_terms": [term for term in medical_indicators if term in user_lower]
            }
        
        return None

    async def _query_external_knowledge(self, user_input: str, detected_terms: List[str]) -> List[Dict[str, Any]]:
        """Query the Super-Graph for relevant external knowledge."""
        try:
            async with self.db_manager.get_session() as session:
                # Build search terms from user input and detected terms
                search_terms = detected_terms + user_input.lower().split()
                search_terms = [term.strip() for term in search_terms if len(term) > 2]
                
                if not search_terms:
                    return []
                
                # Query external knowledge with fuzzy matching
                knowledge_results = []
                
                for term in search_terms[:5]:  # Limit to first 5 terms
                    # Search entity names
                    stmt = select(ExternalKnowledge).where(
                        ExternalKnowledge.entity_name.ilike(f"%{term}%")
                    ).limit(5)
                    
                    result = await session.execute(stmt)
                    entities = result.scalars().all()
                    
                    for entity in entities:
                        knowledge_results.append({
                            "entity_name": entity.entity_name,
                            "entity_type": entity.entity_type,
                            "relationship_type": entity.relationship_type,
                            "related_entity_name": entity.related_entity_name,
                            "source_kg": entity.source_kg,
                            "source_link": entity.source_link,
                            "metadata": entity.metadata or {},
                            "relevance_term": term
                        })
                
                # Also search related entity names
                for term in search_terms[:3]:  # Fewer for related entities
                    stmt = select(ExternalKnowledge).where(
                        ExternalKnowledge.related_entity_name.ilike(f"%{term}%")
                    ).limit(3)
                    
                    result = await session.execute(stmt)
                    entities = result.scalars().all()
                    
                    for entity in entities:
                        knowledge_results.append({
                            "entity_name": entity.entity_name,
                            "entity_type": entity.entity_type,
                            "relationship_type": entity.relationship_type,
                            "related_entity_name": entity.related_entity_name,
                            "source_kg": entity.source_kg,
                            "source_link": entity.source_link,
                            "metadata": entity.metadata or {},
                            "relevance_term": term
                        })
                
                # Remove duplicates and limit results
                seen_fingerprints = set()
                unique_results = []
                
                for result in knowledge_results:
                    fingerprint = f"{result['entity_name']}:{result.get('relationship_type', '')}:{result.get('related_entity_name', '')}"
                    if fingerprint not in seen_fingerprints:
                        seen_fingerprints.add(fingerprint)
                        unique_results.append(result)
                        
                        if len(unique_results) >= 10:  # Limit total results
                            break
                
                logger.debug(f"Found {len(unique_results)} external knowledge items for terms: {search_terms}")

class GoldenStandardValidator:
    """Validates AI performance against expert dialogues and generates corrections."""
    
    def __init__(self, api_base_url: str = "http://localhost:8000"):
        self.api_base_url = api_base_url
        self.jinja_env = Environment(loader=FileSystemLoader("configs/prompts"))
        self.client = httpx.AsyncClient(timeout=30.0)
        
    async def load_golden_dialogues(self, file_path: Path) -> List[Dict[str, Any]]:
        """Load golden standard dialogues from JSON file."""
        try:
            async with aiofiles.open(file_path, 'r', encoding='utf-8') as f:
                content = await f.read()
                data = json.loads(content)
                
            # Handle different JSON structures
            if isinstance(data, list):
                dialogues = data
            elif isinstance(data, dict) and 'dialogues' in data:
                dialogues = data['dialogues']
            elif isinstance(data, dict) and 'turns' in data:
                # Single dialogue format
                dialogues = [data]
            else:
                # Assume it's a dict with dialogue objects as values
                dialogues = list(data.values()) if isinstance(data, dict) else []
            
            logger.info(f"Loaded {len(dialogues)} golden standard dialogues")
            return dialogues
            
        except Exception as e:
            logger.error(f"Failed to load golden dialogues: {e}")
            return []

    async def simulate_conversation(self, dialogue: Dict[str, Any], user_id: str) -> List[Dict[str, Any]]:
        """Simulate a conversation with HelloKoa API using golden dialogue."""
        try:
            conversation_history = []
            ai_responses = []
            
            turns = dialogue.get('turns', [])
            if not turns:
                logger.warning(f"No turns found in dialogue {dialogue.get('dialogue_id', 'unknown')}")
                return []
            
            for i, turn in enumerate(turns):
                if turn.get('speaker') == 'client':
                    client_input = turn.get('text', '').strip()
                    if not client_input:
                        continue
                    
                    # Send to HelloKoa API
                    try:
                        response = await self.client.post(
                            f"{self.api_base_url}/api/v1/chat",
                            json={
                                "message": client_input,
                                "user_id": user_id,
                                "conversation_history": conversation_history
                            }
                        )
                        
                        if response.status_code == 200:
                            ai_response_data = response.json()
                            ai_response_text = ai_response_data.get('response_text', '')
                            
                            # Find corresponding therapist response in golden standard
                            expert_response = None
                            for j in range(i + 1, len(turns)):
                                if turns[j].get('speaker') == 'therapist':
                                    expert_response = turns[j].get('text', '').strip()
                                    break
                            
                            if expert_response:
                                ai_responses.append({
                                    'client_input': client_input,
                                    'ai_response': ai_response_text,
                                    'expert_response': expert_response,
                                    'turn_number': i,
                                    'interaction_id': ai_response_data.get('interaction_id'),
                                    'metadata': ai_response_data.get('metadata', {})
                                })
                            
                            # Update conversation history
                            conversation_history.extend([client_input, ai_response_text])
                            
                        else:
                            logger.warning(f"API request failed with status {response.status_code}")
                            
                    except Exception as e:
                        logger.error(f"Failed to get AI response for turn {i}: {e}")
                        continue
                        
                    # Add small delay to avoid overwhelming the API
                    await asyncio.sleep(0.5)
            
            return ai_responses
            
        except Exception as e:
            logger.error(f"Failed to simulate conversation: {e}")
            return []

    async def critique_response(self, client_input: str, ai_response: str, expert_response: str) -> Optional[Dict[str, Any]]:
        """Use LLM to critique AI response against expert response."""
        try:
            template = self.jinja_env.get_template("critique_and_correct_prompt.txt")
            prompt = template.render(
                client_input=sanitize_text(client_input),
                ai_response=sanitize_text(ai_response),
                expert_response=sanitize_text(expert_response)
            )
            
            # For this script, we'll use the HelloKoa API to generate the critique
            # In a production system, you might use a separate model
            response = await self.client.post(
                f"{self.api_base_url}/api/v1/chat",
                json={
                    "message": prompt,
                    "user_id": "critique_system",
                    "conversation_history": []
                }
            )
            
            if response.status_code == 200:
                critique_text = response.json().get('response_text', '')
                
                # Try to parse as JSON
                try:
                    # Extract JSON from response if it's wrapped in other text
                    start_idx = critique_text.find('{')
                    end_idx = critique_text.rfind('}') + 1
                    
                    if start_idx >= 0 and end_idx > start_idx:
                        json_str = critique_text[start_idx:end_idx]
                        critique_data = json.loads(json_str)
                        return critique_data
                    
                except json.JSONDecodeError:
                    logger.warning(f"Failed to parse critique as JSON: {critique_text}")
                
                # Fallback: create structured response from text
                return {
                    "critique": critique_text,
                    "identified_skill_name": "Unknown",
                    "dora_category": "relational_depth"
                }
            
            return None
            
        except Exception as e:
            logger.error(f"Failed to generate critique: {e}")
            return None

    async def generate_corrective_examples(self, ai_responses: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
        """Generate corrective training examples from critiqued responses."""
        corrective_examples = []
        
        for response_data in ai_responses:
            try:
                critique = await self.critique_response(
                    response_data['client_input'],
                    response_data['ai_response'],
                    response_data['expert_response']
                )
                
                if not critique:
                    continue
                
                # Create training example using expert response as the target
                example = {
                    "input_text": response_data['client_input'],
                    "output_text": response_data['expert_response'],
                    "quality_score": 9.5,  # High quality since it's from expert
                    "source_type": "golden_standard_correction",
                    "dora_category": critique.get('dora_category', 'relational_depth'),
                    "component_id": None,  # Will be assigned later if component exists
                    "user_id": None,
                    "weight_multiplier": 1.5,  # Higher weight for corrections
                    "metadata": {
                        "ai_response": response_data['ai_response'],
                        "critique": critique.get('critique', ''),
                        "identified_skill": critique.get('identified_skill_name', ''),
                        "turn_number": response_data.get('turn_number'),
                        "original_interaction_id": response_data.get('interaction_id')
                    }
                }
                
                corrective_examples.append(example)
                
            except Exception as e:
                logger.error(f"Failed to generate corrective example: {e}")
                continue
        
        return corrective_examples

    async def close(self):
        """Close HTTP client."""
        await self.client.aclose()

async def validate_with_golden_standards(golden_file: Path, output_file: Path, max_dialogues: int = 10):
    """Main validation function."""
    print(f"🔍 Validating HelloKoa against golden standards from {golden_file}")
    
    validator = GoldenStandardValidator()
    
    try:
        # Load golden dialogues
        dialogues = await validator.load_golden_dialogues(golden_file)
        if not dialogues:
            print("❌ No dialogues loaded")
            return
        
        # Limit number of dialogues to process
        dialogues = dialogues[:max_dialogues]
        print(f"📝 Processing {len(dialogues)} dialogues")
        
        all_corrective_examples = []
        processed_count = 0
        
        for i, dialogue in enumerate(dialogues):
            try:
                dialogue_id = dialogue.get('dialogue_id', f'dialogue_{i}')
                user_id = f"validation_user_{generate_id(dialogue_id)}"
                
                print(f"🗣️ Processing dialogue {i+1}/{len(dialogues)}: {dialogue_id}")
                
                # Simulate conversation
                ai_responses = await validator.simulate_conversation(dialogue, user_id)
                
                if ai_responses:
                    # Generate corrective examples
                    corrective_examples = await validator.generate_corrective_examples(ai_responses)
                    all_corrective_examples.extend(corrective_examples)
                    
                    print(f"✅ Generated {len(corrective_examples)} corrective examples from {len(ai_responses)} responses")
                else:
                    print(f"⚠️ No valid responses generated for dialogue {dialogue_id}")
                
                processed_count += 1
                
                # Add delay between dialogues
                await asyncio.sleep(1.0)
                
            except Exception as e:
                logger.error(f"Failed to process dialogue {i}: {e}")
                continue
        
        # Save corrective examples
        output_data = {
            "validation_metadata": {
                "total_dialogues_processed": processed_count,
                "total_corrective_examples": len(all_corrective_examples),
                "golden_standard_file": str(golden_file),
                "validation_timestamp": "2025-06-07T12:00:00Z"
            },
            "examples": all_corrective_examples
        }
        
        async with aiofiles.open(output_file, 'w', encoding='utf-8') as f:

            print("\n📈 DoRA Category Distribution:")
            for category, count in dora_categories.items():
                print(f"  {category}: {count} examples")
    
    except Exception as e:
        logger.error(f"Validation failed: {e}")
        print(f"❌ Validation failed: {e}")
        sys.exit(1)
    
    finally:
        await validator.close()

class ExternalKGService:
    """Service for querying and managing external knowledge graph data."""
    
    def __init__(self, db_manager: DatabaseManager, cache_manager: CacheManager):
        self.db_manager = db_manager
        self.cache_manager = cache_manager
        self.config = get_config()
        self.cache_namespace = "external_kg"
        
        # Common medical/therapeutic terms that suggest factual queries
        self.factual_indicators = {
            "drugs": ["medication", "drug", "medicine", "pill", "tablet", "adderall", "ritalin", "concerta", "vyvanse"],
            "conditions": ["adhd", "depression", "anxiety", "bipolar", "ptsd", "ocd", "autism"],
            "symptoms": ["side effects", "symptoms", "adverse", "reaction", "effectiveness", "efficacy"],
            "queries": ["what is", "what are", "how does", "does", "can", "will", "causes", "treats", "helps"],
            "research": ["study", "studies", "research", "evidence", "clinical", "trial", "effective"]
        }

    async def detect_factual_query(self, user_input: str) -> Optional[Dict[str, Any]]:
        """Detect if user input contains factual medical/therapeutic queries."""
        try:
            user_lower = user_input.lower()
            
            # Check for factual query patterns
            query_score = 0
            detected_categories = []
            detected_terms = []
            
            for category, terms in self.factual_indicators.items():
                for term in terms:
                    if term in user_lower:
                        query_score += 1
                        if category not in detected_categories:
                            detected_categories.append(category)
                        detected_terms.append(term)
            
            # Calculate confidence based on matches
            if query_score > 0:
                confidence = min(0.9, 0.3 + (query_score * 0.1))
                
                return {
                    "is_factual": True,
                    "confidence": confidence,
                    "detected_categories": detected_categories,
                    "detected_terms": detected_terms,
                    "query_score": query_score
                }
            
            return None
            
        except Exception as e:
            logger.error(f"Failed to detect factual query: {e}")
            return None

    async def query_kg(self, source_kg: Optional[str], search_terms: List[str]) -> List[Dict[str, Any]]:
        """Query external knowledge graph for relevant information."""
        try:
            if not search_terms:
                return []
            
            cache_key = f"{source_kg or 'all'}:{':'.join(sorted(search_terms))}"
            cached_result = await self.cache_manager.get(self.cache_namespace, cache_key)
            
            if cached_result:
                logger.debug(f"Cache hit for KG query: {cache_key}")
                return cached_result
            
            async with self.db_manager.get_session() as session:
                # Build query based on search terms
                results = []
                
                for term in search_terms[:5]:  # Limit to prevent overly complex queries
                    # Query entity names
                    stmt = select(ExternalKnowledge).where(
                        ExternalKnowledge.entity_name.ilike(f"%{term}%")
                    )
                    
                    if source_kg:
                        stmt = stmt.where(ExternalKnowledge.source_kg == source_kg)
                    
                    stmt = stmt.limit(5)
                    
                    result = await session.execute(stmt)
                    entities = result.scalars().all()
                    
                    for entity in entities:
                        entity_dict = {
                            "entity_id": entity.entity_id,
                            "entity_name": entity.entity_name,
                            "entity_type": entity.entity_type,
                            "relationship_type": entity.relationship_type,
                            "related_entity_name": entity.related_entity_name,
                            "source_kg": entity.source_kg,
                            "source_link": entity.source_link,
                            "metadata": entity.metadata or {},
                            "search_term": term,
                            "match_type": "entity_name"
                        }
                        results.append(entity_dict)
                
                # Also search related entity names
                for term in search_terms[:3]:
                    stmt = select(ExternalKnowledge).where(
                        ExternalKnowledge.related_entity_name.ilike(f"%{term}%")
                    )
                    
                    if source_kg:
                        stmt = stmt.where(ExternalKnowledge.source_kg == source_kg)
                    
                    stmt = stmt.limit(3)
                    
                    result = await session.execute(stmt)
                    entities = result.scalars().all()
                    
                    for entity in entities:
                        entity_dict = {
                            "entity_id": entity.entity_id,
                            "entity_name": entity.entity_name,
                            "entity_type": entity.entity_type,
                            "relationship_type": entity.relationship_type,
                            "related_entity_name": entity.related_entity_name,
                            "source_kg": entity.source_kg,
                            "source_link": entity.source_link,
                            "metadata": entity.metadata or {},
                            "search_term": term,
                            "match_type": "related_entity"
                        }
                        results.append(entity_dict)
                
                # Remove duplicates and limit results
                unique_results = []
                seen_entities = set()
                
                for result in results:
                    entity_key = f"{result['entity_id']}:{result.get('relationship_type', '')}"
                    if entity_key not in seen_entities:
                        seen_entities.add(entity_key)
                        unique_results.append(result)
                        
                        if len(unique_results) >= self.config.external_kg.max_external_results:
                            break
                
                # Cache results
                await self.cache_manager.set(
                    self.cache_namespace,
                    cache_key,
                    unique_results,
                    ttl=self.config.external_kg.kg_cache_ttl
                )
                
                logger.debug(f"Found {len(unique_results)} external knowledge items for terms: {search_terms}")
                return unique_results
                
        except Exception as e:
            logger.error(f"Failed to query external KG: {e}")
            return []
