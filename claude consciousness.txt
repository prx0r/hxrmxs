import yaml
import torch
from pathlib import Path
import logging

logger = logging.getLogger("config")

class CQEConfig:
    """Configuration manager for the Cosmic Quantum Ecosystem."""
    
    def __init__(self, config_path=None):
        # Default configurations
        self._initialize_defaults()
        
        # Load from file if provided
        if config_path:
            self._load_from_file(config_path)
            
        # Post-initialization processing
        self._post_init()
        
        logger.info(f"Configuration initialized with N={self.N}, k={self.k}, d={self.d}")
    
    def _initialize_defaults(self):
        """Initialize default configuration values."""
        # System dimensions
        self.N = 100  # Number of neurons/agents
        self.m = 10   # Sensory input dimension
        self.d = 8    # Field dimension
        self.k = 8    # Number of qubits
        
        # Time parameters
        self.dt = 0.01  # Time step (s)
        self.T = 10.0   # Simulation duration (s)
        
        # Network parameters
        self.p_sparse = 0.2  # Edge sparsity
        
        # ... (initialize all other parameters from the YAML structure)
        
        # Output settings
        self.checkpoint_dir = "./cosmic_quantum_checkpoints"
        self.output_dir = "./cosmic_quantum_output"
        
        # Device
        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    
    def _load_from_file(self, config_path):
        """Load configuration from YAML file."""
        try:
            with open(config_path, 'r') as file:
                config_data = yaml.safe_load(file)
            
            # Process each section
            if 'dimensions' in config_data:
                self.N = config_data['dimensions'].get('N', self.N)
                self.m = config_data['dimensions'].get('m', self.m)
                self.d = config_data['dimensions'].get('d', self.d)
                self.k = config_data['dimensions'].get('k', self.k)
            
            # ... (load all other sections)
            
            if 'system' in config_data:
                if 'output_dir' in config_data['system']:
                    self.output_dir = config_data['system']['output_dir']
                if 'checkpoint_dir' in config_data['system']:
                    self.checkpoint_dir = config_data['system']['checkpoint_dir']
                if 'device' in config_data['system']:
                    self.device = torch.device(config_data['system']['device'])
            
            logger.info(f"Loaded configuration from {config_path}")
            
        except Exception as e:
            logger.error(f"Error loading configuration: {e}")
            logger.warning("Using default configuration")
    
    def _post_init(self):
        """Post-initialization processing."""
        # Create output directories
        Path(self.checkpoint_dir).mkdir(exist_ok=True)
        Path(self.output_dir).mkdir(exist_ok=True)
        
        # Normalize weights
        total_weight = sum(self.metric_weights.values())
        if abs(total_weight - 1.0) > 1e-6:
            logger.warning(f"Metric weights sum to {total_weight}, normalizing")
            factor = 1.0 / total_weight
            for key in self.metric_weights:
                self.metric_weights[key] *= factor
        
        # Set random seed if specified
        if hasattr(self, 'seed') and self.seed is not None:
            torch.manual_seed(self.seed)
            import numpy as np
            np.random.seed(self.seed)
            import random
            random.seed(self.seed)
    
    def to_dict(self):
        """Convert configuration to dictionary."""
        return {key: getattr(self, key) for key in dir(self) 
                if not key.startswith('_') and not callable(getattr(self, key))}
    
    def save(self, filepath):
        """Save configuration to YAML file."""
        try:
            # Convert to serializable format
            config_dict = self.to_dict()
            
            # Handle non-serializable types
            for key, value in config_dict.items():
                if isinstance(value, torch.device):
                    config_dict[key] = str(value)
            
            with open(filepath, 'w') as file:
                yaml.dump(config_dict, file, default_flow_style=False)
                
            logger.info(f"Configuration saved to {filepath}")
            
        except Exception as e:
            logger.error(f"Failed to save configuration: {e}")

import torch

class CQEAnalyzer:
    """Analysis tools for the Cosmic Quantum Ecosystem."""
    
    def __init__(self, output_dir):
        self.output_dir = Path(output_dir)
        self.output_dir.mkdir(exist_ok=True)
        
    def load_metrics_history(self, filepath):
        """Load metrics history from a file."""
        try:
            data = torch.load(filepath)
            return data.get('metrics_history', [])
        except Exception as e:
            logger.error(f"Failed to load metrics history: {e}")
            return []
    
    def analyze_consciousness_evolution(self, metrics_history):
        """Analyze the evolution of consciousness metrics."""
        if not metrics_history:
            logger.warning("No metrics history to analyze")
            return {}
        
        # Extract metrics over time
        times = [m['time'] for m in metrics_history]
        metrics = {}
        
        # Get all metrics in the history
        all_keys = set()
        for m in metrics_history:
            all_keys.update(m.keys())
        
        # Remove non-metric keys
        non_metrics = {'time'}
        metric_keys = all_keys - non_metrics
        
        # Extract each metric
        for key in metric_keys:
            metrics[key] = [m.get(key, 0) for m in metrics_history]
        
        # Calculate trends
        trends = {}
        for key, values in metrics.items():
            if len(values) > 1:
                slope, intercept, r_value, p_value, std_err = stats.linregress(times, values)
                trends[key] = {
                    'slope': slope,
                    'intercept': intercept,
                    'r_value': r_value,
                    'p_value': p_value,
                    'std_err': std_err
                }
        
        # Calculate stability (inverse variance)
        stability = {}
        for key, values in metrics.items():
            if len(values) > 1:
                stability[key] = 1.0 / (np.var(values) + 1e-10)
        
        # Detect periodic patterns
        periodicity = {}
        for key, values in metrics.items():
            if len(values) > 20:
                try:
                    # Use FFT to find dominant frequencies
                    fft = np.abs(np.fft.rfft(values))
                    freqs = np.fft.rfftfreq(len(values), times[1] - times[0])
                    
                    # Find peaks
                    from scipy.signal import find_peaks
                    peaks, _ = find_peaks(fft)
                    
                    if peaks.size > 0:
                        # Get the dominant frequency
                        dominant_idx = peaks[np.argmax(fft[peaks])]
                        dominant_freq = freqs[dominant_idx]
                        dominant_period = 1.0 / dominant_freq if dominant_freq > 0 else 0
                        
                        periodicity[key] = {
                            'dominant_frequency': dominant_freq,
                            'dominant_period': dominant_period,
                            'amplitude': fft[dominant_idx] / len(values)
                        }
                except Exception as e:
                    logger.error(f"Error in periodicity analysis for {key}: {e}")
        
        return {
            'metrics': metrics,
            'trends': trends,
            'stability': stability,
            'periodicity': periodicity
        }

def setup_logging(log_file=None, log_level=logging.INFO):
    """Set up logging configuration."""
    handlers = [logging.StreamHandler()]
    
    if log_file:
        handlers.append(logging.FileHandler(log_file))
        
    logging.basicConfig(
        level=log_level,
        format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
        handlers=handlers
    )

def explore_parameters():
    """Explore the parameter space by running multiple simulations."""
    logger.info("Starting parameter exploration")
    
    # Parameters to explore
    parameters = {
        'N': [20, 50, 100],         # System size
        'gamma_q': [0.005, 0.01, 0.02],  # Quantum coupling
        'kappa': [0.05, 0.1, 0.2]   # Field reaction strength
    }
    
    # Create output directory
    output_dir = Path("./example_output/parameter_exploration")
    output_dir.mkdir(parents=True, exist_ok=True)
    
    # Store results
    results = []
    
    # Generate all parameter combinations
    combinations = list(itertools.product(*parameters.values()))
    param_names = list(parameters.keys())
    
    logger.info(f"Exploring {len(combinations)} parameter combinations")
    
    # Run simulations for each combination
    for i, values in enumerate(combinations):
        params = dict(zip(param_names, values))
        logger.info(f"Combination {i+1}/{len(combinations)}: {params}")
        
        # Create configuration with these parameters
        config = CQEConfig()
        for key, value in params.items():
            setattr(config, key, value)
        
        # Set shorter duration for exploration
        config.T = 2.0

        combo_dir = output_dir / f"combo_{i+1}"
        combo_dir.mkdir(exist_ok=True)

            'betti_numbers': final_metrics.get('betti_numbers', 0),

        """Override to add recurrent processing."""
        # First run standard update
        action_potential = super().update_neural(S, field_state, quantum_state, t, dt, epsilon)
        
        # Update memory buffer with new field state
        self.memory_buffer = torch.cat(
            [self.memory_buffer[:, 1:, :], field_state.unsqueeze(1)], 
            dim=1

# Function timing decorator
def time_function(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        start_time = time.time()
        result = func(*args, **kwargs)
        end_time = time.time()
        elapsed = end_time - start_time
        logger.debug(f"Function {func.__name__} executed in {elapsed:.6f} seconds")
        return result
    return wrapper

class PerformanceTracker:
    """Track and analyze system performance metrics."""
    
    def __init__(self, output_dir=None):
        self.timing_data = {}

    def start_timing(self, component_name):
        """Start timing a component."""
        if component_name not in self.timing_data:
            self.timing_data[component_name] = {'starts': [], 'ends': [], 'durations': []}
        
        self.timing_data[component_name]['starts'].append(time.time())
    
    def end_timing(self, component_name):
        """End timing a component and record duration."""
        if component_name not in self.timing_data:
            logger.warning(f"No timing start found for {component_name}")
            return
        
        end_time = time.time()
        self.timing_data[component_name]['ends'].append(end_time)
        
        # Calculate duration
        start_time = self.timing_data[component_name]['starts'][-1]
        duration = end_time - start_time
        self.timing_data[component_name]['durations'].append(duration)
        
        return duration
    
    def record_memory_usage(self, component_name):
        """Record current GPU and CPU memory usage for a component."""
        if component_name not in self.memory_usage:
            self.memory_usage[component_name] = {
                'gpu': [],
                'cpu': []
            }
        
        # Get CPU memory
        import psutil
        cpu_memory = psutil.Process().memory_info().rss / (1024 * 1024)  # MB
        
        # Get GPU memory if available
        gpu_memory = 0
        if torch.cuda.is_available():
            try:
                gpu_memory = torch.cuda.memory_allocated() / (1024 * 1024)  # MB
            except:
                pass
        
        self.memory_usage[component_name]['cpu'].append(cpu_memory)
        self.memory_usage[component_name]['gpu'].append(gpu_memory)
        
        return cpu_memory, gpu_memory

def setup_distributed(rank, world_size):

    dist.destroy_process_group()
    logger.info("Cleaned up distributed environment")

def run_distributed(fn, world_size, *args, **kwargs):
    """Run a function in a distributed setting."""
    mp.spawn(
        distributed_worker,
        args=(world_size, fn, args, kwargs),
        nprocs=world_size,
        join=True
    )

def distributed_worker(rank, world_size, fn, args, kwargs):
    """Worker function for distributed computing."""
    # Set up distributed environment
    setup_distributed(rank, world_size)
    
    # Update device based on rank
    if torch.cuda.is_available() and torch.cuda.device_count() >= world_size:
        device = torch.device(f"cuda:{rank}")
    else:
        device = torch.device("cpu")
    
    # Run the function
    try:
        if 'config' in kwargs:
            kwargs['config'].device = device
            
        result = fn(rank=rank, world_size=world_size, *args, **kwargs)
        
        # Wait for all processes
        dist.barrier()
        
        return result
    finally:
        # Clean up
        cleanup_distributed()

class NeuralDynamics(nn.Module):
    """
    Neural dynamics module implementing Hodgkin-Huxley dynamics with 
    excitonic coupling and adaptive learning.
    """
    def __init__(self, config):
        super().__init__()
        self.config = config
        
        # Neural state variables
        self.V = nn.Parameter(torch.full((config.N,), config.V_rest, device=device))
        self.m = nn.Parameter(torch.full((config.N,), 0.05, device=device))
        self.h = nn.Parameter(torch.full((config.N,), 0.6, device=device))
        self.n = nn.Parameter(torch.full((config.N,), 0.32, device=device))
        
        # Synaptic weights
        self.W = nn.Parameter(torch.normal(0, 0.01, (config.N, config.N), device=device))
        
        # Memory systems
        self.M_s = nn.Parameter(torch.zeros(config.N, config.d, dtype=torch.complex64, device=device))
        self.M_l = nn.Parameter(torch.zeros(config.N, config.d, device=device))
        
        # Semantic and apobetic states
        self.S_sem = nn.Parameter(torch.zeros(config.N, config.d, device=device))
        self.P = nn.Parameter(torch.zeros(config.N, config.d, device=device))
        
        # Thresholds
        self.V_th = nn.Parameter(torch.full((config.N,), config.V_th, device=device))
        
        # Action potentials (spikes)
        self.action_potential = torch.zeros(config.N, device=device)
        
        # Spike history
        self.spike_times = [[] for _ in range(config.N)]
        self.firing_history = []
        
        logger.info(f"Initialized NeuralDynamics with {config.N} neurons")

    def compute_gates(self, V):
        """Compute gating variables for Hodgkin-Huxley model"""
        alpha_m = 0.1 * (V + 40.0) / (1.0 - torch.exp(-(V + 40.0) / 10.0))
        beta_m = 4.0 * torch.exp(-(V + 65.0) / 18.0)
        alpha_h = 0.07 * torch.exp(-(V + 65.0) / 20.0)
        beta_h = 1.0 / (1.0 + torch.exp(-(V + 35.0) / 10.0))
        alpha_n = 0.01 * (V + 55.0) / (1.0 - torch.exp(-(V + 55.0) / 10.0))
        beta_n = 0.125 * torch.exp(-(V + 65.0) / 80.0)
        return alpha_m, beta_m, alpha_h, beta_h, alpha_n, beta_n

    def update_rhythms(self, t):
        """Generate cortical rhythms, emphasizing 40Hz gamma"""
        rhythms = torch.zeros(self.config.N, device=device)
        for f, a in zip(self.config.f, self.config.a_f):
            rhythms += a * torch.sin(2 * np.pi * f * t)
        return rhythms

    def compute_inputs(self, S, field_state, quantum_state, excitonic_state, rhythms, t):
        """Compute total inputs to neurons from all sources"""
        # Sensory input
        I_sensory = S.mean(dim=0)
        
        # Synaptic input
        I_synaptic = torch.matmul(self.W, self.action_potential)
        
        # Field input
        I_field = self.config.kappa * field_state.sum(dim=1)
        
        # Quantum input
        I_quantum = self.config.gamma_q * quantum_state
        
        # Excitonic input
        I_excitonic = self.config.kappa_e * excitonic_state.real.sum(dim=1)
        
        # Rhythm input
        I_rhythm = rhythms
        
        # Total input
        return I_sensory + I_synaptic + I_field + I_quantum + I_excitonic + I_rhythm

    def update_neural(self, S, field_state, quantum_state, excitonic_state, t, dt, epsilon=None):
        """Update neural state variables"""
        # Generate rhythms
        rhythms = self.update_rhythms(t)
        
        # Calculate total inputs
        I = self.compute_inputs(S, field_state, quantum_state, excitonic_state, rhythms, t)
        
        # Calculate gating rates
        alpha_m, beta_m, alpha_h, beta_h, alpha_n, beta_n = self.compute_gates(self.V)
        
        # Update gating variables
        dm_dt = alpha_m * (1 - self.m) - beta_m * self.m
        dh_dt = alpha_h * (1 - self.h) - beta_h * self.h
        dn_dt = alpha_n * (1 - self.n) - beta_n * self.n
        
        # Calculate ionic currents
        I_Na = self.config.g_Na * self.m ** 3 * self.h * (self.V - self.config.E_Na)
        I_K = self.config.g_K * self.n ** 4 * (self.V - self.config.E_K)
        I_L = self.config.g_L * (self.V - self.config.E_L)
        
        # Generate noise
        if epsilon is None:
            epsilon = self.config.epsilon_0
            
        xi = torch.normal(0, self.config.sigma, size=(self.config.N,), device=device)
        
        # Update membrane potential
        dV_dt = (I - I_Na - I_K - I_L) / self.config.C_m + epsilon * xi
        
        # Use Euler method to update state variables
        V_new = self.V + dt * dV_dt
        m_new = self.m + dt * dm_dt
        h_new = self.h + dt * dh_dt
        n_new = self.n + dt * dn_dt
        
        # Update neural states
        self.V.data = V_new
        self.m.data = m_new
        self.h.data = h_new
        self.n.data = n_new
        
        # Calculate action potentials
        self.action_potential = (self.V > self.V_th).float()
        
        # Update spike times
        for i in range(self.config.N):
            if self.action_potential[i] > 0:
                self.spike_times[i].append(t)
                
        # Update firing history
        self.firing_history.append(self.action_potential.clone())
        if len(self.firing_history) > 1000:
            self.firing_history.pop(0)
            
        return self.action_potential

    def compute_stdp(self):
        """Compute spike-timing dependent plasticity"""
        stdp = torch.zeros((self.config.N, self.config.N), device=device)
        
        # For each pair of neurons, calculate STDP
        for i in range(self.config.N):
            if not self.spike_times[i]:
                continue
                
            for j in range(self.config.N):
                if not self.spike_times[j]:
                    continue
                    
                # Get latest spike times
                t_i = self.spike_times[i][-1]
                t_j = self.spike_times[j][-1]
                
                # Calculate time difference
                delta_t = t_i - t_j
                
                # Calculate STDP effect
                stdp[i, j] = torch.exp(-abs(delta_t) / self.config.tau_stdp) * torch.sign(torch.tensor(delta_t))
                
        return stdp

    def update_weights(self, dt):
        """Update synaptic weights based on STDP and Hebbian learning"""
        # Compute STDP
        stdp = self.compute_stdp()
        
        # Hebbian term
        hebbian = torch.outer(self.action_potential, self.action_potential)
        
        # Weight update
        dW = self.config.eta * hebbian + self.config.eta_stdp * stdp - self.config.lambda_decay * self.W
        
        # Apply update
        W_new = self.W + dt * dW
        
        # Update weights
        self.W.data = W_new
        
        return self.W

    def update_memory(self, S, excitonic_state, dt):
        """Update short-term and long-term memory"""
        # Short-term memory update
        dM_s = self.action_potential.unsqueeze(1) * (S.mean(dim=0).unsqueeze(1) + self.config.gamma_e * excitonic_state) - self.config.alpha * self.M_s
        
        # Long-term memory update
        dM_l = torch.real(self.M_s * self.P) - self.config.alpha * self.M_l
        
        # Apply updates
        self.M_s.data = self.M_s + dt * dM_s
        self.M_l.data = self.M_l + dt * dM_l
        
        return self.M_s, self.M_l

    def update_semantic(self, field_state, excitonic_state, phi, dt):
        """Update semantic and apobetic states"""
        # Semantic state update
        dS_sem = self.config.eta_s * (torch.sum(field_state.unsqueeze(1) * self.M_l, dim=2) + 
                                      self.config.gamma_e * excitonic_state.real + 
                                      self.config.gamma_phi * phi) - self.S_sem
        
        # Apply update
        self.S_sem.data = self.S_sem + dt * dS_sem
        
        return self.S_sem

    def update_apobetic(self, graph, excitonic_state, phi, dt):
        """Update apobetic (goal-oriented) states"""
        # Create adjacency matrix from graph
        adj = torch.zeros((self.config.N, self.config.N), device=device)
        for i, j in graph.edges():
            adj[i, j] = graph[i][j]['weight']
            adj[j, i] = graph[i][j]['weight']
            
        # Apobetic state update
        dP = self.config.beta_p * self.S_sem + self.config.gamma_p * torch.matmul(adj, self.S_sem) + self.config.gamma_e * excitonic_state.real + self.config.gamma_phi * phi - self.config.alpha * self.P
        
        # Apply update
        self.P.data = self.P + dt * dP
        
        return self.P

    def compute_coherence(self):
        """Compute neural coherence"""
        if self.V.shape[0] <= 1:
            return 0.995
            
        return torch.mean(torch.abs(torch.corrcoef(self.V.unsqueeze(0)))).item()

    def compute_lyapunov(self):
        """Compute Lyapunov exponent to measure metastability"""
        if len(self.firing_history) < 2:
            return 0.0
            
        dv_dt = (self.firing_history[-1] - self.firing_history[-2]) / self.config.dt
        norm = torch.norm(dv_dt).item()
        return np.log(max(norm, 1e-10))

    def meta_update(self, Psi, t):
        """Implement meta-learning for neural parameters"""
        # Calculate error signals
        Psi_error = Psi.mean() - self.config.target_Psi
        spike_rate = self.action_potential.mean()
        spike_rate_error = spike_rate - self.config.target_spike_rate
        
        # Update threshold based on spike rate
        dV_th = self.config.eta_meta * spike_rate_error
        self.V_th.data = torch.clamp(self.V_th - dV_th, -50.0, -30.0)
        
        # Calculate Lyapunov exponent
        lyapunov = self.compute_lyapunov()
        
        # Adjust noise amplitude based on Lyapunov exponent
        epsilon = None
        if abs(lyapunov) < self.config.lambda_0:
            epsilon = self.config.epsilon_0 * torch.exp(torch.tensor(abs(lyapunov) / self.config.lambda_0))
            
        return epsilon

    def generate_decision(self, quantum_state):
        """Generate decision based on neural and quantum states"""
        # Combine neural and quantum states
        decision_input = self.action_potential + self.config.gamma_q * quantum_state + self.M_s.mean(dim=1).real
        
        # Calculate probabilities
        decision_probs = F.softmax(decision_input, dim=0)
        
        # Select action
        decision = torch.argmax(decision_probs).item()
        
        return decision, decision_probs


class QuantumProcessor(nn.Module):
    """
    Quantum processor implementing quantum circuits, Lindblad evolution,
    and quantum-classical coupling.
    """
    def __init__(self, config):
        super().__init__()
        self.config = config
        
        # Initialize quantum device
        self.num_qubits = config.k
        self.qml_device = qml.device("default.qubit", wires=self.num_qubits)
        
        # Initialize quantum statistical operator (density matrix)
        self.rho = nn.Parameter(torch.eye(2**self.num_qubits, dtype=torch.complex64, device=device) / 2**self.num_qubits)
        
        # Initialize quantum state vector
        self.psi = nn.Parameter(torch.ones(2**self.num_qubits, dtype=torch.complex64, device=device) / np.sqrt(2**self.num_qubits))
        
        # Input projection
        self.input_projection = nn.Linear(config.d, self.num_qubits, dtype=torch.float32, device=device)
        
        # Quantum layer
        self.quantum_layer = qml.qnn.TorchLayer(self.quantum_circuit, {"weights": (2 * self.num_qubits,)})
        self.weights = nn.Parameter(torch.randn(2 * self.num_qubits, device=device) * 0.01)
        
        logger.info(f"Initialized QuantumProcessor with {self.num_qubits} qubits")

    @qml.qnode(device="default.qubit", interface="torch", diff_method="backprop")
    def quantum_circuit(self, inputs, weights):
        """Quantum circuit for entanglement generation and measurement"""
        # Encode inputs
        for i in range(self.num_qubits):
            qml.RY(inputs[i], wires=i)
            
        # Entangling layers
        for layer in range(2):
            # Rotation gates
            for i in range(self.num_qubits):
                qml.RZ(weights[layer * self.num_qubits + i], wires=i)
                
            # Entangling gates
            for i in range(self.num_qubits - 1):
                qml.CNOT(wires=[i, i + 1])
                
            # Close the loop
            if self.num_qubits > 2:
                qml.CNOT(wires=[self.num_qubits - 1, 0])
                
        # Measurement
        return [qml.expval(qml.PauliZ(i)) for i in range(self.num_qubits)]

    def lindblad_evolution(self, hamiltonian, rho, dt):
        """Evolve quantum state using Lindblad master equation"""
        # Calculate commutator [H, rho]
        commutator = 1j * (torch.matmul(hamiltonian, rho) - torch.matmul(rho, hamiltonian))
        
        # Create Lindblad operators (amplitude damping)
        lindblad_ops = []
        for i in range(self.num_qubits):
            # Create lowering operator for qubit i
            L_i = torch.zeros((2**self.num_qubits, 2**self.num_qubits), dtype=torch.complex64, device=device)
            
            # This is a simplified version - in a full implementation we would need to 
            # construct the actual lowering operator with proper tensor products
            idx = 2**i
            L_i[0, idx] = np.sqrt(self.config.gamma_d)
            lindblad_ops.append(L_i)
            
        # Calculate dissipation term
        dissipation = torch.zeros_like(rho)
        for L in lindblad_ops:
            L_dag = L.conj().T
            term1 = torch.matmul(torch.matmul(L, rho), L_dag)
            term2 = 0.5 * (torch.matmul(torch.matmul(L_dag, L), rho) + torch.matmul(rho, torch.matmul(L_dag, L)))
            dissipation += term1 - term2
            
        # Calculate derivative
        drho_dt = -commutator + dissipation
        
        # Update density matrix
        rho_new = rho + dt * drho_dt
        
        # Ensure the density matrix remains valid (trace 1, positive semidefinite)
        trace = torch.trace(rho_new).real
        rho_new = rho_new / trace
        
        return rho_new

    def update_quantum_state(self, field_state, neural_state, t, dt):
        """Update quantum state based on field and neural dynamics"""
        # Project field state to quantum inputs
        field_projection = self.input_projection(field_state.mean(dim=0))
        
        # Create inputs for quantum circuit
        inputs = torch.cat([field_projection, neural_state.mean().unsqueeze(0)])
        
        # Get quantum circuit output
        quantum_outputs = self.quantum_layer(inputs)
        
        # Create Hamiltonian
        H = torch.zeros((2**self.num_qubits, 2**self.num_qubits), dtype=torch.complex64, device=device)
        for i in range(self.num_qubits):
            # Diagonal terms
            H[i, i] = quantum_outputs[i].item()
            
        # Add coupling between qubits
        for i in range(self.num_qubits):
            for j in range(i+1, self.num_qubits):
                coupling = self.config.gamma_q * field_state.mean().sum().item()
                H[i, j] = coupling
                H[j, i] = coupling
                
        # Include Bohmian quantum potential
        q_bohm = -(self.config.hbar**2 / (2 * self.config.mass)) * (torch.norm(self.psi) / torch.norm(self.psi))
        H = H + q_bohm * torch.eye(2**self.num_qubits, dtype=torch.complex64, device=device)
        
        # Update density matrix using Lindblad evolution
        self.rho.data = self.lindblad_evolution(H, self.rho, dt)
        
        # Update wave function (approximation from density matrix)
        eigenvalues, eigenvectors = torch.linalg.eigh(self.rho)
        max_idx = torch.argmax(eigenvalues)
        self.psi.data = eigenvectors[:, max_idx]
        
        # Compute entanglement entropy
        S = -torch.trace(torch.matmul(self.rho, torch.log(self.rho + 1e-10))).real
        
        return quantum_outputs, S

    def compute_entanglement(self):
        """Compute entanglement measures"""
        # Partial trace for first qubit
        rho_A = torch.zeros((2, 2), dtype=torch.complex64, device=device)
        rho_A[0, 0] = self.rho[0, 0] + self.rho[2, 2]
        rho_A[0, 1] = self.rho[0, 1] + self.rho[2, 3]
        rho_A[1, 0] = self.rho[1, 0] + self.rho[3, 2]
        rho_A[1, 1] = self.rho[1, 1] + self.rho[3, 3]
        
        # Calculate entanglement entropy
        S = -torch.trace(torch.matmul(rho_A, torch.log(rho_A + 1e-10))).real
        
        # Calculate negativity (sum of negative eigenvalues)
        eigenvalues = torch.linalg.eigvalsh(self.rho)
        negativity = torch.sum(torch.abs(eigenvalues[eigenvalues < 0]))
        
        return S, negativity


class ExcitonicDynamics(nn.Module):
    """
    Excitonic dynamics module implementing quantum coherent transfer
    between neural elements.
    """
    def __init__(self, config):
        super().__init__()
        self.config = config
        
        # Initialize excitonic states
        self.psi_E = nn.Parameter(torch.randn(config.N, config.d, dtype=torch.complex64, device=device))
        
        # Initialize Hamiltonian
        self.H0 = nn.Parameter(torch.eye(config.d, dtype=torch.complex64, device=device) * 0.5)
        
        # Initialize dissipation matrix
        self.G = nn.Parameter(torch.eye(config.d, dtype=torch.complex64, device=device) * 0.01)
        
        logger.info(f"Initialized ExcitonicDynamics with {config.N} neurons and {config.d} states")

    def update_excitonic(self, field_state, neural_state, dt):
        """Update excitonic states"""
        # For each neuron
        for i in range(self.config.N):
            # Create local Hamiltonian with field coupling
            H_i = self.H0.clone()
            for a in range(self.config.d):
                H_i[a, a] = H_i[a, a] + field_state[i, a] * 0.1
                
            # Add random fluctuations (environmental coupling)
            delta = torch.randn(self.config.d, self.config.d, dtype=torch.complex64, device=device) * 0.01
            delta = delta + delta.conj().transpose(0, 1)  # Ensure Hermitian
            H_i = H_i + delta
            
            # Calculate derivative
            dpsi_dt = -1j * torch.matmul(H_i, self.psi_E[i]) - 0.5 * torch.matmul(self.G, self.psi_E[i])
            
            # Update excitonic state
            self.psi_E[i] = self.psi_E[i] + dt * dpsi_dt
            
            # Normalize
            self.psi_E[i] = self.psi_E[i] / torch.norm(self.psi_E[i])
            
        return self.psi_E


class FieldDynamics(nn.Module):
    """
    Field dynamics module implementing PDEs for vitality field evolution,
    with topological feedback and resource diffusion.
    """
    def __init__(self, config):
        super().__init__()
        self.config = config
        
        # Initialize field values
        self.psi = nn.Parameter(torch.normal(0, 0.1, (config.N, config.d), device=device))
        
        # Initialize coupling tensor (adjacency × field interactions)
        self.coupling = nn.Parameter(torch.normal(0, 0.01, (config.N, config.N), device=device))
        
        logger.info(f"Initialized FieldDynamics with {config.N} nodes and {config.d} dimensions")

    def compute_laplacian(self, graph):
        """Compute graph Laplacian for diffusion"""
        # Create adjacency matrix
        adj = torch.zeros((self.config.N, self.config.N), device=device)
        
        # Fill adjacency matrix
        for i, j in graph.edges():
            weight = graph[i][j]['weight']
            adj[i, j] = weight
            adj[j, i] = weight  # Ensure symmetry
            
        # Compute degree matrix
        deg = torch.sum(adj, dim=1)
        
        # Compute Laplacian
        L = torch.diag(deg) - adj
        
        return L

    def compute_ricci_curvature(self, graph):
        """Compute discrete Ricci curvature"""
        ricci = torch.zeros((self.config.N, self.config.N), device=device)
        
        # For each edge
        for i, j in graph.edges():
            # Get common neighbors
            i_neighbors = set(graph.neighbors(i))
            j_neighbors = set(graph.neighbors(j))
            common = i_neighbors.intersection(j_neighbors)
            
            # Compute curvature
            curvature = len(common) / max(1, min(len(i_neighbors), len(j_neighbors)))
            
            # Store
            ricci[i, j] = curvature
            ricci[j, i] = curvature
            
        return ricci

    def update_field(self, graph, neural_state, quantum_state, t, dt):
        """Update field dynamics using reaction-diffusion PDE"""
        # Get Laplacian for diffusion
        L = self.compute_laplacian(graph)
        
        # Get Ricci curvature
        ricci = self.compute_ricci_curvature(graph)
        
        # Compute field components
        
        # 1. Diffusion term
        diffusion = self.config.D * torch.matmul(L, self.psi)
        
        # 2. Reaction term (Primal Pulse)
        reaction = self.config.gamma * self.psi * (1 - torch.sum(self.psi**2, dim=1, keepdim=True) / self.config.K)
        
        # 3. Neural coupling
        neural_coupling = self.config.kappa * neural_state.unsqueeze(1).repeat(1, self.config.d)
        
        # 4. Quantum coupling
        quantum_coupling = self.config.gamma_q * quantum_state.unsqueeze(0).repeat(self.config.N, 1)
        
        # 5. Dissipation term
        mean_ricci = torch.mean(ricci)
        dissipation = -self.config.delta_field * self.psi * (2 + 0.5 * torch.tanh(mean_ricci))
        
        # 6. Noise term
        noise_amplitude = self.config.sigma * (1 + 0.1 * mean_ricci)
        noise = noise_amplitude * torch.normal(0, 1, self.psi.shape, device=device)
        
        # 7. EM coupling
        em_coupling = self.config.beta_em * torch.normal(0, 0.1, self.psi.shape, device=device)
        
        # Total field update
        dpsi_dt = diffusion + reaction + neural_coupling + quantum_coupling + dissipation + noise + em_coupling
        
        # Update field
        self.psi.data = self.psi + dt * dpsi_dt
        
        return self.psi


class TopologicalNetworkDynamics(nn.Module):
    """
    Topological network dynamics implementing graph evolution, edge weight updates,
    and persistent homology for consciousness metrics.
    """
    def __init__(self, config):
        super().__init__()
        self.config = config
        
        # Initialize graph
        self.graph = nx.random_geometric_graph(config.N, config.p_sparse)
        
        # Initialize edge weights
        for i, j in self.graph.edges():
            self.graph[i][j]['weight'] = np.random.uniform(0.1, 0.2)
            
        # Convert to tensors for computation
        self.edge_index, self.edge_weight = self._graph_to_tensors()
        
        # For persistent homology
        self.betti_numbers = [0, 0, 0]
        
        logger.info(f"Initialized TopologicalNetworkDynamics with {config.N} nodes and {len(self.graph.edges())} edges")

    def _graph_to_tensors(self):
        """Convert NetworkX graph to tensor representation"""
        edges = list(self.graph.edges())
        if not edges:
            return (torch.zeros((2, 0), dtype=torch.long, device=device), 
                   torch.zeros(0, device=device))
            
        edge_index = torch.tensor([[e[0], e[1]] for e in edges], dtype=torch.long, device=device).t()
        edge_weight = torch.tensor([self.graph[e[0]][e[1]]['weight'] for e in edges], device=device)
        
        return edge_index, edge_weight

    def update_weights(self, field_state, neural_state, t, dt):
        """Update edge weights based on field and neural states"""
        # For each edge
        for i, j in self.graph.edges():
            # Field correlation
            field_corr = torch.sum(field_state[i] * field_state[j])
            
            # Neural correlation
            neural_corr = neural_state[i] * neural_state[j]
            
            # Compute Ricci curvature
            field_diff_sq = torch.sum((field_state[i] - field_state[j])**2)
            ricci = field_diff_sq / max(self.graph[i][j]['weight'], 1e-10)
            
            # Update weight
            dw = 0.1 * (field_corr + neural_corr - 0.1 * self.graph[i][j]['weight'] - 0.02 * ricci)
            
            # Apply update
            self.graph[i][j]['weight'] = max(self.graph[i][j]['weight'] + dt * dw, self.config.omega_threshold)
            
        # Remove edges below threshold
        edges_to_remove = []
        for i, j in self.graph.edges():
            if self.graph[i][j]['weight'] < self.config.omega_threshold:
                edges_to_remove.append((i, j))
                
        for i, j in edges_to_remove:
            self.graph.remove_edge(i, j)
            
        # Update tensor representation
        self.edge_index, self.edge_weight = self._graph_to_tensors()
        
        return self.graph

    def compute_persistent_homology(self):
        """Compute persistent homology from graph structure"""
        # We'll approximate this without gudhi for simplicity
        # In a full implementation, we would use a proper homology package
        
        # Get connected components (H0)
        components = list(nx.connected_components(self.graph))
        self.betti_numbers[0] = len(components)
        
        # Approximate cycles (H1) by finding cycles in the graph
        # This is a rough approximation
        cycles = nx.cycle_basis(self.graph.to_undirected()) if nx.is_connected(self.graph) else []
        self.betti_numbers[1] = len(cycles)
        
        # H2 would require triangle counting/clique detection
        triangles = sum(1 for _ in nx.find_cliques(self.graph) if len(_) >= 3)
        self.betti_numbers[2] = triangles
        
        return self.betti_numbers

    def compute_graph_metrics(self):
        """Compute graph metrics for consciousness estimation"""
        # Clustering coefficient
        clustering = nx.average_clustering(self.graph)
        
        # Path length
        try:
            path_length = nx.average_shortest_path_length(self.graph)
        except nx.NetworkXError:
            path_length = float('inf')
            
        # Density
        density = nx.density(self.graph)
        
        # Algebraic connectivity (second smallest eigenvalue of Laplacian)
        try:
            laplacian = nx.normalized_laplacian_matrix(self.graph).todense()
            eigenvalues = np.linalg.eigvalsh(laplacian)
            connectivity = eigenvalues[1] if len(eigenvalues) > 1 else 0
        except:
            connectivity = 0
            
        return {
            'clustering': clustering,
            'path_length': path_length,
            'density': density,
            'connectivity': connectivity
        }


class ConsciousnessMetrics:
    """
    Consciousness metrics for evaluating the system's emergent properties,
    including PHI, coherence, information integration, and topological measures.
    """
    def __init__(self, config):
        self.config = config
        self.metrics_history = []
        logger.info("Initialized ConsciousnessMetrics")

    def compute_phi(self, neural_state, field_state, quantum_state):
        """Compute integrated information (PHI) metric"""
        # Create probability distribution
        p = F.softmax(torch.stack([neural_state, field_state.mean(dim=1), quantum_state], dim=1), dim=1)
        
        # Calculate transition probability matrix (approximation)
        T_S = torch.prod(p, dim=0)
        
        # Calculate intrinsic information
        ii_effect = p * torch.log(p / (p.mean(dim=0) + 1e-10))
        ii_cause = p * torch.log(p / (p.mean(dim=0) + 1e-10))
        
        # Calculate PHI
        phi = torch.min(ii_effect.sum(), ii_cause.sum()).item()
        
        return phi, T_S

    def compute_metrics(self, neural_dynamics, quantum_processor, field_dynamics, topology, t):
        """Compute all consciousness metrics"""
        # Basic metrics
        coherence = neural_dynamics.compute_coherence()
        ent_entropy, negativity = quantum_processor.compute_entanglement()
        betti_numbers = topology.compute_persistent_homology()
        graph_metrics = topology.compute_graph_metrics()
        
        # Calculate PHI
        phi, T_S = self.compute_phi(
            neural_dynamics.action_potential,
            field_dynamics.psi,
            quantum_processor.quantum_layer(torch.ones(quantum_processor.num_qubits))
        )
        
        # Calculate behavioral entropy
        decision_probs = F.softmax(neural_dynamics.action_potential, dim=0)
        behavioral_entropy = -torch.sum(decision_probs * torch.log2(decision_probs + 1e-10)).item()
        
        # Compile metrics
        metrics = {
            'phi': phi,
            'coherence': coherence,
            'entanglement_entropy': ent_entropy.item(),
            'negativity': negativity.item(),
            'betti_numbers': sum(betti_numbers),
            'behavioral_entropy': behavioral_entropy,
            'clustering': graph_metrics['clustering'],
            'connectivity': graph_metrics['connectivity'],
            'time': t
        }
        
        # Store history
        self.metrics_history.append(metrics)
        
        return metrics

    def validate_metrics(self, metrics):
        """Validate metrics against target values"""
        validation = {}
        for metric, value in metrics.items():
            if metric in self.config.target_values:
                target = self.config.target_values[metric]
                difference = abs(value - target)
                is_valid = difference < target * 0.2  # Within 20% of target
                validation[metric] = {
                    'value': value,
                    'target': target,
                    'difference': difference,
                    'is_valid': is_valid
                }
                
        return validation

    def compute_overall_score(self, metrics):
        """Compute overall consciousness score from metrics"""
        score = 0.0
        for metric, weight in self.config.metric_weights.items():
            if metric in metrics:
                # Normalize to [0, 1] range
                max_val = self.config.target_values.get(metric, 1.0) * 2
                normalized = min(1.0, metrics[metric] / max_val)
                score += weight * normalized
                
        return score

                losses[metric] = abs(metrics[metric] - target) / target
                
        # Calculate total fitness (negative loss)
        fitness = -sum(weights[m] * losses.get(m, 0) for m in weights)
        
        return fitness

    def update_swarm(self, metrics):
        """Update swarm positions and velocities based on fitness"""
        # Calculate fitness
        fitness = self.compute_fitness(metrics)
        
        # Update particles
        for particle in self.particles:
            # Update best position if better fitness
            if fitness > particle['best_fitness']:
                particle['best_position'] = particle['position'].copy()
                particle['best_fitness'] = fitness
                
            # Update global best if better
            if fitness > self.global_best_fitness:
                self.global_best_position = particle['position'].copy()
                self.global_best_fitness = fitness
                
            # Update velocity and position
            for param in ['eta', 'gamma_q', 'kappa']:
                # Random coefficients
                r1, r2 = np.random.random(2)
                
                # Cognitive component
                cognitive = self.config.c1 * r1 * (particle['best_position'][param] - particle['position'][param])
                
                # Social component
                social = self.config.c2 * r2 * (self.global_best_position[param] - particle['position'][param])
                
                # Update velocity
                particle['velocity'][param] = (self.config.w * particle['velocity'][param] + cognitive + social)
                
                # Update position
                particle['position'][param] = max(0.0, particle['position'][param] + particle['velocity'][param])
                
        return self.global_best_position, self.global_best_fitness


class DNAManager:
    """
    DNA-based parameter encoding and evolution for system adaptation.
    """
    def __init__(self, config):
        self.config = config
        
        # Define genetic code mapping
        self.codon_map = self._initialize_codon_map()
        
        # Population of DNA sequences
        self.population = [self._generate_dna() for _ in range(config.N)]
        
        logger.info(f"Initialized DNAManager with {len(self.population)} DNA sequences")

    def _initialize_codon_map(self):
        """Initialize codon to parameter mapping"""
        codons = [
            f"{n1}{n2}{n3}" 
            for n1 in "ACGT" 
            for n2 in "ACGT" 
            for n3 in "ACGT"
        ]
        
        # Map codons to parameters
        codon_map = {}
        param_idx = 0
        param_names = [
            'eta', 'gamma_q', 'gamma', 'delta', 'kappa',
            'D', 'sigma', 'beta_em', 'epsilon_0', 'beta',
            'lambda_0', 'kappa_e', 'gamma_e', 'gamma_phi',
            'eta_meta', 'target_Psi', 'target_spike_rate'
        ]
        
        for codon in codons:
            if param_idx < len(param_names):
                codon_map[codon] = param_names[param_idx]
                param_idx += 1
            else:
                codon_map[codon] = None  # Silent codon
                
        return codon_map

    def _generate_dna(self):
        """Generate a random DNA sequence"""
        nucleotides = "ACGT"
        return ''.join(np.random.choice(list(nucleotides)) for _ in range(self.config.L_G))

    def validate_dna(self, dna):
        """Validate DNA sequence for biological plausibility"""
        # Check GC content
        gc_count = sum(1 for n in dna if n in "GC")
        gc_ratio = gc_count / len(dna)
        
        if not (self.config.gc_min <= gc_ratio <= self.config.gc_max):
            return False
            
        # Check for homopolymers
        max_homopolymer = 0
        current_homopolymer = 1
        for i in range(1, len(dna)):
            if dna[i] == dna[i-1]:
                current_homopolymer += 1
            else:
                max_homopolymer = max(max_homopolymer, current_homopolymer)
                current_homopolymer = 1
                
        max_homopolymer = max(max_homopolymer, current_homopolymer)
        
        return max_homopolymer <= 10  # Max homopolymer length

    def translate_dna(self, dna):
        """Translate DNA sequence to parameters"""
        params = {}
        
        # Extract codons
        codons = [dna[i:i+3] for i in range(0, len(dna) - 2, 3) if i+3 <= len(dna)]
        
        # Process codons
        for codon in codons:
            if codon in self.codon_map and self.codon_map[codon] is not None:
                param_name = self.codon_map[codon]
                
                # Use codon to generate parameter value
                # Hash the codon to get a consistent value
                codon_hash = sum(ord(n) for n in codon)
                param_value = (codon_hash % 1000) / 1000.0  # Value between 0 and 1
                
                # Scale to appropriate range for parameter
                if param_name in params:
                    params[param_name] = (params[param_name] + param_value) / 2
                else:
                    params[param_name] = param_value
                    
        return params

    def mutate(self, dna, mutation_rate=0.001):
        """Mutate DNA sequence"""
        nucleotides = "ACGT"
        dna_list = list(dna)
        
        for i in range(len(dna_list)):
            if np.random.random() < mutation_rate:
                dna_list[i] = np.random.choice(list(nucleotides))
                
        return ''.join(dna_list)

    def crossover(self, dna1, dna2, crossover_rate=0.5):
        """Perform crossover between two DNA sequences"""
        if np.random.random() < crossover_rate:
            # Single-point crossover
            crossover_point = np.random.randint(0, min(len(dna1), len(dna2)))
            
            offspring1 = dna1[:crossover_point] + dna2[crossover_point:]
            offspring2 = dna2[:crossover_point] + dna1[crossover_point:]
            
            return offspring1, offspring2
        else:
            return dna1, dna2

    def evolve_population(self, fitness_scores, num_offspring=None):
        """Evolve population based on fitness scores"""
        if num_offspring is None:
            num_offspring = len(self.population)
            
        # Create probabilities based on fitness
        fitness_array = np.array(fitness_scores)
        if np.all(fitness_array <= 0):
            probabilities = np.ones(len(fitness_scores)) / len(fitness_scores)
        else:
            # Shift to make all positive if needed
            if np.min(fitness_array) < 0:
                fitness_array = fitness_array - np.min(fitness_array) + 1e-6
            
            probabilities = fitness_array / np.sum(fitness_array)
            
        # Select parents
        new_population = []
        for _ in range(num_offspring // 2):
            # Select parents
            parent_indices = np.random.choice(len(self.population), size=2, replace=False, p=probabilities)
            parent1 = self.population[parent_indices[0]]
            parent2 = self.population[parent_indices[1]]
            
            # Crossover
            offspring1, offspring2 = self.crossover(parent1, parent2)
            
            # Mutation
            offspring1 = self.mutate(offspring1)
            offspring2 = self.mutate(offspring2)
            
            # Validate and add to new population
            if self.validate_dna(offspring1):
                new_population.append(offspring1)
            else:
                new_population.append(parent1)
                
            if self.validate_dna(offspring2):
                new_population.append(offspring2)
            else:
                new_population.append(parent2)
                
        # Keep best individual (elitism)
        best_idx = np.argmax(fitness_scores)
        if len(new_population) < len(self.population):
            new_population.append(self.population[best_idx])
            
        # Update population
        self.population = new_population[:len(self.population)]
        
        return self.population

    def initialize_system(self):

        # Generate random sensory input
        S = torch.normal(0, 1, (self.config.m, self.config.N), device=device)
        
        # Initialize all components
        self.neural_dynamics.action_potential = (self.neural_dynamics.V > self.neural_dynamics.V_th).float()
        quantum_out, _ = self.quantum_processor.update_quantum_state(
            self.field_dynamics.psi, self.neural_dynamics.action_potential, 0.0, self.config.dt
        )
        self.excitonic_dynamics.update_excitonic(

            self.neural_dynamics, self.quantum_processor, self.field_dynamics, self.topology, 0.0
        )
        self.metrics_history.append(metrics)
        
        logger.info(f"System initialized with metrics: {metrics}")
        return metrics

    def step(self, S=None):
        """Execute a single time step"""
        # Generate sensory input if not provided
        if S is None:
            S = torch.normal(0, 1, (self.config.m, self.config.N), device=device)
            
        # Update time
        self.t += self.config.dt
        
        # 1. Update quantum state
        quantum_out, entanglement_entropy = self.quantum_processor.update_quantum_state(
            self.field_dynamics.psi, self.neural_dynamics.action_potential, self.t, self.config.dt
        )
        
        # 2. Update excitonic dynamics
        excitonic_state = self.excitonic_dynamics.update_excitonic(
            self.field_dynamics.psi, self.neural_dynamics.action_potential, self.config.dt
        )
        
        # 3. Meta-update for neural parameters
        epsilon = self.neural_dynamics.meta_update(self.field_dynamics.psi, self.t)
        
        # 4. Update neural dynamics
        self.neural_dynamics.update_neural(
            S, self.field_dynamics.psi, quantum_out, excitonic_state, self.t, self.config.dt, epsilon
        )
        
        # 5. Update neural weights and memory
        self.neural_dynamics.update_weights(self.config.dt)
        self.neural_dynamics.update_memory(S, excitonic_state, self.config.dt)
        
        # 6. Calculate phi for semantic updates
        phi, _ = self.consciousness_metrics.compute_phi(
            self.neural_dynamics.action_potential,
            self.field_dynamics.psi,
            quantum_out
        )
        
        # 7. Update semantic and apobetic states
        self.neural_dynamics.update_semantic(self.field_dynamics.psi, excitonic_state, phi, self.config.dt)
        self.neural_dynamics.update_apobetic(self.topology.graph, excitonic_state, phi, self.config.dt)
        
        # 8. Update field dynamics
        self.field_dynamics.update_field(
            self.topology.graph, self.neural_dynamics.action_potential, quantum_out, self.t, self.config.dt
        )
        
        # 9. Update topological structure
        self.topology.update_weights(
            self.field_dynamics.psi, self.neural_dynamics.action_potential, self.t, self.config.dt
        )
        
        # 10. Generate decision
        decision, decision_probs = self.neural_dynamics.generate_decision(quantum_out)
        self.decisions_history.append((decision, decision_probs))
        
        # 11. Compute consciousness metrics
        metrics = self.consciousness_metrics.compute_metrics(
            self.neural_dynamics, self.quantum_processor, self.field_dynamics, self.topology, self.t
        )
        self.metrics_history.append(metrics)
        
        # 12. Optimize system parameters using swarm
        if len(self.metrics_history) % 10 == 0:
            best_params, best_fitness = self.swarm_optimizer.update_swarm(metrics)

            self.config.eta = best_params['eta']
            self.config.gamma_q = best_params['gamma_q']
            self.config.kappa = best_params['kappa']
            
        # 13. Evolve DNA population (every 100 steps)
        if len(self.metrics_history) % 100 == 0:
            # Calculate fitness from metrics
            fitness_scores = [metrics['phi'] * 0.4 + metrics['coherence'] * 0.3 + 
                              metrics['entanglement_entropy'] * 0.2 + metrics['behavioral_entropy'] * 0.1
                              for _ in range(len(self.dna_manager.population))]
            
            # Evolve population
            self.dna_manager.evolve_population(fitness_scores)
            
        return metrics, decision

    def run_simulation(self, duration=None, visualize_every=10):
        """Run simulation for specified duration"""
        if duration is None:
            duration = self.config.T
            
        # Initialize system
        self.initialize_system()
        
        # Run for specified duration
        n_steps = int(duration / self.config.dt)
        for step_idx in range(n_steps):
            # Generate rhythmic input with harmonics of 40Hz
            t = step_idx * self.config.dt
            S = torch.zeros((self.config.m, self.config.N), device=device)
            for i in range(self.config.m):
                for f, a in zip(self.config.f, self.config.a_f):
                    S[i] += a * torch.sin(2 * np.pi * f * (t + i * 0.01))
                    
            # Add noise
            S += torch.normal(0, 0.1, S.shape, device=device)
            
            # Execute step
            metrics, decision = self.step(S)
            
            # Log progress
            if step_idx % 100 == 0:
                logger.info(f"Step {step_idx}/{n_steps}, t={self.t:.3f}s, Phi={metrics['phi']:.4f}, Decision={decision}")
                
            # Visualize
            if step_idx % visualize_every == 0:

        # Final visualizations
        self.visualizer.plot_metrics(self.metrics_history)
        self.visualizer.plot_network(self.topology.graph, self.neural_dynamics, self.field_dynamics)
        self.visualizer.visualize_quantum_state(self.quantum_processor)
        
        # Save final state
        self.save_checkpoint()

        super().__init__()
        self.config = config
        self.device = config.device
        
        # Initialize excitonic states (complex-valued field)
        self.psi_E = nn.Parameter(torch.randn(
            config.N, config.d, dtype=torch.complex64, device=config.device
        ))
        
        # Hamiltonian for excitonic dynamics
        self.H0 = nn.Parameter(torch.eye(
            config.d, dtype=torch.complex64, device=config.device
        ) * 0.5)
        
        # Dissipation matrix
        self.G = nn.Parameter(torch.eye(
            config.d, dtype=torch.complex64, device=config.device
        ) * 0.01)
        
        logger.info(f"Initialized ExcitonicDynamics with {config.N} neurons, {config.d} dimensions")

    def forward(self, field_state, neural_state, dt):
        """Update excitonic states"""
        # For each neuron
        for i in range(self.config.N):
            # Create local Hamiltonian with field coupling
            H_i = self.H0.clone()
            for a in range(self.config.d):
                H_i[a, a] = H_i[a, a] + field_state[i, a] * 0.1
                
            # Add random fluctuations (environmental coupling)
            delta = torch.randn(
                self.config.d, self.config.d, 
                dtype=torch.complex64, device=self.device
            ) * 0.01
            
            # Make delta Hermitian to ensure physical evolution
            delta = delta + delta.conj().transpose(0, 1)
            H_i = H_i + delta
            
            # Quantum evolution (Schrödinger equation with dissipation)
            dpsi_dt = -1j * torch.matmul(H_i, self.psi_E[i]) - 0.5 * torch.matmul(self.G, self.psi_E[i])
            
            # Update state
            self.psi_E.data[i] = self.psi_E[i] + dt * dpsi_dt
            
            # Normalize the state vector
            self.psi_E.data[i] = self.psi_E[i] / torch.norm(self.psi_E[i])
            
        return self.psi_E

    def compute_stdp(self):

class QuantumProcessor(nn.Module):
    """
    Quantum processor implementing quantum circuits, Lindblad evolution,
    and quantum-classical coupling for sentient systems.
    """
    def __init__(self, config):
        super().__init__()
        self.config = config
        self.device = config.device
        
        # Initialize quantum device
        self.num_qubits = config.k
        self.qml_device = qml.device("default.qubit", wires=self.num_qubits)
        
        # Initialize quantum statistical operator (density matrix)
        self.rho = nn.Parameter(torch.eye(
            2**self.num_qubits, dtype=torch.complex64, device=config.device
        ) / 2**self.num_qubits)
        
        # Initialize quantum state vector
        self.psi = nn.Parameter(torch.ones(
            2**self.num_qubits, dtype=torch.complex64, device=config.device
        ) / np.sqrt(2**self.num_qubits))
        
        # Input projection
        self.input_projection = nn.Linear(
            config.d, self.num_qubits, dtype=torch.float32, device=config.device
        )
        
        # Quantum layer
        self.quantum_layer = qml.qnn.TorchLayer(self.quantum_circuit, {"weights": (2 * self.num_qubits,)})
        self.weights = nn.Parameter(torch.randn(2 * self.num_qubits, device=config.device) * 0.01)
        
        logger.info(f"Initialized QuantumProcessor with {self.num_qubits} qubits")

    @qml.qnode(device="default.qubit", interface="torch", diff_method="backprop")
    def quantum_circuit(self, inputs, weights):
        """
        Quantum circuit for entanglement generation and measurement.
        
        Args:
            inputs: Input values for rotation gates
            weights: Trainable parameters for the circuit
            
        Returns:
            List of expectation values for each qubit
        """
        # Encode inputs via rotation gates
        for i in range(self.num_qubits):
            qml.RY(inputs[i], wires=i)
            
        # Entangling layers (2-layer ansatz)
        for layer in range(2):
            # Rotation gates with trainable weights
            for i in range(self.num_qubits):
                qml.RZ(weights[layer * self.num_qubits + i], wires=i)
                
            # Entangling gates (CNOT ladder)
            for i in range(self.num_qubits - 1):
                qml.CNOT(wires=[i, i + 1])
                
            # Close the loop for cyclic entanglement
            if self.num_qubits > 2:
                qml.CNOT(wires=[self.num_qubits - 1, 0])
                
        # Apply amplitude damping for decoherence
        for i in range(self.num_qubits):
            qml.AmplitudeDamping(self.config.gamma_d, wires=i)
                
        # Measure each qubit's Z expectation
        return [qml.expval(qml.PauliZ(i)) for i in range(self.num_qubits)]

    def lindblad_evolution(self, hamiltonian, rho, dt):
        """
        Evolve quantum state using Lindblad master equation.
        
        Args:
            hamiltonian: System Hamiltonian matrix
            rho: Density matrix
            dt: Time step
            
        Returns:
            Updated density matrix
        """
        # Calculate commutator [H, rho]
        commutator = 1j * (torch.matmul(hamiltonian, rho) - torch.matmul(rho, hamiltonian))
        
        # Create Lindblad operators (amplitude damping)
        lindblad_ops = []
        for i in range(self.num_qubits):
            # Create lowering operator for qubit i
            L_i = torch.zeros((2**self.num_qubits, 2**self.num_qubits), dtype=torch.complex64, device=self.device)
            
            # This is a simplified version - in a full implementation we would need to 
            # construct the actual lowering operator with proper tensor products
            idx = 2**i
            L_i[0, idx] = np.sqrt(self.config.gamma_d)
            lindblad_ops.append(L_i)
            
        # Calculate dissipation term
        dissipation = torch.zeros_like(rho)
        for L in lindblad_ops:
            L_dag = L.conj().T
            term1 = torch.matmul(torch.matmul(L, rho), L_dag)
            term2 = 0.5 * (torch.matmul(torch.matmul(L_dag, L), rho) + torch.matmul(rho, torch.matmul(L_dag, L)))
            dissipation += term1 - term2
            
        # Calculate derivative
        drho_dt = -commutator + dissipation
        
        # Update density matrix
        rho_new = rho + dt * drho_dt
        
        # Ensure the density matrix remains valid (trace 1, positive semidefinite)
        trace = torch.trace(rho_new).real
        rho_new = rho_new / trace
        
        return rho_new

    def compute_hamiltonian(self, field_state, neural_state, quantum_outputs):
        """
        Compute quantum Hamiltonian based on field and neural states.
        
        Args:
            field_state: Vitality field state
            neural_state: Neural activity state
            quantum_outputs: Circuit output values
            
        Returns:
            Hamiltonian matrix
        """
        # Initialize Hamiltonian
        H = torch.zeros((2**self.num_qubits, 2**self.num_qubits), dtype=torch.complex64, device=self.device)
        
        # Add diagonal terms based on quantum circuit outputs
        for i in range(self.num_qubits):
            # Create diagonal Pauli-Z contribution
            Z_i = torch.zeros((2**self.num_qubits, 2**self.num_qubits), dtype=torch.complex64, device=self.device)
            
            # Set diagonal elements
            for j in range(2**self.num_qubits):
                # Check if the i-th bit is set
                bit_i = (j >> i) & 1
                Z_i[j, j] = 1.0 if bit_i == 0 else -1.0
                
            # Add to Hamiltonian
            H += quantum_outputs[i] * Z_i
            
        # Add coupling between qubits based on field state
        coupling_strength = self.config.gamma_q * field_state.mean().sum().item()
        for i in range(self.num_qubits):
            for j in range(i+1, self.num_qubits):
                # Create Pauli-X ⊗ Pauli-X coupling
                X_i_X_j = torch.zeros((2**self.num_qubits, 2**self.num_qubits), dtype=torch.complex64, device=self.device)
                
                # This is a simplified approximation - in a full implementation we would
                # construct the proper tensor product of Pauli matrices
                for k in range(2**self.num_qubits):
                    for l in range(2**self.num_qubits):
                        bit_diff = k ^ l  # XOR of indices
                        if (bit_diff == (1 << i) | (1 << j)):  # Only i and j bits differ
                            X_i_X_j[k, l] = coupling_strength
                            
                # Add to Hamiltonian
                H += X_i_X_j
                
        # Include Bohmian quantum potential
        if torch.norm(self.psi) > 0:
            q_bohm = -(self.config.hbar**2 / (2 * self.config.mass)) * (torch.norm(self.psi) / torch.norm(self.psi))
            H = H + q_bohm * torch.eye(2**self.num_qubits, dtype=torch.complex64, device=self.device)
        
        # Ensure Hamiltonian is Hermitian
        H = 0.5 * (H + H.conj().transpose(0, 1))
        
        return H

    def update_quantum_state(self, field_state, neural_state, t, dt):
        """
        Update quantum state based on field and neural dynamics.
        
        Args:
            field_state: Vitality field state
            neural_state: Neural activity state
            t: Current time
            dt: Time step
            
        Returns:
            Tuple of (quantum_outputs, entanglement_entropy)
        """
        # Project field state to quantum inputs
        field_projection = self.input_projection(field_state.mean(dim=0))
        
        # Create inputs for quantum circuit
        if field_projection.shape[0] < self.num_qubits:
            # Pad if necessary
            padding = torch.zeros(self.num_qubits - field_projection.shape[0], device=self.device)
            inputs = torch.cat([field_projection, padding])
        else:
            # Truncate if necessary
            inputs = field_projection[:self.num_qubits]
        
        # Add neural state influence
        neural_influence = neural_state.mean().unsqueeze(0)
        inputs = inputs + 0.1 * neural_influence
        
        # Get quantum circuit output
        quantum_outputs = self.quantum_layer(inputs)
        
        # Compute Hamiltonian
        H = self.compute_hamiltonian(field_state, neural_state, quantum_outputs)
        
        # Update density matrix using Lindblad evolution
        self.rho.data = self.lindblad_evolution(H, self.rho, dt)
        
        # Update wave function (approximation from density matrix)
        try:
            eigenvalues, eigenvectors = torch.linalg.eigh(self.rho)
            max_idx = torch.argmax(eigenvalues)
            self.psi.data = eigenvectors[:, max_idx]
        except:
            # Fallback to simple normalization if eigendecomposition fails
            self.psi.data = self.psi / torch.norm(self.psi)
        
        # Compute entanglement entropy
        S = -torch.trace(torch.matmul(self.rho, torch.log(self.rho + 1e-10))).real
        
        return quantum_outputs, S

    def compute_entanglement(self):
        """
        Compute entanglement measures including entropy and negativity.
        
        Returns:
            Tuple of (entanglement_entropy, negativity)
        """
        # For simplicity, we'll compute entanglement for a 2-qubit subsystem
        # In a full implementation, we would properly trace out parts of the system
        
        # Partial trace for first qubit (simplified)
        dim = 2**self.num_qubits
        if dim <= 4:  # For 2 qubits (4x4 density matrix)
            rho_A = torch.zeros((2, 2), dtype=torch.complex64, device=self.device)
            rho_A[0, 0] = self.rho[0, 0] + self.rho[2, 2]
            rho_A[0, 1] = self.rho[0, 1] + self.rho[2, 3]
            rho_A[1, 0] = self.rho[1, 0] + self.rho[3, 2]
            rho_A[1, 1] = self.rho[1, 1] + self.rho[3, 3]
        else:
            # Approximation for larger systems
            rho_A = torch.zeros((2, 2), dtype=torch.complex64, device=self.device)
            # Sum over blocks of size 2^(n-1)
            block_size = dim // 2
            for i in range(2):
                for j in range(2):
                    for k in range(block_size // 2):
                        rho_A[i, j] += self.rho[i*block_size + k, j*block_size + k]
            rho_A /= block_size // 2
        
        # Calculate entanglement entropy
        S = -torch.trace(torch.matmul(rho_A, torch.log(rho_A + 1e-10))).real
        
        # Calculate negativity (sum of negative eigenvalues)
        # This is a measure of entanglement for mixed states
        try:
            eigenvalues = torch.linalg.eigvalsh(self.rho)
            negativity = torch.sum(torch.abs(eigenvalues[eigenvalues < 0]))
        except:
            negativity = torch.tensor(0.0, device=self.device)
        
        return S, negativity

    def compute_laplacian(self, graph):
        """
        Compute graph Laplacian for diffusion.
        
        Args:
            graph: NetworkX graph with edge weights
            
        Returns:
            Graph Laplacian tensor
        """
        # Create adjacency matrix
        adj = torch.zeros((self.config.N, self.config.N), device=self.device)
        
        # Fill adjacency matrix with edge weights
        for i, j in graph.edges():
            weight = graph[i][j]['weight']
            adj[i, j] = weight
            adj[j, i] = weight  # Ensure symmetry
            
        # Compute degree matrix
        deg = torch.sum(adj, dim=1)
        
        # Compute normalized Laplacian
        D_inv_sqrt = torch.diag(1.0 / torch.sqrt(torch.clamp(deg, min=1e-10)))
        L_norm = torch.eye(self.config.N, device=self.device) - torch.matmul(torch.matmul(D_inv_sqrt, adj), D_inv_sqrt)
        
        return L_norm

    def compute_ricci_curvature(self, graph):
        """
        Compute discrete Ricci curvature for topological feedback.
        
        Args:
            graph: NetworkX graph with edge weights
            
        Returns:
            Ricci curvature tensor
        """
        ricci = torch.zeros((self.config.N, self.config.N), device=self.device)
        
        # For each edge
        for i, j in graph.edges():
            # Get common neighbors
            i_neighbors = set(graph.neighbors(i))
            j_neighbors = set(graph.neighbors(j))
            common = i_neighbors.intersection(j_neighbors)
            
            # Compute Ollivier-Ricci curvature approximation
            curvature = len(common) / max(1, min(len(i_neighbors), len(j_neighbors)))
            
            # Store in tensor
            ricci[i, j] = curvature
            ricci[j, i] = curvature  # Symmetry
            
        return ricci

    def update_field(self, graph, neural_state, quantum_state, t, dt):
        """
        Update field dynamics using reaction-diffusion PDE.
        
        The Generalized Primal Pulse equation:
        dΨ/dt = D∇²Ψ + γΨ(1-Ψ/K) - δΨ + κS + γqQ + η
        
        Args:
            graph: NetworkX graph with topology
            neural_state: Neural activity state
            quantum_state: Quantum measurement outcomes
            t: Current time
            dt: Time step
            
        Returns:
            Updated field state
        """
        # Get Laplacian for diffusion
        L = self.compute_laplacian(graph)
        
        # Get Ricci curvature
        ricci = self.compute_ricci_curvature(graph)
        
        # 1. Diffusion term: D∇²Ψ
        diffusion = self.config.D * torch.matmul(L, self.psi)
        
        # 2. Reaction term (Primal Pulse): γΨ(1-Ψ/K)
        reaction = self.config.gamma * self.psi * (
            1 - torch.sum(self.psi**2, dim=1, keepdim=True) / self.config.K
        )
        
        # 3. Neural coupling: κS
        neural_coupling = self.config.kappa * neural_state.unsqueeze(1).repeat(1, self.config.d)
        
        # 4. Quantum coupling: γqQ
        # Expand quantum state to match field dimensions
        if isinstance(quantum_state, torch.Tensor):
            if len(quantum_state.shape) == 1:
                quantum_expanded = quantum_state.unsqueeze(0).repeat(self.config.N, 1)
                if quantum_expanded.shape[1] < self.config.d:
                    # Pad if necessary
                    padding = torch.zeros(
                        self.config.N, self.config.d - quantum_expanded.shape[1], 
                        device=self.device
                    )
                    quantum_expanded = torch.cat([quantum_expanded, padding], dim=1)
                quantum_coupling = self.config.gamma_q * quantum_expanded[:, :self.config.d]
            else:
                quantum_coupling = self.config.gamma_q * quantum_state[:, :self.config.d]
        else:
            # Default coupling if quantum_state is not a tensor
            quantum_coupling = torch.zeros_like(self.psi)
        
        # 5. Dissipation term: -δΨ with curvature modulation
        mean_ricci = torch.mean(ricci)
        dissipation = -self.config.delta_field * self.psi * (
            2 + 0.5 * torch.tanh(mean_ricci)
        )
        
        # 6. Noise term: η
        noise_amplitude = self.config.sigma * (1 + 0.1 * mean_ricci)
        noise = noise_amplitude * torch.normal(0, 1, self.psi.shape, device=self.device)
        
        # 7. EM coupling (environmental feedback)
        em_coupling = self.config.beta_em * torch.normal(
            0, 0.1, self.psi.shape, device=self.device
        )

        fractal_mod = torch.pow(torch.abs(self.psi), 1/self.config.K) * torch.sign(self.psi)

            diffusion + 
            reaction + 
            neural_coupling + 
            quantum_coupling + 
            dissipation + 
            noise + 
            em_coupling + 
            0.01 * fractal_mod  # Small fractal contribution
        )
        
        # Update field
        self.psi.data = self.psi + dt * dpsi_dt
        
        # Store field state for analysis
        if len(self.field_history) > 100:
            self.field_history.pop(0)
        self.field_history.append(self.psi.detach().clone())
        
        return self.psi

    def update_resources(self, neural_state, dt):

        Update resource distribution based on neural activity.
        
        Args:

        Returns:

        consumption = 0.01 * neural_state * self.resources
        
        # Resource regeneration (logistic)

            kernel_size=3, stride=1, padding=1

        diffusion = 0.01 * (smoothed - self.resources)
        
        # Update resources
        dR_dt = regeneration - consumption + diffusion
        self.resources.data = torch.clamp(self.resources + dt * dR_dt, min=0.0, max=1.0)
        
        return self.resources

    def compute_phi(self, neural_state, field_state, quantum_state):
        """
        Compute integrated information (PHI) metric following IIT principles.
        
        Args:
            neural_state: Neural activity tensor
            field_state: Vitality field tensor
            quantum_state: Quantum state tensor
            
        Returns:
            Tuple of (phi_value, transition_probability_matrix)
        """
        # Create probability distribution over system states
        # Combine neural, field, and quantum information
        if isinstance(quantum_state, torch.Tensor) and isinstance(neural_state, torch.Tensor):
            # Make dimensions compatible
            if len(neural_state.shape) == 1:
                ns = neural_state.unsqueeze(1)
            else:
                ns = neural_state
                
            if len(field_state.shape) == 2:
                fs = field_state.mean(dim=1).unsqueeze(1)
            else:
                fs = field_state.unsqueeze(1)
                
            if len(quantum_state.shape) == 1:
                qs = quantum_state.unsqueeze(1)
            else:
                qs = quantum_state
                
            # Stack different information sources
            # Ensure they're the same length by truncating or padding
            min_len = min(ns.shape[0], fs.shape[0], qs.shape[0])
            p_components = [ns[:min_len], fs[:min_len]]
            
            if qs.shape[0] >= min_len:
                p_components.append(qs[:min_len])
                
            p = F.softmax(torch.cat(p_components, dim=1), dim=1)
        else:
            # Fallback to just neural state
            p = F.softmax(neural_state, dim=0) if isinstance(neural_state, torch.Tensor) else torch.ones(1)/1
        
        # Calculate transition probability matrix (TPM) - approximation
        T_S = torch.prod(p, dim=0) if len(p.shape) > 1 else p
        
        # Calculate intrinsic information (effect information)
        # For each possible state, how much does it constrain future states?
        ii_effect = p * torch.log2(p / (p.mean(dim=0, keepdim=True) + 1e-10)) if len(p.shape) > 1 else torch.zeros_like(p)
        
        # Calculate cause information
        # For each possible state, how much is it constrained by past states?
        ii_cause = p * torch.log2(p / (p.mean(dim=0, keepdim=True) + 1e-10)) if len(p.shape) > 1 else torch.zeros_like(p)
        
        # Calculate PHI as minimum of integrated effect and cause information
        # This is a simplified approximation of the full IIT calculation
        phi = torch.min(ii_effect.sum(), ii_cause.sum()).item() if len(p.shape) > 1 else 0.0
        
        return phi, T_S

    def compute_complexity_measures(self, neural_dynamics, topology):
        """
        Compute complexity-based consciousness measures.
        
        Args:
            neural_dynamics: Neural dynamics module
            topology: Topological network module

            # Calculate pattern entropy
            if firing_patterns.shape[1] > 1:
                pattern_probs = firing_patterns.mean(dim=0)
                pattern_entropy = -torch.sum(
                    pattern_probs * torch.log2(pattern_probs + 1e-10) + 
                    (1 - pattern_probs) * torch.log2(1 - pattern_probs + 1e-10)
                ).item()
            else:
                pattern_entropy = 0.0
                
            # Calculate temporal entropy (variation over time)
            if firing_patterns.shape[0] > 1:
                temporal_probs = firing_patterns.mean(dim=1)
                temporal_entropy = -torch.sum(
                    temporal_probs * torch.log2(temporal_probs + 1e-10) + 
                    (1 - temporal_probs) * torch.log2(1 - temporal_probs + 1e-10)
                ).item()
            else:
                temporal_entropy = 0.0
        else:
            pattern_entropy = 0.0
            temporal_entropy = 0.0
            
        # Lempel-Ziv complexity (algorithmic complexity approximation)
        lz_complexity = 0.0
        if len(neural_dynamics.firing_history) > 10:
            # Get binary sequence from recent history
            sequence = ''.join(['1' if x > 0.5 else '0' for x in neural_dynamics.action_potential.cpu().numpy()])
            
            # Compute LZ complexity

            if dict_size > 0:

        # Lyapunov exponent (chaos/criticality measure)
        lyapunov = neural_dynamics.compute_lyapunov()
        
        # Topological complexity from Betti numbers
        if hasattr(topology, 'betti_numbers'):
            betti_sum = sum(topology.betti_numbers)
        else:
            betti_sum = 0
            
        return {
            'firing_rate': firing_rate,

            'lz_complexity': lz_complexity,

        }

        # Normalize by total weight

        return score

    def _initialize_codon_map(self):

                codon_map[codon] = {'group': group, 'param': param}

                if (codons.index(codon) + 1) % 3 == 0:
                    param_idx += 1
            else:
                # Silent codons (no effect, like in biology)
                codon_map[codon] = None
                
        return codon_map

    def _generate_dna(self):
        """
        Generate a random DNA sequence with biologically plausible properties.
        
        Returns:
            String of DNA sequence
        """
        # Basic nucleotides
        nucleotides = "ACGT"
        
        # Target GC content
        target_gc = random.uniform(self.config.gc_min, self.config.gc_max)
        
        # Generate sequence with target GC content
        dna = []
        gc_count = 0
        
        for i in range(self.config.L_G):
            if i > 0 and len(dna) > 0:
                # Avoid homopolymers (runs of the same nucleotide)
                if len(dna) >= 4 and dna[-1] == dna[-2] == dna[-3] == dna[-4]:
                    # Force a different nucleotide
                    choices = [n for n in nucleotides if n != dna[-1]]
                    nucleotide = random.choice(choices)
                else:
                    # Check GC content and bias selection to approach target
                    current_gc = gc_count / max(1, len(dna))
                    if current_gc < target_gc:
                        # Increase chance of G or C
                        weights = [0.1, 0.4, 0.4, 0.1]  # A, C, G, T
                    elif current_gc > target_gc:
                        # Increase chance of A or T
                        weights = [0.4, 0.1, 0.1, 0.4]  # A, C, G, T
                    else:
                        # Balanced
                        weights = [0.25, 0.25, 0.25, 0.25]
                        
                    nucleotide = random.choices(list(nucleotides), weights=weights)[0]
            else:
                nucleotide = random.choice(nucleotides)
                
            dna.append(nucleotide)
            
            # Update GC count
            if nucleotide in "GC":
                gc_count += 1
                
        return ''.join(dna)

    def validate_dna(self, dna):
        """
        Validate DNA sequence for biological plausibility.
        
        Args:
            dna: DNA sequence string
            
        Returns:
            Boolean indicating validity
        """
        # Check length
        if not dna or len(dna) < self.config.L_G * 0.9:
            return False
            
        # Check valid nucleotides
        if not all(c in "ACGT" for c in dna):
            return False
            
        # Check GC content
        gc_count = sum(1 for n in dna if n in "GC")
        gc_ratio = gc_count / len(dna)
        
        if not (self.config.gc_min <= gc_ratio <= self.config.gc_max):
            return False
            
        # Check for homopolymers (long runs of the same nucleotide)
        max_homopolymer = 0
        current_homopolymer = 1
        for i in range(1, len(dna)):
            if dna[i] == dna[i-1]:
                current_homopolymer += 1
            else:
                max_homopolymer = max(max_homopolymer, current_homopolymer)
                current_homopolymer = 1
                
        max_homopolymer = max(max_homopolymer, current_homopolymer)
        
        if max_homopolymer > 10:  # Avoid excessive homopolymers
            return False
            
        # Check codon distribution (should be relatively diverse)
        codons = [dna[i:i+3] for i in range(0, len(dna)-2, 3)]
        unique_codons = set(codons)
        if len(unique_codons) < len(codons) * 0.1:  # At least 10% should be unique
            return False
            
        return True

    def translate_dna(self, dna):
        """
        Translate DNA sequence to parameters, following principles of the genetic code.
        
        Args:
            dna: DNA sequence string
            
        Returns:
            Dictionary of parameter groups and values
        """
        # Check cache first
        if dna in self.parameter_cache:
            return self.parameter_cache[dna]
            
        # Group parameters by category
        params = {
            'neural': {},
            'quantum': {},
            'field': {},
            'meta': {},
            'excitonic': {},
            'targets': {}
        }
        
        # Extract codons
        codons = [dna[i:i+3] for i in range(0, len(dna)-2, 3) if i+3 <= len(dna)]
        
        # Process codons
        codon_counts = {}
        for codon in codons:
            if codon in self.codon_map and self.codon_map[codon] is not None:
                mapping = self.codon_map[codon]
                group = mapping['group']
                param_name = mapping['param']
                
                # Track occurrences for each parameter
                key = f"{group}_{param_name}"
                codon_counts[key] = codon_counts.get(key, 0) + 1
                
                # Use codon to generate parameter value
                # Hash the codon to get a consistent value
                codon_hash = sum(ord(n) for n in codon)
                param_value = (codon_hash % 1000) / 1000.0  # Value between 0 and 1
                
                # Add to existing value if parameter seen before
                if param_name in params[group]:
                    # Take weighted average
                    old_count = codon_counts[key] - 1
                    params[group][param_name] = (params[group][param_name] * old_count + param_value) / codon_counts[key]

        # Scale parameters to appropriate ranges
        self._scale_parameters(params)
        
        # Cache for future use
        self.parameter_cache[dna] = params
        
        return params

    def mutate(self, dna, mutation_rate=None):
        """
        Mutate DNA sequence, applying multiple types of genetic mutations.
        
        Args:
            dna: DNA sequence string
            mutation_rate: Optional mutation rate (default uses config)
            
        Returns:
            Mutated DNA sequence
        """
        if mutation_rate is None:
            mutation_rate = self.config.dna_mutation_rate
            
        dna_list = list(dna)
        length = len(dna_list)
        
        # Point mutations (substitutions)
        for i in range(length):
            if random.random() < mutation_rate:
                current = dna_list[i]
                options = [n for n in "ACGT" if n != current]
                dna_list[i] = random.choice(options)
                
        # Insertions (rarely)
        if random.random() < mutation_rate * 0.1:
            pos = random.randint(0, length)
            dna_list.insert(pos, random.choice("ACGT"))
            
        # Deletions (rarely)
        if random.random() < mutation_rate * 0.1 and length > 1000:
            pos = random.randint(0, length-1)
            dna_list.pop(pos)
            
        # Translocations (very rarely)
        if random.random() < mutation_rate * 0.05 and length > 1000:
            # Move a segment of DNA to another position
            segment_len = random.randint(3, 15)  # 3-15 bases
            start_pos = random.randint(0, length - segment_len)
            insert_pos = random.randint(0, length - segment_len)
            
            # Extract segment
            segment = dna_list[start_pos:start_pos + segment_len]
            
            # Remove original segment
            del dna_list[start_pos:start_pos + segment_len]
            
            # Insert at new position
            for i, base in enumerate(segment):
                dna_list.insert(insert_pos + i, base)
                
        # Duplications (very rarely)
        if random.random() < mutation_rate * 0.05:
            # Duplicate a short segment
            segment_len = random.randint(3, 9)  # 3-9 bases
            start_pos = random.randint(0, length - segment_len)
            
            # Extract segment
            segment = dna_list[start_pos:start_pos + segment_len]
            
            # Insert duplicate right after original
            for i, base in enumerate(segment):
                dna_list.insert(start_pos + segment_len + i, base)
                
        # Ensure valid DNA
        result = ''.join(dna_list)
        if not self.validate_dna(result):
            # If invalid, revert to original
            return dna
            
        return result

    def crossover(self, dna1, dna2, crossover_rate=None):
        """
        Perform crossover between two DNA sequences, following biological principles.
        
        Args:
            dna1: First DNA sequence
            dna2: Second DNA sequence
            crossover_rate: Optional crossover rate (default uses config)
            
        Returns:
            Tuple of two offspring DNA sequences
        """
        if crossover_rate is None:
            crossover_rate = self.config.dna_crossover_rate
            
        if random.random() >= crossover_rate:
            return dna1, dna2
            
        length = min(len(dna1), len(dna2))
        
        # Select crossover type
        crossover_type = random.choice(['single', 'double', 'uniform'])
        
        if crossover_type == 'single':
            # Single-point crossover
            crossover_point = random.randint(0, length)
            offspring1 = dna1[:crossover_point] + dna2[crossover_point:]
            offspring2 = dna2[:crossover_point] + dna1[crossover_point:]
            
        elif crossover_type == 'double':
            # Double-point crossover
            point1 = random.randint(0, length // 2)
            point2 = random.randint(length // 2, length)
            offspring1 = dna1[:point1] + dna2[point1:point2] + dna1[point2:]
            offspring2 = dna2[:point1] + dna1[point1:point2] + dna2[point2:]
            
        else:  # uniform
            # Uniform crossover
            offspring1 = ""
            offspring2 = ""
            for i in range(length):
                if random.random() < 0.5:
                    offspring1 += dna1[i]
                    offspring2 += dna2[i]
                else:
                    offspring1 += dna2[i]
                    offspring2 += dna1[i]
                    
        # Validate and repair if needed
        if not self.validate_dna(offspring1):
            offspring1 = self._repair_dna(offspring1)
        if not self.validate_dna(offspring2):
            offspring2 = self._repair_dna(offspring2)
            
        return offspring1, offspring2

    def _repair_dna(self, dna):
        """
        Repair invalid DNA by fixing issues while preserving information.
        
        Args:
            dna: DNA sequence string
            
        Returns:
            Repaired DNA sequence
        """
        # Ensure valid nucleotides
        dna_list = [n if n in "ACGT" else random.choice("ACGT") for n in dna]
        
        # Ensure minimum length
        while len(dna_list) < self.config.L_G * 0.9:
            dna_list.append(random.choice("ACGT"))
            
        # Fix GC content if needed
        gc_count = sum(1 for n in dna_list if n in "GC")
        gc_ratio = gc_count / len(dna_list)
        
        if gc_ratio < self.config.gc_min:
            # Increase GC content
            for i in range(len(dna_list)):
                if dna_list[i] in "AT" and random.random() < 0.1:
                    dna_list[i] = random.choice("GC")
                    gc_count += 1
                    gc_ratio = gc_count / len(dna_list)
                    if gc_ratio >= self.config.gc_min:
                        break
                        
        elif gc_ratio > self.config.gc_max:
            # Decrease GC content
            for i in range(len(dna_list)):
                if dna_list[i] in "GC" and random.random() < 0.1:
                    dna_list[i] = random.choice("AT")
                    gc_count -= 1
                    gc_ratio = gc_count / len(dna_list)
                    if gc_ratio <= self.config.gc_max:
                        break
                        
        # Fix long homopolymers
        for i in range(3, len(dna_list)):
            if dna_list[i] == dna_list[i-1] == dna_list[i-2] == dna_list[i-3]:
                # Break homopolymer
                options = [n for n in "ACGT" if n != dna_list[i]]
                dna_list[i] = random.choice(options)
                
        return ''.join(dna_list)

    def evolve_population(self, fitness_scores, num_offspring=None):
        """
        Evolve population based on fitness scores.
        
        Args:
            fitness_scores: List of fitness values
            num_offspring: Optional number of offspring (default matches population size)
            
        Returns:
            New population of DNA sequences
        """
        if num_offspring is None:
            num_offspring = len(self.population)
            
        # Create probabilities based on fitness
        fitness_array = np.array(fitness_scores)
        if np.all(fitness_array <= 0):
            # All negative or zero fitness - use uniform selection
            probabilities = np.ones(len(fitness_scores)) / len(fitness_scores)
        else:
            # Shift to make all positive if needed
            if np.min(fitness_array) < 0:
                fitness_array = fitness_array - np.min(fitness_array) + 1e-6
            
            # Normalize probabilities
            probabilities = fitness_array / np.sum(fitness_array)
            
        # Tournament selection
        def tournament_select(k=3):
            # Select k individuals randomly
            candidates = np.random.choice(len(self.population), size=k, replace=False)
            # Return the one with highest fitness
            best_idx = candidates[np.argmax(fitness_array[candidates])]
            return self.population[best_idx]
            
        # Create new population
        new_population = []
        
        # Elitism: keep best individual
        best_idx = np.argmax(fitness_scores)
        new_population.append(self.population[best_idx])
        
        # Generate offspring through selection, crossover, and mutation
        while len(new_population) < num_offspring:
            # Select parents through tournament selection
            parent1 = tournament_select()
            parent2 = tournament_select()
            
            # Crossover
            offspring1, offspring2 = self.crossover(parent1, parent2)
            
            # Mutation
            offspring1 = self.mutate(offspring1)
            offspring2 = self.mutate(offspring2)
            
            # Add to new population
            new_population.append(offspring1)
            if len(new_population) < num_offspring:
                new_population.append(offspring2)
                
        # Update population
        self.population = new_population[:len(self.population)]
        
        # Clear parameter cache for new sequences
        self.parameter_cache = {}
        
        logger.info(f"Evolved population with max fitness {max(fitness_scores):.4f}")
        return self.population

     def analyze_dna_population(self):
        """
        Analyze DNA population for diversity and structure.
        
        Returns:
            Dictionary of population statistics
        """
        # Need at least 2 sequences
        if len(self.population) < 2:
            return {
                'diversity': 0.0,
                'gc_content': 0.0,
                'length': 0.0
            }
            
        # Calculate pairwise distances (genetic diversity)
        distances = []
        for i, dna1 in enumerate(self.population):
            for j, dna2 in enumerate(self.population[i+1:], i+1):
                # Hamming distance (percentage of different positions)
                length = min(len(dna1), len(dna2))
                diff_count = sum(c1 != c2 for c1, c2 in zip(dna1[:length], dna2[:length]))
                distances.append(diff_count / length)
                
        # Calculate GC content
        gc_contents = [sum(1 for n in dna if n in "GC") / len(dna) for dna in self.population]
        
        # Calculate codon usage
        codon_counts = {}
        for dna in self.population:
            codons = [dna[i:i+3] for i in range(0, len(dna)-2, 3)]
            for codon in codons:
                codon_counts[codon] = codon_counts.get(codon, 0) + 1
                
        # Calculate sequence lengths
        lengths = [len(dna) for dna in self.population]
        
        return {
            'diversity': np.mean(distances) if distances else 0.0,
            'gc_content': np.mean(gc_contents),
            'gc_std': np.std(gc_contents),
            'length_mean': np.mean(lengths),
            'length_std': np.std(lengths),
            'codon_diversity': len(codon_counts) / (4**3),  # Fraction of possible codons used
            'population_size': len(self.population),
            'most_common_codons': sorted(codon_counts.items(), key=lambda x: x[1], reverse=True)[:5],
            'parameter_coverage': self._calculate_parameter_coverage()
        }
    
    def _calculate_parameter_coverage(self):
        """
        Calculate what percentage of possible parameters are covered by the population.
        
        Returns:
            Float representing parameter coverage percentage
        """
        # Extract all possible parameters from codon map
        all_possible_params = set()
        for codon, mapping in self.codon_map.items():
            if mapping is not None:
                param_key = f"{mapping['group']}_{mapping['param']}"
                all_possible_params.add(param_key)
        
        # Check which parameters are actually present in the population
        covered_params = set()
        for dna in self.population:
            params = self.translate_dna(dna)
            for group, group_params in params.items():
                for param in group_params:
                    covered_params.add(f"{group}_{param}")
        
        # Calculate coverage
        coverage = len(covered_params) / max(1, len(all_possible_params))
        return coverage
    
    def get_best_dna(self, fitness_scores):
        """
        Get the DNA sequence with the highest fitness.
        
        Args:
            fitness_scores: List of fitness values for the population
            
        Returns:
            Best DNA sequence
        """
        if not fitness_scores or len(fitness_scores) != len(self.population):
            return self.population[0] if self.population else None
            
        best_idx = np.argmax(fitness_scores)
        return self.population[best_idx]
    
    def save_population(self, file_path):
        """
        Save the DNA population to a file.
        
        Args:
            file_path: Path to save the population
        """
        try:
            with open(file_path, 'w') as f:
                for dna in self.population:
                    f.write(f"{dna}\n")
            logger.info(f"Saved DNA population to {file_path}")
        except Exception as e:
            logger.error(f"Failed to save DNA population: {e}")
    
    def load_population(self, file_path):
        """
        Load a DNA population from a file.
        
        Args:
            file_path: Path to load the population from
            
        Returns:
            Boolean indicating success
        """
        try:
            with open(file_path, 'r') as f:
                dna_sequences = [line.strip() for line in f if line.strip()]
                
            # Validate sequences
            valid_sequences = [dna for dna in dna_sequences if self.validate_dna(dna)]
            
            if valid_sequences:
                self.population = valid_sequences
                self.parameter_cache = {}  # Clear cache
                logger.info(f"Loaded {len(valid_sequences)} valid DNA sequences from {file_path}")
                return True
            else:
                logger.warning(f"No valid DNA sequences found in {file_path}")
                return False
        except Exception as e:
            logger.error(f"Failed to load DNA population: {e}")
            return False

class SwarmOptimizer:
    """
    Swarm-based optimizer for tuning system parameters based on 
    consciousness metrics and evolutionary dynamics.
    """
    def __init__(self, config):
        self.config = config
        self.device = config.device
        
        # Initialize particles
        self.particles = []
        for _ in range(config.swarm_size):
            self.particles.append({
                'position': {
                    'eta': config.eta,
                    'gamma_q': config.gamma_q,
                    'kappa': config.kappa,
                    'epsilon_0': config.epsilon_0,
                    'beta': config.beta,
                    'gamma': config.gamma,
                    'delta': config.delta_field
                },
                'velocity': {
                    'eta': 0.0,
                    'gamma_q': 0.0,
                    'kappa': 0.0,
                    'epsilon_0': 0.0,
                    'beta': 0.0,
                    'gamma': 0.0,
                    'delta': 0.0
                },
                'best_position': {
                    'eta': config.eta,
                    'gamma_q': config.gamma_q,
                    'kappa': config.kappa,
                    'epsilon_0': config.epsilon_0,
                    'beta': config.beta,
                    'gamma': config.gamma,
                    'delta': config.delta_field
                },
                'best_fitness': float('-inf')
            })
            
        # Global best
        self.global_best_position = {
            'eta': config.eta,
            'gamma_q': config.gamma_q,
            'kappa': config.kappa,
            'epsilon_0': config.epsilon_0,
            'beta': config.beta,
            'gamma': config.gamma,
            'delta': config.delta_field
        }
        self.global_best_fitness = float('-inf')
        
        # Track optimization history
        self.fitness_history = []
        self.parameter_history = []
        
        logger.info(f"Initialized SwarmOptimizer with {config.swarm_size} particles")

    def compute_fitness(self, metrics):
        """
        Compute fitness based on consciousness metrics.
        
        Args:
            metrics: Dictionary of consciousness metrics
            
        Returns:
            Float fitness value
        """
        # Weights for different metrics (use config values if available)
        weights = self.config.metric_weights
        
        # Target values
        targets = self.config.target_values
        
        # Calculate losses from targets
        losses = {}
        for metric, target in targets.items():
            if metric in metrics:
                losses[metric] = abs(metrics[metric] - target) / max(1e-6, target)
                
        # Calculate total fitness (negative loss)
        fitness = -sum(weights.get(m, 0.1) * losses.get(m, 0) for m in losses)
        
        return fitness

    def update_swarm(self, metrics):
        """
        Update swarm positions and velocities based on fitness.
        
        Args:
            metrics: Dictionary of consciousness metrics
            
        Returns:
            Tuple of (best_position, best_fitness)
        """
        # Calculate fitness
        fitness = self.compute_fitness(metrics)
        
        # Store history
        self.fitness_history.append(fitness)
        
        # Update particles
        for particle in self.particles:
            # Update best position if better fitness
            if fitness > particle['best_fitness']:
                particle['best_position'] = particle['position'].copy()
                particle['best_fitness'] = fitness
                
            # Update global best if better
            if fitness > self.global_best_fitness:
                self.global_best_position = particle['position'].copy()
                self.global_best_fitness = fitness
                self.parameter_history.append((self.global_best_position.copy(), fitness))
                
            # Update velocity and position for each parameter
            for param in particle['position'].keys():
                # Random coefficients
                r1, r2 = np.random.random(2)
                
                # Cognitive component
                cognitive = self.config.c1 * r1 * (particle['best_position'][param] - particle['position'][param])
                
                # Social component
                social = self.config.c2 * r2 * (self.global_best_position[param] - particle['position'][param])
                
                # Update velocity with inertia
                particle['velocity'][param] = (self.config.w * particle['velocity'][param] + cognitive + social)
                
                # Add sinusoidal oscillation to escape local minima
                oscillation = 0.01 * np.sin(2 * np.pi * 0.1 * len(self.fitness_history))
                particle['velocity'][param] += oscillation
                
                # Update position with constraints
                param_min, param_max = self._get_param_bounds(param)
                particle['position'][param] = np.clip(
                    particle['position'][param] + particle['velocity'][param],
                    param_min, param_max
                )
                
        return self.global_best_position, self.global_best_fitness
    
    def _get_param_bounds(self, param):
        """
        Get min and max bounds for parameters.
        
        Args:
            param: Parameter name
            
        Returns:
            Tuple of (min, max) bounds
        """
        bounds = {
            'eta': (0.001, 0.05),
            'gamma_q': (0.001, 0.05),
            'kappa': (0.01, 0.3),
            'epsilon_0': (0.001, 0.05),
            'beta': (0.1, 1.0),
            'gamma': (0.01, 0.3),
            'delta': (0.01, 0.2)
        }
        
        return bounds.get(param, (0.001, 1.0))
    
    def apply_best_parameters(self, config):
        """
        Apply best parameters to configuration.
        
        Args:
            config: Configuration object
            
        Returns:
            Updated configuration
        """
        # Apply best parameters found
        for param, value in self.global_best_position.items():
            if hasattr(config, param):
                setattr(config, param, value)
                
        return config
    
    def get_optimization_report(self):
        """
        Generate report on optimization progress.
        
        Returns:
            Dictionary with optimization statistics
        """
        if not self.fitness_history:
            return {
                'best_fitness': float('-inf'),
                'best_parameters': self.global_best_position,
                'convergence': 0.0
            }
            
        # Calculate convergence (how stable the fitness has been)
        if len(self.fitness_history) > 10:
            recent_variance = np.var(self.fitness_history[-10:])
            convergence = 1.0 / (1.0 + recent_variance)
        else:
            convergence = 0.0
            
        # Parameter evolution
        param_evolution = {}
        if len(self.parameter_history) > 0:
            for param in self.global_best_position.keys():
                param_evolution[param] = [p[0][param] for p in self.parameter_history]
                
        return {
            'best_fitness': self.global_best_fitness,
            'best_parameters': self.global_best_position,
            'convergence': convergence,
            'fitness_history': self.fitness_history,
            'parameter_evolution': param_evolution
        }

    def plot_network(self, graph, neural_dynamics, field_dynamics):
        """
        Plot network structure with neural and field states.
        
        Args:
            graph: NetworkX graph
            neural_dynamics: Neural dynamics module
            field_dynamics: Field dynamics module
        """
        if not graph or not hasattr(neural_dynamics, 'action_potential') or not hasattr(field_dynamics, 'psi'):
            logger.warning("Missing data for network visualization")
            return
            
        fig, ax = plt.subplots(figsize=(10, 8))
        
        # Get node positions (2D projection using spring layout)
        try:
            pos = nx.spring_layout(graph, seed=42)
        except:
            # Fallback for empty graph
            pos = {i: (np.random.random(), np.random.random()) for i in range(self.config.N)}
        
        # Get node states
        node_states = neural_dynamics.action_potential.cpu().numpy()
        field_states = field_dynamics.psi.mean(dim=1).cpu().numpy()
        
        # Normalize field states for color mapping
        field_norm = (field_states - field_states.min()) / max(1e-6, field_states.max() - field_states.min())
        
        # Node colors based on field state
        node_colors = [plt.cm.viridis(f) for f in field_norm]
        
        # Node sizes based on neural state (activation)
        node_sizes = 100 + 300 * node_states
        
        # Edge weights for line thickness
        edge_weights = []
        for u, v in graph.edges():
            weight = graph[u][v].get('weight', 0.1)
            edge_weights.append(weight * 5)
        
        # Draw network
        nx.draw_networkx_nodes(graph, pos, node_color=node_colors, node_size=node_sizes, alpha=0.8)
        nx.draw_networkx_edges(graph, pos, width=edge_weights, alpha=0.5, edge_color='w')
        
        # Add node labels for a subset of nodes
        if len(graph) <= 20:
            nx.draw_networkx_labels(graph, pos, font_size=8, font_color='w')
        
        plt.title('Neural Network Structure with Field and Neural States')
        plt.axis('off')
        plt.tight_layout()
        
        output_file = self.output_dir / 'network_visualization.png'
        plt.savefig(output_file)
        plt.close()
        logger.info(f"Saved network plot to {output_file}")

    def visualize_quantum_state(self, quantum_processor):
        """
        Visualize quantum state density matrix.
        
        Args:
            quantum_processor: Quantum processor module
        """
        if not hasattr(quantum_processor, 'rho'):
            logger.warning("Missing quantum state for visualization")
            return
            
        # Get density matrix
        rho = quantum_processor.rho.detach().cpu().numpy()
        
        # Extract real and imaginary parts
        rho_real = np.real(rho)
        rho_imag = np.imag(rho)
        
        # Create figure
        fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(12, 5))
        
        # Plot real part
        im1 = ax1.imshow(rho_real, cmap='viridis')
        ax1.set_title('Quantum Density Matrix (Real Part)')
        plt.colorbar(im1, ax=ax1)
        
        # Plot imaginary part
        im2 = ax2.imshow(rho_imag, cmap='plasma')
        ax2.set_title('Quantum Density Matrix (Imaginary Part)')
        plt.colorbar(im2, ax=ax2)
        
        # Add labels
        ax1.set_xlabel('Qubit State Index')
        ax1.set_ylabel('Qubit State Index')
        ax2.set_xlabel('Qubit State Index')
        
        plt.tight_layout()
        output_file = self.output_dir / 'quantum_state.png'
        plt.savefig(output_file)
        plt.close()
        logger.info(f"Saved quantum state plot to {output_file}")

    def visualize_neural_activity(self, neural_dynamics, t):
        """
        Visualize neural activity patterns.
        
        Args:
            neural_dynamics: Neural dynamics module
            t: Current time
        """
        if not hasattr(neural_dynamics, 'V') or not hasattr(neural_dynamics, 'action_potential'):
            logger.warning("Missing neural data for visualization")
            return
            
        # Get neural data
        V = neural_dynamics.V.detach().cpu().numpy()
        spikes = neural_dynamics.action_potential.detach().cpu().numpy()
        
        # Create figure
        fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(12, 8))
        
        # Plot membrane potentials
        for i in range(min(10, len(V))):  # Plot first 10 neurons
            color = plt.cm.tab10(i % 10)
            ax1.plot([-80, V[i]], [i, i], color=color, linewidth=2)
            if spikes[i] > 0:
                ax1.scatter(V[i], i, color='white', s=50, zorder=3)
                
        ax1.set_xlim(-80, 0)
        ax1.set_ylim(-1, 10)
        ax1.set_xlabel('Membrane Potential (mV)')
        ax1.set_ylabel('Neuron Index')
        ax1.set_title(f'Neural Activity at t={t:.3f}s')
        
        # Plot spikes as raster
        if hasattr(neural_dynamics, 'firing_history') and len(neural_dynamics.firing_history) > 0:
            history_length = min(100, len(neural_dynamics.firing_history))
            activity_matrix = np.zeros((self.config.N, history_length))
            
            for i in range(history_length):
                idx = -(history_length - i)
                if abs(idx) < len(neural_dynamics.firing_history):
                    activity_matrix[:, i] = neural_dynamics.firing_history[idx].cpu().numpy()
            
            ax2.imshow(activity_matrix, aspect='auto', cmap='hot', interpolation='nearest')
            ax2.set_xlabel('Time Steps')
            ax2.set_ylabel('Neuron Index')
            ax2.set_title('Spike Raster Plot')
        else:
            ax2.text(0.5, 0.5, 'No firing history available', 
                     horizontalalignment='center',
                     verticalalignment='center',
                     transform=ax2.transAxes)
            
        plt.tight_layout()
        output_file = self.output_dir / f'neural_activity_{int(t*1000)}.png'
        plt.savefig(output_file)
        plt.close()
        logger.info(f"Saved neural activity plot to {output_file}")

    def plot_field_dynamics(self, field_dynamics):
        """
        Visualize field dynamics and patterns.
        
        Args:
            field_dynamics: Field dynamics module
        """
        if not hasattr(field_dynamics, 'psi') or not hasattr(field_dynamics, 'field_history'):
            logger.warning("Missing field data for visualization")
            return
            
        # Create figure
        fig, axs = plt.subplots(2, 2, figsize=(12, 10))
        
        # Get current field state
        field = field_dynamics.psi.detach().cpu().numpy()
        
        # Plot field intensity across nodes
        ax = axs[0, 0]
        field_sum = np.sum(field, axis=1)
        ax.bar(range(len(field_sum)), field_sum, color='cyan', alpha=0.7)
        ax.set_xlabel('Node Index')
        ax.set_ylabel('Field Intensity')
        ax.set_title('Field Distribution Across Nodes')
        
        # Plot field dimensions
        ax = axs[0, 1]
        field_dims = np.mean(field, axis=0)
        ax.bar(range(len(field_dims)), field_dims, color='magenta', alpha=0.7)
        ax.set_xlabel('Field Dimension')
        ax.set_ylabel('Mean Value')
        ax.set_title('Field Dimension Values')
        
        # Plot field evolution if history exists
        ax = axs[1, 0]
        if hasattr(field_dynamics, 'field_history') and len(field_dynamics.field_history) > 0:
            history = torch.stack(field_dynamics.field_history).cpu().numpy()
            # Calculate field energy over time
            energy = np.sum(history**2, axis=(1, 2))
            ax.plot(range(len(energy)), energy, 'g-')
            ax.set_xlabel('Time Steps')
            ax.set_ylabel('Field Energy')
            ax.set_title('Field Energy Evolution')
        else:
            ax.text(0.5, 0.5, 'No field history available', 
                   horizontalalignment='center',
                   verticalalignment='center',
                   transform=ax.transAxes)
        
        # Plot 2D projection of field
        ax = axs[1, 1]
        if field.shape[1] >= 2:
            # Use first two dimensions for visualization
            ax.scatter(field[:, 0], field[:, 1], c=np.sum(field, axis=1), cmap='viridis')
            ax.set_xlabel('Field Dimension 1')
            ax.set_ylabel('Field Dimension 2')
            ax.set_title('2D Field Projection')
        elif field.shape[0] >= 2 and field.shape[1] >= 1:
            # Plot first dimension for multiple nodes
            for i in range(min(10, field.shape[0])):
                ax.bar(i, field[i, 0], color=plt.cm.tab10(i % 10))
            ax.set_xlabel('Node Index')
            ax.set_ylabel('Field Value (First Dimension)')
            ax.set_title('Field Values by Node')
        else:
            ax.text(0.5, 0.5, 'Insufficient field dimensions for visualization', 
                   horizontalalignment='center',
                   verticalalignment='center',
                   transform=ax.transAxes)
        
        plt.tight_layout()
        output_file = self.output_dir / 'field_dynamics.png'
        plt.savefig(output_file)
        plt.close()
        logger.info(f"Saved field dynamics plot to {output_file}")

class CosmicQuantumEcosystem:
    """
    Main class integrating all components into a unified system for quantum-driven
    sentient simulation.
    """
    def __init__(self, config=None):
        # Initialize configuration
        self.config = config if config is not None else CQEConfig()
        
        # Initialize components
        self.neural_dynamics = NeuralDynamics(self.config)
        self.quantum_processor = QuantumProcessor(self.config)
        self.excitonic_dynamics = ExcitonicDynamics(self.config)
        self.field_dynamics = FieldDynamics(self.config)
        self.topology = TopologicalNetworkDynamics(self.config)
        self.consciousness_metrics = ConsciousnessMetrics(self.config)
        self.swarm_optimizer = SwarmOptimizer(self.config)
        self.dna_manager = DNAManager(self.config)
        self.visualizer = DataVisualizer(self.config)
        
        # Runtime variables
        self.t = 0.0
        self.metrics_history = []
        self.decisions_history = []
        
        # Create output directories
        self.output_dir = Path(self.config.output_dir)
        self.checkpoint_dir = Path(self.config.checkpoint_dir)
        self.output_dir.mkdir(exist_ok=True)
        self.checkpoint_dir.mkdir(exist_ok=True)
        
        # System state history
        self.state_history = []
        
        logger.info("Initialized CosmicQuantumEcosystem with all components")

    def initialize_system(self):
        """
        Initialize the system with random states.
        
        Returns:
            Dictionary of initial metrics
        """
        # Generate random sensory input
        S = torch.normal(0, 1, (self.config.m, self.config.N), device=self.config.device)
        
        # Initialize neural dynamics
        self.neural_dynamics.action_potential = (self.neural_dynamics.V > self.neural_dynamics.V_th).float()
        
        # Initialize quantum processor
        quantum_out, _ = self.quantum_processor.update_quantum_state(
            self.field_dynamics.psi, 
            self.neural_dynamics.action_potential, 
            0.0, 
            self.config.dt
        )
        
        # Initialize excitonic dynamics
        self.excitonic_dynamics.update_excitonic(
            self.field_dynamics.psi, 
            self.neural_dynamics.action_potential, 
            self.config.dt
        )
        
        # Initialize field dynamics
        self.field_dynamics.update_field(
            self.topology.graph, 
            self.neural_dynamics.action_potential, 
            quantum_out, 
            0.0, 
            self.config.dt
        )
        
        # Initialize topology
        self.topology.update_weights(
            self.field_dynamics.psi, 
            self.neural_dynamics.action_potential, 
            0.0, 
            self.config.dt
        )
        
        # Compute initial metrics
        metrics = self.consciousness_metrics.compute_metrics(
            self.neural_dynamics, 
            self.quantum_processor, 
            self.field_dynamics, 
            self.topology, 
            0.0
        )
        self.metrics_history.append(metrics)
        
        logger.info(f"System initialized with metrics: {metrics}")
        return metrics

    def step(self, S=None):
        """
        Execute a single time step of system evolution.
        
        Args:
            S: Optional sensory input tensor (will generate random input if None)
            
        Returns:
            Tuple of (metrics, decision)
        """
        # Generate sensory input if not provided
        if S is None:
            S = torch.normal(0, 1, (self.config.m, self.config.N), device=self.config.device)
            
        # Update time
        self.t += self.config.dt
        
        # 1. Update quantum state
        quantum_out, entanglement_entropy = self.quantum_processor.update_quantum_state(
            self.field_dynamics.psi, 
            self.neural_dynamics.action_potential, 
            self.t, 
            self.config.dt
        )
        
        # 2. Update excitonic dynamics
        excitonic_state = self.excitonic_dynamics.update_excitonic(
            self.field_dynamics.psi, 
            self.neural_dynamics.action_potential, 
            self.config.dt
        )
        
        # 3. Meta-update for neural parameters
        epsilon = self.neural_dynamics.meta_update(self.field_dynamics.psi, self.t)
        
        # 4. Update neural dynamics
        self.neural_dynamics.update_neural(
            S, 
            self.field_dynamics.psi, 
            quantum_out, 
            self.t, 
            self.config.dt, 
            epsilon
        )
        
        # 5. Update neural weights and memory
        self.neural_dynamics.update_weights(self.config.dt)
        self.neural_dynamics.update_memory(S, excitonic_state, self.config.dt)
        
        # 6. Calculate phi for semantic updates
        phi, _ = self.consciousness_metrics.compute_phi(
            self.neural_dynamics.action_potential,
            self.field_dynamics.psi,
            quantum_out
        )
        
        # 7. Update semantic and apobetic states
        self.neural_dynamics.update_semantic(
            self.field_dynamics.psi, 
            excitonic_state, 
            phi, 
            self.config.dt
        )
        self.neural_dynamics.update_apobetic(
            self.topology.graph, 
            excitonic_state, 
            phi, 
            self.config.dt
        )
        
        # 8. Update field dynamics
        self.field_dynamics.update_field(
            self.topology.graph, 
            self.neural_dynamics.action_potential, 
            quantum_out, 
            self.t, 
            self.config.dt
        )
        
        # 9. Update topological structure
        self.topology.update_weights(
            self.field_dynamics.psi, 
            self.neural_dynamics.action_potential, 
            self.t, 
            self.config.dt
        )
        
        # 10. Generate decision
        decision, decision_probs = self.neural_dynamics.generate_decision(quantum_out)
        self.decisions_history.append((decision, decision_probs))
        
        # 11. Compute consciousness metrics
        metrics = self.consciousness_metrics.compute_metrics(
            self.neural_dynamics, 
            self.quantum_processor, 
            self.field_dynamics, 
            self.topology, 
            self.t,
            self.decisions_history
        )
        self.metrics_history.append(metrics)
        
        # 12. Store system state for analysis (every 10 steps)
        if int(self.t / self.config.dt) % 10 == 0:
            state = self.get_system_state()
            self.state_history.append(state)
            if len(self.state_history) > 100:  # Keep only recent states
                self.state_history.pop(0)
                
        # 13. Optimize system parameters using swarm (every 10 steps)
        if len(self.metrics_history) % 10 == 0:
            best_params, best_fitness = self.swarm_optimizer.update_swarm(metrics)
            
            # Apply best parameters
            self.config.eta = best_params['eta']
            self.config.gamma_q = best_params['gamma_q']
            self.config.kappa = best_params['kappa']
            
            # Log optimization progress
            logger.info(f"Swarm optimization: t={self.t:.2f}s, fitness={best_fitness:.4f}")
            
        # 14. Evolve DNA population (every 100 steps)
        if len(self.metrics_history) % 100 == 0:
            # Calculate fitness from metrics
            fitness_scores = [metrics['phi'] * 0.4 + metrics['coherence'] * 0.3 + 
                              metrics['entanglement_entropy'] * 0.2 + 
                              metrics['behavioral_entropy'] * 0.1
                              for _ in range(len(self.dna_manager.population))]
            
            # Evolve population
            self.dna_manager.evolve_population(fitness_scores)
            
            # Apply DNA parameters occasionally for exploration
            if random.random() < 0.2:  # 20% chance to apply DNA parameters
                modules_dict = {
                    'neural_dynamics': self.neural_dynamics,
                    'quantum_processor': self.quantum_processor,
                    'field_dynamics': self.field_dynamics
                }
                self.dna_manager.apply_dna_parameters(modules_dict)
                logger.info(f"Applied DNA parameters at t={self.t:.2f}s")
            
        return metrics, decision

    def get_system_state(self):
        """
        Get complete system state for analysis.
        
        Returns:
            Dictionary containing all system state information
        """
        return {
            'time': self.t,
            'neural_dynamics': self.neural_dynamics,
            'quantum_processor': self.quantum_processor,
            'excitonic_dynamics': self.excitonic_dynamics,
            'field_dynamics': self.field_dynamics,
            'topology': self.topology,
            'consciousness_metrics': self.consciousness_metrics,
            'latest_metrics': self.metrics_history[-1] if self.metrics_history else {},
            'latest_decision': self.decisions_history[-1] if self.decisions_history else None
        }

def run_simulation(self, duration=None, visualize_every=10):
    """
    Run simulation for specified duration.
    
    Args:
        duration: Optional simulation duration (defaults to config value)
        visualize_every: Number of steps between visualizations
        
    Returns:
        List of metrics history
    """
    if duration is None:
        duration = self.config.T
        
    # Initialize system
    self.initialize_system()
    
    # Run for specified duration
    n_steps = int(duration / self.config.dt)
    progress_interval = max(1, n_steps // 20)  # Report progress 20 times
    
    for step_idx in range(n_steps):
        # Generate rhythmic input with harmonics of 40Hz
        t = step_idx * self.config.dt
        S = torch.zeros((self.config.m, self.config.N), device=self.config.device)
        
        # Create input with multiple frequency components
        for i in range(self.config.m):
            for f, a in zip(self.config.f, self.config.a_f):
                S[i] += a * torch.sin(2 * np.pi * f * (t + i * 0.01))
                
        # Add noise
        S += torch.normal(0, 0.1, S.shape, device=self.config.device)
        
        # Execute step
        metrics, decision = self.step(S)
        
        # Log progress
        if step_idx % progress_interval == 0 or step_idx == n_steps - 1:
            progress = (step_idx + 1) / n_steps * 100
            phi = metrics.get('phi', 0)
            coherence = metrics.get('coherence', 0)
            logger.info(f"Progress: {progress:.1f}%, t={self.t:.3f}s, "
                       f"Phi={phi:.4f}, Coherence={coherence:.4f}, Decision={decision}")
            
        # Visualize
        if step_idx % visualize_every == 0 or step_idx == n_steps - 1:
            self.visualizer.visualize_neural_activity(self.neural_dynamics, self.t)
            
            # Create dashboard occasionally
            if step_idx % (visualize_every * 10) == 0 or step_idx == n_steps - 1:
                self.visualizer.create_dashboard(self.get_system_state())
                
        # Save checkpoint occasionally
        if step_idx % (n_steps // 5) == 0 and step_idx > 0:
            self.save_checkpoint()
            
    # Final visualizations
    self.visualizer.plot_metrics(self.metrics_history)
    self.visualizer.plot_network(self.topology.graph, self.neural_dynamics, self.field_dynamics)
    self.visualizer.visualize_quantum_state(self.quantum_processor)
    self.visualizer.plot_field_dynamics(self.field_dynamics)
    self.visualizer.visualize_consciousness_metrics(self.consciousness_metrics)
    
    # Save final state
    self.save_checkpoint()
    
    # Generate and save summary report
    self.generate_summary_report()
    
    return self.metrics_history

def save_checkpoint(self):
    """
    Save system state to checkpoint file.
    """
    checkpoint = {
        't': self.t,
        'metrics_history': self.metrics_history,
        'decisions_history': self.decisions_history,
        'neural_dynamics': self.neural_dynamics.state_dict(),
        'quantum_processor': self.quantum_processor.state_dict(),
        'excitonic_dynamics': self.excitonic_dynamics.state_dict(),
        'field_dynamics': self.field_dynamics.state_dict()
    }
    
    checkpoint_path = self.checkpoint_dir / f'checkpoint_{int(self.t*1000)}.pt'
    torch.save(checkpoint, checkpoint_path)
    logger.info(f"Saved checkpoint to {checkpoint_path}")

def load_checkpoint(self, checkpoint_path):
    """
    Load system state from checkpoint file.
    
    Args:
        checkpoint_path: Path to checkpoint file
    """
    checkpoint = torch.load(checkpoint_path, map_location=self.config.device)
    
    # Load time and history
    self.t = checkpoint['t']
    self.metrics_history = checkpoint['metrics_history']
    self.decisions_history = checkpoint['decisions_history']
    
    # Load component states
    self.neural_dynamics.load_state_dict(checkpoint['neural_dynamics'])
    self.quantum_processor.load_state_dict(checkpoint['quantum_processor'])
    self.excitonic_dynamics.load_state_dict(checkpoint['excitonic_dynamics'])
    self.field_dynamics.load_state_dict(checkpoint['field_dynamics'])
    
    logger.info(f"Loaded checkpoint from {checkpoint_path} at t={self.t:.3f}s")

                    return obj.tolist()

            logging.StreamHandler()
        ]
    )
    
    logger.info("Starting Cosmic Quantum Ecosystem simulation")
    
    # Parse command line arguments
    import argparse
    parser = argparse.ArgumentParser(description='Run Cosmic Quantum Ecosystem simulation')
    parser.add_argument('--duration', type=float, default=None, help='Simulation duration in seconds')
    parser.add_argument('--nodes', type=int, default=None, help='Number of neural nodes')
    parser.add_argument('--qubits', type=int, default=None, help='Number of qubits')
    parser.add_argument('--dimensions', type=int, default=None, help='Field dimensions')
    parser.add_argument('--checkpoint', type=str, default=None, help='Checkpoint file to load')
    parser.add_argument('--output', type=str, default=None, help='Output directory')
    args = parser.parse_args()
    
    # Create configuration
    config = CQEConfig()
    
    # Override configuration with command line arguments
    if args.nodes:
        config.N = args.nodes
    if args.qubits:
        config.k = args.qubits
    if args.dimensions:
        config.d = args.dimensions
    if args.output:
        config.output_dir = args.output
        config.checkpoint_dir = os.path.join(args.output, "checkpoints")
    
    # Create system
    cqe = CosmicQuantumEcosystem(config)
    
    # Load checkpoint if specified
    if args.checkpoint:
        try:
            cqe.load_checkpoint(args.checkpoint)
            logger.info(f"Loaded checkpoint from {args.checkpoint}")
        except Exception as e:
            logger.error(f"Failed to load checkpoint: {e}")
            return
    
    # Run simulation
    try:
        metrics_history = cqe.run_simulation(duration=args.duration)
        
        # Calculate overall score
        final_metrics = metrics_history[-1] if metrics_history else {}
        score = cqe.consciousness_metrics.compute_overall_score(final_metrics)
        status = cqe.consciousness_metrics.get_consciousness_status(final_metrics)
        
        # Print final results
        logger.info("Simulation completed successfully")
        logger.info(f"Final time: {cqe.t:.3f}s")
        logger.info(f"Final consciousness score: {score:.4f}")
        logger.info(f"Consciousness status: {status}")
        logger.info(f"Output saved to: {config.output_dir}")
        
    except Exception as e:
        logger.error(f"Simulation failed: {e}")
        import traceback
        traceback.print_exc()

if __name__ == "__main__":
    main()
