def create_weighted_datum(
    messages: List[Dict],
    renderer,
    tokenizer,
    epoch: int = 0,
    episode_metadata: Dict = None
) -> types.Datum:
    """
    Convert messages to weighted datum with differential token weighting.
    
    Args:
        messages: Chat messages with system/user/assistant roles
        renderer: Tinker renderer for formatting
        tokenizer: Tinker tokenizer
        epoch: Current training epoch (for curriculum learning)
        episode_metadata: Episode info for difficulty weighting
    """
    
    # Render conversation
    text = renderer.render(messages)
    tokens = tokenizer.encode(text)
    token_count = len(tokens)
    
    # Initialize all weights to 1.0
    weights = [1.0] * token_count
    
    # Apply field-level weights within [PEDAGOGY] block
    weights = apply_pedagogy_weights(text, tokens, weights, tokenizer)
    
    # Apply prompt loss weight (0.1x for user messages)
    weights = apply_prompt_weights(text, messages, tokens, weights, tokenizer)
    
    # Apply sample-level difficulty multiplier
    if episode_metadata and DIFFICULTY_FACTORS:
        difficulty_mult = compute_difficulty_multiplier(episode_metadata)
        weights = [w * difficulty_mult for w in weights]
    
    # Apply curriculum schedule
    if epoch < len(CURRICULUM_SCHEDULE):
        schedule = CURRICULUM_SCHEDULE.get(epoch, {})
        pedagogy_mult = schedule.get("pedagogy_base_multiplier", 1.0)
        # Apply multiplier to pedagogy tokens only
        weights = apply_curriculum_multiplier(
            text, tokens, weights, tokenizer, pedagogy_mult
        )
    
    # Safety check
    max_weight = max(weights)
    avg_weight = sum(weights) / len(weights)
    
    if max_weight > 5.5:
        print(f"WARNING: Max weight {max_weight:.2f} exceeds safe threshold")
    
    return types.Datum(
        input_ids=tokens,
        loss_weights=weights,
        loss_fn_inputs={"target": tokens}
    )

# ==================================================================
# PEDAGOGY FIELD WEIGHTING
# ==================================================================

def apply_pedagogy_weights(
    text: str,
    tokens: List[int],
    weights: List[float],
    tokenizer
) -> List[float]:
    """Apply differential weights to fields within [PEDAGOGY] block."""
    
    # Find [PEDAGOGY] block boundaries
    pedagogy_pattern = r'\[PEDAGOGY\](.*?)\[/PEDAGOGY\]'
    pedagogy_matches = list(re.finditer(pedagogy_pattern, text, re.DOTALL))
    
    if not pedagogy_matches:
        return weights
    
    for match in pedagogy_matches:
        pedagogy_start = match.start()
        pedagogy_end = match.end()
        pedagogy_content = match.group(1)
        
        # Apply field-specific weights
        for field_name, weight_value in TOKEN_WEIGHTS.items():
            if field_name in ["response_text", "user_message"]:
                continue  # Handle separately
            
            # Pattern to match field and its content
            # Handles multi-line fields (e.g., accumulated_insight, my_thoughts)
            field_pattern = rf'{field_name}:\s*([^\n]+(?:\n(?![a-z_]+:).*)*)'
            field_matches = re.finditer(field_pattern, pedagogy_content, re.MULTILINE)
            
            for field_match in field_matches:
                # Get absolute position in text
                field_start = pedagogy_start + field_match.start()
                field_end = pedagogy_start + field_match.end()
                
                # Convert to token indices
                field_start_token = len(tokenizer.encode(text[:field_start]))
                field_end_token = len(tokenizer.encode(text[:field_end]))
                
                # Apply weight
                for i in range(field_start_token, min(field_end_token, len(weights))):
                    weights[i] = weight_value
    
    return weights

# ==================================================================
# PROMPT LOSS WEIGHT (0.1x for user messages)
# ==================================================================

def apply_prompt_weights(
    text: str,
    messages: List[Dict],
    tokens: List[int],
    weights: List[float],
    tokenizer
) -> List[float]:
    """Apply 0.1x weight to user messages (research-backed PLW)."""
    
    for msg in messages:
        if msg['role'] == 'user':
            user_text = msg['content']
            user_start = text.find(user_text)
            
            if user_start != -1:
                user_start_token = len(tokenizer.encode(text[:user_start]))
                user_end_token = len(tokenizer.encode(text[:user_start + len(user_text)]))
                
                for i in range(user_start_token, min(user_end_token, len(weights))):
                    weights[i] = 0.1
    
    return weights

# SAMPLE DIFFICULTY COMPUTATION
# ==================================================================

def compute_difficulty_multiplier(episode_metadata: Dict) -> float:
    """
    Compute difficulty multiplier for this episode.
    Harder episodes get weighted more heavily.
    """
    
    difficulty_score = 1.0
    
    # Factor 1: Mechanism complexity
    mechanism = episode_metadata.get('mechanism_shape', 'none')
    mechanism_weights = DIFFICULTY_FACTORS.get('mechanism_complexity', {})
    difficulty_score *= mechanism_weights.get(mechanism, 1.0)
    
    # Factor 2: Register intensity
    register = episode_metadata.get('register', '')
    intensity_match = re.search(r'intensity=PR_(\d+)', register)
    if intensity_match:
        intensity_level = f"PR_0{intensity_match.group(1)}"
        intensity_weights = DIFFICULTY_FACTORS.get('register_intensity', {})
        difficulty_score *= intensity_weights.get(intensity_level, 1.0)
    
    # Factor 3: Confidence penalty (lower confidence = harder)
    if DIFFICULTY_FACTORS.get('confidence_penalty'):
        confidence = episode_metadata.get('impact_confidence', 0.5)
        # Invert: low confidence (0.2) → higher multiplier (1.6)
        difficulty_score *= (1.0 + (1.0 - confidence) * 0.8)
    
    # Factor 4: Multi-turn bonus
    turn_count = episode_metadata.get('turn_count', 1)
    if turn_count > 1 and DIFFICULTY_FACTORS.get('multi_turn_bonus'):
        difficulty_score *= DIFFICULTY_FACTORS['multi_turn_bonus']
    
    # Normalize to 0.5 - 2.0 range
    return min(2.0, max(0.5, difficulty_score))

# ==================================================================
# CURRICULUM MULTIPLIER
# ==================================================================

def apply_curriculum_multiplier(
    text: str,
    tokens: List[int],
    weights: List[float],
    tokenizer,
    multiplier: float
) -> List[float]:
    """Apply curriculum multiplier to pedagogy tokens only."""
    
    pedagogy_pattern = r'\[PEDAGOGY\](.*?)\[/PEDAGOGY\]'
    pedagogy_matches = list(re.finditer(pedagogy_pattern, text, re.DOTALL))
    
    for match in pedagogy_matches:
        pedagogy_start = match.start()
        pedagogy_end = match.end()
        
        pedagogy_start_token = len(tokenizer.encode(text[:pedagogy_start]))
        pedagogy_end_token = len(tokenizer.encode(text[:pedagogy_end]))
        
        for i in range(pedagogy_start_token, min(pedagogy_end_token, len(weights))):
            weights[i] *= multiplier
    
    return weights

def extract_episode_metadata(episode: Dict) -> Dict:
    """Extract metadata from episode for difficulty weighting."""
    
    # Get last assistant message (has the final [PEDAGOGY])
    assistant_messages = [
        msg for msg in episode['messages'] 
        if msg['role'] == 'assistant'
    ]
    
    if not assistant_messages:
        return {}
    
    last_message = assistant_messages[-1]['content']
    
    # Parse [PEDAGOGY] block
    pedagogy_match = re.search(
        r'\[PEDAGOGY\](.*?)\[/PEDAGOGY\]',
        last_message,
        re.DOTALL
    )
    
    if not pedagogy_match:
        return {}
    
    pedagogy_text = pedagogy_match.group(1)
    
    # Extract fields
    metadata = {
        'turn_count': len(assistant_messages),
    }
    
    # Extract specific fields
    for field in ['mechanism_shape', 'register', 'impact_confidence']:
        pattern = rf'{field}:\s*(.+?)(?:\n|$)'
        match = re.search(pattern, pedagogy_text)
        if match:
            value = match.group(1).strip()
            if field == 'impact_confidence':
                try:
                    value = float(value)
                except:
                    value = 0.5
            metadata[field] = value
    
    return metadata

def load_training_data(path: str) -> Tuple[List[Dict], List[Dict]]:
    """
    Load training episodes and split into train/val.
    
    Returns:
        (train_episodes, val_episodes)
    """
    
    with open(path, 'r') as f:
        all_episodes = json.load(f)
    
    # Shuffle
    import random
    random.seed(42)
    random.shuffle(all_episodes)
    
    # Split
    split_idx = int(len(all_episodes) * (1 - VALIDATION_SPLIT))
    train_episodes = all_episodes[:split_idx]
    val_episodes = all_episodes[split_idx:]
    
    print(f"Loaded {len(train_episodes)} training episodes")
    print(f"Loaded {len(val_episodes)} validation episodes")
    
    return train_episodes, val_episodes

def episode_to_messages(episode: Dict, system_prompt: str) -> List[Dict]:
    """
    Convert episode to messages format with system prompt.
    
    Replaces 'USE_HXRMXS_TRAINING_PROMPT' marker with actual prompt.
    """
    
    messages = []
    
    for msg in episode['messages']:
        if msg['role'] == 'system':
            # Replace marker with actual prompt
            content = msg['content']
            if content == 'USE_HXRMXS_TRAINING_PROMPT':
                content = system_prompt
            messages.append({'role': 'system', 'content': content})
        else:
            messages.append(msg)
    
    return messages

def train_hxrmxs():
    """Main training function with all optimizations."""
    
    print("="*70)
    print("HXRMXS TEACHER v1 TRAINING")
    print("="*70)
    
    # Initialize Tinker
    print("\n[1/6] Initializing Tinker client...")
    service_client = tinker.ServiceClient()
    
    # Create training client
    print(f"\n[2/6] Creating LoRA training client...")
    print(f"  Base model: {BASE_MODEL}")
    print(f"  LoRA rank: {LORA_RANK}, alpha: {LORA_ALPHA}")

        rank=LORA_RANK,
        alpha=LORA_ALPHA,

    )
    
    # Get tokenizer and renderer
    tokenizer = training_client.get_tokenizer()
    
    # Simple renderer (Tinker provides this)

    # Load system prompt
    print(f"\n[3/6] Loading system prompt...")
    system_prompt = load_system_prompt(SYSTEM_PROMPT_PATH)
    print(f"  Loaded {len(system_prompt)} characters")
    
    # Load training data
    print(f"\n[4/6] Loading training data...")
    train_episodes, val_episodes = load_training_data(TRAINING_DATA_PATH)
    
    # Get optimal learning rate
    if USE_AUTO_LR:
        print(f"\n[5/6] Computing optimal learning rate...")
        learning_rate = get_lr(BASE_MODEL)
        print(f"  Auto LR: {learning_rate}")
    else:
        learning_rate = MANUAL_LR
        print(f"  Manual LR: {learning_rate}")
    
    # Training state
    print(f"\n[6/6] Starting training...")
    print(f"  Epochs: {NUM_EPOCHS}")
    print(f"  Batch size: {BATCH_SIZE}")
    print(f"  Warmup steps: {WARMUP_STEPS}")
    
    best_val_loss = float('inf')
    patience_counter = 0
    global_step = 0
    
    # Training loop
    for epoch in range(NUM_EPOCHS):
        print(f"\n{'='*70}")
        print(f"EPOCH {epoch + 1}/{NUM_EPOCHS}")
        print(f"{'='*70}")
        
        # Shuffle training data
        random.shuffle(train_episodes)
        
        # Create batches
        batches = [
            train_episodes[i:i+BATCH_SIZE]
            for i in range(0, len(train_episodes), BATCH_SIZE)
        ]
        
        epoch_loss = 0.0
        
        for batch_idx, batch in enumerate(batches):
            # Create weighted data for batch
            weighted_batch = []
            
            for episode in batch:
                # Convert to messages
                messages = episode_to_messages(episode, system_prompt)
                
                # Extract metadata for difficulty weighting
                metadata = extract_episode_metadata(episode)
                
                # Create weighted datum
                datum = create_weighted_datum(
                    messages=messages,
                    renderer=renderer,
                    tokenizer=tokenizer,
                    epoch=epoch,
                    episode_metadata=metadata
                )
                
                weighted_batch.append(datum)
            
            # Forward-backward pass
            result = training_client.forward_backward(
                weighted_batch,
                learning_rate=learning_rate,
                loss_fn="cross_entropy"
            )
            
            # Optimizer step
            training_client.optim_step()
            
            # Track loss
            batch_loss = result.loss_fn_outputs.get('loss', 0.0)
            epoch_loss += batch_loss
            global_step += 1
            
            # Log every 10 steps
            if batch_idx % 10 == 0:
                print(f"  Step {global_step} | Batch {batch_idx}/{len(batches)} | Loss: {batch_loss:.4f}")
            
            # Validation check
            if global_step % VALIDATION_FREQUENCY == 0:
                print(f"\n  Running validation at step {global_step}...")
                val_loss = evaluate(
                    training_client,
                    val_episodes,
                    system_prompt,
                    renderer,
                    tokenizer,
                    epoch
                )
                print(f"  Validation loss: {val_loss:.4f}")

                if val_loss < best_val_loss:
                    best_val_loss = val_loss
                    patience_counter = 0
                    
                    # Save best checkpoint
                    print(f"  ✓ New best model! Saving checkpoint...")
                    save_path = training_client.save_state(
                        name=f"best_e{epoch}_s{global_step}"
                    ).result().path
                    print(f"  Saved to: {save_path}")
                    
                else:
                    patience_counter += 1
                    print(f"  No improvement ({patience_counter}/{PATIENCE})")
                
                # Early stopping
                if EARLY_STOPPING_ENABLED and patience_counter >= PATIENCE:
                    print(f"\n  Early stopping triggered at step {global_step}")
                    break
            
            # Regular checkpoint
            if global_step % CHECKPOINT_FREQUENCY == 0:
                checkpoint_name = f"checkpoint_e{epoch}_s{global_step}"
                training_client.save_state(name=checkpoint_name)

            print(f"\nStopping training early after epoch {epoch + 1}")

def evaluate(
    training_client,
    val_episodes: list,
    system_prompt: str,
    renderer,
    tokenizer,
    epoch: int
) -> float:
    """Run validation and return average loss."""
    
    total_loss = 0.0
    
    # Sample 20 validation episodes (or all if fewer)
    sample_size = min(20, len(val_episodes))
    val_sample = random.sample(val_episodes, sample_size)
    
    for episode in val_sample:
        messages = episode_to_messages(episode, system_prompt)
        metadata = extract_episode_metadata(episode)
        
        datum = create_weighted_datum(
            messages=messages,
            renderer=renderer,
            tokenizer=tokenizer,
            epoch=epoch,
            episode_metadata=metadata
        )
        
        # Forward pass only (no backward, no optim step)
        result = training_client.forward_backward(
            [datum],
            learning_rate=0.0,  # No update
            loss_fn="cross_entropy"
        )
        
        total_loss += result.loss_fn_outputs.get('loss', 0.0)
    
    return total_loss / sample_size
