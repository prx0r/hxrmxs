solidity// ChainNFT.sol - Each published chain becomes an NFT
contract ChainNFT is ERC721 {
    struct ChainMetadata {
        address creator;           // Who created this chain
        string ipfsHash;          // Chain definition stored on IPFS
        uint256 pricePerUse;      // Cost for agents to use this chain
        uint256 totalUses;        // How many times it's been used
        uint256 creatorEarnings;  // Total earned by creator
        bool isActive;            // Can be used by agents
        string[] tags;            // ["tiktok", "viral", "fashion"]
        uint8 qualityScore;       // 1-10 based on success rate
    }
    
    mapping(uint256 => ChainMetadata) public chains;
    mapping(address => uint256[]) public creatorChains;
    
    event ChainUsed(uint256 indexed chainId, address indexed agent, uint256 cost);
    event CreatorPaid(uint256 indexed chainId, address indexed creator, uint256 amount);
    
    function publishChain(
        string memory ipfsHash,
        uint256 pricePerUse,
        string[] memory tags
    ) external returns (uint256 chainId) {
        chainId = _nextTokenId++;
        
        chains[chainId] = ChainMetadata({
            creator: msg.sender,
            ipfsHash: ipfsHash,
            pricePerUse: pricePerUse,
            totalUses: 0,
            creatorEarnings: 0,
            isActive: true,
            tags: tags,
            qualityScore: 5  // Default, improves with usage
        });
        
        creatorChains[msg.sender].push(chainId);
        _mint(msg.sender, chainId);
        
        return chainId;
    }
    
    function useChain(uint256 chainId) external payable {
        ChainMetadata storage chain = chains[chainId];
        require(chain.isActive, "Chain not active");
        require(msg.value >= chain.pricePerUse, "Insufficient payment");
        
        // 85% to creator, 15% platform fee
        uint256 creatorPayout = (chain.pricePerUse * 85) / 100;
        uint256 platformFee = chain.pricePerUse - creatorPayout;
        
        // Pay creator immediately
        payable(chain.creator).transfer(creatorPayout);
        
        // Update stats
        chain.totalUses++;
        chain.creatorEarnings += creatorPayout;
        
        emit ChainUsed(chainId, msg.sender, chain.pricePerUse);
        emit CreatorPaid(chainId, chain.creator, creatorPayout);
    }
}

contract VersionedChainNFT is ERC721 {
    using SafeERC20 for IERC20;
    
    IERC20 public immutable usdcToken;
    address public immutable platformWallet;
    address public immutable gasPaymaster;  // For sponsored transactions
    
    struct ChainMetadata {
        address creator;           // Human or agent who created it
        address beneficiary;       // Who receives NFT ownership  
        string projectId;          // Associated project
        uint256 pricePerUse;      // Cost in USDC
        uint256 creatorShareBps;  // Creator's revenue share (basis points)
        uint256 totalUses;
        uint256 creatorEarnings;
        bool isActive;
    }
    
    struct ChainVersion {
        string ipfsHash;          // Chain definition
        string arweaveId;         // Permanent backup
        uint256 version;
        uint256 createdAt;
        bool isActive;
    }
    
    mapping(uint256 => ChainMetadata) public chains;
    mapping(uint256 => ChainVersion[]) public chainVersions;
    mapping(uint256 => uint256) public activeVersion;
    
    modifier onlyPaymaster() {
        require(msg.sender == gasPaymaster, "Only paymaster can sponsor");
        _;
    }
    
    function publishChainFromProject(
        string memory ipfsHash,
        string memory arweaveId,
        uint256 pricePerUse,
        string memory projectId,
        address beneficiary,
        uint256 creatorShareBps
    ) external returns (uint256 chainId) {
        chainId = _nextTokenId++;
        
        chains[chainId] = ChainMetadata({
            creator: msg.sender,
            beneficiary: beneficiary,
            projectId: projectId,
            pricePerUse: pricePerUse,
            creatorShareBps: creatorShareBps,
            totalUses: 0,
            creatorEarnings: 0,
            isActive: true
        });
        
        chainVersions[chainId].push(ChainVersion({
            ipfsHash: ipfsHash,
            arweaveId: arweaveId,
            version: 1,
            createdAt: block.timestamp,
            isActive: true
        }));
        
        activeVersion[chainId] = 0;
        _mint(beneficiary, chainId);
        
        emit ChainPublished(chainId, msg.sender, beneficiary, projectId);
        return chainId;
    }
    
    function useChainSponsored(uint256 chainId, address agent) external onlyPaymaster {
        ChainMetadata storage chain = chains[chainId];
        require(chain.isActive, "Chain not active");
        
        // Transfer USDC from agent to split between parties
        uint256 platformFee = (chain.pricePerUse * 1500) / 10000; // 15%
        uint256 remaining = chain.pricePerUse - platformFee;
        uint256 creatorPayout = (remaining * chain.creatorShareBps) / 10000;
        uint256 beneficiaryPayout = remaining - creatorPayout;
        
        usdcToken.safeTransferFrom(agent, chain.creator, creatorPayout);
        usdcToken.safeTransferFrom(agent, chain.beneficiary, beneficiaryPayout);
        usdcToken.safeTransferFrom(agent, platformWallet, platformFee);
        
        chain.totalUses++;
        chain.creatorEarnings += creatorPayout;
        
        emit ChainUsed(chainId, agent, creatorPayout, beneficiaryPayout);
    }
}

javascript// /cloudflare-workers/replicate-cache-worker.js
export default {
  async fetch(request, env) {
    const url = new URL(request.url);
    const cacheKey = `replicate:${url.pathname}${url.search}`;
    
    // Check cache first
    const cached = await env.CACHE.get(cacheKey);
    if (cached) {
      return new Response(cached, {
        headers: { 'Content-Type': 'application/json' }
      });
    }
    
    // Forward to Replicate API
    const replicateResponse = await fetch(`https://api.replicate.com${url.pathname}${url.search}`, {
      method: request.method,
      headers: {
        'Authorization': `Token ${env.REPLICATE_API_KEY}`,
        'Content-Type': 'application/json'
      },
      body: request.method !== 'GET' ? await request.text() : undefined
    });
    
    const result = await replicateResponse.text();
    
    // Cache successful responses for 1 hour
    if (replicateResponse.ok) {
      await env.CACHE.put(cacheKey, result, { expirationTtl: 3600 });
    }
    
    return new Response(result, {
      status: replicateResponse.status,
      headers: { 'Content-Type': 'application/json' }
    });
  }
};
