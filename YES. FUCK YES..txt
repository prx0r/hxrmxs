pythonclass GeometricAnalyst(nn.Module):
    """
    Small, fast, specialized neural network for geometric reasoning
    """
    
    def __init__(self):
        # Student state encoder
        self.student_encoder = nn.Sequential(
            nn.Linear(student_features, 256),
            nn.ReLU(),
            nn.Linear(256, 128)
        )
        
        # Context encoder (conversation history)
        self.context_encoder = nn.GRU(
            input_size=128,
            hidden_size=128,
            num_layers=2
        )
        
        # Graph encoder (move/bridge/shape library)
        self.graph_encoder = GNN(
            in_channels=move_features,
            hidden_channels=128,
            num_layers=3
        )
        
        # Decision heads
        self.move_selector = nn.Linear(256, num_moves)
        self.register_predictor = nn.Linear(256, 6)  # 6 register axes
        self.success_predictor = nn.Linear(256, 3)   # success/partial/fail
        
    def forward(self, student_state, conversation_history, move_graph):
        # Encode student
        s = self.student_encoder(student_state)
        
        # Encode conversation
        c, _ = self.context_encoder(conversation_history)
        c = c[-1]  # Last hidden state
        
        # Encode geometric library
        g = self.graph_encoder(move_graph)
        
        # Combine
        combined = torch.cat([s, c], dim=-1)
        
        # Predict
        move_logits = self.move_selector(combined)
        register = self.register_predictor(combined)
        success_prob = self.success_predictor(combined)
        
        return move_logits, register, success_prob
    
    def select_move(self, student_state, context, move_library):
        """Main decision function"""
        
        # Get valid moves (symbolic filtering)
        valid_moves = self.get_valid_moves(
            phase=context.phase,
            progress=context.progress,
            student_state=student_state
        )
        
        # Encode and predict
        move_logits, register, success_prob = self.forward(
            student_state, context.history, move_library.graph
        )
        
        # Mask invalid moves
        move_logits[~valid_moves] = -inf
        
        # Select best
        move_idx = torch.argmax(move_logits)
        move = move_library[move_idx]
        
        # Build directive
        directive = GeometricDirective(
            move_id=move.move_id,
            move_tag=move.move_tag,
            geometry_context=context.to_dict(),
            register_directive=register.to_dict(),
            content_directive=self.get_content_directive(move, context),
            reasoning=self.generate_reasoning(move, student_state),
            constraints=self.get_constraints(move, context)
        )
        
        return directive, success_prob

pythonimport torch
import torch.nn as nn

class MinimalGeometricAnalyst(nn.Module):
    """
    Tiny specialist: 5M parameters
    Just does move selection + register prediction
    """
    
    def __init__(self):
        # Student state → 128
        self.student_enc = nn.Linear(50, 128)  # 50 features
        
        # Move selector: 128 → 50 moves
        self.move_head = nn.Linear(128, 50)
        
        # Register predictor: 128 → 6 register axes
        self.register_head = nn.Linear(128, 6)
    
    def forward(self, student_features):
        x = torch.relu(self.student_enc(student_features))
        
        move_logits = self.move_head(x)
        register_values = torch.sigmoid(self.register_head(x))
        
        return move_logits, register_values
    
    def select(self, student_state, valid_moves):
        # Encode student
        features = encode_student(student_state)
        
        # Predict
        move_logits, register = self.forward(features)
        
        # Mask invalid
        move_logits[~valid_moves] = -1e9
        
        # Select
        move_idx = torch.argmax(move_logits)
        
        return move_idx, register
