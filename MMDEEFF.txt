class DNAEncodingSpec:
    """Shared DNA encoding specification for all variants"""
    def __init__(self, config):
        self.gc_min = config.gc_min
        self.gc_max = config.gc_max
        self.max_homopolymer = config.max_homopolymer
        self.target_density = config.target_density
        self.primer_length = config.primer_length
        self.error_correction_method = config.error_correction_method
        self.interleavers = [
            lambda x: x,
            lambda x: x[::-1],
            lambda x: ''.join([x[i] for i in range(0, len(x), 2)] + [x[i] for i in range(1, len(x), 2)]),
            lambda x: ''.join([x[i] for i in range(0, len(x), 3)] + [x[i] for i in range(1, len(x), 3)] + [x[i] for i in range(2, len(x), 3)])
        ]
        logger.info(f"Initialized DNA encoding spec: GC={self.gc_min}-{self.gc_max}, max homopolymer={self.max_homopolymer}")

    def validate_sequence(self, sequence):
        if not sequence:
            return False
        if not all(c in 'ACGT' for c in sequence):
            return False
        gc_count = sum(1 for c in sequence if c in 'GC')
        gc_ratio = gc_count / len(sequence)
        if not (self.gc_min <= gc_ratio <= self.gc_max):
            return False
        prev_nt = None
        run_length = 1
        for nt in sequence:
            if nt == prev_nt:
                run_length += 1
                if run_length > self.max_homopolymer:
                    return False
            else:
                run_length = 1
            prev_nt = nt
        return True

    def fix_sequence(self, sequence):
        if not sequence:
            return ""
        sequence = ''.join(c if c in 'ACGT' else np.random.choice(['A', 'C', 'G', 'T']) for c in sequence)
        fixed_seq = ""
        prev_nt = None
        run_length = 1
        for i, nt in enumerate(sequence):
            if nt == prev_nt:
                run_length += 1
                if run_length > self.max_homopolymer:
                    replacements = [n for n in 'ACGT' if n != prev_nt]
                    nt = replacements[i % len(replacements)]
                    run_length = 1
            else:
                run_length = 1
            fixed_seq += nt
            prev_nt = nt
        if self.validate_sequence(fixed_seq):
            return fixed_seq
        for interleaver in self.interleavers:
            interleaved = interleaver(fixed_seq)
            if self.validate_sequence(interleaved):
                return interleaved
        gc_count = sum(1 for c in fixed_seq if c in 'GC')
        gc_ratio = gc_count / len(fixed_seq)
        if gc_ratio < self.gc_min:
            needed = int((self.gc_min - gc_ratio) * len(fixed_seq)) + 1
            at_positions = [i for i, c in enumerate(fixed_seq) if c in 'AT']
            if at_positions:
                to_replace = np.random.choice(at_positions, size=min(needed, len(at_positions)), replace=False)
                fixed_seq = list(fixed_seq)
                for pos in to_replace:
                    fixed_seq[pos] = np.random.choice(['G', 'C'])
                fixed_seq = ''.join(fixed_seq)
        elif gc_ratio > self.gc_max:
            needed = int((gc_ratio - self.gc_max) * len(fixed_seq)) + 1
            gc_positions = [i for i, c in enumerate(fixed_seq) if c in 'GC']
            if gc_positions:
                to_replace = np.random.choice(gc_positions, size=min(needed, len(gc_positions)), replace=False)
                fixed_seq = list(fixed_seq)
                for pos in to_replace:
                    fixed_seq[pos] = np.random.choice(['A', 'T'])
                fixed_seq = ''.join(fixed_seq)
        return fixed_seq

class Evo2DEWE(AlgorithmVariant):

    def __init__(self, config):

        self.hyper_optimizer = None
        self.metrics = {'info_density': [], 'error_correction': [], 'random_access': [], 'constraints': []}

    def initialize(self):
        self.evo_graph = EvolutionaryGraph(self.config)
        self.model = Evo2Model(d=64).to(device)
        self.meta_optimizer = optim.Adam(self.model.parameters(), lr=self.config.meta_lr)

        np.random.seed(self.config.seed)
        torch.manual_seed(self.config.seed)
        logger.info(f"Initialized Evo2-DEWE variant: {self.name}")

    def run_iteration(self, compute_allocation):
        generations = max(1, int(compute_allocation * self.config.generations / self.config.total_compute_per_iteration))
        dataset = DNADataset(self.config)
        R = torch.randn(256, 64, device=device, requires_grad=True)
        E = torch.zeros(256, 256, device=device)
        G = nx.Graph()
        G.add_nodes_from(range(256))
        task_dist = TaskDistribution(self.config)
        metrics_history = {'info_density': [], 'error_correction': [], 'random_access': [], 'constraints': []}
        fitness_scores = []

        for gen in range(generations):
            tasks = [dataset[i] for i in np.random.choice(len(dataset), size=min(len(dataset), 20), replace=False)]
            metrics_batch = []
            for i, graph in enumerate(self.evo_graph.population):
                E = torch.zeros(256, 256, device=device)
                for u, v in graph.edges():
                    E[u, v] = E[v, u] = graph[u][v].get('weight', 0.5)
                metrics = {'info_density': [], 'error_correction': [], 'random_access': [], 'constraints': []}
                for task in tasks:
                    fast_weights, task_loss = self.adapt_to_task(task, R, E, C_env, graph)
                    R_new = self.model_with_params(R, E, C_env, fast_weights)
                    info_density, error_correction, random_access, constraints = self.compute_metrics(graph, E, R_new, C_env)
                    metrics['info_density'].append(info_density)
                    metrics['error_correction'].append(error_correction)
                    metrics['random_access'].append(random_access)
                    metrics['constraints'].append(constraints)
                fitness = self._calculate_fitness(metrics)
                fitness_scores.append(fitness)
                metrics_batch.append(metrics)
            self.evo_graph.fitness_scores = fitness_scores
            self.evo_graph.evolve(lambda g: self._calculate_fitness(self._evaluate_graph_metrics(g, tasks)))
            avg_metrics = self._average_metrics(metrics_batch)
            metrics_history['info_density'].append(avg_metrics['info_density'])
            metrics_history['error_correction'].append(avg_metrics['error_correction'])
            metrics_history['random_access'].append(avg_metrics['random_access'])
            metrics_history['constraints'].append(avg_metrics['constraints'])
            R = self.model(R, E, C_env)
            E = self.update_edges(R, E, gen, self.evo_graph.best_graph)
            G.clear()
            G.add_nodes_from(range(256))
            G.add_edges_from([(i, j) for i in range(256) for j in range(i + 1, 256) if E[i, j] > self.config.connectivity_threshold])
            meta_loss = sum(task_dist.sample()[0]["weight"] * self.task_loss(task, R, self.evo_graph.best_graph) for task in tasks)
            self.meta_optimizer.zero_grad()
            if meta_loss.requires_grad:
                meta_loss.backward(retain_graph=True)
                self.meta_optimizer.step()
            self.hyper_optimizer.zero_grad()
            if meta_loss.requires_grad:
                meta_loss.backward()
                self.hyper_optimizer.step()
            C_env.add_(self.config.env_scale * torch.randn(64, device=device))
        result = {
            "fitness": self.evo_graph.best_fitness,
            "compute_used": generations,
            "metrics": self._average_metrics(metrics_batch),
            "best_graph": self.evo_graph.best_graph,
            "metrics_history": metrics_history
        }
        if result["fitness"] > self.best_fitness:
            self.best_fitness = result["fitness"]
            self.best_solution = result
        return result

    def adapt_to_task(self, task, R, E, C_env, graph):
        fast_weights = {name: param.clone() for name, param in self.model.named_parameters()}
        for _ in range(self.config.inner_steps):
            R_new = self.model_with_params(R, E, C_env, fast_weights)
            loss = self.task_loss(task, R_new, graph)
            grads = torch.autograd.grad(loss, fast_weights.values(), create_graph=True, allow_unused=True)
            fast_weights = {
                name: param - self.inner_lr * (grad if grad is not None else torch.zeros_like(param))
                for (name, param), grad in zip(fast_weights.items(), grads)
            }
        return fast_weights, loss.item()

    def model_with_params(self, R, E, C_env, params):
        state_dict = self.model.state_dict()
        for name, param in params.items():
            state_dict[name] = param
        temp_model = Evo2Model(d=64).to(device)
        temp_model.load_state_dict(state_dict)
        return temp_model(R, E, C_env)

    def task_loss(self, task, R, graph):
        task_type = task["name"]
        density = nx.density(graph)
        if task_type == "info_density":
            return torch.tensor(2.0 - density * 2.0, device=device)
        elif task_type == "error_correction":
            return torch.tensor(0.1 + (1 - density) * 0.1, device=device)
        elif task_type == "random_access":
            return torch.tensor(0.2 + density * 0.1, device=device)
        else:
            return torch.var(R)

    def update_edges(self, R, E, t, graph):
        E_np = E.cpu().numpy()
        freqs = np.array(self.config.resonance_frequencies)
        r = np.max([np.cos(2 * np.pi * f * t) for f in freqs], axis=0)
        for u, v in graph.edges():
            sim = 1 - cosine(R[u].cpu().detach().numpy(), R[v].cpu().detach().numpy())
            noise = self.config.mutation_rate * np.random.randn()
            E_np[u, v] = E_np[v, u] = 1 / (1 + np.exp(-(self.config.alpha * sim * (1 + r) - self.config.lambda_ * E_np[u, v] + noise)))
        return torch.tensor(E_np, device=device)

    def compute_metrics(self, G, E, R, C_env):
        G_nx = nx.from_numpy_array(E.cpu().numpy() > self.config.connectivity_threshold)
        info_density = nx.density(G_nx) * 2.0
        error_correction = 1 - nx.density(G_nx) * 0.1
        random_access = nx.density(G_nx) * 0.1
        constraints = torch.var(R).item()
        return info_density, error_correction, random_access, constraints

    def get_metrics(self):
        if self.best_solution and "metrics" in self.best_solution:
            return self.best_solution["metrics"]
        return {}

    def save_checkpoint(self, path):

    def _calculate_fitness(self, metrics):
        info_density = np.mean(metrics['info_density'])
        error_correction = np.mean(metrics['error_correction'])
        random_access = np.mean(metrics['random_access'])
        constraints = np.mean(metrics['constraints'])
        norm_density = min(1.0, info_density / 2.0)
        norm_error = 1.0 - min(1.0, error_correction / 0.1)
        norm_access = 1.0 - min(1.0, random_access / 0.1)
        norm_constraints = 1.0 - min(1.0, constraints / 0.1)
        return (
            self.config.info_density_weight * norm_density +
            self.config.error_correction_weight * norm_error +
            self.config.random_access_weight * norm_access +
            self.config.constraints_weight * norm_constraints
        )

    def _evaluate_graph_metrics(self, graph, tasks):
        connectivity = nx.density(graph)
        info_density = connectivity * 2.0
        error_correction = 1 - connectivity * 0.1
        random_access = connectivity * 0.1
        constraints = 0.1
        return {
            "info_density": [info_density] * len(tasks),
            "error_correction": [error_correction] * len(tasks),
            "random_access": [random_access] * len(tasks),
            "constraints": [constraints] * len(tasks),
            "task_losses": [0.5] * len(tasks)
        }

    def _average_metrics(self, metrics_batch):
        return {
            "info_density": np.mean([np.mean(m['info_density']) for m in metrics_batch]),
            "error_correction": np.mean([np.mean(m['error_correction']) for m in metrics_batch]),
            "random_access": np.mean([np.mean(m['random_access']) for m in metrics_batch]),
            "constraints": np.mean([np.mean(m['constraints']) for m in metrics_batch])
        }

class Evo2Model(nn.Module):
    """Model mimicking Evo2's StripedHyena 2 for DNA encoding"""
    def __init__(self, d):
        super().__init__()
        self.conv = nn.Conv1d(d, d, kernel_size=3, padding=1)
        self.attention = nn.MultiheadAttention(d, num_heads=4)
        self.similarity = nn.Linear(d, d)
        self.activation = nn.Tanh()

    def forward(self, R, E, C_env):
        R_conv = R.unsqueeze(0).transpose(1, 2)
        R_conv = self.activation(self.conv(R_conv)).transpose(1, 2).squeeze(0)
        R_attn, _ = self.attention(R_conv.unsqueeze(0), R_conv.unsqueeze(0), R_conv.unsqueeze(0))
        R_attn = R_attn.squeeze(0)
        neighbor_sum = torch.einsum('ij,jk->ik', E, R_attn)
        dR_dt = -R_attn + self.activation(R_attn + neighbor_sum + C_env)
        R_new = R_attn + dR_dt * 0.01
        R_sim = self.similarity(R_new)
        return R_sim

class EvolutionaryGraph:
    def __init__(self, config):
        self.config = config
        self.population = []
        self.fitness_scores = []
        self.best_graph = None
        self.best_fitness = -float('inf')
        self._initialize_population()

    def _initialize_population(self):
        for _ in range(self.config.population_size):
            graph = nx.Graph()
            for i in range(256):
                graph.add_node(i, type="parameter", index=i)
            for i in range(256):
                for j in range(i + 1, min(i + 5, 256)):
                    if np.random.random() < 0.3:
                        graph.add_edge(i, j, weight=np.random.uniform(0.1, 0.5))
            self._mutate_graph(graph, rate=self.config.mutation_rate * 2)
            self.population.append(graph)
            self.fitness_scores.append(0.0)

    def evolve(self, fitness_function):
        for i, graph in enumerate(self.population):
            self.fitness_scores[i] = fitness_function(graph)
            if self.fitness_scores[i] > self.best_fitness:
                self.best_fitness = self.fitness_scores[i]
                self.best_graph = graph.copy()
        new_population = []
        for _ in range(self.config.population_size):
            parent1 = self._tournament_selection()
            parent2 = self._tournament_selection()
            child = self._crossover(self.population[parent1], self.population[parent2]) if np.random.random() < self.config.crossover_rate else self.population[parent1].copy()
            self._mutate_graph(child, rate=self.config.mutation_rate)
            new_population.append(child)
        self.population = new_population

    def _tournament_selection(self, tournament_size=3):
        indices = np.random.choice(range(len(self.population)), size=tournament_size, replace=False)
        tournament_fitness = [self.fitness_scores[i] for i in indices]
        return indices[np.argmax(tournament_fitness)]

    def _crossover(self, graph1, graph2):
        child = nx.Graph()
        for node, data in graph1.nodes(data=True):
            child.add_node(node, **data)
        for node, data in graph2.nodes(data=True):
            if node not in child:
                child.add_node(node, **data)
        all_edges = set(graph1.edges()) | set(graph2.edges())
        for u, v in all_edges:
            if u not in child or v not in child:
                continue
            include_prob = weight = 0.0
            if graph1.has_edge(u, v):
                include_prob += 0.5
                weight += 0.5 * graph1[u][v].get('weight', 0.5)
            if graph2.has_edge(u, v):
                include_prob += 0.5
                weight += 0.5 * graph2[u][v].get('weight', 0.5)
            if np.random.random() < include_prob:
                child.add_edge(u, v, weight=weight)
        return child

    def _mutate_graph(self, graph, rate=0.05):
        for u, v, data in list(graph.edges(data=True)):
            if np.random.random() < rate:
                delta = np.random.normal(0, 0.1)
                new_weight = max(0.1, min(1.0, data.get('weight', 0.5) + delta))
                graph[u][v]['weight'] = new_weight
        nodes = list(graph.nodes())
        for _ in range(int(rate * len(nodes))):
            u = np.random.choice(nodes)
            v = np.random.choice(nodes)
            if u != v and not graph.has_edge(u, v):
                graph.add_edge(u, v, weight=np.random.uniform(0.1, 0.5))
        for u, v in list(graph.edges()):
            if np.random.random() < rate * 0.5:
                graph.remove_edge(u, v)

        return len(self.data)

    def __getitem__(self, idx):
        return self.data[idx]

            {"name": "random_access", "weight": torch.tensor(0.2, requires_grad=True, device=device)},
            {"name": "constraints", "weight": torch.tensor(0.1, requires_grad=True, device=device)}
        ]
        self.hyper_optimizer = optim.Adam([task["weight"] for task in self.tasks], lr=config.hyper_lr)

    def sample(self, num_tasks=1):
        weights = torch.softmax(torch.stack([task["weight"] for task in self.tasks]), dim=0)
        return np.random.choice(self.tasks, size=num_tasks, replace=True, p=weights.cpu().detach().numpy())

class MetricStandardization:
    """Standardizes and normalizes metrics across variants"""
    def __init__(self, config):
        self.config = config
        self.metric_ranges = {}
        self.known_metrics = set()
        self.metric_directions = config.metric_directions
        logger.info("Initialized metric standardization")

    def update_metric_ranges(self, all_metrics):
        metric_values = {}
        for variant_metrics in all_metrics.values():
            for metric_dict in variant_metrics:
                for name, value in metric_dict.items():
                    if isinstance(value, (int, float)):
                        metric_values.setdefault(name, []).append(value)
        for name, values in metric_values.items():
            if values:
                min_val = min(values)
                max_val = max(values)
                self.metric_ranges[name] = {"min": min_val, "max": max_val}
                self.known_metrics.add(name)

    def standardize_metrics(self, metrics_dict):
        standardized = {}
        for name, value in metrics_dict.items():
            if not isinstance(value, (int, float)):
                continue
            if name in self.metric_ranges:
                min_val = self.metric_ranges[name]["min"]
                max_val = self.metric_ranges[name]["max"]
                if max_val > min_val:
                    norm_value = (value - min_val) / (max_val - min_val)
                    if name in self.metric_directions and not self.metric_directions[name]:
                        norm_value = 1.0 - norm_value
                    standardized[name] = norm_value
                else:
                    standardized[name] = 0.5
            else:
                standardized[name] = value
        return standardized

    def calculate_fitness(self, standardized_metrics):
        fitness = 0.0
        total_weight = 0.0
        for name, value in standardized_metrics.items():
            if not isinstance(value, (int, float)):
                continue
            weight = self.config.metric_weights.get(name, 1.0)
            fitness += weight * value
            total_weight += weight
        if total_weight > 0:
            fitness /= total_weight
        return fitness

class AdaptiveSelectionPolicy:
    """Selects variants to run in each iteration"""
    def __init__(self, config):
        self.config = config
        self.exploration_rate = config.exploration_rate_init
        self.min_selection = max(1, int(config.min_selection_fraction * config.meta_population_size))
        self.max_selection = max(self.min_selection, int(config.max_selection_fraction * config.meta_population_size))
        self.last_best_fitness = -float('inf')
        self.stagnation_counter = 0
        logger.info(f"Initialized selection policy with exploration rate {self.exploration_rate}")

    def select(self, variants, performance_data):
        if not variants:
            return []
        n_select = min(self.max_selection, len(variants))
        current_best_fitness = max(
            (max(perf["fitness"]) if perf["fitness"] else -float('inf'))
            for perf in performance_data.values()
        )
        if current_best_fitness <= self.last_best_fitness:
            self.stagnation_counter += 1
        else:
            self.stagnation_counter = 0
            self.last_best_fitness = current_best_fitness
        if self.stagnation_counter >= self.config.stagnation_threshold:
            self.exploration_rate = min(0.9, self.exploration_rate * 1.5)
            self.stagnation_counter = 0
            logger.info(f"Stagnation detected, exploration rate: {self.exploration_rate}")
        if np.random.random() < self.exploration_rate:
            selected_indices = np.random.choice(len(variants), size=n_select, replace=False)
            selected = [variants[i] for i in selected_indices]
            logger.info(f"Exploration: selected {len(selected)} variants")
        else:
            scores = []
            for variant in variants:
                perf = performance_data[variant.name]
                if not perf["fitness"]:
                    scores.append(0.0)
                else:
                    recent_fitness = perf["fitness"][-self.config.lookback_window:]
                    recent_compute = perf["compute"][-self.config.lookback_window:]
                    efficiency = [f/(c+1e-10) for f, c in zip(recent_fitness, recent_compute)]
                    weights = [0.5 + 0.5 * i/len(recent_fitness) for i in range(len(recent_fitness))]
                    weighted_efficiency = sum(w*e for w, e in zip(weights, efficiency)) / sum(weights)
                    scores.append(weighted_efficiency)
            indices = np.argsort(scores)[::-1]
            top_k = max(1, n_select // 2)
            selected_indices = list(indices[:top_k])
            if n_select > top_k and len(indices) > top_k:
                remaining = list(indices[top_k:])
                additional = np.random.choice(remaining, size=min(n_select - top_k, len(remaining)), replace=False)
                selected_indices.extend(additional)
            selected = [variants[i] for i in selected_indices]
            logger.info(f"Exploitation: selected {len(selected)} variants")
        self.exploration_rate *= self.config.exploration_decay
        if len(selected) < self.min_selection:
            remaining = [v for v in variants if v not in selected]
            if remaining:
                additional = np.random.choice(remaining, size=min(self.min_selection - len(selected), len(remaining)), replace=False)
                selected.extend(additional)
        return selected

class ResourceAllocator:
    """Allocates resources to selected variants"""
    def __init__(self, config):
        self.config = config
        self.allocation_history = {}
        self.performance_history = {}
        logger.info("Initialized resource allocator")

    def allocate(self, selected_variants, performance_data):
        if not selected_variants:
            return []
        total_compute = self.config.total_compute_per_iteration
        scores = []
        names = []
        for variant in selected_variants:
            name = variant.name
            perf = performance_data[name]
            names.append(name)
            if not perf["fitness"]:
                scores.append(1.0)
                continue
            window = min(self.config.lookback_window, len(perf["fitness"]))
            recent_fitness = perf["fitness"][-window:]
            recent_compute = perf["compute"][-window:]
            avg_fitness = sum(recent_fitness) / window
            if window > 1:
                x = np.arange(window)
                trend = np.linalg.lstsq(x.reshape(-1, 1), recent_fitness, rcond=None)[0][0]
            else:
                trend = 0
            efficiency = sum(recent_fitness) / (sum(recent_compute) + 1e-10)
            score = (
                self.config.weight_fitness * avg_fitness +
                self.config.weight_trend * max(0, trend * 10) +
                self.config.weight_efficiency * efficiency
            )
            scores.append(max(0.1, score))
        total_score = sum(scores) + 1e-10
        norm_scores = [s / total_score for s in scores]
        min_per_variant = total_compute * self.config.min_allocation_fraction / len(selected_variants)
        remaining = total_compute - min_per_variant * len(selected_variants)
        allocations = [min_per_variant + ns * remaining for ns in norm_scores]
        for name, allocation in zip(names, allocations):
            self.allocation_history.setdefault(name, []).append(allocation)
            logger.info(f"Allocated {allocation:.2f} to {name}")
        return allocations

class ConvergenceDetector:
    """Detects convergence of the meta-meta process"""
    def __init__(self, config):
        self.config = config
        self.best_fitness_history = []
        self.iterations_without_improvement = 0
        logger.info(f"Convergence detector: threshold={self.config.convergence_threshold}, patience={self.config.convergence_patience}")

    def has_converged(self, performance_data):
        current_best = -float('inf')
        best_variant = None
        for name, perf in performance_data.items():
            if perf["fitness"]:
                max_fitness = max(perf["fitness"])
                if max_fitness > current_best:
                    current_best = max_fitness
                    best_variant = name
        if current_best == -float('inf'):
            return False
        self.best_fitness_history.append(current_best)
        if len(self.best_fitness_history) < self.config.patience:
            return False
        previous_best = max(self.best_fitness_history[:-1])
        if current_best > previous_best + self.config.convergence_threshold:
            self.iterations_without_improvement = 0
            logger.info(f"Improvement: {current_best:.6f} > {previous_best:.6f} by {best_variant}")
        else:
            self.iterations_without_improvement += 1
            logger.info(f"No improvement for {self.iterations_without_improvement} iterations")
        return self.iterations_without_improvement >= self.config.convergence_patience

class MetaMetaEvolution:
    """Meta-meta evolutionary framework for DNA encoding algorithms"""
    def __init__(self, config):
        self.config = config
        self.algorithm_variants = []
        self.variant_performance = {}
        self.selection_policy = AdaptiveSelectionPolicy(config)
        self.resource_allocator = ResourceAllocator(config)
        self.convergence_detector = ConvergenceDetector(config)
        self.metric_standardizer = MetricStandardization(config)
        self.iteration = 0
        logging.basicConfig(
            level=getattr(logging, config.log_level),
            format='%(asctime)s - %(levelname)s - %(message)s',
            handlers=[
                logging.FileHandler(os.path.join(config.output_dir, "meta_evolution.log")),
                logging.StreamHandler()
            ]
        )
        self.logger = logging.getLogger("MetaMetaEvolution")

    def register_variant(self, variant_class, variant_config=None, name=None):
        if variant_config is None:
            variant_config = self.config.variant_configs.get(variant_class.__name__, {})
        for param in ["gc_min", "gc_max", "max_homopolymer", "target_density", "primer_length", "error_correction_method"]:
            variant_config[param] = getattr(self.config, param)
        variant = variant_class(variant_config)
        variant.name = name or f"{variant_class.__name__}_{len(self.algorithm_variants)}"
        variant.initialize()
        self.algorithm_variants.append(variant)
        self.variant_performance[variant.name] = {
            "fitness": [], "compute": [], "metrics": [], "selected": 0
        }
        self.logger.info(f"Registered variant: {variant.name}")

    def run(self, max_iterations=None):
        if max_iterations is None:
            max_iterations = self.config.max_iterations
        self.logger.info(f"Starting meta-meta evolution with {len(self.algorithm_variants)} variants")
        for iteration in range(max_iterations):
            self.iteration = iteration
            self.logger.info(f"Iteration {iteration+1}/{max_iterations}")
            selected_variants = self.selection_policy.select(self.algorithm_variants, self.variant_performance)
            self.logger.info(f"Selected variants: {[v.name for v in selected_variants]}")
            allocations = self.resource_allocator.allocate(selected_variants, self.variant_performance)
            results = {}
            for variant, allocation in zip(selected_variants, allocations):
                self.logger.info(f"Running {variant.name} with allocation {allocation:.2f}")
                self.variant_performance[variant.name]["selected"] += 1
                result = variant.run_iteration(allocation)
                results[variant.name] = result
                self.logger.info(f"{variant.name} fitness: {result['fitness']:.4f}")
            for name, result in results.items():
                self.variant_performance[name]["fitness"].append(result["fitness"])
                self.variant_performance[name]["compute"].append(result["compute_used"])
                self.variant_performance[name]["metrics"].append(result.get("metrics", {}))
            self.metric_standardizer.update_metric_ranges({name: [r["metrics"]] for name, r in results.items()})
            if iteration % 3 == 0:
                self._evolve_variants()
            if iteration % self.config.checkpoint_frequency == 0:
                self._save_checkpoint()
            if self.convergence_detector.has_converged(self.variant_performance):
                self.logger.info(f"Convergence at iteration {iteration+1}")
                break
        self._save_checkpoint()
        best_result = self._get_best_result()
        self.logger.info(f"Completed. Best: {best_result['name']} fitness {best_result['fitness']:.4f}")
        if self.config.generate_analysis:
            self.analyze_results()
        return best_result

    def _evolve_variants(self):
        self.logger.info("Evolving variants...")
        variant_fitness = []
        for variant in self.algorithm_variants:
            perf = self.variant_performance[variant.name]
            if perf["fitness"]:
                fitness = np.mean(perf["fitness"][-3:])
                variant_fitness.append((variant, fitness))
        if len(variant_fitness) < 2:
            self.logger.info("Not enough variants for evolution")
            return
        variant_fitness.sort(key=lambda x: x[1], reverse=True)
        top_variants = variant_fitness[:len(variant_fitness)//2]
        for i in range(min(2, len(top_variants))):
            idx1, idx2 = np.random.choice(len(top_variants), 2, replace=False)
            parent1, _ = top_variants[idx1]
            parent2, _ = top_variants[idx2]
            if np.random.random() < self.config.meta_crossover_rate:
                new_variant = self._crossover_variants(parent1, parent2)
                name = f"Cross_{parent1.name}_{parent2.name}_{self.iteration}"
                self.register_variant(new_variant.__class__, new_variant.get_config(), name)
        for i in range(min(1, len(top_variants))):
            idx = np.random.choice(len(top_variants))
            parent, _ = top_variants[idx]
            if np.random.random() < self.config.meta_mutation_rate:
                new_variant = self._mutate_variant(parent)
                name = f"Mutate_{parent.name}_{self.iteration}"
                self.register_variant(new_variant.__class__, new_variant.get_config(), name)

    def _crossover_variants(self, parent1, parent2):
        config1 = parent1.get_config()
        config2 = parent2.get_config()
        new_config = {}
        common_params = set(config1.keys()).intersection(set(config2.keys()))
        for param in common_params:
            value1 = config1[param]
            value2 = config2[param]
            if isinstance(value1, (int, float)) and isinstance(value2, (int, float)):
                crossover_type = np.random.choice(["uniform", "blend"])
                if crossover_type == "uniform":
                    new_config[param] = value1 if np.random.random() < 0.5 else value2
                else:
                    alpha = np.random.uniform(-0.3, 1.3)
                    new_value = value1 + alpha * (value2 - value1)
                    if isinstance(value1, int) and isinstance(value2, int):
                        new_value = int(round(new_value))
                    new_config[param] = new_value
            else:
                new_config[param] = value1 if np.random.random() < 0.5 else value2
        for param in set(config1.keys()) - common_params:
            if np.random.random() < 0.5:
                new_config[param] = config1[param]
        for param in set(config2.keys()) - common_params:
            if np.random.random() < 0.5:
                new_config[param] = config2[param]
        variant_class = parent1.__class__ if np.random.random() < 0.5 else parent2.__class__
        new_variant = variant_class(new_config)
        new_variant.name = f"Cross_{parent1.name}_{parent2.name}"
        self.logger.info(f"Created variant {new_variant.name} via crossover")
        return new_variant

    def _mutate_variant(self, parent):
        config = copy.deepcopy(parent.get_config())
        for key, value in config.items():
            if isinstance(value, (int, float)):
                if isinstance(value, int):
                    if value != 0:
                        perturbation = int(value * np.random.uniform(-0.2, 0.2))
                        config[key] = max(1, value + perturbation)
                    else:
                        config[key] = np.random.choice([-1, 0, 1])
                else:
                    perturbation = value * np.random.uniform(-0.2, 0.2)
                    config[key] = value + perturbation
        new_variant = parent.__class__(config)
        new_variant.name = f"Mutate_{parent.name}"
        self.logger.info(f"Created variant {new_variant.name} via mutation")
        return new_variant

    def _save_checkpoint(self):
        checkpoint_dir = os.path.join(self.config.output_dir, f"checkpoint_{self.iteration}")
        os.makedirs(checkpoint_dir, exist_ok=True)
        meta_state = {
            "iteration": self.iteration,
            "variant_performance": {
                name: {
                    "fitness": perf["fitness"],
                    "compute": perf["compute"],
                    "metrics": perf["metrics"],
                    "selected": perf["selected"]
                } for name, perf in self.variant_performance.items()
            },
            "exploration_rate": self.selection_policy.exploration_rate
        }
        with open(os.path.join(checkpoint_dir, "meta_state.json"), "w") as f:
            json.dump(meta_state, f, indent=2)
        for variant in self.algorithm_variants:
            variant_path = os.path.join(checkpoint_dir, f"{variant.name}.pt")
            variant.save_checkpoint(variant_path)
        self.logger.info(f"Saved checkpoint to {checkpoint_dir}")

    def _get_best_result(self):
        best_name = None
        best_fitness = -float('inf')
        best_solution = None
        for variant in self.algorithm_variants:
            solution = variant.get_solution()
            fitness = variant.get_fitness()
            if fitness > best_fitness:
                best_fitness = fitness
                best_solution = solution
                best_name = variant.name
        return {
            "name": best_name,
            "fitness": best_fitness,
            "solution": best_solution
        }

    def analyze_results(self):
        meta_state_path = os.path.join(self.config.output_dir, f"checkpoint_{self.iteration}", "meta_state.json")
        if not os.path.exists(meta_state_path):
            self.logger.warning("No final checkpoint for analysis")
            return
        with open(meta_state_path, "r") as f:
            meta_state = json.load(f)
        variant_performance = meta_state["variant_performance"]
        plt.figure(figsize=(12, 8))
        plt.subplot(2, 2, 1)
        for variant, perf in variant_performance.items():
            if perf["fitness"]:
                plt.plot(perf["fitness"], label=variant)
        plt.title("Fitness Evolution")
        plt.xlabel("Iteration")
        plt.ylabel("Fitness")
        plt.legend()
        plt.subplot(2, 2, 2)
        variants = list(variant_performance.keys())
        selections = [perf["selected"] for perf in variant_performance.values()]
        plt.bar(variants, selections)
        plt.title("Selection Frequency")
        plt.xlabel("Variant")
        plt.ylabel("Times Selected")
        plt.xticks(rotation=45)
        plt.subplot(2, 2, 3)
        for variant, perf in variant_performance.items():
            if perf["fitness"] and perf["compute"]:
                efficiency = [f/c for f, c in zip(perf["fitness"], perf["compute"])]
                plt.plot(efficiency, label=variant)
        plt.title("Compute Efficiency")
        plt.xlabel("Iteration")
        plt.ylabel("Fitness / Compute")
        plt.legend()
        best_variant = max(variant_performance.items(), 
                          key=lambda x: max(x[1]["fitness"]) if x[1]["fitness"] else -float('inf'))[0]
        plt.subplot(2, 2, 4)
        metrics = variant_performance[best_variant]["metrics"]
        if metrics:
            for metric in set().union(*[m.keys() for m in metrics]):
                values = [m.get(metric, 0) for m in metrics]
                plt.plot(values, label=metric)
            plt.title(f"Metrics for {best_variant}")
            plt.xlabel("Iteration")
            plt.ylabel("Value")
            plt.legend()
        plt.tight_layout()
        plt.savefig(os.path.join(self.config.output_dir, "performance_analysis.png"))
        plt.close()
        self.logger.info(f"Analysis saved to {self.config.output_dir}/performance_analysis.png")
