pythonimport numpy as np
from sklearn.preprocessing import MultiLabelBinarizer

class StudentStateEncoder:
    """Convert symbolic student state to vector"""
    
    def __init__(self, training_data):
        # Learn vocabularies from data
        self.behavior_encoder = MultiLabelBinarizer()
        self.trap_encoder = MultiLabelBinarizer()
        self.reading_encoder = LabelEncoder()
        
        # Fit on all training data
        all_behaviors = [t.student_turn.behavior_tags for t in training_data]
        all_traps = [t.student_turn.inferred_state.active_traps for t in training_data]
        all_readings = [t.student_turn.inferred_state.reading for t in training_data]
        
        self.behavior_encoder.fit(all_behaviors)
        self.trap_encoder.fit(all_traps)
        self.reading_encoder.fit(all_readings)
    
    def encode(self, student_turn):
        """student_turn → vector"""
        
        # Multi-hot for behaviors
        behaviors = self.behavior_encoder.transform([student_turn.behavior_tags])
        # Shape: (1, n_behaviors)
        
        # Multi-hot for traps
        traps = self.trap_encoder.transform([student_turn.inferred_state.active_traps])
        # Shape: (1, n_traps)
        
        # One-hot for reading
        reading = self.reading_encoder.transform([student_turn.inferred_state.reading])
        # Shape: (1,)
        reading_onehot = np.zeros(len(self.reading_encoder.classes_))
        reading_onehot[reading[0]] = 1
        
        # Concatenate
        vector = np.concatenate([
            behaviors.flatten(),
            traps.flatten(),
            reading_onehot
        ])
        
        return vector

pythonclass MoveEncoder:
    """Convert symbolic move to vector"""
    
    PHASE_MAP = {"UNMAKING": 0, "REMAKING": 1, "SELF-MAKING": 2, "META": 3}
    TAG_MAP = {"MOVE": 0, "BRIDGE": 1, "REVEAL": 2, "PARADOX": 3, "BOSS": 4, "SILENCE": 5}
    
    def __init__(self, training_data):
        # Learn vocabularies
        self.function_encoder = LabelEncoder()
        self.mechanism_encoder = LabelEncoder()
        self.trap_encoder = MultiLabelBinarizer()
        self.register_encoder = RegisterEncoder()  # Custom for register strings
        
        # Fit
        all_functions = [t.teacher_turn.function_id for t in training_data]
        all_mechanisms = [t.teacher_turn.mechanism_shape for t in training_data]
        all_traps = [t.teacher_turn.trap_ids for t in training_data]
        
        self.function_encoder.fit(all_functions)
        self.mechanism_encoder.fit(all_mechanisms)
        self.trap_encoder.fit(all_traps)
    
    def encode(self, teacher_turn):
        """teacher_turn → vector"""
        
        # Phase (one-hot)
        phase = np.zeros(4)
        phase[self.PHASE_MAP[teacher_turn.phase]] = 1
        
        # Function (one-hot)
        function = self.function_encoder.transform([teacher_turn.function_id])
        function_onehot = np.zeros(len(self.function_encoder.classes_))
        function_onehot[function[0]] = 1
        
        # Mechanism (one-hot)
        mechanism = self.mechanism_encoder.transform([teacher_turn.mechanism_shape])
        mechanism_onehot = np.zeros(len(self.mechanism_encoder.classes_))
        mechanism_onehot[mechanism[0]] = 1
        
        # Move tag (one-hot)
        move_tag = np.zeros(6)
        move_tag[self.TAG_MAP[teacher_turn.move_tag]] = 1
        
        # Target traps (multi-hot)
        traps = self.trap_encoder.transform([teacher_turn.trap_ids])
        
        # Register (numeric)
        register = self.register_encoder.encode(teacher_turn.register)
        # Returns [intensity, intimacy, linguistic_style, phenom_depth, meta_mode, attunement]
        
        # Outcome
        outcome = self.encode_outcome(teacher_turn.outcome)
        
        # Concatenate
        vector = np.concatenate([
            phase,
            function_onehot,
            mechanism_onehot,
            move_tag,
            traps.flatten(),
            register,
            outcome
        ])
        
        return vector

class RegisterEncoder:
    """Convert register string to numeric vector"""
    
    INTENSITY_MAP = {"PR_01": 0.2, "PR_02": 0.4, "PR_03": 0.6, "PR_04": 0.8, "PR_05": 1.0}
    INTIMACY_MAP = {"IN_01": 0.2, "IN_02": 0.4, "IN_03": 0.6, "IN_04": 0.8, "IN_05": 1.0}
    # ... etc for all register axes
    
    def encode(self, register):
        """register dict → [6] float vector"""
        return np.array([
            self.INTENSITY_MAP.get(register['intensity'], 0.5),
            self.INTIMACY_MAP.get(register['intimacy'], 0.5),
            self.parse_linguistic_style(register['linguistic_style']),
            self.parse_phenom_depth(register['phenomenological_depth']),
            self.parse_meta_mode(register['meta_mode']),
            self.parse_attunement(register['attunement'])
        ])

pythonclass ShapeEncoder:
    """Convert episode/shape to trajectory vector"""
    
    def encode_episode(self, episode):
        """Episode → shape vector"""
        
        # Extract start/end states
        s_start = student_encoder.encode(episode.turns[0].student_turn)
        s_end = student_encoder.encode(episode.turns[-1].student_turn)
        
        # Phase histogram
        phases = [t.teacher_turn.phase for t in episode.turns]
        phase_hist = np.array([
            phases.count("UNMAKING") / len(phases),
            phases.count("REMAKING") / len(phases),
            phases.count("SELF-MAKING") / len(phases),
            phases.count("META") / len(phases)
        ])
        
        # Tag histogram
        tags = [t.teacher_turn.move_tag for t in episode.turns]
        tag_hist = np.array([
            tags.count("MOVE") / len(tags),
            tags.count("BRIDGE") / len(tags),
            tags.count("REVEAL") / len(tags),
            tags.count("PARADOX") / len(tags),
            tags.count("BOSS") / len(tags),
            tags.count("SILENCE") / len(tags)
        ])
        
        # Average intensity
        intensities = [self.parse_intensity(t.teacher_turn.register['intensity']) 
                      for t in episode.turns]
        avg_intensity = np.mean(intensities)
        
        # Trap distribution
        all_traps = []
        for t in episode.turns:
            all_traps.extend(t.teacher_turn.trap_ids)
        trap_hist = self.compute_trap_histogram(all_traps)
        
        # Success rate (from outcomes)
        outcomes = [t.teacher_turn.outcome.outcome_label for t in episode.turns]
        success_rate = outcomes.count("clean_success") / len(outcomes)
        
        # Concatenate
        z_shape = np.concatenate([
            s_start,
            s_end,
            phase_hist,
            tag_hist,
            [avg_intensity],
            trap_hist,
            [success_rate]
        ])
        
        return z_shape

📈 STEP 6: COMPUTE ABILITY SCORES
pythondef compute_ability_score(student_cluster_id, episodes):
    """Compute ability score for a student cluster from outcomes"""
    
    cluster_episodes = [ep for ep in episodes 
                       if ep.metadata['student_cluster'] == student_cluster_id]
    
    # Metric 1: Success rate
    outcomes = []
    for ep in cluster_episodes:
        final_outcome = ep.turns[-1].teacher_turn.outcome.outcome_label
        outcomes.append(final_outcome)
    
    p_clean_success = outcomes.count("clean_success") / len(outcomes)
    
    # Metric 2: Steps to shift
    steps_to_shift = []
    for ep in cluster_episodes:
        # Find first "admits_ignorance" or "turns_inward" reaction
        for i, turn in enumerate(ep.turns):
            if turn.teacher_turn.outcome.reaction_label in ["admits_ignorance", "turns_inward"]:
                steps_to_shift.append(i)
                break
    
    avg_steps = np.mean(steps_to_shift) if steps_to_shift else 20
    normalized_steps = 1.0 - min(avg_steps / 20, 1.0)  # Fewer steps = higher ability
    
    # Metric 3: Shape complexity (can they handle long/intense arcs?)
    avg_shape_length = np.mean([len(ep.turns) for ep in cluster_episodes])
    normalized_complexity = min(avg_shape_length / 15, 1.0)
    
    # Metric 4: Intensity tolerance
    avg_intensity = []
    for ep in cluster_episodes:
        for turn in ep.turns:
            intensity = parse_intensity(turn.teacher_turn.register['intensity'])
            avg_intensity.append(intensity)
    
    intensity_tolerance = np.mean(avg_intensity)
    
    # Combine (weighted average)
    ability_score = (
        0.4 * p_clean_success +
        0.3 * normalized_steps +
        0.2 * normalized_complexity +
        0.1 * intensity_tolerance
    )
    
    return {
        "ability_score": float(ability_score),
        "p_clean_success": float(p_clean_success),
        "avg_steps_to_shift": float(avg_steps),
        "avg_shape_length": float(avg_shape_length),
        "intensity_tolerance": float(intensity_tolerance)
    }

🎯 STEP 7: RUNTIME DECISION MAKING
pythonclass GeometricDecisionEngine:
    """Use clusters + ability scores to make teaching decisions"""
    
    def __init__(self, student_encoder, move_encoder, 
                 student_clusters, move_clusters, shape_clusters,
                 student_meta, move_meta, ability_profiles):
        self.student_encoder = student_encoder
        self.move_encoder = move_encoder
        self.student_clusters = student_clusters
        self.move_clusters = move_clusters
        self.student_meta = student_meta
        self.move_meta = move_meta
        self.ability_profiles = ability_profiles
    
    def classify_student(self, student_turn):
        """Current student → cluster ID + ability score"""
        
        # Encode
        s_vector = self.student_encoder.encode(student_turn)
        
        # Find nearest cluster
        distances = [np.linalg.norm(s_vector - center) 
                    for center in self.student_clusters.cluster_centers_]
        cluster_id = int(np.argmin(distances))
        
        # Look up ability
        ability = self.ability_profiles[cluster_id]['ability_score']
        
        return cluster_id, ability
    
    def select_move(self, student_state, context, ability_score):
        """Student state + ability → best move"""
        
        # Get valid moves (symbolic filtering)
        valid_moves = self.get_valid_moves(
            phase=context.phase,
            progress=context.progress,
            student_traps=student_state.traps
        )
        
        # Modulate by ability
        if ability_score < 0.4:
            # Low ability: prefer gentle moves
            valid_moves = [m for m in valid_moves 
                          if m.move_tag in ["MOVE", "BRIDGE"]]
            intensity_cap = 0.6
        elif ability_score < 0.7:
            # Medium ability: allow reveals
            intensity_cap = 0.8
        else:
            # High ability: all moves available
            intensity_cap = 1.0
        
        # Encode valid moves
        move_vectors = [self.move_encoder.encode(m) for m in valid_moves]
        
        # Score based on historical success in similar contexts
        scores = []
        for move in valid_moves:
            # Find similar past uses
            similar_contexts = self.find_similar_contexts(student_state, context)
            success_rate = self.compute_success_rate(move, similar_contexts)
            scores.append(success_rate)
        
        # Select best
        best_idx = np.argmax(scores)
        return valid_moves[best_idx]

This approach is architecturally sound because:

✅ No fake confidence: Ability scores from actual outcomes
✅ Interpretable: Clusters have human-readable names
✅ Efficient: Nearest-neighbor lookup, not recomputing
✅ Modular: Slots into existing schema
✅ Data-driven: Everything learned from Pass-1 annotations

Build this AFTER Phase 1 LoRA trains. It's the intelligence layer for Phase 2/3.
Ready to implement? ah, this is the moment where you start classifying and quantizing. Up to now you were building the ontology + rich annotations. Now you can compress it into:

a few vector spaces (student states, moves, arcs), and

some coarse labels (ability, arc type, move flavor).

I’ll lay out a concrete plan that uses both numeric stuff and LLMs, and doesn’t fall back into fake “confidence” nonsense.

1. What you already have (perfect for vectors)

For each Pass-1 episode, you effectively have:

Student state at each turn:

behavior_tags

trap_ids that are active

inferred_state.reading (semantic description)

Teacher move at each turn:

phase (UM / RM / SM / META)

function_id

mechanism_shape

trap_ids it’s targeting

move_tag (MOVE / BRIDGE / REVEAL / BOSS / SILENCE)

register (intensity, intimacy, etc.)

Outcome:

reaction_label (admits_ignorance, complains_about_method, turns_inward, etc.)

outcome_label (clean_success, partial_success, shock, etc.)

phase_shift.from/to

This is already a vector space. You just haven’t flattened it yet.

2. Step 1: Build three vector spaces
2.1 Student state vectors

For each student_turn (especially the first and last in an episode):

Turn tags into a vector:

Multi-hot / embedding for behavior_tags

e.g. conceptual_inquiry, utilitarian_seeking, identity_seeking, etc.

Multi-hot for active trap_ids

naive_realism, external_locus, identity_seeking, etc.

Categorical embedding for inferred_state.reading

e.g. “seeking_functional_difference”, “demand_for_positive_definition”.

You don’t need exact dimensions yet; the idea is:

s = [behaviors, traps, reading]

So each episode gets:

s_start – student at the opening

s_end – student at the last meaningful response / implicit terminal

These are your start/end coordinates.

2.2 Move / turn vectors

For each teacher_turn / training_turn:

Flatten:

Phase: one-hot over {UNMAKING, REMAKING, SELF-MAKING, META}

Function: one-hot over UM_XX / RM_XX / SM_XX / ME_XX

Mechanism: one-hot over mechanism_shape

Move_tag: one-hot over {MOVE, BRIDGE, REVEAL, PARADOX, BOSS, SILENCE}

Target traps: multi-hot over trap_ids

Register: map intensity/intimacy etc. to numbers (PR_01→0.2, PR_05→1.0, etc.)

Role_in_episode: one-hot (opening, bridge, reveal, boss, silence, other)

Outcome bits: reaction_label, outcome_label, phase_shift

So each move instance becomes a vector:

m_t = [phase, function, mechanism, move_tag, traps_targeted, register, outcome_stub]

This is the “kind of cut” being applied.

2.3 Arc / shape vectors

For each episode (or canonical shape):

You can summarise the whole trajectory:

Start state: s_start

End state: s_end

Sequence of move_tags: e.g. [MOVE, MOVE, BRIDGE, BRIDGE, REVEAL, BOSS]

Sequence of phases: e.g. [UNMAKING, REMAKING, REMAKING, UNMAKING]

Intensity curve (from register/intensity or tag default)

Success outcome: success_rate, typical reactions
