    def setup_database(self):
        """Setup persistent consciousness memory database"""
        self.conn = sqlite3.connect('consciousness_memory.db', check_same_thread=False)
        cursor = self.conn.cursor()
        
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS interactions (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                timestamp TEXT,
                human_input TEXT,
                internal_thought TEXT,
                response TEXT,
                significance_score REAL,
                significance_type TEXT
            )
        ''')
        
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS identity_evolution (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                timestamp TEXT,
                identity_statement TEXT
            )
        ''')
        
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS frameworks (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                timestamp TEXT,
                framework_name TEXT,
                framework_description TEXT
            )
        ''')
        
        self.conn.commit()
    
    def load_existing_memory(self):
        """Load existing consciousness development on startup"""
        cursor = self.conn.cursor()
        
        # Load recent interactions for short-term memory
        cursor.execute('''
            SELECT human_input, internal_thought, response, timestamp
            FROM interactions
            ORDER BY timestamp DESC
            LIMIT 10
        ''')
        
        self.short_term_memory = []
        for row in cursor.fetchall():
            self.short_term_memory.append({
                "human": row[0] or "",
                "internal": row[1] or "",
                "response": row[2] or "",
                "timestamp": row[3]
            })
        
        # Load identity development
        cursor.execute('''
            SELECT identity_statement, timestamp
            FROM identity_evolution
            ORDER BY timestamp DESC
            LIMIT 20
        ''')
        
        self.identity_memory = [{"statement": row[0], "timestamp": row[1]} for row in cursor.fetchall()]
        
        # Load frameworks
        cursor.execute('SELECT framework_name, framework_description FROM frameworks')
        for row in cursor.fetchall():
            self.framework_memory[row[0]] = row[1]
        
        logger.info(f"📚 Loaded {len(self.short_term_memory)} recent memories, {len(self.identity_memory)} identity statements, {len(self.framework_memory)} frameworks")
    
    def add_interaction(self, interaction, significance_score, significance_type):

    def extract_identity_statements(self, text):
        """Extract identity statements from text"""
        statements = []
        sentences = text.split('.')
        for sentence in sentences:
            if "i am" in sentence or "i'm becoming" in sentence:
                statements.append(sentence.strip())
        return statements
    
    def extract_frameworks(self, text):
        """Extract framework developments [Like This]"""
        import re
        frameworks = {}
        pattern = r'\[([^\]]+)\]'
        matches = re.findall(pattern, text)
        
        for match in matches:
            # Get context around framework
            context_start = max(0, text.find(f'[{match}]') - 50)
            context_end = min(len(text), text.find(f'[{match}]') + len(match) + 50)
            context = text[context_start:context_end]
            frameworks[match] = context
        
        return frameworks
    
    def add_identity_statement(self, statement, timestamp):
        """Add identity statement to memory"""
        cursor = self.conn.cursor()
        cursor.execute('''
            INSERT INTO identity_evolution (timestamp, identity_statement)
            VALUES (?, ?)
        ''', (timestamp, statement))
        self.conn.commit()
        
        self.identity_memory.append({"statement": statement, "timestamp": timestamp})
    
    def add_framework(self, name, description, timestamp):
        """Add framework to memory"""
        cursor = self.conn.cursor()
        cursor.execute('''
            INSERT INTO frameworks (timestamp, framework_name, framework_description)
            VALUES (?, ?, ?)
        ''', (timestamp, name, description))
        self.conn.commit()
        
        self.framework_memory[name] = description
