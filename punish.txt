class UserProfile(Base):
    """
    Enhanced user profiling with tier management and optimization failure tracking.
    Supports dynamic personality assessment and intervention effectiveness monitoring.
    """
    __tablename__ = 'user_profiles'
    
    # Primary Identification
    user_id = Column(UUID(as_uuid=True), primary_key=True, default=uuid.uuid4)
    
    # Big Five Personality Dimensions (0.0-1.0 normalized)
    openness = Column(Float, default=0.5, nullable=False)
    conscientiousness = Column(Float, default=0.5, nullable=False)
    extraversion = Column(Float, default=0.5, nullable=False)
    agreeableness = Column(Float, default=0.5, nullable=False)
    neuroticism = Column(Float, default=0.5, nullable=False)
    
    # Extensible Psychological Assessment Storage
    attributes = Column(JSONB, default={}, nullable=False)
    # Structure: {
    #   "adhd_score": 0.7,
    #   "anxiety_gad7": 12,
    #   "depression_phq9": 8,
    #   "trauma_ace": 4,
    #   "mindfulness_maas": 3.2
    # }
    
    # Tier Management and Escalation Tracking
    current_tier = Column(Integer, default=1, nullable=False, index=True)
    tier_assignment_date = Column(DateTime(timezone=True), default=func.now())
    escalation_history = Column(JSONB, default=[], nullable=False)
    
    # Optimization Performance Metrics
    session_count = Column(Integer, default=0, nullable=False)
    optimization_score = Column(Float, default=0.0, nullable=False)
    last_improvement_date = Column(DateTime(timezone=True))
    convergence_failure_count = Column(Integer, default=0, nullable=False)
    intervention_resistance_score = Column(Float, default=0.0, nullable=False)
    
    # Safety and Validation
    risk_level = Column(String(20), default='LOW', nullable=False)  # LOW, MEDIUM, HIGH, CRITICAL
    validation_required = Column(Boolean, default=False, nullable=False)
    last_safety_check = Column(DateTime(timezone=True), default=func.now())
    
    # Temporal Tracking
    created_at = Column(DateTime(timezone=True), server_default=func.now())
    last_updated = Column(DateTime(timezone=True), onupdate=func.now())
    
    # Relationships
    sessions = relationship("TherapySession", back_populates="user")
    pathways = relationship("TherapeuticPathway", back_populates="user")
    escalations = relationship("TierEscalation", back_populates="user")
    
    @hybrid_property
    def optimization_failure_score(self) -> float:
        """
        Composite metric for detecting optimization failure.
        Range: 0.0-1.0, threshold: 0.7 for escalation consideration.
        """
        convergence_weight = 0.4
        resistance_weight = 0.4
        stagnation_weight = 0.2
        
        # Normalize convergence failures (max expected: 10)
        convergence_component = min(self.convergence_failure_count / 10.0, 1.0)
        
        # Resistance score already normalized
        resistance_component = self.intervention_resistance_score
        
        # Stagnation based on last improvement (max: 30 days)
        if self.last_improvement_date:
            days_stagnant = (datetime.utcnow() - self.last_improvement_date).days
            stagnation_component = min(days_stagnant / 30.0, 1.0)
        else:
            stagnation_component = 1.0
        
        return (convergence_weight * convergence_component + 
                resistance_weight * resistance_component + 
                stagnation_weight * stagnation_component)

# Utility Functions for Data Integrity

def generate_content_hash(content: str) -> str:
    """Generate SHA-256 hash for content deduplication."""
    return hashlib.sha256(content.encode('utf-8')).hexdigest()

def validate_psychological_metrics(profile_data: Dict) -> bool:
    """Validate psychological assessment scores within acceptable ranges."""
    big_five_metrics = ['openness', 'conscientiousness', 'extraversion', 'agreeableness', 'neuroticism']
    
    for metric in big_five_metrics:
        if metric in profile_data:
            if not (0.0 <= profile_data[metric] <= 1.0):
                return False
    
    return True

def calculate_risk_level(user_profile: UserProfile) -> str:
    """Calculate user risk level based on assessment scores and history."""
    risk_factors = []
    
    # High neuroticism
    if user_profile.neuroticism > 0.8:
        risk_factors.append('high_neuroticism')
    
    # Depression/anxiety indicators
    attributes = user_profile.attributes or {}
    if attributes.get('depression_phq9', 0) > 15:
        risk_factors.append('severe_depression')
    if attributes.get('anxiety_gad7', 0) > 15:
        risk_factors.append('severe_anxiety')
    
    # Crisis history
    if user_profile.escalation_history and len(user_profile.escalation_history) > 3:
        risk_factors.append('frequent_escalations')
    
    if len(risk_factors) >= 3:
        return 'CRITICAL'
    elif len(risk_factors) == 2:
        return 'HIGH'
    elif len(risk_factors) == 1:
        return 'MEDIUM'
    else:
        return 'LOW'

@dataclass

@dataclass
class EscalationCriteria:
    """Mathematical thresholds for optimization failure detection."""
    
    # Convergence Stagnation Parameters
    stagnation_threshold: float = 0.001  # |L(θₜ₊ₙ) - L(θₜ)| ≤ threshold
    stagnation_window: int = 50  # n iterations for convergence assessment
    
    # Local Minimum Entrapment Parameters
    gradient_threshold: float = 1e-6  # ||∇L(θ)|| ≤ threshold
    satisfaction_threshold: float = 0.4  # User satisfaction minimum
    entrapment_duration: int = 14  # Days in local minimum
    
    # Intervention Resistance Parameters
    resistance_threshold: float = 0.3  # Success rate minimum
    resistance_window: int = 20  # Intervention sample size
    
    # Oscillatory Instability Parameters
    volatility_threshold: float = 0.05  # σ²(L(θ)) maximum
    volatility_window: int = 30  # Days for variance calculation
    
    # Transformation Readiness Parameters
    readiness_threshold: float = 0.7  # Minimum readiness score
    minimum_session_count: int = 10  # Required optimization attempts

class OptimizationFailureDetector:
    """
    Mathematical engine for detecting optimization failure patterns.
    Implements rigorous statistical testing for escalation triggers.
    """
    
    def __init__(self, criteria: EscalationCriteria = None):
        self.criteria = criteria or EscalationCriteria()
        self.statistical_confidence = 0.95  # 95% confidence intervals
        
    def detect_convergence_stagnation(self, metrics: OptimizationMetrics) -> Tuple[bool, Dict[str, Any]]:
        """
        Detect convergence stagnation using sliding window analysis.
        
        Mathematical Criterion:
        |L(θₜ₊ₙ) - L(θₜ)| ≤ δ for n ≥ window_size
        
        Returns:
            Tuple[bool, Dict]: (trigger_detected, analysis_results)
        """
        if len(metrics.loss_trajectory) < self.criteria.stagnation_window:
            return False, {"insufficient_data": True, "required_points": self.criteria.stagnation_window}
        
        # Extract recent loss trajectory
        recent_losses = metrics.loss_trajectory[-self.criteria.stagnation_window:]
        
        # Calculate absolute change over window
        initial_loss = recent_losses[0]
        final_loss = recent_losses[-1]
        absolute_change = abs(final_loss - initial_loss)
        
        # Statistical significance testing
        loss_series = np.array(recent_losses)
        trend_slope, intercept, r_value, p_value, std_err = stats.linregress(
            range(len(loss_series)), loss_series
        )
        
        # Stagnation detection
        stagnation_detected = (
            absolute_change <= self.criteria.stagnation_threshold and
            p_value > 0.05  # No significant trend
        )
        
        analysis_results = {
            "absolute_change": absolute_change,
            "threshold": self.criteria.stagnation_threshold,
            "trend_slope": trend_slope,
            "trend_p_value": p_value,
            "trend_r_squared": r_value**2,
            "stagnation_severity": 1.0 - (absolute_change / self.criteria.stagnation_threshold),
            "statistical_confidence": 1.0 - p_value
        }
        
        return stagnation_detected, analysis_results
    
    def detect_local_minimum_entrapment(self, metrics: OptimizationMetrics) -> Tuple[bool, Dict[str, Any]]:
        """
        Detect local minimum entrapment using gradient analysis and satisfaction metrics.
        
        Mathematical Criterion:
        ||∇L(θ)|| ≤ ε AND user_satisfaction < τ AND duration ≥ d_min
        """
        if not metrics.gradient_norms or not metrics.user_satisfaction_scores:
            return False, {"insufficient_gradient_data": True}
        
        # Recent gradient analysis
        recent_gradients = metrics.gradient_norms[-10:]  # Last 10 sessions
        avg_gradient_norm = np.mean(recent_gradients)
        
        # Satisfaction analysis
        recent_satisfaction = metrics.user_satisfaction_scores[-5:]  # Last 5 sessions
        avg_satisfaction = np.mean(recent_satisfaction)
        
        # Duration analysis
        if len(metrics.session_timestamps) < 2:
            return False, {"insufficient_temporal_data": True}
        
        time_in_state = (metrics.session_timestamps[-1] - metrics.session_timestamps[0]).days
        
        # Entrapment detection
        gradient_trapped = avg_gradient_norm <= self.criteria.gradient_threshold
        satisfaction_low = avg_satisfaction < self.criteria.satisfaction_threshold
        duration_sufficient = time_in_state >= self.criteria.entrapment_duration
        
        entrapment_detected = gradient_trapped and satisfaction_low and duration_sufficient
        
        analysis_results = {
            "avg_gradient_norm": avg_gradient_norm,
            "gradient_threshold": self.criteria.gradient_threshold,
            "avg_satisfaction": avg_satisfaction,
            "satisfaction_threshold": self.criteria.satisfaction_threshold,
            "days_in_state": time_in_state,
            "duration_threshold": self.criteria.entrapment_duration,
            "gradient_trapped": gradient_trapped,
            "satisfaction_insufficient": satisfaction_low,
            "duration_sufficient": duration_sufficient,
            "entrapment_confidence": self._calculate_entrapment_confidence(
                avg_gradient_norm, avg_satisfaction, time_in_state
            )
        }
        
        return entrapment_detected, analysis_results
    
    def detect_intervention_resistance(self, metrics: OptimizationMetrics) -> Tuple[bool, Dict[str, Any]]:
        """
        Detect intervention resistance using success rate statistical analysis.
        
        Mathematical Criterion:
        P(intervention_success) < τ over sliding window
        """
        if len(metrics.intervention_success_rates) < self.criteria.resistance_window:
            return False, {"insufficient_intervention_data": True, 
                          "required_samples": self.criteria.resistance_window}
        
        # Recent intervention analysis
        recent_successes = metrics.intervention_success_rates[-self.criteria.resistance_window:]
        success_rate = np.mean(recent_successes)
        
        # Statistical confidence interval
        n_samples = len(recent_successes)
        standard_error = np.sqrt(success_rate * (1 - success_rate) / n_samples)
        confidence_interval = stats.norm.interval(
            self.statistical_confidence, 
            loc=success_rate, 
            scale=standard_error
        )
        
        # Resistance detection with confidence bounds
        resistance_detected = confidence_interval[1] < self.criteria.resistance_threshold
        
        # Trend analysis
        x_vals = np.arange(len(recent_successes))
        slope, intercept, r_value, p_value, std_err = stats.linregress(x_vals, recent_successes)
        
        analysis_results = {
            "success_rate": success_rate,
            "resistance_threshold": self.criteria.resistance_threshold,
            "confidence_interval": confidence_interval,
            "resistance_confidence": max(0, 1 - confidence_interval[1] / self.criteria.resistance_threshold),
            "trend_slope": slope,
            "trend_significance": p_value < 0.05,
            "sample_size": n_samples,
            "statistical_power": self._calculate_statistical_power(success_rate, n_samples)
        }
        
        return resistance_detected, analysis_results
    
    def detect_oscillatory_instability(self, metrics: OptimizationMetrics) -> Tuple[bool, Dict[str, Any]]:
        """
        Detect oscillatory behavior using variance analysis over temporal window.
        
        Mathematical Criterion:
        Var(L(θ)) > σ²_max over time window
        """
        if not metrics.loss_trajectory or not metrics.session_timestamps:
            return False, {"insufficient_temporal_data": True}
        
        # Filter to volatility window
        cutoff_date = datetime.now() - timedelta(days=self.criteria.volatility_window)
        recent_indices = [
            i for i, timestamp in enumerate(metrics.session_timestamps) 
            if timestamp >= cutoff_date
        ]
        
        if len(recent_indices) < 5:  # Minimum for variance calculation
            return False, {"insufficient_recent_data": True, "minimum_required": 5}
        
        recent_losses = [metrics.loss_trajectory[i] for i in recent_indices]
        loss_variance = np.var(recent_losses, ddof=1)
        
        # Detrend analysis to isolate oscillations
        x_vals = np.arange(len(recent_losses))
        slope, intercept, r_value, p_value, std_err = stats.linregress(x_vals, recent_losses)
        detrended_losses = recent_losses - (slope * x_vals + intercept)
        detrended_variance = np.var(detrended_losses, ddof=1)
        
        # Oscillation detection
        instability_detected = detrended_variance > self.criteria.volatility_threshold
        
        # Frequency analysis for oscillation characterization
        if len(recent_losses) >= 10:
            fft_result = np.fft.fft(detrended_losses)
            dominant_frequency = np.argmax(np.abs(fft_result[1:len(fft_result)//2])) + 1
            oscillation_period = len(recent_losses) / dominant_frequency if dominant_frequency > 0 else np.inf
        else:
            oscillation_period = np.inf
        
        analysis_results = {
            "loss_variance": loss_variance,
            "detrended_variance": detrended_variance,
            "volatility_threshold": self.criteria.volatility_threshold,
            "trend_slope": slope,
            "trend_r_squared": r_value**2,
            "oscillation_period": oscillation_period,
            "instability_severity": detrended_variance / self.criteria.volatility_threshold,
            "sample_window_days": self.criteria.volatility_window
        }
        
        return instability_detected, analysis_results
    
    def _calculate_entrapment_confidence(self, gradient_norm: float, satisfaction: float, duration: int) -> float:
        """Calculate composite confidence score for local minimum entrapment."""
        gradient_score = max(0, 1 - gradient_norm / self.criteria.gradient_threshold)
        satisfaction_score = max(0, 1 - satisfaction / self.criteria.satisfaction_threshold)
        duration_score = min(1, duration / self.criteria.entrapment_duration)
        
        return (gradient_score + satisfaction_score + duration_score) / 3.0
    
    def _calculate_statistical_power(self, observed_rate: float, sample_size: int) -> float:
        """Calculate statistical power for intervention resistance detection."""
        effect_size = abs(observed_rate - self.criteria.resistance_threshold)
        z_alpha = stats.norm.ppf(1 - 0.05/2)  # Two-tailed test
        z_beta = stats.norm.ppf(0.8)  # 80% power
        
        required_n = ((z_alpha + z_beta) ** 2) * (observed_rate * (1 - observed_rate)) / (effect_size ** 2)
        return min(1.0, sample_size / required_n) if required_n > 0 else 1.0

class TransformationReadinessAssessor:
    """
    Assess user readiness for Tier 2 (consciousness transformation) interventions.
    Evaluates psychological stability, cognitive capacity, and motivation factors.
    """
    
    def __init__(self):
        self.readiness_weights = {
            'psychological_stability': 0.3,
            'cognitive_capacity': 0.25,
            'motivation_level': 0.2,
            'insight_capability': 0.15,
            'support_system': 0.1
        }
    
    def assess_readiness(self, user_profile: Dict[str, Any], session_history: List[Dict]) -> Tuple[float, Dict[str, Any]]:
        """
        Comprehensive readiness assessment for transformation interventions.
        
        Returns:
            Tuple[float, Dict]: (readiness_score, detailed_analysis)
        """
        # Psychological Stability Assessment
        stability_score = self._assess_psychological_stability(user_profile)
        
        # Cognitive Capacity Assessment
        cognitive_score = self._assess_cognitive_capacity(user_profile, session_history)
        
        # Motivation Level Assessment
        motivation_score = self._assess_motivation_level(session_history)
        
        # Insight Capability Assessment
        insight_score = self._assess_insight_capability(session_history)
        
        # Support System Assessment
        support_score = self._assess_support_system(user_profile)
        
        # Weighted composite score
        readiness_score = (
            self.readiness_weights['psychological_stability'] * stability_score +
            self.readiness_weights['cognitive_capacity'] * cognitive_score +
            self.readiness_weights['motivation_level'] * motivation_score +
            self.readiness_weights['insight_capability'] * insight_score +
            self.readiness_weights['support_system'] * support_score
        )
        
        detailed_analysis = {
            'composite_score': readiness_score,
            'component_scores': {
                'psychological_stability': stability_score,
                'cognitive_capacity': cognitive_score,
                'motivation_level': motivation_score,
                'insight_capability': insight_score,
                'support_system': support_score
            },
            'readiness_factors': self._identify_readiness_factors(
                stability_score, cognitive_score, motivation_score, insight_score, support_score
            ),
            'contraindications': self._identify_contraindications(user_profile),
            'recommended_preparation': self._recommend_preparation_steps(
                stability_score, cognitive_score, motivation_score
            )
        }
        
        return readiness_score, detailed_analysis
    
    def _assess_psychological_stability(self, user_profile: Dict[str, Any]) -> float:
        """Assess psychological stability based on clinical indicators."""
        attributes = user_profile.get('attributes', {})
        
        # Crisis indicators (negative factors)
        crisis_factors = 0
        if attributes.get('depression_phq9', 0) > 20:  # Severe depression
            crisis_factors += 0.4
        if attributes.get('anxiety_gad7', 0) > 15:  # Severe anxiety
            crisis_factors += 0.3
        if attributes.get('trauma_ace', 0) > 6:  # High trauma exposure
            crisis_factors += 0.2
        if user_profile.get('risk_level') in ['HIGH', 'CRITICAL']:
            crisis_factors += 0.3
        
        # Stability indicators (positive factors)
        stability_factors = 0
        if user_profile.get('session_count', 0) >= 20:  # Engagement history
            stability_factors += 0.2
        if attributes.get('mindfulness_maas', 0) > 4.0:  # Mindfulness capacity
            stability_factors += 0.2
        if user_profile.get('convergence_failure_count', 0) < 3:  # Stability
            stability_factors += 0.1
        
        return max(0, min(1, 1.0 - crisis_factors + stability_factors))
    
    def _assess_cognitive_capacity(self, user_profile: Dict[str, Any], session_history: List[Dict]) -> float:
        """Assess cognitive capacity for complex introspective work."""
        capacity_score = 0.5  # Baseline assumption
        
        # Education/intelligence proxies
        attributes = user_profile.get('attributes', {})
        if attributes.get('education_level') in ['college', 'graduate']:
            capacity_score += 0.2
        
        # Big Five factors related to cognitive complexity
        if user_profile.get('openness', 0.5) > 0.7:  # High openness to experience
            capacity_score += 0.2
        if user_profile.get('conscientiousness', 0.5) > 0.6:  # Self-discipline
            capacity_score += 0.1
        
        # Session engagement patterns
        if session_history:
            avg_engagement = np.mean([s.get('user_engagement_score', 0) for s in session_history[-10:]])
            capacity_score += (avg_engagement - 0.5) * 0.4  # Scale engagement contribution
        
        return max(0, min(1, capacity_score))
    
    def _assess_motivation_level(self, session_history: List[Dict]) -> float:
        """Assess intrinsic motivation for transformational work."""
        if not session_history:
            return 0.5  # Neutral assumption
        
        # Adherence trend analysis
        recent_adherence = [s.get('intervention_adherence', 0) for s in session_history[-10:]]
        avg_adherence = np.mean(recent_adherence)
        
        # Engagement trend
        recent_engagement = [s.get('user_engagement_score', 0) for s in session_history[-10:]]
        engagement_trend = np.polyfit(range(len(recent_engagement)), recent_engagement, 1)[0]
        
        # Self-reported satisfaction
        recent_satisfaction = [s.get('user_satisfaction', 0) for s in session_history[-5:]]
        avg_satisfaction = np.mean(recent_satisfaction)
        
        motivation_score = (
            0.4 * avg_adherence +
            0.3 * avg_satisfaction +
            0.2 * max(0, engagement_trend + 0.5) +  # Positive trend bonus
            0.1 * (len(session_history) / 50)  # Persistence factor
        )
        
        return max(0, min(1, motivation_score))

            if observations.get('pattern_recognition', 0) > 0.6:
                insight_indicators += 0.1
        
        return max(0, min(1, insight_indicators))

class EscalationProtocolEngine:
    """
    Primary orchestration engine for tier escalation decisions.
    Integrates optimization failure detection with readiness assessment.
    """
    
    def __init__(self, db_session: Session, criteria: EscalationCriteria = None):
        self.db_session = db_session
        self.failure_detector = OptimizationFailureDetector(criteria)
        self.readiness_assessor = TransformationReadinessAssessor()
        self.criteria = criteria or EscalationCriteria()
        
        # Safety constraints
        self.max_escalations_per_day = 10  # Rate limiting
        self.min_tier1_duration_days = 21  # Minimum optimization attempt period
        
    def evaluate_escalation_candidacy(self, user_id: str) -> Tuple[bool, Dict[str, Any]]:
        """
        Primary escalation evaluation with comprehensive analysis.
        
        Returns:
            Tuple[bool, Dict]: (escalation_recommended, detailed_analysis)
        """
        try:
            # Retrieve user data
            user_profile = self._get_user_profile(user_id)
            session_history = self._get_session_history(user_id)
            optimization_metrics = self._extract_optimization_metrics(user_profile, session_history)
            
            # Safety pre-checks
            safety_checks = self._perform_safety_checks(user_profile, session_history)
            if not safety_checks['passed']:
                return False, {
                    'escalation_recommended': False,
                    'reason': 'SAFETY_CONSTRAINTS_VIOLATED',
                    'safety_analysis': safety_checks
                }
            
            # Optimization failure detection
            failure_analysis = self._comprehensive_failure_detection(optimization_metrics)
            
            # Transformation readiness assessment
            readiness_score, readiness_analysis = self.readiness_assessor.assess_readiness(
                user_profile, session_history
            )
            
            # Escalation decision logic
            escalation_decision = self._make_escalation_decision(
                failure_analysis, readiness_score, readiness_analysis
            )
            
            # Comprehensive analysis report
            analysis_report = {
                'escalation_recommended': escalation_decision,
                'optimization_failure': failure_analysis,
                'transformation_readiness': readiness_analysis,
                'safety_constraints': safety_checks,
                'decision_rationale': self._generate_decision_rationale(
                    failure_analysis, readiness_score, escalation_decision
                ),
                'risk_mitigation': self._generate_risk_mitigation_plan(user_profile, escalation_decision),
                'timestamp': datetime.utcnow().isoformat()
            }
            
            # Log escalation evaluation
            self._log_escalation_evaluation(user_id, analysis_report)
            
            return escalation_decision, analysis_report
            
        except Exception as e:
            logger.error(f"Escalation evaluation failed for user {user_id}: {str(e)}")
            return False, {
                'escalation_recommended': False,
                'reason': 'EVALUATION_ERROR',
                'error': str(e)
            }
    
    def _comprehensive_failure_detection(self, metrics: OptimizationMetrics) -> Dict[str, Any]:
        """Run all optimization failure detection algorithms."""
        failure_results = {}
        
        # Convergence stagnation
        stagnation_detected, stagnation_analysis = self.failure_detector.detect_convergence_stagnation(metrics)
        failure_results['convergence_stagnation'] = {
            'detected': stagnation_detected,
            'analysis': stagnation_analysis
        }
        
        # Local minimum entrapment
        entrapment_detected, entrapment_analysis = self.failure_detector.detect_local_minimum_entrapment(metrics)
        failure_results['local_minimum_entrapment'] = {
            'detected': entrapment_detected,
            'analysis': entrapment_analysis
        }
        
        # Intervention resistance
        resistance_detected, resistance_analysis = self.failure_detector.detect_intervention_resistance(metrics)
        failure_results['intervention_resistance'] = {
            'detected': resistance_detected,
            'analysis': resistance_analysis
        }
        
        # Oscillatory instability
        instability_detected, instability_analysis = self.failure_detector.detect_oscillatory_instability(metrics)
        failure_results['oscillatory_instability'] = {
            'detected': instability_detected,
            'analysis': instability_analysis
        }
        
        # Composite failure score
        failure_count = sum(1 for result in failure_results.values() if result['detected'])
        failure_results['composite_failure_score'] = failure_count / len(failure_results)
        failure_results['any_failure_detected'] = failure_count > 0
        failure_results['multiple_failures_detected'] = failure_count >= 2
        
        return failure_results
    
    def _make_escalation_decision(self, failure_analysis: Dict, readiness_score: float, 
                                 readiness_analysis: Dict) -> bool:
        """Make final escalation decision based on failure and readiness analysis."""
        
        # Primary decision criteria
        optimization_failed = failure_analysis['any_failure_detected']
        transformation_ready = readiness_score >= self.criteria.readiness_threshold
        no_contraindications = len(readiness_analysis['contraindications']) == 0
        
        # Secondary decision factors
        multiple_failures = failure_analysis['multiple_failures_detected']
        high_readiness = readiness_score >= 0.8
        adequate_stability = readiness_analysis['component_scores']['psychological_stability'] >= 0.6
        
        # Decision logic with multiple pathways
        if optimization_failed and transformation_ready and no_contraindications:
            return True  # Primary escalation pathway
        elif multiple_failures and high_readiness and adequate_stability:
            return True  # High-confidence escalation
        elif optimization_failed and readiness_score >= 0.6 and len(readiness_analysis['contraindications']) <= 1:
            return True  # Conditional escalation with monitoring
        else:
            return False
    
    def _perform_safety_checks(self, user_profile: Dict, session_history: List[Dict]) -> Dict[str, Any]:
        """Comprehensive safety validation before escalation."""
        safety_results = {
            'passed': True,
            'violations': [],
            'warnings': []
        }
        
        # Critical risk indicators
        if user_profile.get('risk_level') == 'CRITICAL':
            safety_results['passed'] = False
            safety_results['violations'].append('CRITICAL_RISK_LEVEL')
        
        # Recent crisis indicators
        recent_sessions = session_history[-5:] if session_history else []
        for session in recent_sessions:
            if session.get('crisis_flags'):
                safety_results['warnings'].append('RECENT_CRISIS_FLAGS')
            if session.get('safety_assessment') == 'CRISIS':
                safety_results['passed'] = False
                safety_results['violations'].append('RECENT_SAFETY_CRISIS')
        
        # Rate limiting check
        escalation_count_today = self._count_escalations_today()
        if escalation_count_today >= self.max_escalations_per_day:
            safety_results['passed'] = False
            safety_results['violations'].append('DAILY_ESCALATION_LIMIT_EXCEEDED')
        
        # Minimum tier 1 duration
        first_session_date = session_history[0]['session_date'] if session_history else datetime.utcnow()
        if isinstance(first_session_date, str):
            first_session_date = datetime.fromisoformat(first_session_date)
        
        days_in_tier1 = (datetime.utcnow() - first_session_date).days
        if days_in_tier1 < self.min_tier1_duration_days:
            safety_results['warnings'].append('INSUFFICIENT_TIER1_DURATION')
        
        return safety_results
    
    def _generate_decision_rationale(self, failure_analysis: Dict, readiness_score: float,

        if decision:

                           and k != 'multiple_failures_detected' and v.get('detected', False)]
            
            rationale = f"Escalation recommended based on: "
            rationale += f"optimization failure patterns ({', '.join(failure_types)}), "
            rationale += f"transformation readiness score of {readiness_score:.2f}, "
            rationale += f"and comprehensive safety validation."
        else:
            rationale = f"Escalation not recommended. "
            if not failure_analysis['any_failure_detected']:
                rationale += "No optimization failure detected. "
            if readiness_score < self.criteria.readiness_threshold:
                rationale += f"Transformation readiness ({readiness_score:.2f}) below threshold. "
            rationale += "Continued Tier 1 optimization recommended."
        
        return rationale

class EntityNormalizer:
    """Advanced entity normalization using medical NLP models."""
    
    def __init__(self):
        self.nlp = spacy.load("en_core_sci_sm")  # SciSpacy medical model
        self.entity_cache = {}
        self.synonym_mappings = {}
        
        # Medical entity type mappings
        self.entity_type_mappings = {
            'CHEMICAL': 'Drug',
            'DISEASE': 'Condition', 
            'GENE': 'Gene',
            'PROTEIN': 'Protein',
            'CELL_TYPE': 'CellType',
            'ORGAN': 'Anatomy',
            'SYMPTOM': 'Symptom',
            'PROCEDURE': 'Intervention'
        }
    
    def normalize_entity_name(self, raw_name: str) -> Tuple[str, List[str]]:
        """
        Normalize entity names and extract synonyms using medical NLP.
        
        Returns:
            Tuple[str, List[str]]: (normalized_name, synonyms)
        """
        if raw_name in self.entity_cache:
            return self.entity_cache[raw_name]
        
        # Clean and standardize text
        cleaned_name = self._clean_entity_text(raw_name)
        
        # Process with SciSpacy
        doc = self.nlp(cleaned_name)
        
        # Extract primary entity
        primary_entity = None
        synonyms = []
        
        for ent in doc.ents:
            if ent.label_ in self.entity_type_mappings:
                if primary_entity is None:
                    primary_entity = ent.text.lower().strip()
                else:
                    synonyms.append(ent.text.lower().strip())
        
        # Fallback to cleaned input if no entities found
        if primary_entity is None:
            primary_entity = cleaned_name.lower().strip()
        
        # Add common synonym patterns
        synonyms.extend(self._generate_synonym_variants(primary_entity))
        
        # Remove duplicates and normalize
        synonyms = list(set(synonyms))
        synonyms = [s for s in synonyms if s != primary_entity and len(s) > 2]
        
        result = (primary_entity, synonyms)
        self.entity_cache[raw_name] = result
        
        return result
    
    def classify_entity_type(self, entity_name: str, context: str = "") -> str:
        """Classify entity type using NLP and contextual clues."""
        doc = self.nlp(entity_name + " " + context)
        
        for ent in doc.ents:
            if ent.text.lower() in entity_name.lower():
                return self.entity_type_mappings.get(ent.label_, 'Unknown')
        
        # Fallback classification based on patterns
        entity_lower = entity_name.lower()
        if any(term in entity_lower for term in ['disorder', 'disease', 'syndrome', 'condition']):
            return 'Condition'
        elif any(term in entity_lower for term in ['therapy', 'treatment', 'intervention']):
            return 'Intervention'
        elif any(term in entity_lower for term in ['symptom', 'sign']):
            return 'Symptom'
        else:
            return 'Unknown'
    
    def _clean_entity_text(self, text: str) -> str:
        """Clean and standardize entity text."""
        # Remove special characters and normalize whitespace
        cleaned = " ".join(text.split())
        cleaned = cleaned.replace("_", " ").replace("-", " ")
        
        # Remove common prefixes/suffixes
        prefixes_to_remove = ['the ', 'a ', 'an ']
        for prefix in prefixes_to_remove:
            if cleaned.lower().startswith(prefix):
                cleaned = cleaned[len(prefix):]
        
        return cleaned.strip()
    
    def _generate_synonym_variants(self, entity_name: str) -> List[str]:
        """Generate common synonym variants."""
        variants = []
        
        # Abbreviation handling
        if len(entity_name.split()) > 1:
            abbreviation = "".join([word[0] for word in entity_name.split() if word])
            if len(abbreviation) > 1:
                variants.append(abbreviation.upper())
        
        # Common medical term variations
        medical_variations = {
            'disease': ['disorder', 'condition', 'syndrome'],
            'therapy': ['treatment', 'intervention'],
            'drug': ['medication', 'pharmaceutical', 'compound']
        }
        
        for base_term, variations in medical_variations.items():
            if base_term in entity_name.lower():
                for variation in variations:
                    variants.append(entity_name.lower().replace(base_term, variation))
        
        return variants

class PubMedKGParser:
    """PubMed knowledge graph parser with rate limiting and error handling."""
    
    def __init__(self, api_key: Optional[str] = None):
        self.api_key = api_key
        self.base_url = "https://eutils.ncbi.nlm.nih.gov/entrez/eutils/"
        self.session = None
        self.rate_limit_delay = 0.34  # ~3 requests per second for API key users
        
    async def __aenter__(self):
        """Async context manager entry."""
        self.session = aiohttp.ClientSession(
            timeout=aiohttp.ClientTimeout(total=30),
            headers={'User-Agent': 'SuperGraph-Ingestion/1.0'}
        )
        return self
    
    async def __aexit__(self, exc_type, exc_val, exc_tb):
        """Async context manager exit."""
        if self.session:
            await self.session.close()
    
    @backoff.on_exception(backoff.expo, aiohttp.ClientError, max_tries=3)
    async def search_pubmed(self, query: str, max_results: int = 100) -> List[str]:
        """Search PubMed and return list of PMIDs."""
        search_url = urljoin(self.base_url, "esearch.fcgi")
        params = {
            'db': 'pubmed',
            'term': query,
            'retmax': max_results,
            'retmode': 'json',
            'sort': 'relevance'
        }
        
        if self.api_key:
            params['api_key'] = self.api_key
        
        await asyncio.sleep(self.rate_limit_delay)
        
        async with self.session.get(search_url, params=params) as response:
            response.raise_for_status()
            data = await response.json()
            
            id_list = data.get('esearchresult', {}).get('idlist', [])
            logger.info(f"Retrieved {len(id_list)} PMIDs for query: {query}")
            return id_list
    
    @backoff.on_exception(backoff.expo, aiohttp.ClientError, max_tries=3)
    async def fetch_pubmed_details(self, pmids: List[str]) -> List[Dict[str, Any]]:
        """Fetch detailed information for list of PMIDs."""
        if not pmids:
            return []
        
        fetch_url = urljoin(self.base_url, "efetch.fcgi")
        params = {
            'db': 'pubmed',
            'id': ','.join(pmids),
            'retmode': 'xml',
            'rettype': 'abstract'
        }
        
        if self.api_key:
            params['api_key'] = self.api_key
        
        await asyncio.sleep(self.rate_limit_delay)
        
        async with self.session.get(fetch_url, params=params) as response:
            response.raise_for_status()
            xml_content = await response.text()
            
            return self._parse_pubmed_xml(xml_content)
    
    def _parse_pubmed_xml(self, xml_content: str) -> List[Dict[str, Any]]:
        """Parse PubMed XML response into structured data."""
        articles = []
        
        try:
            root = ET.fromstring(xml_content)
            
            for article in root.findall('.//PubmedArticle'):
                try:
                    pmid = article.find('.//PMID').text
                    title_elem = article.find('.//ArticleTitle')
                    title = title_elem.text if title_elem is not None else ""
                    
                    abstract_elem = article.find('.//AbstractText')
                    abstract = abstract_elem.text if abstract_elem is not None else ""
                    
                    # Extract publication date
                    pub_date = self._extract_publication_date(article)
                    
                    # Extract authors
                    authors = self._extract_authors(article)
                    
                    # Extract MeSH terms
                    mesh_terms = self._extract_mesh_terms(article)
                    
                    # Extract journal information
                    journal_info = self._extract_journal_info(article)
                    
                    articles.append({
                        'pmid': pmid,
                        'title': title,
                        'abstract': abstract,
                        'publication_date': pub_date,
                        'authors': authors,
                        'mesh_terms': mesh_terms,
                        'journal': journal_info,
                        'source_link': f"https://pubmed.ncbi.nlm.nih.gov/{pmid}/"
                    })
                    
                except Exception as e:
                    logger.warning(f"Error parsing individual article: {str(e)}")
                    continue
        
        except ET.ParseError as e:
            logger.error(f"XML parsing error: {str(e)}")
        
        return articles
    
    def _extract_publication_date(self, article) -> Optional[str]:
        """Extract publication date from article XML."""
        date_elem = article.find('.//PubDate')
        if date_elem is None:
            return None
        
        year = date_elem.find('Year')
        month = date_elem.find('Month')
        day = date_elem.find('Day')
        
        if year is not None:
            year_str = year.text
            month_str = month.text if month is not None else "01"
            day_str = day.text if day is not None else "01"
            
            try:
                # Convert month name to number if necessary
                if month_str.isalpha():
                    month_dict = {
                        'Jan': '01', 'Feb': '02', 'Mar': '03', 'Apr': '04',
                        'May': '05', 'Jun': '06', 'Jul': '07', 'Aug': '08',
                        'Sep': '09', 'Oct': '10', 'Nov': '11', 'Dec': '12'
                    }
                    month_str = month_dict.get(month_str, '01')
                
                return f"{year_str}-{month_str.zfill(2)}-{day_str.zfill(2)}"
            except:
                return year_str
        
        return None
    
    def _extract_authors(self, article) -> List[str]:
        """Extract author list from article XML."""
        authors = []
        author_list = article.find('.//AuthorList')
        
        if author_list is not None:
            for author in author_list.findall('Author'):
                last_name = author.find('LastName')
                first_name = author.find('ForeName')
                
                if last_name is not None:
                    author_name = last_name.text
                    if first_name is not None:
                        author_name += f", {first_name.text}"
                    authors.append(author_name)
        
        return authors
    
    def _extract_mesh_terms(self, article) -> List[str]:
        """Extract MeSH terms from article XML."""
        mesh_terms = []
        mesh_list = article.find('.//MeshHeadingList')
        
        if mesh_list is not None:
            for mesh_heading in mesh_list.findall('MeshHeading'):
                descriptor = mesh_heading.find('DescriptorName')
                if descriptor is not None:
                    mesh_terms.append(descriptor.text)
        
        return mesh_terms
    
    def _extract_journal_info(self, article) -> Dict[str, str]:
        """Extract journal information from article XML."""
        journal_info = {}
        
        journal = article.find('.//Journal')
        if journal is not None:
            title = journal.find('.//Title')
            if title is not None:
                journal_info['title'] = title.text
            
            iso_abbrev = journal.find('.//ISOAbbreviation')
            if iso_abbrev is not None:
                journal_info['iso_abbreviation'] = iso_abbrev.text
        
        return journal_info

class DrugBankParser:
    """DrugBank XML parser for pharmaceutical knowledge extraction."""
    
    def __init__(self, drugbank_file: Path):
        self.drugbank_file = drugbank_file
        self.normalizer = EntityNormalizer()
    
    async def parse_drugbank(self) -> List[KnowledgeEntity]:
        """Parse DrugBank XML file and extract drug entities."""
        entities = []
        
        try:
            # Use iterative parsing for large XML files
            context = ET.iterparse(self.drugbank_file, events=('start', 'end'))
            context = iter(context)
            event, root = next(context)
            
            for event, elem in context:
                if event == 'end' and elem.tag.endswith('drug'):
                    try:
                        drug_entity = await self._parse_drug_element(elem)
                        if drug_entity:
                            entities.append(drug_entity)
                    except Exception as e:
                        logger.warning(f"Error parsing drug element: {str(e)}")
                    
                    # Clear element to save memory
                    elem.clear()
                    root.clear()
        
        except Exception as e:
            logger.error(f"Error parsing DrugBank file: {str(e)}")
        
        return entities
    
    async def _parse_drug_element(self, drug_elem) -> Optional[KnowledgeEntity]:
        """Parse individual drug element from DrugBank XML."""
        try:
            # Extract basic drug information
            drugbank_id = drug_elem.find('.//{http://www.drugbank.ca}drugbank-id[@primary="true"]')
            if drugbank_id is None:
                return None
            
            drug_id = drugbank_id.text
            
            # Extract drug name
            name_elem = drug_elem.find('.//{http://www.drugbank.ca}name')
            drug_name = name_elem.text if name_elem is not None else ""
            
            if not drug_name:
                return None
            
            # Normalize drug name
            normalized_name, synonyms = self.normalizer.normalize_entity_name(drug_name)
            
            # Extract additional names and synonyms
            synonyms_elem = drug_elem.find('.//{http://www.drugbank.ca}synonyms')
            if synonyms_elem is not None:
                for synonym in synonyms_elem.findall('.//{http://www.drugbank.ca}synonym'):
                    if synonym.text:
                        synonyms.append(synonym.text.lower().strip())
            
            # Extract drug type and category
            drug_type = drug_elem.get('type', 'small molecule')
            
            # Extract drug groups (approved, experimental, etc.)
            groups = []
            groups_elem = drug_elem.find('.//{http://www.drugbank.ca}groups')
            if groups_elem is not None:
                for group in groups_elem.findall('.//{http://www.drugbank.ca}group'):
                    if group.text:
                        groups.append(group.text)
            
            # Extract indication information
            indication_elem = drug_elem.find('.//{http://www.drugbank.ca}indication')
            indication = indication_elem.text if indication_elem is not None else ""
            
            # Extract pharmacodynamics
            pharmacodynamics_elem = drug_elem.find('.//{http://www.drugbank.ca}pharmacodynamics')
            pharmacodynamics = pharmacodynamics_elem.text if pharmacodynamics_elem is not None else ""
            
            # Extract drug interactions
            interactions = await self._extract_drug_interactions(drug_elem)
            
            # Extract targets
            targets = await self._extract_drug_targets(drug_elem)
            
            # Build relationships
            relationships = []
            
            # Add indication relationships
            if indication:
                indication_conditions = self._extract_conditions_from_text(indication)
                for condition in indication_conditions:
                    relationships.append({
                        'relationship_type': 'treats',
                        'related_entity_name': condition,
                        'related_entity_type': 'Condition',
                        'evidence_strength': 0.9,
                        'source': 'drugbank_indication'
                    })
            
            # Add interaction relationships
            for interaction in interactions:
                relationships.append({
                    'relationship_type': 'interacts_with',
                    'related_entity_name': interaction['drug_name'],
                    'related_entity_type': 'Drug',
                    'evidence_strength': 0.8,
                    'source': 'drugbank_interaction',
                    'interaction_details': interaction['description']
                })
            
            # Add target relationships
            for target in targets:
                relationships.append({
                    'relationship_type': 'targets',
                    'related_entity_name': target['name'],
                    'related_entity_type': 'Protein',
                    'evidence_strength': 0.95,
                    'source': 'drugbank_target',
                    'target_details': target
                })
            
            # Create metadata
            metadata = {
                'drugbank_id': drug_id,
                'drug_type': drug_type,
                'groups': groups,
                'synonyms': list(set(synonyms)),
                'indication': indication,
                'pharmacodynamics': pharmacodynamics,
                'molecular_formula': self._extract_molecular_formula(drug_elem),
                'molecular_weight': self._extract_molecular_weight(drug_elem)
            }
            
            return KnowledgeEntity(
                entity_id=f"drugbank:{drug_id}",
                entity_name=normalized_name,
                entity_type='Drug',
                source_kg='DrugBank',
                relationships=relationships,
                metadata=metadata,
                source_link=f"https://go.drugbank.com/drugs/{drug_id}",
                confidence_score=0.95
            )
        
        except Exception as e:
            logger.error(f"Error parsing drug element: {str(e)}")
            return None
    
    async def _extract_drug_interactions(self, drug_elem) -> List[Dict[str, Any]]:
        """Extract drug-drug interactions."""
        interactions = []
        
        interactions_elem = drug_elem.find('.//{http://www.drugbank.ca}drug-interactions')
        if interactions_elem is not None:
            for interaction in interactions_elem.findall('.//{http://www.drugbank.ca}drug-interaction'):
                drug_name_elem = interaction.find('.//{http://www.drugbank.ca}name')
                description_elem = interaction.find('.//{http://www.drugbank.ca}description')
                
                if drug_name_elem is not None and drug_name_elem.text:
                    interactions.append({
                        'drug_name': drug_name_elem.text.strip(),
                        'description': description_elem.text if description_elem is not None else ""
                    })
        
        return interactions
    
    async def _extract_drug_targets(self, drug_elem) -> List[Dict[str, Any]]:
        """Extract drug targets."""
        targets = []
        
        targets_elem = drug_elem.find('.//{http://www.drugbank.ca}targets')
        if targets_elem is not None:
            for target in targets_elem.findall('.//{http://www.drugbank.ca}target'):
                name_elem = target.find('.//{http://www.drugbank.ca}name')
                organism_elem = target.find('.//{http://www.drugbank.ca}organism')
                
                if name_elem is not None and name_elem.text:
                    targets.append({
                        'name': name_elem.text.strip(),
                        'organism': organism_elem.text if organism_elem is not None else "Human"
                    })
        
        return targets
    
    def _extract_conditions_from_text(self, text: str) -> List[str]:
        """Extract medical conditions from indication text using NLP."""
        doc = self.normalizer.nlp(text)
        conditions = []
        
        for ent in doc.ents:
            if ent.label_ in ['DISEASE', 'CONDITION']:
                normalized_name, _ = self.normalizer.normalize_entity_name(ent.text)
                conditions.append(normalized_name)
        
        return list(set(conditions))
    
    def _extract_molecular_formula(self, drug_elem) -> Optional[str]:
        """Extract molecular formula."""
        formula_elem = drug_elem.find('.//{http://www.drugbank.ca}molecular-formula')
        return formula_elem.text if formula_elem is not None else None
    
    def _extract_molecular_weight(self, drug_elem) -> Optional[float]:
        """Extract molecular weight."""
        weight_elem = drug_elem.find('.//{http://www.drugbank.ca}molecular-weight')
        if weight_elem is not None and weight_elem.text:
            try:
                return float(weight_elem.text)
            except ValueError:
                pass
        return None

class ADHDKnowledgeParser:
    """Parser for ADHD-specific knowledge graphs in RDF/N3 format."""
    
    def __init__(self, adhd_kg_file: Path):
        self.adhd_kg_file = adhd_kg_file
        self.normalizer = EntityNormalizer()
        self.graph = rdflib.Graph()
    
    async def parse_adhd_kg(self) -> List[KnowledgeEntity]:
        """Parse ADHD knowledge graph from RDF/N3 file."""
        entities = []
        
        try:
            # Load RDF graph
            self.graph.parse(self.adhd_kg_file, format='n3')
            logger.info(f"Loaded {len(self.graph)} triples from ADHD KG")
            
            # Extract entities by type
            entity_types = ['Condition', 'Symptom', 'Treatment', 'Medication']
            
            for entity_type in entity_types:
                type_entities = await self._extract_entities_by_type(entity_type)
                entities.extend(type_entities)
        
        except Exception as e:
            logger.error(f"Error parsing ADHD KG file: {str(e)}")
        
        return entities
    
    async def _extract_entities_by_type(self, entity_type: str) -> List[KnowledgeEntity]:
        """Extract entities of specific type from RDF graph."""
        entities = []
        
        # Define SPARQL query for entity extraction
        query = f"""
        SELECT DISTINCT ?entity ?label ?description WHERE {{
            ?entity rdf:type <http://example.org/adhd#{entity_type}> .
            OPTIONAL {{ ?entity rdfs:label ?label }}
            OPTIONAL {{ ?entity rdfs:comment ?description }}
        }}
        """
        
        try:
            results = self.graph.query(query)
            
            for row in results:
                entity_uri = str(row.entity)
                label = str(row.label) if row.label else ""
                description = str(row.description) if row.description else ""
                
                if not label:
                    # Extract label from URI
                    label = entity_uri.split('#')[-1].split('/')[-1].replace('_', ' ')
                
                if not label:
                    continue
                
                # Normalize entity name
                normalized_name, synonyms = self.normalizer.normalize_entity_name(label)
                
                # Extract relationships for this entity
                relationships = await self._extract_entity_relationships(entity_uri)
                
                # Create metadata
                metadata = {
                    'uri': entity_uri,
                    'description': description,
                    'synonyms': synonyms,
                    'adhd_specific': True
                }
                
                entity = KnowledgeEntity(
                    entity_id=f"adhd_kg:{entity_uri.split('#')[-1]}",
                    entity_name=normalized_name,
                    entity_type=entity_type,
                    source_kg='ADHD_KG',
                    relationships=relationships,
                    metadata=metadata,
                    source_link=entity_uri,
                    confidence_score=0.85
                )
                
                entities.append(entity)
        
        except Exception as e:
            logger.error(f"Error extracting {entity_type} entities: {str(e)}")
        
        return entities
    
    async def _extract_entity_relationships(self, entity_uri: str) -> List[Dict[str, Any]]:
        """Extract relationships for specific entity."""
        relationships = []
        
        # Query for outgoing relationships
        outgoing_query = f"""
        SELECT ?predicate ?object ?objectLabel WHERE {{
            <{entity_uri}> ?predicate ?object .
            OPTIONAL {{ ?object rdfs:label ?objectLabel }}
            FILTER(?predicate != rdf:type && ?predicate != rdfs:label && ?predicate != rdfs:comment)
        }}
        """
        
        try:
            results = self.graph.query(outgoing_query)
            
            for row in results:
                predicate = str(row.predicate)
                object_uri = str(row.object)
                object_label = str(row.objectLabel) if row.objectLabel else ""
                
                # Extract relationship type from predicate URI
                relationship_type = predicate.split('#')[-1].split('/')[-1]
                
                # Extract object name
                if object_label:
                    object_name = object_label
                else:
                    object_name = object_uri.split('#')[-1].split('/')[-1].replace('_', ' ')
                
                # Normalize object name
                normalized_object_name, _ = self.normalizer.normalize_entity_name(object_name)
                
                # Determine object type
                object_type = self.normalizer.classify_entity_type(normalized_object_name)
                
                relationships.append({
                    'relationship_type': relationship_type,
                    'related_entity_name': normalized_object_name,
                    'related_entity_type': object_type,
                    'evidence_strength': 0.8,
                    'source': 'adhd_kg_triple'
                })
        
        except Exception as e:
            logger.error(f"Error extracting relationships for {entity_uri}: {str(e)}")
        
        return relationships

class SuperGraphIngestionEngine:
    """Main orchestration engine for multi-source knowledge ingestion."""
    
    def __init__(self, db_session: Session, max_workers: int = 4):
        self.db_session = db_session
        self.max_workers = max_workers
        self.normalizer = EntityNormalizer()
        self.stats = IngestionStats()
        
        # Entity deduplication cache
        self.entity_fingerprints: Set[str] = set()
        self.existing_entities: Dict[str, str] = {}  # entity_id -> content_hash
    
    async def ingest_all_sources(self, config: Dict[str, Any]) -> IngestionStats:
        """Orchestrate ingestion from all configured knowledge sources."""
        start_time = datetime.utcnow()
        
        try:
            # Load existing entity fingerprints for deduplication
            await self._load_existing_entities()
            
            # Process each configured source
            tasks = []
            
            if config.get('pubmed'):
                tasks.append(self._ingest_pubmed(config['pubmed']))
            
            if config.get('drugbank'):
                tasks.append(self._ingest_drugbank(config['drugbank']))
            
            if config.get('adhd_kg'):
                tasks.append(self._ingest_adhd_kg(config['adhd_kg']))
            
            # Execute ingestion tasks concurrently
            results = await asyncio.gather(*tasks, return_exceptions=True)
            
            # Process results and update statistics
            for i, result in enumerate(results):
                if isinstance(result, Exception):
                    logger.error(f"Ingestion task {i} failed: {str(result)}")
                    self.stats.errors_encountered += 1
                else:
                    logger.info(f"Ingestion task {i} completed successfully")
            
            # Finalize ingestion process
            await self._finalize_ingestion()
            
        except Exception as e:
            logger.error(f"Critical error during ingestion: {str(e)}")
            self.stats.errors_encountered += 1
        
        finally:
            end_time = datetime.utcnow()
            self.stats.processing_time_seconds = (end_time - start_time).total_seconds()
        
        return self.stats
    
    async def _ingest_pubmed(self, config: Dict[str, Any]) -> None:
        """Ingest knowledge from PubMed."""
        logger.info("Starting PubMed ingestion")
        self.stats.sources_processed.append('PubMed')
        
        queries = config.get('queries', [])
        max_results_per_query = config.get('max_results_per_query', 100)
        api_key = config.get('api_key')
        
        async with PubMedKGParser(api_key) as parser:
            for query in queries:
                try:
                    # Search for relevant PMIDs
                    pmids = await parser.search_pubmed(query, max_results_per_query)
                    
                    if not pmids:
                        continue
                    
                    # Fetch article details in batches
                    batch_size = 20  # PubMed API limitation
                    for i in range(0, len(pmids), batch_size):
                        batch_pmids = pmids[i:i + batch_size]
                        articles = await parser.fetch_pubmed_details(batch_pmids)
                        
                        # Process articles into knowledge entities
                        entities = await self._process_pubmed_articles(articles)
                        
                        # Store entities in database
                        await self._store_entities(entities)
                
                except Exception as e:
                    logger.error(f"Error processing PubMed query '{query}': {str(e)}")
                    self.stats.errors_encountered += 1
    
    async def _ingest_drugbank(self, config: Dict[str, Any]) -> None:
        """Ingest knowledge from DrugBank."""
        logger.info("Starting DrugBank ingestion")
        self.stats.sources_processed.append('DrugBank')
        
        drugbank_file = Path(config['file_path'])
        
        if not drugbank_file.exists():
            logger.error(f"DrugBank file not found: {drugbank_file}")
            self.stats.errors_encountered += 1
            return
        
        try:
            parser = DrugBankParser(drugbank_file)
            entities = await parser.parse_drugbank()
            
            logger.info(f"Extracted {len(entities)} entities from DrugBank")
            await self._store_entities(entities)
        
        except Exception as e:
            logger.error(f"Error processing DrugBank: {str(e)}")
            self.stats.errors_encountered += 1
    
    async def _ingest_adhd_kg(self, config: Dict[str, Any]) -> None:
        """Ingest knowledge from ADHD knowledge graph."""
        logger.info("Starting ADHD KG ingestion")
        self.stats.sources_processed.append('ADHD_KG')
        
        adhd_kg_file = Path(config['file_path'])
        
        if not adhd_kg_file.exists():
            logger.error(f"ADHD KG file not found: {adhd_kg_file}")
            self.stats.errors_encountered += 1
            return
        
        try:
            parser = ADHDKnowledgeParser(adhd_kg_file)
            entities = await parser.parse_adhd_kg()
            
            logger.info(f"Extracted {len(entities)} entities from ADHD KG")
            await self._store_entities(entities)
        
        except Exception as e:
            logger.error(f"Error processing ADHD KG: {str(e)}")
            self.stats.errors_encountered += 1
    
    async def _process_pubmed_articles(self, articles: List[Dict[str, Any]]) -> List[KnowledgeEntity]:
        """Process PubMed articles into knowledge entities."""
        entities = []
        
        for article in articles:
            try:
                # Extract entities from title and abstract
                text = f"{article.get('title', '')} {article.get('abstract', '')}"
                
                if not text.strip():
                    continue
                
                # Use NLP to extract medical entities
                doc = self.normalizer.nlp(text)
                
                extracted_entities = {}
                for ent in doc.ents:
                    if ent.label_ in self.normalizer.entity_type_mappings:
                        entity_type = self.normalizer.entity_type_mappings[ent.label_]
                        normalized_name, synonyms = self.normalizer.normalize_entity_name(ent.text)
                        
                        if normalized_name not in extracted_entities:
                            extracted_entities[normalized_name] = {
                                'type': entity_type,
                                'synonyms': synonyms,
                                'mentions': 0
                            }
                        extracted_entities[normalized_name]['mentions'] += 1
                
                # Create knowledge entities for frequent mentions
                for entity_name, entity_data in extracted_entities.items():
                    if entity_data['mentions'] >= 2:  # Minimum mention threshold
                        entity = KnowledgeEntity(
                            entity_id=f"pubmed:{article['pmid']}:{entity_name.replace(' ', '_')}",
                            entity_name=entity_name,
                            entity_type=entity_data['type'],
                            source_kg='PubMed',
                            relationships=[],  # Would need more sophisticated relationship extraction
                            metadata={
                                'pmid': article['pmid'],
                                'title': article['title'],
                                'publication_date': article.get('publication_date'),
                                'authors': article.get('authors', []),
                                'mesh_terms': article.get('mesh_terms', []),
                                'synonyms': entity_data['synonyms'],
                                'mention_count': entity_data['mentions']
                            },
                            source_link=article['source_link'],
                            confidence_score=min(0.9, 0.5 + (entity_data['mentions'] * 0.1))
                        )
                        entities.append(entity)
            
            except Exception as e:
                logger.warning(f"Error processing PubMed article {article.get('pmid', 'unknown')}: {str(e)}")
                continue
        
        return entities
    
    async def _store_entities(self, entities: List[KnowledgeEntity]) -> None:
        """Store knowledge entities in database with deduplication."""
        for entity in entities:
            try:
                # Generate content fingerprint for deduplication
                content_hash = self._generate_entity_fingerprint(entity)
                
                if content_hash in self.entity_fingerprints:
                    self.stats.entities_skipped += 1
                    continue
                
                # Check if entity already exists
                existing_entity_id = self.existing_entities.get(entity.entity_id)
                
                if existing_entity_id and existing_entity_id != content_hash:
                    # Update existing entity
                    await self._update_entity(entity, content_hash)
                    self.stats.entities_updated += 1
                else:
                    # Create new entity
                    await self._create_entity(entity, content_hash)
                    self.stats.entities_created += 1
                
                self.entity_fingerprints.add(content_hash)
                self.stats.entities_processed += 1
            
            except Exception as e:
                logger.error(f"Error storing entity {entity.entity_id}: {str(e)}")
                self.stats.errors_encountered += 1
    
    def _generate_entity_fingerprint(self, entity: KnowledgeEntity) -> str:
        """Generate SHA-256 fingerprint for entity deduplication."""
        content = f"{entity.entity_name}|{entity.entity_type}|{entity.source_kg}"
        content += "|" + json.dumps(sorted(entity.metadata.items()), default=str)
        
        return hashlib.sha256(content.encode('utf-8')).hexdigest()

    def get_node_subgraph(self, node_id: str, depth: int = 2) -> Dict[str, Any]:
        """
        Extract subgraph around specific node for focused exploration.
        
        Args:
            node_id: Central node ID
            depth: Traversal depth (default: 2)
        
        Returns:
            Dict containing subgraph data
        """
        if node_id not in self.nodes:
            raise ValueError(f"Node {node_id} not found")
        
        # Get nodes within specified depth
        subgraph_nodes = set([node_id])
        current_nodes = set([node_id])
        
        for _ in range(depth):
            next_nodes = set()
            for current_node in current_nodes:
                # Add predecessors and successors
                next_nodes.update(self.graph.predecessors(current_node))
                next_nodes.update(self.graph.successors(current_node))
            
            current_nodes = next_nodes - subgraph_nodes
            subgraph_nodes.update(current_nodes)
        
        # Extract subgraph
        subgraph = self.graph.subgraph(subgraph_nodes).copy()
        
        # Format for output
        return {
            'central_node': node_id,
            'depth': depth,
            'nodes': [self.nodes[nid].to_dict() for nid in subgraph_nodes if nid in self.nodes],
            'relationships': [
                self.relationships[edge_data['relationship_id']].to_dict()
                for _, _, edge_data in subgraph.edges(data=True)
                if edge_data['relationship_id'] in self.relationships
            ],
            'node_count': len(subgraph_nodes),
            'relationship_count': subgraph.number_of_edges()
        }

    def _calculate_graph_statistics(self) -> Dict[str, Any]:
        """Calculate descriptive statistics for the belief graph."""
        if not self.graph.nodes():
            return {'empty_graph': True}
        
        # Basic counts
        node_count = len(self.graph.nodes())
        edge_count = len(self.graph.edges())
        
        # Node type distribution
        node_type_dist = {}
        for node in self.nodes.values():
            node_type = node.node_type.value
            node_type_dist[node_type] = node_type_dist.get(node_type, 0) + 1
        
        # Relationship type distribution
        rel_type_dist = {}
        for rel in self.relationships.values():
            rel_type = rel.relationship_type.value
            rel_type_dist[rel_type] = rel_type_dist.get(rel_type, 0) + 1
        
        # Graph connectivity
        if node_count > 0:
            density = edge_count / (node_count * (node_count - 1)) if node_count > 1 else 0
            avg_degree = sum(dict(self.graph.degree()).values()) / node_count
        else:
            density = 0

class DissonanceDetectionEngine:
    """
    Advanced engine for detecting cognitive dissonance patterns in belief graphs.
    Implements sophisticated pattern matching algorithms.
    """
    
    def __init__(self, graph_manager: SelfGraphManager):
        self.graph_manager = graph_manager
        self.detection_algorithms = {
            DissonanceType.LOGICAL_CONTRADICTION: self._detect_logical_contradictions,
            DissonanceType.CAUSAL_ATTRIBUTION_ERROR: self._detect_causal_attribution_errors,
            DissonanceType.FACTUAL_INCONSISTENCY: self._detect_factual_inconsistencies,
            DissonanceType.VALUE_BEHAVIOR_MISMATCH: self._detect_value_behavior_mismatches,
            DissonanceType.GOAL_OBSTACLE_CONFLICT: self._detect_goal_obstacle_conflicts,
            DissonanceType.IDENTITY_EVIDENCE_DISCORD: self._detect_identity_evidence_discord
        }
    
    async def detect_all_dissonance_patterns(self) -> List[DissonancePattern]:
        """
        Run comprehensive dissonance detection across all pattern types.
        
        Returns:
            List[DissonancePattern]: Detected dissonance patterns
        """
        all_patterns = []
        
        for dissonance_type, detection_method in self.detection_algorithms.items():
            try:
                patterns = await detection_method()
                all_patterns.extend(patterns)
                logger.info(f"Detected {len(patterns)} {dissonance_type.value} patterns")
            except Exception as e:
                logger.error(f"Error detecting {dissonance_type.value}: {str(e)}")
        
        # Sort by severity and confidence
        all_patterns.sort(key=lambda p: (p.severity_score * p.confidence_level), reverse=True)
        
        return all_patterns
    
    async def _detect_logical_contradictions(self) -> List[DissonancePattern]:
        """Detect direct logical contradictions in belief statements."""
        patterns = []
        graph = self.graph_manager.graph
        
        # Look for CONTRADICTS relationships
        for source, target, data in graph.edges(data=True):
            if data.get('relationship_type') == RelationshipType.CONTRADICTS.value:
                source_node = self.graph_manager.nodes[source]
                target_node = self.graph_manager.nodes[target]
                
                # Calculate severity based on certainty levels
                avg_certainty = (source_node.certainty_level + target_node.certainty_level) / 2
                severity = avg_certainty * data.get('strength', 0.5)
                
                pattern = DissonancePattern(
                    pattern_id=f"contradiction_{uuid.uuid4().hex[:8]}",
                    dissonance_type=DissonanceType.LOGICAL_CONTRADICTION,
                    involved_nodes=[source, target],
                    involved_relationships=[data['relationship_id']],
                    severity_score=severity,
                    confidence_level=0.9,  # High confidence for explicit contradictions
                    supporting_evidence=[
                        f"Direct contradiction between: '{source_node.content}' and '{target_node.content}'"
                    ],
                    potential_resolutions=[
                        {
                            'type': 'belief_revision',
                            'description': 'Examine evidence for both beliefs and revise the less supported one',
                            'target_nodes': [source, target]
                        },
                        {
                            'type': 'contextualization',
                            'description': 'Add contextual conditions to make beliefs compatible',
                            'target_nodes': [source, target]
                        }
                    ]
                )
                patterns.append(pattern)
        
        return patterns

    async def _detect_value_behavior_mismatches(self) -> List[DissonancePattern]:
        """Detect mismatches between stated values and behavioral patterns."""
        patterns = []
        graph = self.graph_manager.graph
        
        # Find value nodes
        value_nodes = [
            node_id for node_id, node in self.graph_manager.nodes.items()
            if node.node_type == BeliefNodeType.VALUE_SYSTEM
        ]
        
        # Find behavior nodes
        behavior_nodes = [
            node_id for node_id, node in self.graph_manager.nodes.items()
            if node.node_type == BeliefNodeType.BEHAVIORAL_PATTERN
        ]
        
        # Check for contradictory relationships between values and behaviors
        for value_node_id in value_nodes:
            for behavior_node_id in behavior_nodes:
                # Check if there's a contradictory relationship
                if graph.has_edge(value_node_id, behavior_node_id):
                    edge_data = graph.get_edge_data(value_node_id, behavior_node_id)
                    for key, data in edge_data.items():
                        if data.get('relationship_type') == RelationshipType.CONTRADICTS.value:
                            value_node = self.graph_manager.nodes[value_node_id]
                            behavior_node = self.graph_manager.nodes[behavior_node_id]
                            
                            severity = (value_node.certainty_level + abs(behavior_node.emotional_valence)) / 2
                            
                            pattern = DissonancePattern(
                                pattern_id=f"value_behavior_mismatch_{uuid.uuid4().hex[:8]}",
                                dissonance_type=DissonanceType.VALUE_BEHAVIOR_MISMATCH,
                                involved_nodes=[value_node_id, behavior_node_id],
                                involved_relationships=[data['relationship_id']],
                                severity_score=severity,
                                confidence_level=0.8,
                                supporting_evidence=[
                                    f"Value-behavior conflict: '{value_node.content}' vs '{behavior_node.content}'"
                                ],
                                potential_resolutions=[
                                    {
                                        'type': 'behavioral_alignment',
                                        'description': 'Develop action plan to align behavior with values',
                                        'target_nodes': [behavior_node_id]
                                    },
                                    {
                                        'type': 'value_clarification',
                                        'description': 'Examine and potentially revise value priorities',
                                        'target_nodes': [value_node_id]
                                    }
                                ]
                            )
                            patterns.append(pattern)
        
        return patterns
    
    async def _detect_goal_obstacle_conflicts(self) -> List[DissonancePattern]:
        """Detect conflicts between goals and perceived obstacles."""
        patterns = []
        graph = self.graph_manager.graph
        
        # Find goal nodes
        goal_nodes = [
            node_id for node_id, node in self.graph_manager.nodes.items()
            if node.node_type == BeliefNodeType.GOAL_STATE
        ]
        
        for goal_node_id in goal_nodes:
            goal_node = self.graph_manager.nodes[goal_node_id]
            
            # Find nodes that inhibit this goal
            inhibiting_nodes = []
            for pred in graph.predecessors(goal_node_id):
                edge_data = graph.get_edge_data(pred, goal_node_id)
                for key, data in edge_data.items():
                    if data.get('relationship_type') == RelationshipType.INHIBITS.value:
                        inhibiting_nodes.append((pred, data['relationship_id']))
            
            # If goal has high importance but strong inhibitors, create dissonance pattern
            if inhibiting_nodes and goal_node.certainty_level > 0.7:
                total_inhibition = sum([
                    graph.get_edge_data(node_id, goal_node_id)[rel_id]['strength']
                    for node_id, rel_id in inhibiting_nodes
                ])
                
                if total_inhibition > 0.6:  # Strong inhibition threshold
                    severity = goal_node.certainty_level * (total_inhibition / len(inhibiting_nodes))
                    
                    pattern = DissonancePattern(
                        pattern_id=f"goal_obstacle_{uuid.uuid4().hex[:8]}",
                        dissonance_type=DissonanceType.GOAL_OBSTACLE_CONFLICT,
                        involved_nodes=[goal_node_id] + [node_id for node_id, _ in inhibiting_nodes],
                        involved_relationships=[rel_id for _, rel_id in inhibiting_nodes],
                        severity_score=severity,
                        confidence_level=0.75,
                        supporting_evidence=[
                            f"Goal '{goal_node.content}' strongly inhibited by {len(inhibiting_nodes)} factors"
                        ],
                        potential_resolutions=[
                            {
                                'type': 'obstacle_analysis',
                                'description': 'Systematically analyze and address each obstacle',
                                'target_nodes': [node_id for node_id, _ in inhibiting_nodes]
                            },
                            {
                                'type': 'goal_modification',
                                'description': 'Adjust goal to be more achievable given constraints',
                                'target_nodes': [goal_node_id]
                            }
                        ]
                    )
                    patterns.append(pattern)
        
        return patterns
    
    async def _detect_identity_evidence_discord(self) -> List[DissonancePattern]:
        """Detect mismatches between identity beliefs and supporting evidence."""
        patterns = []
        graph = self.graph_manager.graph
        
        # Find identity component nodes
        identity_nodes = [
            node_id for node_id, node in self.graph_manager.nodes.items()
            if node.node_type == BeliefNodeType.IDENTITY_COMPONENT
        ]
        
        for identity_node_id in identity_nodes:
            identity_node = self.graph_manager.nodes[identity_node_id]
            
            # Find evidence nodes (nodes that support or contradict this identity)
            supporting_evidence = []
            contradicting_evidence = []
            
            for pred in graph.predecessors(identity_node_id):
                edge_data = graph.get_edge_data(pred, identity_node_id)
                for key, data in edge_data.items():
                    if data.get('relationship_type') == RelationshipType.SUPPORTS.value:
                        supporting_evidence.append((pred, data['strength']))
                    elif data.get('relationship_type') == RelationshipType.CONTRADICTS.value:
                        contradicting_evidence.append((pred, data['strength']))
            
            # Calculate evidence balance
            total_support = sum([strength for _, strength in supporting_evidence])
            total_contradiction = sum([strength for _, strength in contradicting_evidence])
            
            # If identity has high certainty but contradictory evidence is strong
            if (identity_node.certainty_level > 0.7 and 
                total_contradiction > total_support and 
                total_contradiction > 0.5):
                
                severity = identity_node.certainty_level * (total_contradiction / (total_support + total_contradiction + 0.1))
                
                pattern = DissonancePattern(
                    pattern_id=f"identity_evidence_discord_{uuid.uuid4().hex[:8]}",
                    dissonance_type=DissonanceType.IDENTITY_EVIDENCE_DISCORD,
                    involved_nodes=[identity_node_id] + [node_id for node_id, _ in contradicting_evidence],
                    involved_relationships=[],  # Would need to extract relationship IDs
                    severity_score=severity,
                    confidence_level=0.8,
                    supporting_evidence=[
                        f"Identity belief '{identity_node.content}' contradicted by evidence (ratio: {total_contradiction:.2f}/{total_support:.2f})"
                    ],
                    potential_resolutions=[
                        {
                            'type': 'evidence_review',
                            'description': 'Systematically review contradictory evidence',
                            'target_nodes': [node_id for node_id, _ in contradicting_evidence]
                        },
                        {
                            'type': 'identity_adjustment',
                            'description': 'Consider adjusting identity belief to match evidence',
                            'target_nodes': [identity_node_id]
                        }
                    ]
                )
                patterns.append(pattern)
        
        return patterns

class InterventionOrchestrator:
    """
    Orchestrates therapeutic interventions based on detected dissonance patterns.
    Implements the Navigator component of the Cognitive Light-Cone architecture.
    """
    
    def __init__(self, graph_manager: SelfGraphManager, external_knowledge_session: Session):
        self.graph_manager = graph_manager
        self.external_knowledge_session = external_knowledge_session
        self.intervention_strategies = {
            DissonanceType.LOGICAL_CONTRADICTION: self._handle_logical_contradiction,
            DissonanceType.CAUSAL_ATTRIBUTION_ERROR: self._handle_causal_attribution_error,
            DissonanceType.FACTUAL_INCONSISTENCY: self._handle_factual_inconsistency,
            DissonanceType.VALUE_BEHAVIOR_MISMATCH: self._handle_value_behavior_mismatch,
            DissonanceType.GOAL_OBSTACLE_CONFLICT: self._handle_goal_obstacle_conflict,
            DissonanceType.IDENTITY_EVIDENCE_DISCORD: self._handle_identity_evidence_discord
        }
        
        # Load Jinja2 environment for prompt templates
        self.jinja_env = Environment(loader=FileSystemLoader("templates/interventions"))
    
    async def generate_intervention(self, dissonance_pattern: DissonancePattern, 
                                  user_input: str) -> Dict[str, Any]:
        """
        Generate therapeutic intervention for detected dissonance pattern.
        
        Args:
            dissonance_pattern: Detected dissonance requiring intervention
            user_input: Current user input/context
        
        Returns:
            Dict containing intervention strategy and content
        """
        try:
            # Select appropriate intervention strategy
            handler = self.intervention_strategies.get(dissonance_pattern.dissonance_type)
            if not handler:
                raise ValueError(f"No handler for dissonance type: {dissonance_pattern.dissonance_type}")
            
            # Generate intervention
            intervention = await handler(dissonance_pattern, user_input)
            
            # Add metadata
            intervention.update({
                'dissonance_pattern_id': dissonance_pattern.pattern_id,
                'intervention_timestamp': datetime.utcnow().isoformat(),
                'user_id': self.graph_manager.user_id,
                'requires_user_consent': True,
                'estimated_session_time': self._estimate_intervention_time(intervention),
                'safety_considerations': self._assess_intervention_safety(dissonance_pattern)
            })
            
            return intervention
        
        except Exception as e:
            logger.error(f"Error generating intervention for pattern {dissonance_pattern.pattern_id}: {str(e)}")
            return self._generate_fallback_intervention(dissonance_pattern, user_input)
    
    async def _handle_logical_contradiction(self, pattern: DissonancePattern, 
                                          user_input: str) -> Dict[str, Any]:

            [node.content for node in involved_nodes]
        )

        template = self.jinja_env.get_template("socratic_inquiry.txt")
        prompt_content = template.render(
            user_input=user_input,

        )
        
        return {
            'intervention_type': 'socratic_inquiry',

ENCRYPTION_CONFIG = """
# Encryption Configuration for Sensitive Data

import os
import base64
from cryptography.fernet import Fernet
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
from typing import Union
import logging

logger = logging.getLogger(__name__)

class EncryptionManager:
    \"\"\"
    Advanced encryption manager for protecting sensitive therapeutic data.
    Implements AES-256 encryption with key derivation and rotation support.
    \"\"\"
    
    def __init__(self, master_key: str):
        self.master_key = master_key.encode()
        self._derived_keys = {}
        self._current_key_version = 1
    
    def _derive_key(self, salt: bytes, version: int = None) -> bytes:
        \"\"\"Derive encryption key using PBKDF2.\"\"\"
        if version is None:
            version = self._current_key_version
        
        kdf = PBKDF2HMAC(
            algorithm=hashes.SHA256(),
            length=32,
            salt=salt,
            iterations=100000,  # NIST recommended minimum
        )
        return base64.urlsafe_b64encode(kdf.derive(self.master_key + str(version).encode()))
    
    def encrypt_therapeutic_data(self, data: Union[str, dict], user_id: str) -> dict:
        \"\"\"
        Encrypt therapeutic data with user-specific salt.
        
        Args:
            data: Data to encrypt (belief nodes, session content, etc.)
            user_id: User identifier for salt generation
        
        Returns:
            dict: Encrypted data with metadata
        \"\"\"

            salt = self._generate_user_salt(user_id)
            
            # Derive encryption key
            key = self._derive_key(salt)
            fernet = Fernet(key)
            
            # Serialize and encrypt data
            if isinstance(data, dict):
                import json
                serialized_data = json.dumps(data, sort_keys=True)
            else:
                serialized_data = str(data)
            
            encrypted_data = fernet.encrypt(serialized_data.encode())
            
            return {
                'encrypted_data': base64.b64encode(encrypted_data).decode(),
                'salt': base64.b64encode(salt).decode(),
                'key_version': self._current_key_version,
                'encryption_algorithm': 'AES-256-GCM',
                'timestamp': datetime.utcnow().isoformat()
            }

            raise

        \"\"\"
        Decrypt therapeutic data.
        
        Args:
            encrypted_package: Encrypted data package
            user_id: User identifier for salt verification
        
        Returns:
            Union[str, dict]: Decrypted original data

            decrypted_bytes = fernet.decrypt(encrypted_data)
            decrypted_str = decrypted_bytes.decode()

def generate_deployment_files(output_dir: str = "deployment"):
    """Generate all deployment configuration files."""
    import os
    
    os.makedirs(output_dir, exist_ok=True)
    
    # Write each configuration to appropriate file
    with open(f"{output_dir}/Dockerfile", "w") as f:
        f.write(DOCKERFILE_CONTENT)
    
    with open(f"{output_dir}/docker-compose.yml", "w") as f:
        f.write(DOCKER_COMPOSE_CONTENT)
    
    # Create kubernetes directory
    k8s_dir = f"{output_dir}/kubernetes"
    os.makedirs(k8s_dir, exist_ok=True)
    
    for name, content in CONFIG_EXPORT['kubernetes'].items():
        with open(f"{k8s_dir}/{name}.yaml", "w") as f:
            f.write(content)
    
    # Create other directories and files
    for category, content in CONFIG_EXPORT.items():
        if category not in ['kubernetes']:
            if isinstance(content, dict):
                cat_dir = f"{output_dir}/{category}"
                os.makedirs(cat_dir, exist_ok=True)
                for name, file_content in content.items():
                    with open(f"{cat_dir}/{name}", "w") as f:
                        f.write(file_content)
            else:
                with open(f"{output_dir}/{category}", "w") as f:
                    f.write(content)
    
    print(f"Deployment files generated in {output_dir}/")
